לפרק הקודם * * * תוכן העניינים

איוב פרק ד

א                 וַיַּעַן אֱלִיפַז הַתֵּימָנִי וַיֹּאמַר׃   
ב הֲנִסָּה דָבָר אֵלֶיךָ תִּלְאֶה    וַעְצֹר בְּמִלִּין מִי יוּכָל׃   
ג הִנֵּה יִסַּרְתָּ רַבִּים    וְיָדַיִם רָפוֹת תְּחַזֵּק׃   
ד כּוֹשֵׁל יְקִימוּן מִלֶּיךָ    וּבִרְכַּיִם כֹּרְעוֹת תְּאַמֵּץ׃   
ה כִּי עַתָּה תָּבוֹא אֵלֶיךָ וַתֵּלֶא    תִּגַּע עָדֶיךָ וַתִּבָּהֵל׃   
ו הֲלֹא יִרְאָתְךָ כִּסְלָתֶךָ    תִּקְוָתְךָ וְתֹם דְּרָכֶיךָ׃   
ז זְכָר-נָא מִי הוּא נָקִי אָבָד    וְאֵיפֹה יְשָׁרִים נִכְחָדוּ׃   
ח כַּאֲשֶׁר רָאִיתִי חֹרְשֵׁי אָוֶן    וְזֹרְעֵי עָמָל יִקְצְרֻהוּ׃   
ט מִנִּשְׁמַת אֱלוֹהַּ יֹאבֵדוּ    וּמֵרוּחַ אַפּוֹ יִכְלוּ׃   
י שַׁאֲגַת אַרְיֵה וְקוֹל שָׁחַל    וְשִׁנֵּי כְפִירִים נִתָּעוּ׃   
יא לַיִשׁ אֹבֵד מִבְּלִי-טָרֶף    וּבְנֵי לָבִיא יִתְפָּרָדוּ׃   
יב וְאֵלַי דָּבָר יְגֻנָּב    וַתִּקַּח אָזְנִי שֵׁמֶץ מֶנְהוּ׃   
יג בִּשְׂעִפִּים מֵחֶזְיֹנוֹת לָיְלָה    בִּנְפֹל תַּרְדֵּמָה עַל-אֲנָשִׁים׃   
יד פַּחַד קְרָאַנִי וּרְעָדָה    וְרֹב עַצְמוֹתַי הִפְחִיד׃   
טו וְרוּחַ עַל-פָּנַי יַחֲלֹף    תְּסַמֵּר שַׂעֲרַת בְּשָׂרִי׃   
טז יַעֲמֹד וְלֹא אַכִּיר מַרְאֵהוּ     תְּמוּנָה לְנֶגֶד עֵינָי
דְּמָמָה וָקוֹל    אֶשְׁמָע׃   
יז הַאֱנוֹשׁ מֵאֱלוֹהַּ יִצְדָּק    אִם מֵעֹשֵׂהוּ יִטְהַר-גָּבֶר׃   
יח הֵן בַּעֲבָדָיו לֹא יַאֲמִין    וּבְמַלְאָכָיו יָשִׂים תָּהֳלָה׃   
יט אַף שֹׁכְנֵי בָתֵּי-חֹמֶר אֲשֶׁר-בֶּעָפָר יְסוֹדָם    יְדַכְּאוּם לִפְנֵי-עָשׁ׃   
כ מִבֹּקֶר לָעֶרֶב יֻכַּתּוּ    מִבְּלִי מֵשִׂים לָנֶצַח יֹאבֵדוּ׃   
כא הֲלֹא-נִסַּע יִתְרָם בָּם    יָמוּתוּ וְלֹא בְחָכְמָה׃   

לפרק הבא * * * תוכן העניינים
