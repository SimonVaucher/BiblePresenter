לפרק הקודם * * * תוכן העניינים

איוב פרק לג

א וְאוּלָם שְׁמַע-נָא אִיּוֹב מִלָּי    וְכָל-דְּבָרַי הַאֲזִינָה׃   
ב הִנֵּה-נָא פָּתַחְתִּי פִי    דִּבְּרָה לְשׁוֹנִי בְחִכִּי׃   
ג יֹשֶׁר-לִבִּי אֲמָרָי    וְדַעַת שְׂפָתַי בָּרוּר מִלֵּלוּ׃   
ד רוּחַ-אֵל עָשָׂתְנִי    וְנִשְׁמַת שַׁדַּי תְּחַיֵּנִי׃   
ה אִם-תּוּכַל הֲשִׁיבֵנִי    עֶרְכָה לְפָנַי הִתְיַצָּבָה׃   
ו הֵן-אֲנִי כְפִיךָ לָאֵל    מֵחֹמֶר קֹרַצְתִּי גַם-אָנִי׃   
ז הִנֵּה אֵמָתִי לֹא תְבַעֲתֶךָּ    וְאַכְפִּי עָלֶיךָ לֹא-יִכְבָּד׃   
ח אַךְ אָמַרְתָּ בְאָזְנָי    וְקוֹל מִלִּין אֶשְׁמָע׃   
ט זַךְ אֲנִי בְּלִי-פָשַׁע    חַף אָנֹכִי וְלֹא עָוֹן לִי׃   
י הֵן תְּנוּאוֹת עָלַי יִמְצָא    יַחְשְׁבֵנִי לְאוֹיֵב לוֹ׃   
יא יָשֵׂם בַּסַּד רַגְלָי    יִשְׁמֹר כָּל-אָרְחֹתָי׃   
יב הֶן-זֹאת לֹא-צָדַקְתָּ אֶעֱנֶךָּ    כִּי-יִרְבֶּה אֱלוֹהַּ מֵאֱנוֹשׁ׃   
יג מַדּוּעַ אֵלָיו רִיבוֹתָ    כִּי כָל-דְּבָרָיו לֹא יַעֲנֶה׃   
יד כִּי-בְאַחַת יְדַבֶּר-אֵל    וּבִשְׁתַּיִם לֹא יְשׁוּרֶנָּה׃   
טו בַּחֲלוֹם חֶזְיוֹן לַיְלָה בִּנְפֹל תַּרְדֵּמָה עַל-אֲנָשִׁים    בִּתְנוּמוֹת עֲלֵי מִשְׁכָּב׃   
טז אָז יִגְלֶה אֹזֶן אֲנָשִׁים    וּבְמֹסָרָם יַחְתֹּם׃   
יז לְהָסִיר אָדָם מַעֲשֶׂה    וְגֵוָה מִגֶּבֶר יְכַסֶּה׃   
יח יַחְשֹׂךְ נַפְשׁוֹ מִנִּי-שָׁחַת    וְחַיָּתוֹ מֵעֲבֹר בַּשָּׁלַח׃   
יט וְהוּכַח בְּמַכְאוֹב עַל-מִשְׁכָּבוֹ    וריב (וְרוֹב) עֲצָמָיו אֵתָן׃   
כ וְזִהֲמַתּוּ חַיָּתוֹ לָחֶם    וְנַפְשׁוֹ מַאֲכַל תַּאֲוָה׃   
כא יִכֶל בְּשָׂרוֹ מֵרֹאִי    ושפי (וְשֻׁפּוּ) עַצְמֹתָיו לֹא רֻאּוּ׃   
כב וַתִּקְרַב לַשַּׁחַת נַפְשׁוֹ    וְחַיָּתוֹ לַמְמִתִים׃   
כג אִם-יֵשׁ עָלָיו מַלְאָךְ מֵלִיץ אֶחָד מִנִּי-אָלֶף    לְהַגִּיד לְאָדָם יָשְׁרוֹ׃   
כד וַיְחֻנֶּנּוּ וַיֹּאמֶר פְּדָעֵהוּ מֵרֶדֶת שָׁחַת    מָצָאתִי כֹפֶר׃   
כה רֻטְפַשׁ בְּשָׂרוֹ מִנֹּעַר    יָשׁוּב לִימֵי עֲלוּמָיו׃   
כו יֶעְתַּר אֶל-אֱלוֹהַּ וַיִּרְצֵהוּ    וַיַּרְא פָּנָיו בִּתְרוּעָה
וַיָּשֶׁב לֶאֱנוֹשׁ    צִדְקָתוֹ׃   
כז יָשֹׁר עַל-אֲנָשִׁים וַיֹּאמֶר חָטָאתִי וְיָשָׁר הֶעֱוֵיתִי    וְלֹא-שָׁוָה לִי׃   
כח פָּדָה נפשי (נַפְשׁוֹ) מֵעֲבֹר בַּשָּׁחַת    וחיתי (וְחַיָּתוֹ) בָּאוֹר תִּרְאֶה׃   
כט הֶן-כָּל-אֵלֶּה יִפְעַל-אֵל     פַּעֲמַיִם שָׁלוֹשׁ עִם-גָּבֶר׃   
ל לְהָשִׁיב נַפְשׁוֹ מִנִּי-שָׁחַת     לֵאוֹר בְּאוֹר הַחַיִּים׃   
לא הַקְשֵׁב אִיּוֹב שְׁמַע-לִי    הַחֲרֵשׁ וְאָנֹכִי אֲדַבֵּר׃   
לב אִם-יֵשׁ-מִלִּין הֲשִׁיבֵנִי    דַּבֵּר כִּי-חָפַצְתִּי צַדְּקֶךָּ׃   
לג אִם-אַיִן אַתָּה שְׁמַע-לִי    הַחֲרֵשׁ וַאֲאַלֶּפְךָ חָכְמָה׃   

לפרק הבא * * * תוכן העניינים
