לפרק הקודם * * * תוכן העניינים

איוב פרק כז

א                 וַיֹּסֶף אִיּוֹב שְׂאֵת מְשָׁלוֹ וַיֹּאמַר׃   
ב חַי-אֵל הֵסִיר מִשְׁפָּטִי    וְשַׁדַּי הֵמַר נַפְשִׁי׃   
ג כִּי-כָל-עוֹד נִשְׁמָתִי בִי    וְרוּחַ אֱלוֹהַּ בְּאַפִּי׃   
ד אִם-תְּדַבֵּרְנָה שְׂפָתַי עַוְלָה    וּלְשׁוֹנִי אִם-יֶהְגֶּה רְמִיָּה׃   
ה חָלִילָה לִּי    אִם-אַצְדִּיק אֶתְכֶם
עַד-אֶגְוָע     לֹא-אָסִיר תֻּמָּתִי מִמֶּנִּי׃   
ו בְּצִדְקָתִי הֶחֱזַקְתִּי וְלֹא אַרְפֶּהָ    לֹא-יֶחֱרַף לְבָבִי מִיָּמָי׃   
ז יְהִי כְרָשָׁע אֹיְבִי    וּמִתְקוֹמְמִי כְעַוָּל׃   
ח כִּי מַה-תִּקְוַת חָנֵף כִּי יִבְצָע    כִּי יֵשֶׁל אֱלוֹהַּ נַפְשׁוֹ׃   
ט הֲצַעֲקָתוֹ יִשְׁמַע אֵל     כִּי-תָבוֹא עָלָיו צָרָה׃   
י אִם-עַל-שַׁדַּי יִתְעַנָּג    יִקְרָא אֱלוֹהַּ בְּכָל-עֵת׃   
יא אוֹרֶה אֶתְכֶם בְּיַד-אֵל    אֲשֶׁר עִם-שַׁדַּי לֹא אֲכַחֵד׃   
יב הֵן-אַתֶּם כֻּלְּכֶם חֲזִיתֶם    וְלָמָּה-זֶּה הֶבֶל תֶּהְבָּלוּ׃   
יג זֶה חֵלֶק-אָדָם רָשָׁע עִם-אֵל    וְנַחֲלַת עָרִיצִים מִשַּׁדַּי יִקָּחוּ׃   
יד אִם-יִרְבּוּ בָנָיו לְמוֹ-חָרֶב    וְצֶאֱצָאָיו לֹא יִשְׂבְּעוּ-לָחֶם׃   
טו שְׂרִידָו בַּמָּוֶת יִקָּבֵרוּ    וְאַלְמְנֹתָיו לֹא תִבְכֶּינָה׃   
טז אִם-יִצְבֹּר כֶּעָפָר כָּסֶף    וְכַחֹמֶר יָכִין מַלְבּוּשׁ׃   
יז יָכִין וְצַדִּיק יִלְבָּשׁ    וְכֶסֶף נָקִי יַחֲלֹק׃   
יח בָּנָה כָעָשׁ בֵּיתוֹ    וּכְסֻכָּה עָשָׂה נֹצֵר׃   
יט עָשִׁיר יִשְׁכַּב וְלֹא יֵאָסֵף    עֵינָיו פָּקַח וְאֵינֶנּוּ׃   
כ תַּשִּׂיגֵהוּ כַמַּיִם בַּלָּהוֹת    לַיְלָה גְּנָבַתּוּ סוּפָה׃   
כא יִשָּׂאֵהוּ קָדִים וְיֵלַךְ    וִישָׂעֲרֵהוּ מִמְּקֹמוֹ׃   
כב וְיַשְׁלֵךְ עָלָיו וְלֹא יַחְמֹל    מִיָּדוֹ בָּרוֹחַ יִבְרָח׃   
כג יִשְׂפֹּק עָלֵימוֹ כַפֵּימוֹ    וְיִשְׁרֹק עָלָיו מִמְּקֹמוֹ׃   

לפרק הבא * * * תוכן העניינים
