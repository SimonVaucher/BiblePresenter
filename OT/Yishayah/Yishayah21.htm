לפרק הקודם * * * תוכן העניינים

ישעיהו פרק כא

א מַשָּׂא מִדְבַּר-יָם  כְּסוּפוֹת בַּנֶּגֶב לַחֲלֹף מִמִּדְבָּר בָּא מֵאֶרֶץ נוֹרָאָה׃    ב חָזוּת קָשָׁה הֻגַּד-לִי  הַבּוֹגֵד בּוֹגֵד וְהַשּׁוֹדֵד שׁוֹדֵד עֲלִי עֵילָם צוּרִי מָדַי כָּל-אַנְחָתָה הִשְׁבַּתִּי׃    ג עַל-כֵּן מָלְאוּ מָתְנַי חַלְחָלָה צִירִים אֲחָזוּנִי כְּצִירֵי יוֹלֵדָה נַעֲוֵיתִי מִשְּׁמֹעַ נִבְהַלְתִּי מֵרְאוֹת׃    ד תָּעָה לְבָבִי פַּלָּצוּת בִּעֲתָתְנִי אֵת נֶשֶׁף חִשְׁקִי שָׂם לִי לַחֲרָדָה׃    ה עָרֹךְ הַשֻּׁלְחָן צָפֹה הַצָּפִית אָכוֹל שָׁתֹה קוּמוּ הַשָּׂרִים מִשְׁחוּ מָגֵן׃   ס

ו כִּי כֹה אָמַר אֵלַי אֲדֹנָי  לֵךְ הַעֲמֵד הַמְצַפֶּה אֲשֶׁר יִרְאֶה יַגִּיד׃    ז וְרָאָה רֶכֶב צֶמֶד פָּרָשִׁים רֶכֶב חֲמוֹר רֶכֶב גָּמָל וְהִקְשִׁיב קֶשֶׁב רַב-קָשֶׁב׃    ח וַיִּקְרָא אַרְיֵה עַל-מִצְפֶּה אֲדֹנָי אָנֹכִי עֹמֵד תָּמִיד יוֹמָם וְעַל-מִשְׁמַרְתִּי אָנֹכִי נִצָּב כָּל-הַלֵּילוֹת׃    ט וְהִנֵּה-זֶה בָא רֶכֶב אִישׁ צֶמֶד פָּרָשִׁים וַיַּעַן וַיֹּאמֶר נָפְלָה נָפְלָה בָּבֶל וְכָל-פְּסִילֵי אֱלֹהֶיהָ שִׁבַּר לָאָרֶץ׃    י מְדֻשָׁתִי וּבֶן-גָּרְנִי  אֲשֶׁר שָׁמַעְתִּי מֵאֵת יְהוָה צְבָאוֹת אֱלֹהֵי יִשְׂרָאֵל הִגַּדְתִּי לָכֶם׃   פ

יא מַשָּׂא דּוּמָה  אֵלַי קֹרֵא מִשֵּׂעִיר שֹׁמֵר מַה-מִּלַּיְלָה שֹׁמֵר מַה-מִּלֵּיל׃    יב אָמַר שֹׁמֵר אָתָה בֹקֶר וְגַם-לָיְלָה אִם-תִּבְעָיוּן בְּעָיוּ שֻׁבוּ אֵתָיוּ׃   פ

יג מַשָּׂא בַּעְרָב  בַּיַּעַר בַּעְרַב תָּלִינוּ אֹרְחוֹת דְּדָנִים׃    יד לִקְרַאת צָמֵא הֵתָיוּ מָיִם יֹשְׁבֵי אֶרֶץ תֵּימָא בְּלַחְמוֹ קִדְּמוּ נֹדֵד׃    טו כִּי-מִפְּנֵי חֲרָבוֹת נָדָדוּ מִפְּנֵי חֶרֶב נְטוּשָׁה וּמִפְּנֵי קֶשֶׁת דְּרוּכָה וּמִפְּנֵי כֹּבֶד מִלְחָמָה׃   ס

טז כִּי-כֹה אָמַר אֲדֹנָי אֵלָי  בְּעוֹד שָׁנָה כִּשְׁנֵי שָׂכִיר וְכָלָה כָּל-כְּבוֹד קֵדָר׃    יז וּשְׁאָר מִסְפַּר-קֶשֶׁת גִּבּוֹרֵי בְנֵי-קֵדָר יִמְעָטוּ  כִּי יְהוָה אֱלֹהֵי-יִשְׂרָאֵל דִּבֵּר׃   ס

לפרק הבא * * * תוכן העניינים
