לפרק הקודם * * * תוכן העניינים

תהילים פרק לט

א   לַמְנַצֵּחַ לידיתון (לִידוּתוּן)    מִזְמוֹר לְדָוִד׃   
ב   אָמַרְתִּי     אֶשְׁמְרָה דְרָכַי מֵחֲטוֹא בִלְשׁוֹנִי
אֶשְׁמְרָה לְפִי מַחְסוֹם     בְּעֹד רָשָׁע לְנֶגְדִּי׃   
ג   נֶאֱלַמְתִּי דוּמִיָּה הֶחֱשֵׁיתִי מִטּוֹב    וּכְאֵבִי נֶעְכָּר׃   
ד   חַם-לִבִּי בְּקִרְבִּי בַּהֲגִיגִי תִבְעַר-אֵשׁ    דִּבַּרְתִּי בִּלְשׁוֹנִי׃   
ה   הוֹדִיעֵנִי יְהוָה קִצִּי וּמִדַּת יָמַי מַה-הִיא    אֵדְעָה מֶה-חָדֵל אָנִי׃   
ו   הִנֵּה טְפָחוֹת נָתַתָּה יָמַי     וְחֶלְדִּי כְאַיִן נֶגְדֶּךָ
אַךְ כָּל-הֶבֶל כָּל-אָדָם    נִצָּב סֶלָה׃   
ז   אַךְ-בְּצֶלֶם יִתְהַלֶּךְ-אִישׁ     אַךְ-הֶבֶל יֶהֱמָיוּן
יִצְבֹּר    וְלֹא-יֵדַע מִי-אֹסְפָם׃   
ח   וְעַתָּה מַה-קִּוִּיתִי אֲדֹנָי     תּוֹחַלְתִּי לְךָ הִיא׃   
ט   מִכָּל-פְּשָׁעַי הַצִּילֵנִי    חֶרְפַּת נָבָל אַל-תְּשִׂימֵנִי׃   
י   נֶאֱלַמְתִּי לֹא אֶפְתַּח-פִּי    כִּי אַתָּה עָשִׂיתָ׃   
יא   הָסֵר מֵעָלַי נִגְעֶךָ    מִתִּגְרַת יָדְךָ אֲנִי כָלִיתִי׃   
יב   בְּתוֹכָחוֹת עַל-עָוֹן יִסַּרְתָּ אִישׁ     וַתֶּמֶס כָּעָשׁ חֲמוּדוֹ
אַךְ הֶבֶל כָּל-אָדָם    סֶלָה׃   
יג   שִׁמְעָה תְפִלָּתִי יְהוָה    וְשַׁוְעָתִי הַאֲזִינָה
אֶל-דִּמְעָתִי    אַל-תֶּחֱרַשׁ
כִּי גֵר אָנֹכִי עִמָּךְ    תּוֹשָׁב כְּכָל-אֲבוֹתָי׃   
יד   הָשַׁע מִמֶּנִּי וְאַבְלִיגָה     בְּטֶרֶם אֵלֵךְ וְאֵינֶנִּי׃   

לפרק הבא * * * תוכן העניינים
