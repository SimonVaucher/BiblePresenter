לפרק הקודם * * * תוכן העניינים

תהילים פרק יז

א   תְּפִלָּה לְדָוִד
שִׁמְעָה יְהוָה צֶדֶק     הַקְשִׁיבָה רִנָּתִי הַאֲזִינָה תְפִלָּתִי
בְּלֹא    שִׂפְתֵי מִרְמָה׃   
ב   מִלְּפָנֶיךָ מִשְׁפָּטִי יֵצֵא    עֵינֶיךָ תֶּחֱזֶינָה מֵישָׁרִים׃   
ג   בָּחַנְתָּ לִבִּי פָּקַדְתָּ לַּיְלָה     צְרַפְתַּנִי בַל-תִּמְצָא
זַמֹּתִי    בַּל-יַעֲבָר-פִּי׃   
ד   לִפְעֻלּוֹת אָדָם בִּדְבַר שְׂפָתֶיךָ     אֲנִי שָׁמַרְתִּי אָרְחוֹת פָּרִיץ׃   
ה   תָּמֹךְ אֲשֻׁרַי בְּמַעְגְּלוֹתֶיךָ    בַּל-נָמוֹטּוּ פְעָמָי׃   
ו   אֲנִי-קְרָאתִיךָ כִי-תַעֲנֵנִי אֵל    הַט-אָזְנְךָ לִי שְׁמַע אִמְרָתִי׃   
ז   הַפְלֵה חֲסָדֶיךָ מוֹשִׁיעַ חוֹסִים     מִמִּתְקוֹמְמִים בִּימִינֶךָ׃   
ח   שָׁמְרֵנִי כְּאִישׁוֹן בַּת-עָיִן    בְּצֵל כְּנָפֶיךָ תַּסְתִּירֵנִי׃   
ט   מִפְּנֵי רְשָׁעִים זוּ שַׁדּוּנִי    אֹיְבַי בְּנֶפֶשׁ יַקִּיפוּ עָלָי׃   
י   חֶלְבָּמוֹ סָּגְרוּ    פִּימוֹ דִּבְּרוּ בְגֵאוּת׃   
יא   אַשֻּׁרֵינוּ עַתָּה סבבוני (סְבָבוּנוּ)    עֵינֵיהֶם יָשִׁיתוּ לִנְטוֹת בָּאָרֶץ׃   
יב   דִּמְיֹנוֹ כְּאַרְיֵה יִכְסוֹף לִטְרֹף    וְכִכְפִיר יֹשֵׁב בְּמִסְתָּרִים׃   
יג   קוּמָה יְהוָה     קַדְּמָה פָנָיו הַכְרִיעֵהוּ
פַּלְּטָה נַפְשִׁי    מֵרָשָׁע חַרְבֶּךָ׃   
יד   מִמְתִים יָדְךָ יְהוָה    מִמְתִים מֵחֶלֶד
חֶלְקָם בַּחַיִּים    וצפינך (וּצְפוּנְךָ) תְּמַלֵּא בִטְנָם
יִשְׂבְּעוּ בָנִים     וְהִנִּיחוּ יִתְרָם לְעוֹלְלֵיהֶם׃   
טו   אֲנִי בְּצֶדֶק אֶחֱזֶה פָנֶיךָ    אֶשְׂבְּעָה בְהָקִיץ תְּמוּנָתֶךָ׃   

לפרק הבא * * * תוכן העניינים
