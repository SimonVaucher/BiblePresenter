לפרק הקודם * * * תוכן העניינים

עזרא פרק ז

א וְאַחַר הַדְּבָרִים הָאֵלֶּה בְּמַלְכוּת אַרְתַּחְשַׁסְתְּא מֶלֶךְ-פָּרָס עֶזְרָא בֶּן-שְׂרָיָה בֶּן-עֲזַרְיָה בֶּן-חִלְקִיָּה׃    ב בֶּן-שַׁלּוּם בֶּן-צָדוֹק בֶּן-אֲחִיטוּב׃    ג בֶּן-אֲמַרְיָה בֶן-עֲזַרְיָה בֶּן-מְרָיוֹת׃    ד בֶּן-זְרַחְיָה בֶן-עֻזִּי בֶּן-בֻּקִּי׃    ה בֶּן-אֲבִישׁוּעַ בֶּן-פִּינְחָס בֶּן-אֶלְעָזָר בֶּן-אַהֲרֹן הַכֹּהֵן הָרֹאשׁ׃    ו הוּא עֶזְרָא עָלָה מִבָּבֶל וְהוּא-סֹפֵר מָהִיר בְּתוֹרַת מֹשֶׁה אֲשֶׁר-נָתַן יְהוָה אֱלֹהֵי יִשְׂרָאֵל וַיִּתֶּן-לוֹ הַמֶּלֶךְ כְּיַד-יְהוָה אֱלֹהָיו עָלָיו כֹּל בַּקָּשָׁתוֹ׃   פ

ז וַיַּעֲלוּ מִבְּנֵי-יִשְׂרָאֵל וּמִן-הַכֹּהֲנִים וְהַלְוִיִּם וְהַמְשֹׁרְרִים וְהַשֹּׁעֲרִים וְהַנְּתִינִים אֶל-יְרוּשָׁלִָם  בִּשְׁנַת-שֶׁבַע לְאַרְתַּחְשַׁסְתְּא הַמֶּלֶךְ׃    ח וַיָּבֹא יְרוּשָׁלִַם בַּחֹדֶשׁ הַחֲמִישִׁי  הִיא שְׁנַת הַשְּׁבִיעִית לַמֶּלֶךְ׃    ט כִּי בְּאֶחָד לַחֹדֶשׁ הָרִאשׁוֹן הוּא יְסֻד הַמַּעֲלָה מִבָּבֶל וּבְאֶחָד לַחֹדֶשׁ הַחֲמִישִׁי בָּא אֶל-יְרוּשָׁלִַם כְּיַד-אֱלֹהָיו הַטּוֹבָה עָלָיו׃    י כִּי עֶזְרָא הֵכִין לְבָבוֹ לִדְרֹשׁ אֶת-תּוֹרַת יְהוָה וְלַעֲשֹׂת וּלְלַמֵּד בְּיִשְׂרָאֵל חֹק וּמִשְׁפָּט׃   ס

יא וְזֶה פַּרְשֶׁגֶן הַנִּשְׁתְּוָן אֲשֶׁר נָתַן הַמֶּלֶךְ אַרְתַּחְשַׁסְתְּא לְעֶזְרָא הַכֹּהֵן הַסֹּפֵר  סֹפֵר דִּבְרֵי מִצְוֹת-יְהוָה וְחֻקָּיו עַל-יִשְׂרָאֵל׃   פ

יב  אַרְתַּחְשַׁסְתְּא מֶלֶךְ מַלְכַיָּא  לְעֶזְרָא כָהֲנָא סָפַר דָּתָא דִּי-אֱלָהּ שְׁמַיָּא גְּמִיר וּכְעֶנֶת׃ יב  מִמֶּנִּי הוּשַׂם צַו, שֶׁכָּל מִתְנַדֵּב בְּמַלְכוּתִי מֵעַם יִשְׂרָאֵל וְכֹהֲנָיו וְהַלְּוִיִּים לָלֶכֶת לִירוּשָׁלַיִם עִמְּךָ, יֵלֵךְ.
יג  מִנִּי שִׂים טְעֵם דִּי כָל-מִתְנַדַּב בְּמַלְכוּתִי מִן-עַמָּא יִשְׂרָאֵל וְכָהֲנוֹהִי וְלֵוָיֵא לִמְהָךְ לִירוּשְׁלֶם עִמָּךְ יְהָךְ׃ יג  אֲזַי דָּרְיָוֶשׁ הַמֶּלֶךְ שּׂם צַו, וּבִקְּרוּ בְּבֵית הַסְּפָרִים, אֲשֶׁר אֶת הַגְּנָזִים מוֹרִידִים שָׁמָּה בְּבָבֶל.
יד  כָּל-קֳבֵל דִּי מִן-קֳדָם מַלְכָּא וְשִׁבְעַת יָעֲטֹהִי שְׁלִיחַ לְבַקָּרָה עַל-יְהוּד וְלִירוּשְׁלֶם בְּדָת אֱלָהָךְ דִּי בִידָךְ׃ יד  כָּל-עֻמַּת שֶׁמִּלִּפְנֵי הַמֶּלֶךְ וְשִׁבְעַת יוֹעֲצָיו נִשְׁלַח, לְבַקֵּר אֶל יְהוּדָה וְלִירוּשָׁלַיִם, בְּדָת אֱלֹהֶיךָ אֲשֶׁר בְּיָדֶךָ.
טו  וּלְהֵיבָלָה כְּסַף וּדְהַב דִּי-מַלְכָּא וְיָעֲטוֹהִי הִתְנַדַּבוּ לֶאֱלָהּ יִשְׂרָאֵל דִּי בִירוּשְׁלֶם מִשְׁכְּנֵהּ׃ טו  וּלְהוֹבִיל כֶּסֶף וְזָהָב, אֲשֶׁר הַמֶּלֶךְ וְיוֹעֲצָיו הִתְנַדְּבוּ לֶאֱלֹהֵי יִשְׂרָאֵל, אֲשֶׁר בִּירוּשָׁלַיִם מִשְׁכָּנוֹ.
טז  וְכֹל כְּסַף וּדְהַב דִּי תְהַשְׁכַּח בְּכֹל מְדִינַת בָּבֶל עִם הִתְנַדָּבוּת עַמָּא וְכָהֲנַיָּא מִתְנַדְּבִין לְבֵית אֱלָהֲהֹם דִּי בִירוּשְׁלֶם׃ טז  וְכָל כֶּסֶף וְזָהָב, אֲשֶׁר תִּמְצָא בְּכָל מְדִינַת בָּבֶל, עִם הִתְנַדְּבוּת הָעָם וְהַכֹּהֲנִים, מִתְנַדְּבִים לְבֵית אֱלֹהֵיהֶם אֲשֶׁר בִּירוּשָׁלַיִם.
יז  כָּל-קֳבֵל דְּנָה אָסְפַּרְנָא תִקְנֵא בְּכַסְפָּא דְנָה תּוֹרִין דִּכְרִין אִמְּרִין וּמִנְחָתְהוֹן וְנִסְכֵּיהוֹן וּתְקָרֵב הִמּוֹ עַל-מַדְבְּחָה דִּי בֵּית אֱלָהֲכֹם דִּי בִירוּשְׁלֶם׃ יז  כָּל-עֻמַּת זֶה בְּמֶרֶץ תִּקְנֶה בְּכֶסֶף זֶה פָּרִים, אֵילִים, כְּבָשִׂים וּמִנְחוֹתֵיהֶם וְנִסְכֵּיהֶם, וְתַקְרִיב אוֹתָם עַל הַמִּזְבֵּחַ שֶׁל בֵּית אֱלֹהֵיכֶם שֶׁבִּירוּשָׁלַיִם.
יח  וּמָה דִי עליך (עֲלָךְ) וְעַל-אחיך (אֶחָךְ) יֵיטַב בִּשְׁאָר כַּסְפָּא וְדַהֲבָה לְמֶעְבַּד  כִּרְעוּת אֱלָהֲכֹם תַּעַבְדוּן׃ יח  וּמַה שֶׁעָלֶיךָ וְעַל אַחֶיךָ יִיטַב, בִּשְׁאָר הַכֶּסֶף וְהַזָּהָב לַעֲשׂוֹת, כִּרְצוֹן אֱלֹהֵיכֶם תַּעֲשׂוּ.
יט  וּמָאנַיָּא דִּי-מִתְיַהֲבִין לָךְ לְפָלְחָן בֵּית אֱלָהָךְ הַשְׁלֵם קֳדָם אֱלָהּ יְרוּשְׁלֶם׃ יט  וְהַכֵּלִים, אֲשֶׁר נִתָּנִים לְךָ לַעֲבוֹדַת בֵּית אֱלֹהֶיךָ, מְסֹר לִפְנֵי אֱלֹהֵי יְרוּשָׁלַיִם.
כ  וּשְׁאָר חַשְׁחוּת בֵּית אֱלָהָךְ דִּי יִפֶּל-לָךְ לְמִנְתַּן תִּנְתֵּן מִן-בֵּית גִּנְזֵי מַלְכָּא׃ כ  וּשְׁאָר צֹרֶךְ בֵּית אֱלֹהֶיךָ, אֲשֶׁר יוּטַל עָלֶיךָ לָתֵת, תִּתֵּן מִבֵּית גִּנְזֵי הַמֶּלֶךְ.
כא  וּמִנִּי אֲנָה אַרְתַּחְשַׁסְתְּא מַלְכָּא שִׂים טְעֵם לְכֹל גִּזַּבְרַיָּא דִּי בַּעֲבַר נַהֲרָה  דִּי כָל-דִּי יִשְׁאֲלֶנְכוֹן עֶזְרָא כָהֲנָא סָפַר דָּתָא דִּי-אֱלָהּ שְׁמַיָּא אָסְפַּרְנָא יִתְעֲבִד׃ כא  וּמִמֶּנִּי אֲנִי אַרְתַּחְשַׁסְתְּא הַמֶּלֶךְ, הוּשַׂם צַו לְכָל הַגִּזְבָּרִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, שֶׁכָּל אֲשֶׁר יִשְׁאַלְכֶם עֶזְרָא הַכֹּהֵן, סוֹפֵר הַדָּת שֶׁל אֱלֹהֵי הַשָׁמַיִם, בְּמֶרֶץ יֵעָשֶׂה.
כב  עַד-כְּסַף כַּכְּרִין מְאָה וְעַד-חִנְטִין כּוֹרִין מְאָה וְעַד-חֲמַר בַּתִּין מְאָה וְעַד-בַּתִּין מְשַׁח מְאָה וּמְלַח דִּי-לָא כְתָב׃ כב  עַד כֶּסֶף כִּכָּרִים מֵאָה, וְעַד חִטִּים כּוֹרִים מֵאָה, וְעַד יַיִן בַּתִּים מֵאָה, וְעַד בַּתִּים שֶׁמֶן מֵאָה, וּמֶלַח לְלֹא כְתָב.
כג  כָּל-דִּי מִן-טַעַם אֱלָהּ שְׁמַיָּא יִתְעֲבֵד אַדְרַזְדָּא לְבֵית אֱלָהּ שְׁמַיָּא  דִּי-לְמָה לֶהֱוֵא קְצַף עַל-מַלְכוּת מַלְכָּא וּבְנוֹהִי׃ כג  כָּל אֲשֶׁר מִמִּצְוַת אֱלֹהֵי הַשָׁמַיִם, יֵעָשֶׂה אֶל נָכוֹן לְבֵית אֱלֹהֵי הַשָׁמַיִם, שַׁלָּמָה יִהְיֶה קֶצֶף עַל מַלְכוּת הַמֶּלֶךְ וּבָנָיו.
כד  וּלְכֹם מְהוֹדְעִין דִּי כָל-כָּהֲנַיָּא וְלֵוָיֵא זַמָּרַיָּא תָרָעַיָּא נְתִינַיָּא וּפָלְחֵי בֵּית אֱלָהָא דְנָה מִנְדָּה בְלוֹ וַהֲלָךְ לָא שַׁלִּיט לְמִרְמֵא עֲלֵיהֹם׃ כד  וְלָכֶם מוֹדִיעִים, אֲשֶׁר כָּל הַכֹּהֲנִים וְהַלְּוִיִּים, הַמְּשׁוֹרְרִים, הַשּׁוֹעֲרִים, הַנְּתִינִים וְעוֹבְדֵי בֵּית הָאֱלֹהִים הַזֶּה, מַתָּת מִנְחָה וּמַס אֵין רַשַׁאי לְהַטִּיל עֲלֵיהֶם.
כה  וְאַנְתְּ עֶזְרָא כְּחָכְמַת אֱלָהָךְ דִּי-בִידָךְ מֶנִּי שָׁפְטִין וְדַיָּנִין דִּי-לֶהֱוֹן דאנין (דָּיְנִין) לְכָל-עַמָּא דִּי בַּעֲבַר נַהֲרָה לְכָל-יָדְעֵי דָּתֵי אֱלָהָךְ וְדִי לָא יָדַע תְּהוֹדְעוּן׃ כה  וְאַתָּה עֶזְרָא, כְּחָכְמַת אֱלֹהֶיךָ אֲשֶׁר בְּיָדְךָ מַנֵּה שׁוֹפְטִים וְדַיָּנִים, אֲשֶׁר יִהְיוּ דָנִים אֶת כָּל הָעָם אֲשֶׁר בְּעֵבֶר הַנָּהָר, אֶת כָּל יוֹדְעֵי דָת אֱלֹהֶיךָ, וַאֲשֶׁר אֵינוֹ יוֹדֵעַ -- תּוֹדִיעוֹ.
כו  וְכָל-דִּי-לָא לֶהֱוֵא עָבֵד דָּתָא דִי-אֱלָהָךְ וְדָתָא דִּי מַלְכָּא אָסְפַּרְנָא דִּינָה לֶהֱוֵא מִתְעֲבֵד מִנֵּהּ  הֵן לְמוֹת הֵן לשרשו (לִשְׁרֹשִׁי) הֵן-לַעֲנָשׁ נִכְסִין וְלֶאֱסוּרִין׃   פ כו  וְכָל אֲשֶׁר לֹא יִהְיֶה עוֹשֶׂה דָת אֱלֹהֶיךָ וְדָת הַמֶּלֶךְ, בְּמֶרֶץ הַדִּין יִהְיֶה נַעֲשֶׂה בּוֹ, אִם לְמָוֶת אִם לְגָלוּת, אִם לְעֹנֶשׁ כְּסָפִים וְלֶאֱסוּרִים.   פ

כז בָּרוּךְ יְהוָה אֱלֹהֵי אֲבֹתֵינוּ אֲשֶׁר נָתַן כָּזֹאת בְּלֵב הַמֶּלֶךְ לְפָאֵר אֶת-בֵּית יְהוָה אֲשֶׁר בִּירוּשָׁלִָם׃    כח וְעָלַי הִטָּה-חֶסֶד לִפְנֵי הַמֶּלֶךְ וְיוֹעֲצָיו וּלְכָל-שָׂרֵי הַמֶּלֶךְ הַגִּבֹּרִים וַאֲנִי הִתְחַזַּקְתִּי כְּיַד-יְהוָה אֱלֹהַי עָלַי וָאֶקְבְּצָה מִיִּשְׂרָאֵל רָאשִׁים לַעֲלוֹת עִמִּי׃   פ
