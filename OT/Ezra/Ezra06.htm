לפרק הקודם * * * תוכן העניינים

עזרא פרק ו

א  בֵּאדַיִן דָּרְיָוֶשׁ מַלְכָּא שָׂם טְעֵם וּבַקַּרוּ בְּבֵית סִפְרַיָּא דִּי גִנְזַיָּא מְהַחֲתִין תַּמָּה בְּבָבֶל׃ א  אֲזַי דָּרְיָוֶשׁ הַמֶּלֶךְ שּׂם צַו, וּבִקְּרוּ בְּבֵית הַסְּפָרִים, אֲשֶׁר אֶת הַגְּנָזִים מוֹרִידִים שָׁמָּה בְּבָבֶל.
ב  וְהִשְׁתְּכַח בְּאַחְמְתָא בְּבִירְתָא דִּי בְּמָדַי מְדִינְתָּא מְגִלָּה חֲדָה וְכֵן-כְּתִיב בְּגַוַּהּ דִּכְרוֹנָה׃   פ ב  וְנִמְצָא בְאַחְמְתָא, בַּבִּירָה אֲשֶׁר בְּמָדַי הַמְּדִינָה, מְגִלָּה אַחַת, וְכֵן כָּתוּב בְּתוֹכָהּ: הַזִּכָּרוֹן.   פ

ג  בִּשְׁנַת חֲדָה לְכוֹרֶשׁ מַלְכָּא כּוֹרֶשׁ מַלְכָּא שָׂם טְעֵם בֵּית-אֱלָהָא בִירוּשְׁלֶם בַּיְתָא יִתְבְּנֵא אֲתַר דִּי-דָבְחִין דִּבְחִין וְאֻשּׁוֹהִי מְסוֹבְלִין רוּמֵהּ אַמִּין שִׁתִּין פְּתָיֵהּ אַמִּין שִׁתִּין׃ ג  בִּשְׁנַת אַחַת לְכוֹרֶשׁ הַמֶּלֶךְ, כּוֹרֶשׁ הַמֶּלֶךְ שּׂם צַו: בֵּית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם הַבַּיִת יִבָּנֶה, מָקוֹם שֶׁזּוֹבְחִים זְבָחִים וִיסוֹדָיו מוּקָמִים, רוּמוֹ אַמּוֹת שִׁשִׁים, רָחְבּוֹ אַמּוֹת שִׁשִׁים.
ד  נִדְבָּכִין דִּי-אֶבֶן גְּלָל תְּלָתָא וְנִדְבָּךְ דִּי-אָע חֲדַת וְנִפְקְתָא מִן-בֵּית מַלְכָּא תִּתְיְהִב׃ ד  נִדְבָּכִים שֶׁל אֶבֶן גָּלָל שְׁלשָׁה, וְנִדְבָּךְ שֶׁל עֵץ אֶחָד, וְהַהוֹצָאָה מִבֵּית הַמֶּלֶךְ תִּנָּתֵן.
ה  וְאַף מָאנֵי בֵית-אֱלָהָא דִּי דַהֲבָה וְכַסְפָּא דִּי נְבוּכַדְנֶצַּר הַנְפֵּק מִן-הֵיכְלָא דִי-בִירוּשְׁלֶם וְהֵיבֵל לְבָבֶל יַהֲתִיבוּן וִיהָךְ לְהֵיכְלָא דִי-בִירוּשְׁלֶם לְאַתְרֵהּ וְתַחֵת בְּבֵית אֱלָהָא׃   ס ה  וְאַף כְּלֵי בֵּית הָאֱלֹהִים שֶׁל זָהָב וָכֶסֶף, אֲשֶׁר נְבוּכַדְנֶצַּר הוֹצִיא מִן הַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם וְהוֹבִיל לְבָבֶל, יָשִׁיבוּ, וְיֵלֵךְ לַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם לִמְקוֹמוֹ, וְתוֹרִיד בְּבֵית הָאֱלֹהִים.   ס

ו  כְּעַן תַּתְּנַי פַּחַת עֲבַר-נַהֲרָה שְׁתַר בּוֹזְנַי וּכְנָוָתְהוֹן אֲפַרְסְכָיֵא דִּי בַּעֲבַר נַהֲרָה רַחִיקִין הֲווֹ מִן-תַּמָּה׃ ו  כָּעֵת, תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, שְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, הַפַּרְסִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, מְרֻחָקִים הֱיוּ מִשָׁם.
ז  שְׁבֻקוּ לַעֲבִידַת בֵּית-אֱלָהָא דֵךְ פַּחַת יְהוּדָיֵא וּלְשָׂבֵי יְהוּדָיֵא בֵּית-אֱלָהָא דֵךְ יִבְנוֹן עַל-אַתְרֵהּ׃ ז  הַנִּיחוּ לַעֲבוֹדַת בֵּית הָאֱלֹהִים הַזֶּה, פַּחַת הַיְּהוּדִים וּלְזִקְנֵי הַיְּהוּדִים, בֵּית אֱלֹהִים זֶה יִבְנוּ עַל מְקוֹמוֹ.
ח  וּמִנִּי שִׂים טְעֵם לְמָא דִי-תַעַבְדוּן עִם-שָׂבֵי יְהוּדָיֵא אִלֵּךְ לְמִבְנֵא בֵּית-אֱלָהָא דֵךְ וּמִנִּכְסֵי מַלְכָּא דִּי מִדַּת עֲבַר נַהֲרָה אָסְפַּרְנָא נִפְקְתָא תֶּהֱוֵא מִתְיַהֲבָא לְגֻבְרַיָּא אִלֵּךְ דִּי-לָא לְבַטָּלָא׃ ח  וּמִמֶּנִּי הוּשַׂם צַו, לְמַה שֶׁתַּעֲשׂוּ עִם זִקְנֵי הַיְּהוּדִים הָאֵלֶּה, לִבְנוֹת אֶת בֵּית הָאֱלֹהִים הַזֶּה, וּמִנִּכְסֵי הַמֶּלֶךְ שֶׁמַּתַּת עֵבֶר הַנָּהָר, בִּמְהֵרָה הַהוֹצָאָה תִּהְיֶה נִתֶּנֶת לָאֲנָשִׁים הָאֵלֶּה, שֶׁלֹּא לְבַטֵּל.
ט  וּמָה חַשְׁחָן וּבְנֵי תוֹרִין וְדִכְרִין וְאִמְּרִין לַעֲלָוָן לֶאֱלָהּ שְׁמַיָּא חִנְטִין מְלַח חֲמַר וּמְשַׁח כְּמֵאמַר כָּהֲנַיָּא דִי-בִירוּשְׁלֶם לֶהֱוֵא מִתְיְהֵב לְהֹם יוֹם בְּיוֹם דִּי-לָא שָׁלוּ׃ ט  וּמַה שֶׁצָּרִיךְ, וּבְנֵי בָקָר וְאֵילִים וּכְבָשִׁים לְעוֹלוֹת לֵאלֹהֵי הַשָׁמַיִם, חִטִּים, מֶלַח, יַיִן וָשֶׁמֶן, כְּמַאֲמַר הַכֹּהֲנִים אֲשֶׁר בִּירוּשָׁלַיִם, יְהִי נִתָּן לָהֶם יוֹם בְּיוֹם בְּלֹא מִשְׁגֶּה.
י  דִּי-לֶהֱוֹן מְהַקְרְבִין נִיחוֹחִין לֶאֱלָהּ שְׁמַיָּא וּמְצַלַּיִן לְחַיֵּי מַלְכָּא וּבְנוֹהִי׃ י  אֲשֶׁר יִהְיוּ מַקְרִיבִים נִיחוֹחִים לֵאלֹהֵי הַשָׁמַיִם, וּמִתְפַּלְּלִים לְחַיֵּי הַמֶּלֶךְ וּבָנָיו.
יא  וּמִנִּי שִׂים טְעֵם דִּי כָל-אֱנָשׁ דִּי יְהַשְׁנֵא פִּתְגָמָא דְנָה יִתְנְסַח אָע מִן-בַּיְתֵהּ וּזְקִיף יִתְמְחֵא עֲלֹהִי וּבַיְתֵהּ נְוָלוּ יִתְעֲבֵד עַל-דְּנָה׃ יא  וּמִמֶּנִּי הוּשַׂם צַו, שֶׁכָּל אִישׁ אֲשֶׁר יְשַׁנֶּה דָּבָר זֶה, יֻסַּח עֵץ מִבֵּיתוֹ, וְזָקוּף יֻכֶּה עָלָיו, וּבֵיתוֹ אַשְׁפָּה יֵעָשֶׂה עַל זֶה.
יב  וֵאלָהָא דִּי שַׁכִּן שְׁמֵהּ תַּמָּה יְמַגַּר כָּל-מֶלֶךְ וְעַם דִּי יִשְׁלַח יְדֵהּ לְהַשְׁנָיָה לְחַבָּלָה בֵּית-אֱלָהָא דֵךְ דִּי בִירוּשְׁלֶם אֲנָה דָרְיָוֶשׁ שָׂמֶת טְעֵם אָסְפַּרְנָא יִתְעֲבִד׃   פ יב  וְהָאֱלֹהִים, אֲשֶׁר שִׁכֵּן שְׁמוֹ שָׁם, יְמַגֵּר כָּל מֶלֶךְ וְעַם, אֲשֶׁר יִשְׁלַח יָדוֹ לְשַׁנּוֹת, לְחַבֵּל אֶת בֵּית הָאֱלֹהִים הַזֶּה אֲשֶׁר בִּירוּשָׁלַיִם; אֲנִי דָרְיָוֶשׁ נָתַתִּי צַו, מְהֵרָה יֵעָשֶׂה!   פ

יג  אֱדַיִן תַּתְּנַי פַּחַת עֲבַר-נַהֲרָה שְׁתַר בּוֹזְנַי וּכְנָוָתְהוֹן  לָקֳבֵל דִּי-שְׁלַח דָּרְיָוֶשׁ מַלְכָּא כְּנֵמָא אָסְפַּרְנָא עֲבַדוּ׃ יג  אֲזַי תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, שְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, לְעֻמַּת אֲשֶׁר שָׁלַח דָּרְיָוֶשׁ הַמֶּלֶךְ, כֵּן מְהֵרָה עָשׂוּ.
יד  וְשָׂבֵי יְהוּדָיֵא בָּנַיִן וּמַצְלְחִין בִּנְבוּאַת חַגַּי נְבִיָּאה וּזְכַרְיָה בַּר-עִדּוֹא וּבְנוֹ וְשַׁכְלִלוּ מִן-טַעַם אֱלָהּ יִשְׂרָאֵל וּמִטְּעֵם כּוֹרֶשׁ וְדָרְיָוֶשׁ וְאַרְתַּחְשַׁשְׂתְּא מֶלֶךְ פָּרָס׃ יד  וְזִקְנֵי הַיְּהוּדִים בּוֹנִים וּמַצְלִיחִים, בִּנְבוּאַת חַגַּי הַנָּבִיא וּזְכַרְיָה בֶּן עִדּוֹא, וּבָנוּ וְתִקְּנוּ, בְּמִצְוַת אֱלֹהֵי יִשְׂרָאֵל, וּבְמִצְוַת כּוֹרֶשׁ וְדָרְיָוֶשׁ וְאַרְתַּחְשַׁשְׂתְּא מֶלֶךְ פָּרַס.
טו  וְשֵׁיצִיא בַּיְתָה דְנָה עַד יוֹם תְּלָתָה לִירַח אֲדָר דִּי-הִיא שְׁנַת-שֵׁת לְמַלְכוּת דָּרְיָוֶשׁ מַלְכָּא׃   פ טו  וְנִשְׁלַם בַּיִת זֶה עַד יוֹם שלשה לְחֹדֶשׁ אֲדָר, שֶׁהִיא שְׁנַת שֵׁשׁ לְמַלְכוּת דָּרְיָוֶשׁ הַמֶּלֶךְ.   פ

טז  וַעֲבַדוּ בְנֵי-יִשְׂרָאֵל כָּהֲנַיָּא וְלֵוָיֵא וּשְׁאָר בְּנֵי-גָלוּתָא חֲנֻכַּת בֵּית-אֱלָהָא דְנָה בְּחֶדְוָה׃ טז  וְעָשׂוּ בְנֵי יִשְׂרָאֵל, הַכֹּהֲנִים וְהַלְּוִיִּים וּשְׁאָר בְּנֵי הַגּוֹלָה, חֲנֻכַּת בֵּית הָאֱלֹהִים הַזֶּה בְּחֶדְוָה.
יז  וְהַקְרִבוּ לַחֲנֻכַּת בֵּית-אֱלָהָא דְנָה תּוֹרִין מְאָה דִּכְרִין מָאתַיִן אִמְּרִין אַרְבַּע מְאָה וּצְפִירֵי עִזִּין לחטיא (לְחַטָּאָה) עַל-כָּל-יִשְׂרָאֵל תְּרֵי-עֲשַׂר לְמִנְיָן שִׁבְטֵי יִשְׂרָאֵל׃ יז  וְהִקְרִיבוּ לַחֲנֻכַּת בֵּית הָאֱלֹהִים הַזֶּה, פָּרִים מֵאָה, אֵילִים מָאתַיִם, כְּבָשִׁים אַרְבַּע מֵאוֹת, וּצְפִירֵי עִזִּים, לְחַטֵּא עַל כָּל יִשְׂרָאֵל, שְׁנֵים עָשּׂר, לְמִנְיַן שִׁבְטֵי יִשְׂרָאֵל.
יח  וַהֲקִימוּ כָהֲנַיָּא בִּפְלֻגָּתְהוֹן וְלֵוָיֵא בְּמַחְלְקָתְהוֹן עַל-עֲבִידַת אֱלָהָא דִּי בִירוּשְׁלֶם  כִּכְתָב סְפַר מֹשֶׁה׃   פ יח  18 וְהֵקִימוּ הַכֹּהֲנִים בִּפְלֻגּוֹתֵיהֶם, וְהַלְּוִיִּים בְּמַחְלְקוֹתֵיהֶם, עַל עֲבוֹדַת הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם, כִּכְתָב סֵפֶר מֹשֶׁה.   פ

יט וַיַּעֲשׂוּ בְנֵי-הַגּוֹלָה אֶת-הַפָּסַח בְּאַרְבָּעָה עָשָׂר לַחֹדֶשׁ הָרִאשׁוֹן׃    כ כִּי הִטַּהֲרוּ הַכֹּהֲנִים וְהַלְוִיִּם כְּאֶחָד כֻּלָּם טְהוֹרִים וַיִּשְׁחֲטוּ הַפֶּסַח לְכָל-בְּנֵי הַגּוֹלָה וְלַאֲחֵיהֶם הַכֹּהֲנִים וְלָהֶם׃    כא וַיֹּאכְלוּ בְנֵי-יִשְׂרָאֵל הַשָּׁבִים מֵהַגּוֹלָה וְכֹל הַנִּבְדָּל מִטֻּמְאַת גּוֹיֵ-הָאָרֶץ אֲלֵהֶם לִדְרֹשׁ לַיהוָה אֱלֹהֵי יִשְׂרָאֵל׃    כב וַיַּעֲשׂוּ חַג-מַצּוֹת שִׁבְעַת יָמִים בְּשִׂמְחָה  כִּי שִׂמְּחָם יְהוָה וְהֵסֵב לֵב מֶלֶךְ-אַשּׁוּר עֲלֵיהֶם לְחַזֵּק יְדֵיהֶם בִּמְלֶאכֶת בֵּית-הָאֱלֹהִים אֱלֹהֵי יִשְׂרָאֵל׃   פ
