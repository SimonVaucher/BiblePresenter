לפרק הקודם * * * תוכן העניינים

דניאל פרק ב

א וּבִשְׁנַת שְׁתַּיִם לְמַלְכוּת נְבֻכַדְנֶצַּר חָלַם נְבֻכַדְנֶצַּר חֲלֹמוֹת וַתִּתְפָּעֶם רוּחוֹ וּשְׁנָתוֹ נִהְיְתָה עָלָיו׃    ב וַיֹּאמֶר הַמֶּלֶךְ לִקְרֹא לַחַרְטֻמִּים וְלָאַשָּׁפִים וְלַמְכַשְּׁפִים וְלַכַּשְׂדִּים לְהַגִּיד לַמֶּלֶךְ חֲלֹמֹתָיו וַיָּבֹאוּ וַיַּעַמְדוּ לִפְנֵי הַמֶּלֶךְ׃    ג וַיֹּאמֶר לָהֶם הַמֶּלֶךְ חֲלוֹם חָלָמְתִּי וַתִּפָּעֶם רוּחִי לָדַעַת אֶת-הַחֲלוֹם׃
ד  וַיְדַבְּרוּ הַכַּשְׂדִּים לַמֶּלֶךְ אֲרָמִית  מַלְכָּא לְעָלְמִין חֱיִי אֱמַר חֶלְמָא לעבדיך (לְעַבְדָךְ) וּפִשְׁרָא נְחַוֵּא׃ ד  וַיְדַבְּרוּ הַכַּשְׂדִּים לַמֶּלֶךְ אֲרָמִית: הַמֶּלֶךְ, לְעוֹלָמִים חֱיֵה! אֱמֹר אֶת הַחֲלוֹם לַעֲבָדֶיךָ, וְאֶת הַפִּתְרוֹן נַגִּידָה.
ה  עָנֵה מַלְכָּא וְאָמַר לכשדיא (לְכַשְׂדָּאֵי) מִלְּתָה מִנִּי אַזְדָּא  הֵן לָא תְהוֹדְעוּנַּנִי חֶלְמָא וּפִשְׁרֵהּ הַדָּמִין תִּתְעַבְדוּן וּבָתֵּיכוֹן נְוָלִי יִתְּשָׂמוּן׃ ה  עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר אֶל הַכַּשְׂדִּים: הַדָּבָר וַדַּאי מִמֶּנִּי, אִם לֹא תּוֹדִיעוּנִי אֶת הַחֲלוֹם וְאֶת פִּתְרוֹנוֹ, נְתָחִים תֵּעָשׂוּ, וּבָתֵּיכֶם אַשְׁפָּה יוּשָׂמוּ.
ו  וְהֵן חֶלְמָא וּפִשְׁרֵהּ תְּהַחֲוֹן מַתְּנָן וּנְבִזְבָּה וִיקָר שַׂגִּיא תְּקַבְּלוּן מִן-קֳדָמָי לָהֵן חֶלְמָא וּפִשְׁרֵהּ הַחֲוֹנִי׃ ו  וְאִם אֶת הַחֲלוֹם וְאֶת פִּתְרוֹנוֹ תַּגִּידוּ, מַתָּנוֹת וּתְשׁוּרוֹת וְכָבוֹד רַב תְּקַבְּלוּ מִלְּפָנַי; אֶלָּא אֶת הַחֲלוֹם וְאֶת פִּתְרוֹנוֹ הַגִּידוּ לִי.
ז  עֲנוֹ תִנְיָנוּת וְאָמְרִין  מַלְכָּא חֶלְמָא יֵאמַר לְעַבְדוֹהִי וּפִשְׁרָה נְהַחֲוֵה׃ ז  עָנוּ שֵׁנִית וְאוֹמְרִים: הַמֶּלֶךְ, אֶת הַחֲלוֹם יֹאמַר לַעֲבָדָיו, וְאֶת הַפִּתְרוֹן נַגִּידָה.
ח  עָנֵה מַלְכָּא וְאָמַר מִן-יַצִּיב יָדַע אֲנָה דִּי עִדָּנָא אַנְתּוּן זָבְנִין כָּל-קֳבֵל דִּי חֲזֵיתוֹן דִּי אַזְדָּא מִנִּי מִלְּתָא׃ ח  עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר: נָכוֹן יוֹדֵעַ אָנֹכִי, כִּי זְמַן אַתֶּם קוֹנִים, כָּל-עֻמַּת שֶׁרְאִיתֶם כִּי וַדַּאי מֵאִתִּי הַדָּבָר.
ט  דִּי הֵן-חֶלְמָא לָא תְהוֹדְעֻנַּנִי חֲדָה-הִיא דָתְכוֹן וּמִלָּה כִדְבָה וּשְׁחִיתָה הזמנתון (הִזְדְּמִנְתּוּן) לְמֵאמַר קָדָמַי עַד דִּי עִדָּנָא יִשְׁתַּנֵּא לָהֵן חֶלְמָא אֱמַרוּ לִי וְאִנְדַּע דִּי פִשְׁרֵהּ תְּהַחֲוֻנַּנִי׃ ט  שֶׁאִם אֶת הַחֲלוֹם לֹא תּוֹדִיעוּנִי, אֶחָד הוּא מִשְׁפַּטְכֶם, וְדָבָר כּוֹזֵב וּמָשְׁחָת הִתְכּוֹנַנְתֶּם לֵאמֹר לְפָנַי, עַד שֶׁהַזְּמַן יִשְׁתַּנֶּה; אֶלָּא אֶת הַחֲלוֹם אִמְרוּ לִי, וְאֵדַע כִּי אֶת פִּתְרוֹנוֹ תַּגִּידוּנִי.
י  עֲנוֹ כשדיא (כַשְׂדָּאֵי) קֳדָם-מַלְכָּא וְאָמְרִין לָא-אִיתַי אֱנָשׁ עַל-יַבֶּשְׁתָּא דִּי מִלַּת מַלְכָּא יוּכַל לְהַחֲוָיָה  כָּל-קֳבֵל דִּי כָּל-מֶלֶךְ רַב וְשַׁלִּיט מִלָּה כִדְנָה לָא שְׁאֵל לְכָל-חַרְטֹם וְאָשַׁף וְכַשְׂדָּי׃ י  עָנוּ הַכַּשְׂדִּים לִפְנֵי הַמֶּלֶךְ וְאוֹמְרִים: אֵין אָדָם עַל הַיַּבָּשָׁה אֲשֶׁר אֶת דְּבַר הַמֶּלֶךְ יוּכַל לְהַגִּיד, כָּל-עֻמַּת שֶׁכָּל מֶלֶךְ גָּדוֹל וְשַׁלִּיט, דָּבָר כָּזֶה לֹא שָׁאַל מִכֹּל חַרְטֹם, אַשָׁף וְכַשְׂדִּי.
יא  וּמִלְּתָא דִי-מַלְכָּה שָׁאֵל יַקִּירָה וְאָחֳרָן לָא אִיתַי דִּי יְחַוִּנַּהּ קֳדָם מַלְכָּא לָהֵן אֱלָהִין דִּי מְדָרְהוֹן עִם-בִּשְׂרָא לָא אִיתוֹהִי׃ יא  וְהַדָּבָר אֲשֶׁר הַמֶּלֶךְ שׁוֹאֵל כָּבֵד הוּא, וְאַחֵר אַיִן אֲשֶׁר יַגִּידֶנּוּ לִפְנֵי הַמֶּלֶךְ, אֶלָּא אֱלֹהִים אֲשֶׁר אֵין מְדוֹרָם עִם בָּשָׂר.
יב  כָּל-קֳבֵל דְּנָה מַלְכָּא בְּנַס וּקְצַף שַׂגִּיא וַאֲמַר לְהוֹבָדָה לְכֹל חַכִּימֵי בָבֶל׃ יב  כָּל-עֻמַּת זֶה הִתְאַנַּף הַמֶּלֶךְ וְקָצַף מְאֹד, וַיֹּאמֶר לְהַאֲבִיד אֶת כָּל חַכְמֵי בָבֶל.
יג  וְדָתָא נֶפְקַת וְחַכִּימַיָּא מִתְקַטְּלִין וּבְעוֹ דָּנִיֵּאל וְחַבְרוֹהִי לְהִתְקְטָלָה׃    פ יג  וְהַדָּת יָצְאָה, וְהַחֲכָמִים נֶהֱרָגִים, וְעָמְדוּ דָּנִיֵּאל וַחֲבֵרָיו לְהֵהָרֵג.    פ

יד  בֵּאדַיִן דָּנִיֵּאל הֲתִיב עֵטָא וּטְעֵם לְאַרְיוֹךְ רַב-טַבָּחַיָּא דִּי מַלְכָּא דִּי נְפַק לְקַטָּלָה לְחַכִּימֵי בָּבֶל׃ יד  אֲזַי דָּנִיֵּאל הֵשִׁיב עֵצָה וָטַעַם לְאַרְיוֹךְ שַׁר הַטַּבָּחִים שֶׁל הַמֶּלֶךְ, אֲשֶׁר יָצָא לַהֲרֹג אֶת חַכְמֵי בָבֶל.
טו  עָנֵה וְאָמַר לְאַרְיוֹךְ שַׁלִּיטָא דִי-מַלְכָּא עַל-מָה דָתָא מְהַחְצְפָה מִן-קֳדָם מַלְכָּא אֱדַיִן מִלְּתָא הוֹדַע אַרְיוֹךְ לְדָנִיֵּאל׃ טו  עוֹנֶה וְאוֹמֵר לְאַרְיוֹךְ הַמְּמֻנֶּה שֶׁל הַמֶּלֶךְ: לָמָה הַדָּת עַזָּה מִלִּפְנֵי הַמֶּלֶךְ? אֲזַי הוֹדִיעַ אַרְיוֹךְ אֶת הַדָּבָר לְדָנִיֵּאל.
טז  וְדָנִיֵּאל עַל וּבְעָה מִן-מַלְכָּא  דִּי זְמָן יִנְתִּן-לֵהּ וּפִשְׁרָא לְהַחֲוָיָה לְמַלְכָּא׃    פ טז  וְדָנִיֵּאל נִכְנַס וּבִקֵּשׁ מִן הַמֶּלֶךְ, כִּי יִתֶּן לוֹ זְמָן, וְאֶת הַפִּתְרוֹן יַגִּיד לַמֶּלֶךְ.    פ

יז  אֱדַיִן דָּנִיֵּאל לְבַיְתֵהּ אֲזַל וְלַחֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה חַבְרוֹהִי מִלְּתָא הוֹדַע׃ יז  אֲזַי דָּנִיֵּאל לְבֵיתוֹ הָלַךְ, וְלַחֲנַנְיָה, מִישָׁאֵל וַעֲזַרְיָה חֲבֵרָיו אֶת הַדָּבָר הוֹדִיעַ.
יח  וְרַחֲמִין לְמִבְעֵא מִן-קֳדָם אֱלָהּ שְׁמַיָּא עַל-רָזָא דְּנָה דִּי לָא יְהוֹבְדוּן דָּנִיֵּאל וְחַבְרוֹהִי עִם-שְׁאָר חַכִּימֵי בָבֶל׃ יח  וְרַחֲמִים לְבַקֵּשׁ מִלִּפְנֵי אֱלֹהֵי הַשָׁמַיִם עַל הָרָז הַזֶּה, שֶׁלֹּא יֹאבְדוּ דָּנִיֵּאל וַחֲבֵרָיו עִם שְׁאָר חַכְמֵי בָבֶל.
יט  אֱדַיִן לְדָנִיֵּאל בְּחֶזְוָא דִי-לֵילְיָא רָזָא גְלִי אֱדַיִן דָּנִיֵּאל בָּרִךְ לֶאֱלָהּ שְׁמַיָּא׃ יט  אֲזַי נִגְלָה הָרָז לְדָנִיֵּאל בְּחֶזְיוֹן הַלַּיְלָה, אֲזַי בֵּרַךְ דָּנִיֵּאל אֶת אֱלֹהֵי הַשָׁמַיִם.
כ  עָנֵה דָנִיֵּאל וְאָמַר לֶהֱוֵא שְׁמֵהּ דִּי-אֱלָהָא מְבָרַךְ מִן-עָלְמָא וְעַד עָלְמָא  דִּי חָכְמְתָא וּגְבוּרְתָא דִּי לֵהּ-הִיא׃ כ  עוֹנֶה דָּנִיֵּאל וְאוֹמֵר: יְהִי שֵׁם הָאֱלֹהִים מְבֹרָךְ, מִן הָעוֹלָם וְעַד הָעוֹלָם, אֲשֶׁר הַחָכְמָה וְהַגְּבוּרָה שֶׁלּוֹ הִיא.
כא  וְהוּא מְהַשְׁנֵא עִדָּנַיָּא וְזִמְנַיָּא מְהַעְדֵּה מַלְכִין וּמְהָקֵים מַלְכִין יָהֵב חָכְמְתָא לְחַכִּימִין וּמַנְדְּעָא לְיָדְעֵי בִינָה׃ כא  וְהוּא מְשַׁנֶּה אֶת הָעִתִּים וְאֶת הַזְּמַנִּים, מַעֲבִיר מְלָכִים וּמֵקִים מְלָכִים, נוֹתֵן אֶת הַחָכְמָה לַחֲכָמִים, וְאֶת הַמַּדָּע לְיוֹדְעֵי בִינָה.
כב  הוּא גָּלֵא עַמִּיקָתָא וּמְסַתְּרָתָא יָדַע מָה בַחֲשׁוֹכָא ונהירא (וּנְהוֹרָא) עִמֵּהּ שְׁרֵא׃ כב  הוּא מְגַלֶּה אֶת הָעֲמֻקּוֹת וְאֶת הַנִּסְתָּרוֹת, יוֹדֵעַ מַה שֶׁבַּחֹשֶׁךְ, וְהָאוֹר עִמּוֹ שָׁרוּי.
כג  לָךְ אֱלָהּ אֲבָהָתִי מְהוֹדֵא וּמְשַׁבַּח אֲנָה דִּי חָכְמְתָא וּגְבוּרְתָא יְהַבְתְּ לִי וּכְעַן הוֹדַעְתַּנִי דִּי-בְעֵינָא מִנָּךְ דִּי-מִלַּת מַלְכָּא הוֹדַעְתֶּנָא׃ כג  לְךָ אֱלֹהֵי אֲבוֹתַי, מוֹדֶה וּמְשַׁבֵּחַ אֲנִי, אֲשֶׁר אֶת הַחָכְמָה וְהַגְּבוּרָה נָתַתָּ לִי, וְכָעֵת הוֹדַעְתַּנִי אֵת אֲשֶׁר בִּקַּשְׁנוּ מִמֶּךָּ, שֶׁאֶת דְּבַר הַמֶּלֶךְ הוֹדַעְתָּנוּ.
כד  כָּל-קֳבֵל דְּנָה דָּנִיֵּאל עַל עַל-אַרְיוֹךְ דִּי מַנִּי מַלְכָּא לְהוֹבָדָה לְחַכִּימֵי בָבֶל אֲזַל וְכֵן אֲמַר-לֵהּ לְחַכִּימֵי בָבֶל אַל-תְּהוֹבֵד הַעֵלְנִי קֳדָם מַלְכָּא וּפִשְׁרָא לְמַלְכָּא אֲחַוֵּא׃    ס כד  כָּל-עֻמַּת זֶה דָּנִיֵּאל נִכְנַס אֶל אַרְיוֹךְ, אֲשֶׁר מִנָּה הַמֶּלֶךְ לְהַאֲבִיד אֶת חַכְמֵי בָבֶל, הָלַךְ וְכֵן אָמַר לוֹ: אֶת חַכְמֵי בָבֶל אַל תַּאֲבֵד, הַכְנִיסֵנִי לִפְנֵי הַמֶּלֶךְ, וְאֶת הַפִּתְרוֹן לַמֶּלֶךְ אַגִידָה.    ס

כה  אֱדַיִן אַרְיוֹךְ בְּהִתְבְּהָלָה הַנְעֵל לְדָנִיֵּאל קֳדָם מַלְכָּא וְכֵן אֲמַר-לֵהּ דִּי-הַשְׁכַּחַת גְּבַר מִן-בְּנֵי גָלוּתָא דִּי יְהוּד דִּי פִשְׁרָא לְמַלְכָּא יְהוֹדַע׃ כה  אֲזַי הִכְנִיס אַרְיוֹךְ בְּהִתְבַּהֲלוּת אֶת דָּנִיֵּאל לִפְנֵי הַמֶּלֶךְ, וְכֵן אָמַר לוֹ: מָצָאתִי אִישׁ מִבְּנֵי גָּלוּת יְהוּדָה, אֲשֶׁר יוֹדִיעַ אֶת הַפִּתְרוֹן לַמֶּלֶךְ.
כו  עָנֵה מַלְכָּא וְאָמַר לְדָנִיֵּאל דִּי שְׁמֵהּ בֵּלְטְשַׁאצַּר  האיתיך (הַאִיתָךְ) כָּהֵל לְהוֹדָעֻתַנִי חֶלְמָא דִי-חֲזֵית וּפִשְׁרֵהּ׃ כו  עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר לְדָנִיֵּאל, אֲשֶׁר שְׁמוֹ בֵּלְטְשַׁאצַּר: הֲיֶשְׁךָ יָכוֹל לְהוֹדִיעֵנִי אֶת הַחֲלוֹם אֲשֶׁר רָאִיתִי וְאֶת פִּתְרוֹנוֹ?
כז  עָנֵה דָנִיֵּאל קֳדָם מַלְכָּא וְאָמַר  רָזָא דִּי-מַלְכָּא שָׁאֵל לָא חַכִּימִין אָשְׁפִין חַרְטֻמִּין גָּזְרִין יָכְלִין לְהַחֲוָיָה לְמַלְכָּא׃ כז  עוֹנֶה דָּנִיֵּאל לִפְנֵי הַמֶּלֶךְ וְאוֹמֵר: הָרָז אֲשֶׁר הַמֶּלֶךְ שׁוֹאֵל אֵין חֲכָמִים, אַשָׁפִים, חַרְטֻמִּים וְקוֹסְמִים יְכוֹלִים לְהַגִּיד לַמֶּלֶךְ.
כח  בְּרַם אִיתַי אֱלָהּ בִּשְׁמַיָּא גָּלֵא רָזִין וְהוֹדַע לְמַלְכָּא נְבוּכַדְנֶצַּר מָה דִּי לֶהֱוֵא בְּאַחֲרִית יוֹמַיָּא חֶלְמָךְ וְחֶזְוֵי רֵאשָׁךְ עַל-מִשְׁכְּבָךְ דְּנָה הוּא׃    פ כח  אוּלָם, יֵשׁ אֱלוֹהַּ בַּשָׁמַיִם מְגַלֶּה רָזִים, וְהוֹדִיעַ לַמֶּלֶךְ נְבוּכַדְנֶצַּר, מַה שֶׁיִּהְיֶה בְּאַחֲרִית הַיָּמִים, חֲלוֹמְךָ וְחֶזְיוֹנוֹת רֹאשְׁךָ עַל מִשְׁכָּבְךָ זֶה הוּא.    פ

כט  אנתה (אַנְתְּ) מַלְכָּא רַעְיוֹנָךְ עַל-מִשְׁכְּבָךְ סְלִקוּ מָה דִּי לֶהֱוֵא אַחֲרֵי דְנָה וְגָלֵא רָזַיָּא הוֹדְעָךְ מָה-דִי לֶהֱוֵא׃ כט  אַתָּה, הַמֶּלֶךְ, מַחְשְׁבוֹתֶיךָ עַל מִשְׁכָּבְךָ עָלוּ, מַה שֶׁיִּהְיֶה אַחֲרֵי כֵן, וּמְגַלֵּה הָרָזִים הוֹדִיעֲךָ מַה שֶׁיִּהְיֶה.
ל  וַאֲנָה לָא בְחָכְמָה דִּי-אִיתַי בִּי מִן-כָּל-חַיַּיָּא רָזָא דְנָה גֱּלִי לִי לָהֵן עַל-דִּבְרַת דִּי פִשְׁרָא לְמַלְכָּא יְהוֹדְעוּן וְרַעְיוֹנֵי לִבְבָךְ תִּנְדַּע׃ ל  וַאֲנִי, לֹא בְחָכְמָה שֶׁיֶּשׁ בִּי מִכָּל הַחַיִּים נִגְלָה לִי רָז זֶה, אֶלָּא עַל דִּבְרַת, שֶׁאֶת הַפִּתְרוֹן לַמֶּלֶךְ יוֹדִיעוּ, וְאֶת רַעְיוֹנוֹת לִבְּךָ תֵּדָע.
לא  אנתה (אַנְתְּ) מַלְכָּא חָזֵה הֲוַיְתָ וַאֲלוּ צְלֵם חַד שַׂגִּיא צַלְמָא דִּכֵּן רַב וְזִיוֵהּ יַתִּיר קָאֵם לְקָבְלָךְ וְרֵוֵהּ דְּחִיל׃ לא  אַתָּה הַמֶּלֶךְ, רוֹאֶה הָיִיתָ וְהִנֵּה צֶלֶם אֶחָד גָּדוֹל, צֶלֶם זֶה גָּדוֹל הוּא וַהֲדָרוֹ רַב, עוֹמֵד נֶגְדְּךָ וּמַרְאֵהוּ נוֹרָא.
לב  הוּא צַלְמָא רֵאשֵׁהּ דִּי-דְהַב טָב חֲדוֹהִי וּדְרָעוֹהִי דִּי כְסַף מְעוֹהִי וְיַרְכָתֵהּ דִּי נְחָשׁ׃ לב  הַצֶּלֶם הַהוּא, רֹאשׁוֹ שֶׁל זָהָב טוֹב, חָזֵהוּ וּזְרוֹעוֹתָיו שֶׁל כֶּסֶף, מֵעָיו וִירֵכָיו שֶׁל נְחֹשֶׁת.
לג  שָׁקוֹהִי דִּי פַרְזֶל רַגְלוֹהִי מנהון (מִנְּהֵן) דִּי פַרְזֶל ומנהון (וּמִנְּהֵן) דִּי חֲסַף׃ לג  שׁוֹקָיו שֶׁל בַּרְזֶל, רַגְלָיו, מֵהֶן שֶׁל בַּרְזֶל וּמֵהֶן שֶׁל חָרֶשׂ.
לד  חָזֵה הֲוַיְתָ עַד דִּי הִתְגְּזֶרֶת אֶבֶן דִּי-לָא בִידַיִן וּמְחָת לְצַלְמָא עַל-רַגְלוֹהִי דִּי פַרְזְלָא וְחַסְפָּא וְהַדֵּקֶת הִמּוֹן׃ לד  רוֹאֶה הָיִיתָ, עַד שֶׁנִּכְרְתָה אֶבֶן שֶׁלֹּא בְיָדַיִם, וַתַּךְ אֶת הַצֶּלֶם עַל רַגְלָיו שֶׁל בַּרְזֶל וְחֶרֶשׁ, וַתָּדֶק אוֹתָן.
לה  בֵּאדַיִן דָּקוּ כַחֲדָה פַּרְזְלָא חַסְפָּא נְחָשָׁא כַּסְפָּא וְדַהֲבָא וַהֲווֹ כְּעוּר מִן-אִדְּרֵי-קַיִט וּנְשָׂא הִמּוֹן רוּחָא וְכָל-אֲתַר לָא-הִשְׁתְּכַח לְהוֹן וְאַבְנָא דִּי-מְחָת לְצַלְמָא הֲוָת לְטוּר רַב וּמְלָאת כָּל-אַרְעָא׃ לה  אֲזַי הוּדַקוּ הַבַּרְזֶל, הַחֶרֶשׂ, הַנְּחֹשֶׁת, הַכֶּסֶף וְהַזָּהָב, וְהָיוּ כְמֹץ מִגָּרְנוֹת קַיִץ וַתִּשָׂא אוֹתָם הָרוּחַ, וְכָל מָקוֹם לֹא נִמְצָא לָהֶם, וְהָאֶבֶן אֲשֶׁר הִכְּתָה אֶת הַצֶּלֶם הָיְתָה לְהַר גָּדוֹל, וּמָלְאָה כָּל הָאָרֶץ.
לו  דְּנָה חֶלְמָא וּפִשְׁרֵהּ נֵאמַר קֳדָם-מַלְכָּא׃ לו  זֶה הַחֲלוֹם, וְאֶת פִּתְרוֹנוֹ נֹאמַר לִפְנֵי הַמֶּלֶךְ.
לז  אנתה (אַנְתְּ) מַלְכָּא מֶלֶךְ מַלְכַיָּא  דִּי אֱלָהּ שְׁמַיָּא מַלְכוּתָא חִסְנָא וְתָקְפָּא וִיקָרָא יְהַב-לָךְ׃ לז  אַתָּה הַמֶּלֶךְ, מֶלֶךְ הַמְּלָכִים, אֲשֶׁר אֱלֹהֵי הַשָׁמַיִם נָתַן לְךָ אֶת הַמְּלוּכָה, אֶת הָעֹז, וְאֶת הַתֹּקֶף וְאֶת הַכָּבוֹד.
לח  וּבְכָל-דִּי דארין (דָיְרִין) בְּנֵי-אֲנָשָׁא חֵיוַת בָּרָא וְעוֹף-שְׁמַיָּא יְהַב בִּידָךְ וְהַשְׁלְטָךְ בְּכָלְּהוֹן אנתה- (אַנְתְּ-) הוּא רֵאשָׁה דִּי דַהֲבָא׃ לח  וּבְכָל שֶׁדָּרִים בְּנֵי הָאָדָם, אֶת חַיַּת הַשָׂדֶה וְעוֹף הַשָׁמַיִם נָתַן בְּיָדְךָ וְהִשְׁלִיטְךָ בְּכֻלָּם, אַתָּה הוּא רֹאשׁ הַזָּהָב.
לט  וּבָתְרָךְ תְּקוּם מַלְכוּ אָחֳרִי ארעא (אֲרַע) מִנָּךְ וּמַלְכוּ תליתיא (תְלִיתָאָה) אָחֳרִי דִּי נְחָשָׁא דִּי תִשְׁלַט בְּכָל-אַרְעָא׃ לט  וְאַחֲרֶיךָ תָּקוּם מַלְכוּת אַחֶרֶת שְׁפֵלָה מִמְּךָ, וּמַלְכוּת שְׁלִישִׁית אַחֶרֶת מִנְּחֹשֶׁת, אֲשֶׁר תִּמְשֹׁל בְּכָל הָאָרֶץ.
מ  וּמַלְכוּ רביעיה (רְבִיעָאָה) תֶּהֱוֵא תַקִּיפָה כְּפַרְזְלָא כָּל-קֳבֵל דִּי פַרְזְלָא מְהַדֵּק וְחָשֵׁל כֹּלָּא וּכְפַרְזְלָא דִּי-מְרָעַע כָּל-אִלֵּן תַּדִּק וְתֵרֹעַ׃ מ  וּמַלְכוּת רְבִיעִית תִּהְיֶה חֲזָקָה כַּבַּרְזֶל, כָּל-עֻמַּת שֶׁהַבַּרְזֶל מֵדֵק וְטוֹחֵן אֶת הַכֹּל, וְכַבַּרְזֶל הַרוֹעֵעַ -- אֶת כָּל אֵלֶּה תָּדֵק וְתָרֹעַ.
מא  וְדִי-חֲזַיְתָה רַגְלַיָּא וְאֶצְבְּעָתָא מנהון (מִנְּהֵן) חֲסַף דִּי-פֶחָר ומנהון (וּמִנְּהֵן) פַּרְזֶל מַלְכוּ פְלִיגָה תֶּהֱוֵה וּמִן-נִצְבְּתָא דִי-פַרְזְלָא לֶהֱוֵא-בַהּ כָּל-קֳבֵל דִּי חֲזַיְתָה פַּרְזְלָא מְעָרַב בַּחֲסַף טִינָא׃ מא  וַאֲשֶׁר רָאִיתָ הָרַגְלַיִם וְהָאֶצְבָּעוֹת, מֵהֶן חֶרֶשׂ הַיּוֹצֵר וּמֵהֶן בַּרְזֶל, מַלְכוּת חֲלוּקָה תִּהְיֶה, וּמֵעֹצֶם הַבַּרְזֶל יִהְיֶה בָּהּ, כָּל-עֻמַּת שֶׁרָאִיתָ בַּרְזֶל מְעֹרָב בְּחֶרֶשׂ הַטִּין.
מב  וְאֶצְבְּעָת רַגְלַיָּא מנהון (מִנְּהֵן) פַּרְזֶל ומנהון (וּמִנְּהֵן) חֲסַף מִן-קְצָת מַלְכוּתָא תֶּהֱוֵה תַקִּיפָה וּמִנַּהּ תֶּהֱוֵא תְבִירָה׃ מב  וְאֶצְבְּעוֹת הָרַגְלַיִם, מֵהֶן בַּרְזֶל וּמֵהֶן חֶרֶשׁ, מִקְצָת הַמַּלְכוּת תִּהְיֶה חֲזָקָה, וּמִמֶּנָּה תִּהְיֶה שְׁבוּרָה.
מג  די (וְדִי) חֲזַיְתָ פַּרְזְלָא מְעָרַב בַּחֲסַף טִינָא מִתְעָרְבִין לֶהֱוֹן בִּזְרַע אֲנָשָׁא וְלָא-לֶהֱוֹן דָּבְקִין דְּנָה עִם-דְּנָה הֵא-כְדִי פַרְזְלָא לָא מִתְעָרַב עִם-חַסְפָּא׃ מג  וַאֲשֶׁר רָאִיתָ הַבַּרְזֶל מְעֹרָב בְּחֶרֶשׂ הַטִּין, מִתְעָרְבִים יִהְיוּ בְּזֶרַע הָאֲנָשִׁים, וְלֹא יִהְיוּ דְּבֵקִים זֶה בָזֶה, כְּמוֹ שֶׁהַבַּרְזֶל לֹא יִתְעָרֵב עִם הֶחָרֶשׂ.
מד  וּבְיוֹמֵיהוֹן דִּי מַלְכַיָּא אִנּוּן יְקִים אֱלָהּ שְׁמַיָּא מַלְכוּ דִּי לְעָלְמִין לָא תִתְחַבַּל וּמַלְכוּתָה לְעַם אָחֳרָן לָא תִשְׁתְּבִק תַּדִּק וְתָסֵיף כָּל-אִלֵּין מַלְכְוָתָא וְהִיא תְּקוּם לְעָלְמַיָּא׃ מד  וּבִימֵיהֶם שֶׁל הַמְּלָכִים הָהֵם, יָקִים אֱלֹהֵי הַשָׁמַיִם מַלְכוּת, אֲשֶׁר לְעוֹלָם לֹא תִּשָׁחֵת, וּמַלְכוּתָהּ לְעַם אַחֵר לֹא תֵּעָזֵב, תָּדֵק וּתְכַלֶּה כָּל אֵלֶּה הַמַּלְכֻיּוֹת, וְהִיא תַּעֲמֹד לְעוֹלָמִים.
מה  כָּל-קֳבֵל דִּי-חֲזַיְתָ דִּי מִטּוּרָא אִתְגְּזֶרֶת אֶבֶן דִּי-לָא בִידַיִן וְהַדֵּקֶת פַּרְזְלָא נְחָשָׁא חַסְפָּא כַּסְפָּא וְדַהֲבָא אֱלָהּ רַב הוֹדַע לְמַלְכָּא מָה דִּי לֶהֱוֵא אַחֲרֵי דְנָה וְיַצִּיב חֶלְמָא וּמְהֵימַן פִּשְׁרֵהּ׃    פ מה  כָּל-עֻמַּת שֶׁרָאִיתָ, כִּי מִן הָהָר נִכְרְתָה אֶבֶן שֶׁלֹּא בְיָדַיִם, וְהֵדֵקָּה אֶת הַבַּרְזֶל, וְאֶת הַנְּחֹשֶׁת, וְאֶת הַחֶרֶשׂ, וְאֶת הַכֶּסֶף, וְאֶת הַזָּהָב, הָאֵל הַגָּדוֹל הוֹדִיעַ אֶת הַמֶּלֶךְ, מַה שֶׁיִּהְיֶה אַחֲרֵי זֶה, וְנָכוֹן הַחֲלוֹם וְנֶאֱמָן פִּתְרוֹנוֹ.    פ

מו  בֵּאדַיִן מַלְכָּא נְבוּכַדְנֶצַּר נְפַל עַל-אַנְפּוֹהִי וּלְדָנִיֵּאל סְגִד וּמִנְחָה וְנִיחֹחִין אֲמַר לְנַסָּכָה לֵהּ׃ מו  אֲזַי הַמֶּלֶךְ נְבוּכַדְנֶצַּר נָפַל עַל פָּנָיו, וּלְדָנִיֵּאל סָגַד, וּמִנְחָה וְנִיחוֹחִים אָמַר לְנַסֵּךְ לוֹ.
מז  עָנֵה מַלְכָּא לְדָנִיֵּאל וְאָמַר מִן-קְשֹׁט דִּי אֱלָהֲכוֹן הוּא אֱלָהּ אֱלָהִין וּמָרֵא מַלְכִין וְגָלֵה רָזִין  דִּי יְכֵלְתָּ לְמִגְלֵא רָזָא דְנָה׃ מז  עוֹנֶה הַמֶּלֶךְ לְדָנִיֵּאל וְאוֹמֵר: אֱמֶת כִּי אֱלֹהֵיכֶם הוּא אֱלֹהֵי הָאֱלֹהִים, וַאֲדוֹנֵי הָאֲדוֹנִים וּמְגַלֶּה רָזִים, אֲשֶׁר יָכֹלְתָּ לְגַלּוֹת אֶת הָרָז הַזֶּה.
מח  אֱדַיִן מַלְכָּא לְדָנִיֵּאל רַבִּי וּמַתְּנָן רַבְרְבָן שַׂגִּיאָן יְהַב-לֵהּ וְהַשְׁלְטֵהּ עַל כָּל-מְדִינַת בָּבֶל וְרַב-סִגְנִין עַל כָּל-חַכִּימֵי בָבֶל׃ מח  אֲזַי הַמֶּלֶךְ אֶת דָּנִיֵּאל גִּדֵּל, וּמַתָּנוֹת רַבּוֹת גְּדוֹלוֹת נָתַן לוֹ, וְיַשְׁלִיטֵהוּ עַל כָּל מְדִינַת בָּבֶל, וְרַב סְגָנִים עַל כָּל חַכְמֵי בָבֶל.
מט  וְדָנִיֵּאל בְּעָא מִן-מַלְכָּא וּמַנִּי עַל עֲבִידְתָּא דִּי מְדִינַת בָּבֶל לְשַׁדְרַךְ מֵישַׁךְ וַעֲבֵד נְגוֹ וְדָנִיֵּאל בִּתְרַע מַלְכָּא׃    פ מט  וְדָנִיֵּאל בִּקֵּשׁ מִן הַמֶּלֶךְ, וַיְמַן עַל עֲבוֹדַת מְדִינַת בָּבֶל אֶת שַׁדְרַךְ, מֵישַׁךְ וַעֲבֵד נְגוֹ, וְדָנִיֵּאל בְּשַׁעַר הַמֶּלֶךְ.    פ
