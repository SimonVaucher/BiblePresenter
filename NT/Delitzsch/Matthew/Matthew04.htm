הבשורה על-פי מתי
פרק ד

ישוע עומד בניסיון

(מרקוס א' 13-12;לוקס ד' 13-1)

1 אָז נָשָׂא הָרוּחַ אֶת-יֵשׁוּעַ הַמִּדְבָּרָה לְמַעַן יְנַסֵּהוּ הַשָּׂטָן׃ 2 וַיְהִי אַחֲרֵי צוּמוֹ אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה וַיִּרְעָב׃ 3 וַיִּגַּשׁ אֵלָיו הַמְנַסֶּה וַיֹּאמַר אִם בֶּן-הָאֱלֹהִים אַתָּה דַּבֵּר לָאֲבָנִים הָאֵלֶּה וְתִהְיֶינָה לְלָחֶם׃ 4 וַיַּעַן וַיֹּאמַר הֵן כָּתוּב לֹא עַל-הַלֶּחֶם לְבַדּוֹ יִחְיֶה הָאָדָם כִּי עַל-כָּל-מוֹצָא פִי-יְהוָֹה׃(דברים ח' ג') 5 וַיִּשָּׂאֵהוּ הַשָּׂטָן אֶל-עִיר הַקֹּדֶשׁ וַיַּעֲמִידֵהוּ עַל-פִּנַּת-גַּג בֵּית הַמִּקְדָּשׁ׃ 6 וַיֹּאמֶר אֵלָיו אִם בֶּן-הָאֱלֹהִים אַתָּה נְפֹל לְמָטָּה כִּי כָתוּב כִּי מַלְאָכָיו יְצַוֶּה-לָּךְ וְעַל-כַּפַּיִם יִשָּׂאוּנְךָ פֶּן-תִּגֹּף בָּאֶבֶן רַגְלֶךָ׃(תהילים צא' יא'-יב') 7 וַיֹּאמֶר אֵלָיו יֵשׁוּעַ וְעוֹד כָּתוּב לֹא תְנַסֶּה אֵת יְהוָֹה אֱלֹהֶיךָ׃(דברים ו' טז') 8 וַיּוֹסֶף הַשָּׂטָן וַיִּשָּׂאֵהוּ אֶל-הַר גָּבֹהַּ מְאֹד וַיַּרְאֵהוּ אֶת-כָּל מַמְלְכוֹת תֵּבֵל וּכְבוֹדָן׃ 9 וַיֹּאמֶר אֵלָיו כָּל-זֹאת לְךָ אֶתְּנֶנָּה אִם-תִּקֹּד וְתִשְׁתַּחֲוֶה לִי׃ 10 וַיֹּאמֶר אֵלָיו יֵשׁוּעַ סוּר מִמֶּנִּי הַשָּׂטָן כִּי כָתוּב לַיהוָֹה אֱלֹהֶיךָ תִּשְׁתַּחֲוֶה וְאוֹתוֹ לְבַדּוֹ תַּעֲבֹד׃(דברים ו' יג'; מלכים ב', יז' לה'-לו') 11 וַיִּרֶף מִמֶּנּוּ הַשָּׂטָן וְהִנֵּה נִגְּשׁוּ אֵלָיו מַלְאָכִים וַיְשָׁרֲתוּהוּ׃

ישוע מתחיל את שליחותו בגליל

(מרקוס א' 15-14;לוקס ד' 15-14)

12 וַיְהִי כִּשְׁמֹעַ יֵשׁוּעַ כִּי הִסְגִּירוּ אֶת-יוֹחָנָן וַיֵּלֶךְ אֶל אֶרֶץ הַגָּלִיל׃ 13 וַיֵּצֵא מִנְּצֶרֶת וַיָּבֹא וַיֵּשֵׁב בִּכְפַר-נַחוּם אֲשֶׁר עַל-שְׁפַת הַיָּם בִּגְבוּל זְבֻלוּן וְנַפְתָּלִי׃ 14 לְמַלֹאת הַנֶּאֱמָר עַל-פִּי יְשַׁעְיָהוּ הַנָּבִיא לֵאמֹר׃ 15 אַרְצָה זְבֻלוּן וְאַרְצָה נַפְתָּלִי דֶּרֶךְ הַיָּם עֵבֶר הַיַּרְדֵּן גְלִיל הַגּוֹיִם׃(ישעיהו ח' כג') 16 הָעָם הָהֹלְכִים בַּחֹשֶׁךְ רָאוּ אוֹר גָּדוֹל ישְׁבֵי בְּאֶרֶץ צַלְמָוֶת אוֹר נָגַהּ עֲלֵיהֶם׃(ישעיהו ט' א'-ב') 17 מִן-הָעֵת הַהִיא הֵחֵל יֵשׁוּעַ לִקְרֹא קָרוֹא וְאָמוֹר שׁוּבוּ כִּי מַלְכוּת הַשָׁמַיִם קָרְבָה לָבוֹא׃

הקריאה לארבעה דייגים

(מרקוס א' 20-16;לוקס ה' 11-1)

18 וַיְהִי בְּהִתְהַלְּכוֹ עַל-יַד יָם-הַגָּלִיל וַיַּרְא וְהִנֵּה שְׁנֵי אֲנָשִׁים אַחִים שִׁמְעוֹן הַנִּקְרָא פֶטְרוֹס וְאַנְדְּרַי אָחִיו מַשְׁלִיכִים מְצוֹדָה בַּיָּם כִּי דַיָּגִים הֵם׃ 19 וַיֹּאמֶר אֲלֵיהֶם לְכוּ אַחֲרָי וַאֲשִׂימְכֶם לְדַיְּגֵי אֲנָשִׁים׃ 20 וַיַּעַזְבוּ מְהֵרָה אֵת הַמִּכְמֹרוֹת וַיֵּלְכוּ אַחֲרָיו׃ 21 וַיְהִי כְּעָבְרוֹ מִשָׁם וַיַּרְא שְׁנֵי אֲנָשִׁים אַחִים אֲחֵרִים אֶת-יַעֲקֹב בֶּן-זַבְדַּי וְאֶת-יוֹחָנָן אָחִיו בָּאֳנִיָּה עִם-זַבְדַּי אֲבִיהֶם וְהֵמָּה מְתַקְּנִים אֶת-מִכְמְרוֹתָם וַיִּקְרָא אֲלֵיהֶם׃ 22 וַיַּעַזְבוּ מְהֵרָה אֶת-הָאֳנִיָּה וְאֶת-אֲבִיהֶם וַיֵּלְכוּ אַחֲרָיו׃

ישוע משרת את ההמונים

(לוקס ו' 19-17)

23 וַיָּסָב יֵשׁוּעַ בְּכָל-הַגָּלִיל וַיְלַמֵּד בִּכְנֵסִיּוֹתֵיהֶם וַיְבַשֵּׂר בְּשׂוֹרַת הַמַּלְכוּת וַיְרַפֵּא כָּל-מַחֲלָה וְכָל-מַדְוֶה בָּעָם׃ 24 וְשִׁמְעוֹ יָצָא בְּכָל-אֶרֶץ סוּרְיָא וַיָּבִיאוּ אֵלָיו אֵת כָּל-הַחוֹלִים הַמְעֻנִּים בְּכָל-חֳלָיִם וּמַכְאוֹבִים וַאֲחוּזֵי שֵׁדִים וּמֻכֵּי יָרֵחַ וּנְכֵי אֵבָרִים וַיִּרְפָּאֵם׃ 25 וַיֵּלְכוּ אַחֲרָיו הֲמֹנִים הֲמֹנִים מִן-הַגָּלִיל וּמִן-עֶשֶׂר הֶעָרִים וְמִירוּשָׁלַיִם וִיהוּדָה וּמֵעֵבֶר לַיַּרְדֵּן׃
