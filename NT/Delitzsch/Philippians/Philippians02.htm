אגרת שאול אל הפיליפים
פרק ב

התנהגות המשיח מופת למאמינים

1 לָכֵן אִם-יֵשׁ תּוֹכַחַת בַּמָּשִׁיחַ אִם-תַּנְחוּמוֹת הָאַהֲבָה אִם-חֶבְרַת רוּחַ אִם-רַחֲמִים וְחֶמְלָה׃ 2 הַשְׁלִימוּ-נָא אֶת-שִׂמְחָתִי בִּהְיוֹת לָכֶם לֵב אֶחָד וְאַהֲבָה אַחַת וְנֶפֶשׁ אַחַת וְרָצוֹן אֶחָד׃ 3 וְאַל-תַּעֲשׂוּ דָבָר בְּדֶרֶךְ מְרִיבָה אוֹ כְּבוֹד שָׁוְא כִּי אִם-בְּשִׁפְלוּת רוּחַ יַחְשֹׁב אִישׁ אֶת-רֵעֵהוּ גָּדוֹל מִמֶּנּוּ׃ 4 כָּל-אֶחָד אַל-יִדְאַג לַאֲשֶׁר-לוֹ לְבַדּוֹ כִּי אִם-גַּם לַאֲשֶׁר לַחֲבֵרוֹ׃ 5 כִּי יְהִי לְבַבְכֶם כִּלְבַב הַמָּשִׁיחַ יֵשׁוּעַ׃ 6 אֲשֶׁר אַף כִּי-הָיָה בִּדְמוּת הָאֱלֹהִים לֹא-חָשַׁב לוֹ לְשָׁלָל הֱיוֹתוֹ שָׁוֶה לֵאלֹהִים׃ 7 כִּי אִם-הִפְשִׁיט אֶת-עַצְמוֹ וַיִּלְבַּשׁ דְּמוּת עֶבֶד (ישעיהו נג' ג') וַיְהִי דוֹמֶה לִבְנֵי אָדָם וַיִּמָּצֵא בִתְכוּנָתוֹ כְּבֶן אָדָם׃(ישעיהו מב' א') 8 וַיַּשְׁפֵּל אֶת-נַפְשׁוֹ וַיִּכָּנַע עַד-מָוֶת עַד-מִיתַת הַצְּלִיבָה׃(תהילים מ' ט') 9 עַל-כֵּן הִגְבִּיהוֹ הָאֱלֹהִים מְאֹד וַיִּתֵּן-לוֹ שֵׁם נַעֲלֶה עַל-כָּל-שֵׁם׃ 10 אֲשֶׁר בְּשֵׁם יֵשׁוּעַ תִּכְרַע כָּל-בֶּרֶךְ אֲשֶׁר בַּשָּׁמַיִם וּבָאָרֶץ וּמִתַּחַת לָאָרֶץ׃(תהילים כב' ל'; ישעיהו מה' כג') 11 וְכָל-לָשׁוֹן תּוֹדֶה כִּי אָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ לִכְבוֹד אֱלֹהִים הָאָב׃ 12 לָכֵן חֲבִיבַי כַּאֲשֶׁר שְׁמַעְתֶּם לִי בְּכָל-עֵת כֵּן לֹא בִּהְיוֹתִי אֶצְלְכֶם בִּלְבָד אֶלָּא-עוֹד יֹתֵר עַתָּה כְּשֶׁאֲנִי רָחוֹק מִכֶּם תִּיגְעוּ לִתְשׁוּעַת נַפְשֹׁתֵיכֶם בְּיִרְאָה וּבִרְעָדָה׃(תהילים ב' יא') 13 כִּי הָאֱלֹהִים הוּא הַפֹּעֵל בָּכֶם גַּם-לַחְפֹּץ גַּם-לִפְעֹל כְּפִי רְצוֹנוֹ׃ 14 עֲשׂוּ כָּל-דָּבָר בְּלֹא תְלֻנּוֹת וּבְלֹא מְזִמּוֹת׃ 15 לְמַעַן תִּהְיוּ נְקִיִּים וּטְהוֹרִים בְּנֵי אֱלֹהִים אֵין-מוּם בָּם בְּתוֹךְ דּוֹר עִקֵּשׁ וּפְתַלְתֹּל אֲשֶׁר-תָּאִירוּ בֵינֵיהֶם כִּמְאוֹרֹת בָּעוֹלָם׃(דניאל יב' ג') 16 מַחֲזִיקִים בִּדְבַר הַחַיִּים לִתְהִלָּה לִי בְּיוֹם הַמָּשִׁיחַ אֲשֶׁר לֹא לְחִנָּם רַצְתִּי וְלֹא לָרִיק יָגָעְתִּי׃(ישעיהו סה' כג') 17 אֲבָל גַּם כִּי-אֻסַּךְ עַל-זֶבַח אֱמוּנַתְכֶם וַעֲבוֹדָתָהּ (במדבר טו' ה'-י') הִנְנִי שָׂמֵחַ וְשָׂשׂ עִם-כֻּלְּכֶם׃ 18 וְכָכָה שִׂמְחוּ אַף-אַתֶּם וְשִׂישׂוּ עִמָּדִי׃

טימוטיאוס ואפפרודיטוס

19 וְקִוֵּיתִי בְּיֵשׁוּעַ אֲדֹנֵינוּ לִשְׁלֹחַ אֲלֵיכֶם בִּמְהֵרָה אֶת-טִימוֹתִיּוֹס לְמַעַן תָּנוּחַ דַּעְתִּי בְּהִוָּדַע לִי שְׁלוֹמְכֶם׃ 20 כִּי זוּלָתוֹ אֵין-אִתִּי אִישׁ כִּלְבָבִי אֲשֶׁר בְּלֵב שָׁלֵם יִדְאַג לָכֶם׃ 21 כִּי כֻלָּם אֵת אֲשֶׁר-לָהֶם יִדְרשׁוּ וְלֹא אֶת-אֲשֶׁר לְיֵשׁוּעַ הַמָּשִׁיחַ׃ 22 וְאוֹתוֹ יְדַעְתֶּם כִּי בָחוּן הוּא אֲשֶׁר כַּבֵּן הָעֹבֵד אֶת-אָבִיו כֵּן הָיָה אִתִּי בַּעֲבוֹדַת הַבְּשׂוֹרָה׃ 23 וְאֹתוֹ אֲקַוֶּה לִשְׁלֹחַ אֲלֵיכֶם מַהֵר כַּאֲשֶׁר אֶרְאֶה מַה-יִּהְיֶה לִי׃ 24 וּבָטַחְתִּי בַאֲדֹנֵינוּ אֲשֶׁר גַּם-אָנֹכִי אָבוֹא אֲלֵיכֶם בִּמְהֵרָה׃ 25 וְחָשַׁבְתִּי לְחוֹבָה לִי לִשְׁלֹחַ אֲלֵיכֶם אֶת-אֶפַּפְרוֹדִיטוֹס אָחִי וְעֹזְרִי וַחֲבֵרִי בַצָּבָא וְהוּא שְׁלִיחֲכֶם וּמְשָׁרֲתִי בִּצְרָכָי׃ 26 מִפְּנֵי שֶׁהָיָה נִכְסָף לְכֻלְּכֶם וְנֶעְצָב מְאֹד עַל-אֲשֶׁר שְׁמַעְתֶּם כִּי חָלָה׃ 27 אָמְנָם חָלֹה חָלָה וְנָטָה לָמוּת אֲבָל הָאֱלֹהִים רִחַם עָלָיו וְלֹא עָלָיו בִּלְבַד כִּי גַּם-עָלַי רִחַם שֶׁלּא-יָבוֹא עָלַי יָגוֹן עַל-יָגוֹן׃ 28 לָכֵן מִהַרְתִּי וּשְׁלַחְתִּיו אֲלֵיכֶם לְמַעַן תִּרְאֻהוּ וְתָשׁוּבוּ לִשְׂמוֹחַ וְגַם-יִמְעַט יְגוֹנִי׃ 29 עַל-כֵּן קַבְּלֻהוּ בַאֲדֹנֵינוּ בְּכָל-שִׂמְחָה וְהוֹקִירוּ אֲנָשִׁים כָּמֹהוּ׃ 30 כִּי בְעַד מַעֲשֵׂה הַמָּשִׁיחַ הִגִּיעַ עַד-מָוֶת וַתֵּקַל נַפְשׁוֹ בְּעֵינָיו לְמַעַן יְמַלֵּא אֶת-אֲשֶׁר חֲסַרְתֶּם בְּשָׁרֶתְכֶם אוֹתִי׃
