לפרק הקודם

אגרת אל העברים
פרק יב

מוסר ה'

1 לָכֵן גַּם-אֲנַחְנוּ אֲשֶׁר-עֲנַן עֵדִים רַב כָּזֶה מְסֹבֵב אֹתָנוּ נַשְׁלִיכָה מִמֶּנּוּ כָּל-טֹרַח וְהַחֵטְא הַמַּקִּיף עָלֵינוּ וְנָרוּצָה בְסַבְלָנוּת אֶת-הַמְּרוּצָה הָעֲרוּכָה לְפָנֵינוּ׃ 2 וְנַבִּיטָה אֶל-יֵשׁוּעַ רֹאשׁ הָאֱמוּנָה וּמַשְׁלִימָהּ (תהילים כה' טו') אֲשֶׁר בְּעַד הַשִׂמְחָה הַשְּׁמוּרָה לוֹ קִבֵּל אֶת-הַצְּלוּב וַיִּבֶז הַחֶרְפָּה וַיֵּשֶׁב לִימִין כִּסֵּא הָאֱלֹהִים׃(תהילים קי' א') 3 הִתְבּוֹנֲנוּ אֵלָיו אֲשֶׁר-נָשָׂא כְלִמָּה גְּדוֹלָה כָזֹאת מֵאֵת הַחַטָּאִים לְמַעַן אֲשֶׁר לֹא תִיעֲפוּ וְלֹא תִיגְעוּ בְּנַפְשׁוֹתֵיכֶם׃ 4 עֲדַיִן לֹא עֲמַדְתֶּם בְּמִלְחַמְתְּכֶם עִם-הַחֵטְא עַד-לַדָּם׃ 5 וַתִּשְׁכְּחוּ נִחוּמֵי הַמְדַבֵּר אֲלֵיכֶם כְּדַבֵּר אֶל-בָּנִים לֵאמֹר מוּסַר יְהוָֹה בְּנִי אַל-תִּמְאָס וְאַל-תָּקֹץ בְּתוֹכַחְתּוֹ׃(איוב ה' יז') 6 כִּי אֶת אֲשֶׁר יֶאֱהַב יְהוָֹה יוֹכִיחַ יַכְאִב אֶת-בֵּן יִרְצֶה׃(משלי ג' יא'-יב') 7 אִם-מְקַבְּלִים אַתֶּם מוּסָר דְּעוּ כִּי-כְאָב עִם-בָּנָיו מִתְנַהֵג אֱלֹהִים עִמָּכֶם כִּי מִי הַבֵּן אֲשֶׁר אָבִיו לֹא יְיַסְּרֶנּוּ׃ 8 וְאִם-תִּהְיוּ בְאֵין-מוּסָר אֲשֶׁר הָיָה מְנָת כֻּלָּם אָז מַמְזְרִים אַתֶּם וְלֹא בָנִים׃ 9 וְעוֹד אִם-אֲבוֹת בְּשָׂרֵנוּ הָיוּ מְיַסְּרִים אֹתָנוּ וַנִּירָא מֵהֶם אַף כִּי-נִכָּנַע לִפְנֵי אֲבִי הָרוּחוֹת וְנִחְיֶה׃ 10 כִּי הֵמָּה יִסְּרוּנוּ כַּטּוֹב בְּעֵינֵיהֶם לְשָׁעָה קַלָּה אֲבָל זֶה לְהוֹעִיל לְמַעַן יִהְיֶה-לָּנוּ חֵלֶק בִּקְדֻשָּׁתוֹ׃ 11 וְכָל-מוּסָר כְּשֶׁבָּא עָלֵינוּ אֵינֶנּוּ שִׂמְחָה בְעֵינֵינוּ כִּי אִם-יָגוֹן אֲבָל בְּאַחֲרִיתוֹ יִתֵּן פְּרִי שָׁלוֹם לִצְדָקָה לִמְלֻמָּדָיו׃ 12 עַל-כֵּן חַזְּקוּ יָדַיִם רָפוֹת וּבִרְכַּיִם כֹּשְׂלוֹת׃(ישעיהו לה' ג') 13 וּמַעְגַּל רַגְלֵיכֶם פַּלֵּסוּ (משלי ד' כו') לְמַעַן לֹא-תִטֶּה הַצֹּלֵעָה מִן-הַדֶּרֶךְ כִּי אִם-תֵּרָפֵא׃

הזהרות

14 רִדְפוּ אֶת-הַשָּׁלוֹם עִם-כָּל-אָדָם וְאֵת הַקְדֻשָּׁה אֲשֶּׁר בִּלְעָדֶיהָ לֹא-יִרְאֶה אִישׁ אֶת-הָאָדוֹן׃(תהילים לד' טו') 15 וְהִזָּהֲרוּ פֶּן-יֵשׁ בָּכֶם אִישׁ אֲשֶׁר יֶחְדַּל מֵחֶסֶד אֱלֹהִים פֶּן-יָעִיק לָכֶם שֹׁרֶשׁ פֹּרֶה לַעֲנָה (דברים כט' יז') וְיִטַּמְּאוּ בוֹ רַבִּים׃ 16 פֶּן-יִמָּצֵא בָכֶם זֹנֶה אוֹ חָלָל כְּעֵשָׂו אֲשֶׁר בְּנָזִיד אֶחָד מָכַר אֶת-בְּכוֹרָתוֹ׃(בראשית כה' לג') 17 הֲלֹא יְדַעְתֶּם כִּי נִמְאַס אַחֲרֵי כֵן כַּאֲשֶׁר רָצָה לָרֶשֶׁת אֶת-הַבְּרָכָה כִּי לֹא-מָצָא מָקוֹם לִתְשׁוּבָה אַף כִּי-בִקֵּשׁ אוֹתָהּ בִּדְמָעוֹת׃(בראשית כז' לח') 18 כִּי לֹא-בָאתֶם אֶל-הַר נִמְשָׁשׁ וּבֹעֵר בָּאֵשׁ וְלֹא אֶל-עָנָן וַעֲרָפֶל וּסְעָרָה׃(דברים ד' יא') 19 וְלֹא לְקוֹל שׁוֹפָר וּלְקוֹל הַדְּבָרִים אֲשֶׁר שֹׁמְעָיו בִּקְשׁוּ שֶׁלּא יוֹסִיף לְדַבֵּר עִמָּהֶם עוֹד׃(שמות יט' יט'; כ' טו') 20 כִּי לֹא-יָכְלוּ לָשֵׂאת אֵת אֲשֶׁר צֻוּוּ גַּם-אִם-בְּהֵמָה תִּגַּע בָּהָר סָקֹל תִּסָּקֵל [אוֹ-יָרֹה תִיָּרֶה בַּחִצִּים]׃(שמות יט' יב'-יג') 21 וְהַמַּרְאֶה הָיָה נוֹרָא עַד-מְאֹד וַיֹּאמֶר משֶׁה יָגֹרְתִּי וְחָרָדְתִּי׃(דברים ט' יט') 22 כִּי אִם-בָּאתֶם אֶל-הַר צִיּוֹן וְאֶל-עִיר אֱלֹהִים חַיִּים אֶל-יְרוּשָׁלַיִם שֶׁבַּשָּׁמָיִם׃(דניאל ז' י') 23 וְאֶל-מַקְהֵלַת רִבֲבוֹת הַמַּלְאָכִים וַעֲדַת הַבְּכוֹרִים הַכְּתוּבִים בַּשָּׁמָיִם (דניאל יב' א') וְאֶל-אֱלֹהִים שֹׁפֵט הַכֹּל וְאֶל-רוּחוֹת הַצַּדִּיקִים הַנִּשְׁלָמִים׃(בראשית יח' כה') 24 וְאֶל-יֵשׁוּעַ מְתַוֵּךְ הַבְּרִית הַחֲדָשָׁה וְאֶל-דַּם הַהַזָּאָה הַמֵּיטִיב דַּבֵּר מִדַּם-הָבֶל׃(ירמיהו לא' ל'-לד') 25 לָכֵן רְאוּ פֶּן-תְּמָאֲנוּ לִשְׁמֹעַ אֶל-הַמְדַבֵּר כִּי הֵן לֹא נִמְלְטוּ הַמֵּאֲנִים לִשְׁמֹעַ אֶל-הַמְדַבֵּר עִמָּהֶם בָּאָרֶץ אַף כִּי-אֲנַחְנוּ אִם-נְמָאֵן לִשְׁמֹעַ בְּקוֹל הַמְדַבֵּר מִן-הַשָּׁמָיִם׃ 26 אֲשֶׁר קוֹלוֹ הִרְעִישׁ אָז אֶת-הָאָרֶץ וְעַתָּה זֶה הִבְטִיחַ לֵאמֹר עוֹד אַחַת וַאֲנִי מַרְעִישׁ לֹא אֶת-הָאָרֶץ בִּלְבַד אֶלָּא אַף אֶת-הַשָּׁמָיִם׃(חגי ב' ו') 27 וְזוּ עוֹד אַחַת שֶׁאָמַר מַשְׁמִיעַ שִׁנּוּי הַדְּבָרִים הַמִּתְעַרְעָרִים אֲשֶׁר הֵם עֲשׂוּיִם לְמַעַן יַעֲמֹד אֲשֶׁר אֵינֶנּוּ נִרְעָשׁ׃(דניאל ב' מד') 28 לָכֵן אֲנַחְנוּ הַמְקַבְּלִים מַלְכוּת אֲשֶׁר לֹא תִמּוֹט נָבֹאָה-נָּא בְתוֹדָה וְנַעֲבֹד בָּהּ אֶת-הָאֱלֹהִים כִּרְצוֹנוֹ בִּצְנִיעוּת וּבְיִרְאָה׃ 29 כִּי אֱלֹהֵינוּ אֵשׁ אֹכְלָה הוּא׃(דברים ד' כד')
