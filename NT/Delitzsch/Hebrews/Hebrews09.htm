פרק ט

המקדש בארץ לעומת המקדש בשמים

1 הֵן גַּם-[הַבְּרִית הָרִאשׁוֹנָה גַּם לָהּ] הָיוּ דִּינֵי עֲבוֹדָה וּמִקְדָּשׁ בָּאָרֶץ׃ 2 כִּי-הוּקַם הַמִּשְׁכָּן הַחִיצוֹן אֲשֶׁר-בּוֹ הַמְּנוֹרָה וְהַשֻּׁלְחָן וּמַעֲרֶכֶת הַלֶּחֶם וְהוּא נִקְרָא קֹדֶשׁ׃(שמות מ' כב'-כד') 3 וּמִבֵּית לַפָּרֹכֶת הַשֵּׁנִית מִשְׁכָּן הַנִּקְּרָא קֹדֶשׁ הַקֳּדָשִׁים׃(שמות כו' לא'-לג') 4 אֲשֶׁר-לוֹ מִזְבַּח הַזָּהָב לִקְטֹרֶת וַאֲרוֹן הַבְּרִית מְצֻפֶּה זָהָב כֻּלּוֹ וּבוֹ צִנְצֶנֶת זָהָב אֲשֶׁר הַמָּן בְּתוֹכוֹ וּמַטֵּה אַהֲרֹן אֲשֶׁר פָּרַח וְלוּחוֹת הַבְּרִית׃(ויקרא טז' יב'; שמות טז' לג'; כה' י'; במדבר יז' כג'; דברים י' ה') 5 וּמִמַּעַל לוֹ כְּרוּבֵי הַכָּבוֹד הַסֹּכֲכִים עַל-הַכַּפֹּרֶת(ישעיהו לז' טז') לֹא נְדַבֵּר כָּעֵת עַל-כָּל-אֶחָד מֵהֶם לְבָד׃ 6 וְאַחֲרֵי נַעֲשׂוּ אֵלֶּה כָּכָה בָּאוּ הַכֹּהֲנִים תָּמִיד אֶל-הַמִּשְׁכָּן הַחִיצוֹן לַעֲבֹד שָׁם אֶת-עֲבוֹדָתָם׃(שמות ל' ז'-ח') 7 וְהַמִּשְׁכָּן אֲשֶׁר לִפְנִים מִמֶּנּוּ שָׁמָּה בָּא הַכֹּהֵן הַגָּדוֹל לְבַדּוֹ פַּעַם אַחַת בַּשָּׁנָה לֹא בִּבְלִי-דָם אֲשֶׁר יַקְרִיב בְּעַד נַפְשׁוֹ וּבְעַד שִׁגְגוֹת הָעָם׃(ויקרא טז' כט'-לד') 8 וְרוּחַ הַקֹּדֶשׁ מוֹדִיעַ בָּזֹאת שֶׁלּא-נִגְלָה הַדֶּרֶךְ אֶל-הַקֹּדֶשׁ כָּל הַיָּמִים אֲשֶׁר יַעֲמֹד הַמִּשְׁכָּן הַחִיצוֹן׃ 9 וְהוּא מָשָׁל לַזְּמַן הַזֶּה אֲשֶׁר בּוֹ מַקְרִיבִים מְנָחוֹת וּזְבָחִים אֲשֶׁר אֵין בָּהֶם לְהַשְׁלִים אֶת-לְבַב הָעֹבֵד׃ 10 כִּי אִם-חֻקּוֹת הַבָּשָׂר הֵנָּה עִם-הַמַּאֲכָלוֹת וְהַמַּשְׁקִים וְהַטְּבִילוֹת הַשֹּׁנוֹת אֲשֶׁר נִתְּנוּ עַד-עֵת הַתִּקּוּן׃ 11 וְהַמָּשִׁיחַ בְּבֹאוֹ לִהְיוֹת כֹּהֵן גָּדוֹל לַטֹּבוֹת הָעֲתִידוֹת עָבַר בְּתוֹךְ הַמִּשְׁכָּן הַמְעֻלֶּה בִּגְדֻלָּה וּשְׁלֵמוּת אֲשֶׁר לֹא-נַעֲשָׂה בִּידֵי אָדָם כִּי-אֵינֶנּוּ מֵהַבְּרִיאָה הַזֹּאת׃ 12 גַּם לֹא-בָא בְּדַם שְׂעִירִים וַעֲגָלִים כִּי בְדַם-נַפְשׁוֹ בָּא בְּפַעַם-אַחַת אֶל-הַקֹּדֶשׁ פְּנִימָה וַיִּמְצָא גְּאֻלַּת עוֹלָם׃(ויקרא טז' טו') 13 כִּי אִם-דַּם הַפָּרִים וְהַשְׂעִירִים וְאֵפֶר הַפָּרָה אֲשֶׁר יֻזֶּה עַל-הַטְּמֵאִים יְקַדְּשֵׁם לְטַהֵר בְּשָׂרָם׃ 14 אַף כִּי-דַם הַמָּשִׁיחַ אֲשֶׁר-הִקְרִיב אֶת-עַצְמוֹ לֵאלֹהִים בְּרוּחַ נִצְחִי וּבְלִי-מוּם יְטַהֵר לִבְּכֶם מִמַּעֲשֵׂי מָוֶת לַעֲבֹד אֶת-אֱלֹהִים חַיִּים׃ 15 וּבַעֲבוּר זֹאת הוּא מְתַוֵךְ לִבְרִית חֲדָשָׁה לְמַעַן אֲשֶׁר-יִירְשׁוּ הַמְקֹרָאִים אֶת-הַבְטָחַת נַחֲלַת עוֹלָם אַחֲרֵי אֲשֶׁר-מֵת לִפְדוֹת מִן-הַפְּשָׁעִים אֲשֶׁר נַעֲשׂוּ בִּימֵי הַבְּרִית הָרִאשׁוֹנָה׃(ירמיהו לא' ל'-לב') 16 כִּי בִמְקוֹם שֶׁיֵּשׁ בְּרִית הִיא צַוָּאָה צָרִיךְ שֶׁתִּוָּדַע מִיתַת הַנֹּתֵן׃ 17 כִּי רַק-בְּמוֹת הַמֵּת תִּכּוֹן הַצַּוָּאָה וְאֵין לָהּ תֹּקֶף בְּחַיֵּי הַנֹּתֵן׃ 18 לָכֵן גַּם-הָרִאשׁוֹנָה לֹא חֻנְּכָה בְּלֹא-דָם׃ 19 כִּי כְּכַלּוֹת משֶׁה לְסַפֵּר לְכָל-הָעָם אֶת-כָּל-מִשְׁפְּטֵי הַתּוֹרָה לָקַח דַם הָעֲגָלִים וְהַשְׂעִירִים עִם-מַיִם וְתוֹלַעַת שָׁנִי וְאֵזוֹב וַיִּזְרֹק עַל-הַסֵּפֶר וְעַל כָּל-הָעָם׃ 20 וַיֹּאמַר הִנֵּה דַם-הַבְּרִית אֲשֶׁר צִוָּה אֱלֹהִים אֲלֵיכֶם׃(שמות כד' ו'-ח') 21 וְגַם עַל-הַמִּשְׁכָּן וְעַל כָּל-כְּלֵי הַשָּׁרֵת הִזָּה דָם׃ 22 וְכִמְעַט הַכֹּל יְטֻהַר בַּדָּם עַל-פִּי הַתּוֹרָה וְאֵין כַּפָּרָה בְּלֹא-שְׁפִיכַת דָּם׃(ויקרא יז' יא')

קרבן המשיח מסיר חטאים

23 לָכֵן דִּמְיוֹנֵי הַדְּבָרִים שֶׁבַּשָּׁמַיִם צְרִיכִים שֶׁיִּטָּהֲרוּ בָאֵלֶּה וְהַדְּבָרִים שֶׁבַּשָּׁמַיִם בְּעַצְמָם צְרִיכִים שֶׁיִּטָּהֲרוּ בִּזְבָחִים טוֹבִים מֵאֵלֶּה׃(שמות כה' מ') 24 כִּי הַמָּשִׁיחַ לֹא-בָא אֶל-הַקֹּדֶשׁ הַנַּעֲשֶׂה בִּידֵי אָדָם שֶׁהוּא רַק-דְּמוּת הָאֲמִתִּי כִּי אִם-בָּא אֶל-עֶצֶם הַשָּׁמַיִם לֵרָאוֹת עַתָּה בַעֲדֵנוּ אֶת-פְּנֵי הָאֱלֹהִים׃ 25 אַף-לֹא לְהַקְרִיב אֶת-נַפְשׁוֹ פְּעָמִים רַבּוֹת כַּכֹּהֵן הַגָּדוֹל אֲשֶׁר-בָּא שָׁנָה בְּשָׁנָה אֶל-הַקֹּדֶשׁ בְּדַם אֲחֵרִים׃(***) 26 כִּי אִם-כֵּן הֲלֹא הָיָה-לוֹ לֵעָנוֹת פְּעָמִים רַבּוֹת מֵרֵאשִׁית הָעוֹלָם וְעַתָּה בְּקֵץ הָעִתִּים נִגְלָה בְּפַעַם אַחַת כְּדֵי-לְכַלֵּא אֶת-הַחֵטְא בְּזֶבַח נַפְשׁוֹ׃(דניאל ט' כד') 27 וְכַאֲשֶׁר נִגְזַר עַל-בְּנֵי אָדָם לָמוּת פַּעַם אֶחָת וְאַחֲרֵי-כֵן הַמִּשְׁפָּט׃(בראשית ג' יט'; דניאל יב' ב') 28 כֵּן-הֻקְרַב הַמָּשִׁיחַ פַּעַם אַחַת לָשֵׂאת חֵטְא רַבִּים (ישעיהו נג' יא') וּפַעַם שֵׁנִית יֵרָאֶה בְלִי-חֵטְא לִישׁוּעָה לַמְחַכִּים לוֹ׃
