הבשורה על-פי יוחנן

הנס הראשון

1 וּבַיּוֹם הַשְּׁלִישִׁי הָיְתָה חֲתֻנָּה בְּקָנָה אֲשֶׁר בַּגָּלִיל וְשָׁם אֵם יֵשׁוּעַ׃ 2 וְיֵשׁוּעַ וְתַלְמִידָיו גַּם-הֵם הָיוּ מִן-הַקְּרוּאִים אֶל-הַחֲתֻנָּה׃ 3 וַיְהִי כַּאֲשֶׁר כָּלָה הַיָּיִן וַתֹּאמֶר אֵם יֵשׁוּעַ אֵלָיו אֵין לָהֶם יָיִן׃ 4 וַיֹּאמֶר אֵלֶיהָ יֵשׁוּעַ מַה-לִּי וָלָךְ אִשָּׁה עִתִּי עֲדַיִן לֹא-בָאָה׃ 5 וַתֹּאמֶר אִמּוֹ אֶל-הַמְשָׁרֲתִים כָּל-אֲשֶׁר יֹאמַר לָכֶם תַּעֲשׂוּ׃(בראשית מא' נה') 6 וְשָׁם שִׁשָּׁה כַדֵּי-אֶבֶן עֲרוּכִים כְּמִשְׁפַּט הַיְּהוּדִים לְטָהֳרָתָם שְׁתַּיִם אוֹ-שָׁלשׁ בַּתִּים יָכִיל כָּל-אֶחָד׃ 7 וַיֹּאמֶר אֲלֵיהֶם יֵשׁוּעַ מַלְאוּ הַכַּדִּים מָיִם וַיְמַלְאוּם עַד-לְמָעְלָה׃ 8 וַיֹּאמֶר שַׁאֲבוּ-נָא וְהָבִיאוּ אֶל-רַב הַמְּסִבָּה וַיָּבִיאוּ׃ 9 וַיִּטְעַם רֹאשׁ הַמְּסִבָּה אֶת-הַמַּיִם אֲשֶׁר נֶהֶפְכוּ לְיַיִן וְלֹא יָדַע מֵאַיִן הוּא וְהַמְשָׁרֲתִים אֲשֶׁר-שָׁאֲבוּ אֶת-הַמַּיִם יָדָעוּ וַיִּקְרָא רֹאשׁ-הַמְּסִבָּה אֶל-הֶחָתָן׃ 10 וַיֹּאמֶר אֵלָיו כָּל-אִישׁ יִתֵּן בָּרִאשׁוֹנָה אֶת-הַיַּיִן הַטּוֹב וְאַחֲרֵי שְׁתוֹתָם לִרְוָיָה יִתֵּן לָהֶם אֶת-הַגָּרוּעַ וְאַתָּה צָפַנְתָּ הַיַּיִן הַטּוֹב עַד-עָתָּה׃ 11 זֹאת רֵאשִׁית הָאֹתוֹת אֲשֶׁר עָשָׂה יֵשׁוּעַ בְּקָנָה אֲשֶׁר בְּאֶרֶץ הַגָּלִיל וַיְגַל אֶת-כְּבוֹדוֹ וַיַּאֲמִינוּ בוֹ תַּלְמִידָיו׃(ישעיהו ס' א'; מו' יג') 12 וַיְהִי אַחֲֵרי-כֵן וַיֵּרֶד אֶל-כְּפַר-נַחוּם הוּא וְאִמּוֹ וְאֶחָיו וְתַלְמִידָיו וְלֹא אָרְכוּ לָהֶם שָׁם הַיָּמִים׃

טיהור בית המקדש

13 וַיִּקְרְבוּ יְמֵי חַג-הַפֶּסַח אֲשֶׁר לַיְּהוּדִים וַיַּעַל יֵשׁוּעַ יְרוּשָׁלָיִם׃(שמות יב' יא'; דברים טז' ה'-ו') 14 וַיִּמְצָא בַמִּקְדָּשׁ מֹכְרֵי בָקָר וָצֹאן וּבְנֵי יוֹנָה וּפֹרְטֵי הַכֶּסֶף ישְׁבִים שָׁם׃(דברים יד' כד'-כו') 15 וַיִּקַּח חֲבָלִים וַיְעַבְּתֵם לְשׁוֹט וַיְגָרֶשׁ כֻּלָּם מִן-הַמִּקְדָּשׁ וְאֵת הַצֹּאן וְאֵת הַבָּקָר וַיְפַזֵּר אֶת-מְעוֹת הַשֻּׂלְחָנִים וַיַּהֲפֹך שֻׁלְחֲנֹתֵיהֶם׃(מלאכי ג' א'-ב') 16 וְאֶל-מֹכְרֵי הַיּוֹנִים אָמָר הוֹצִיאוּ אֵלֶּה מִזֶּה וְאַל-תַּעֲשׂוּ אֶת-בֵּית אָבִי לְבֵית מִסְחָר׃ 17 וַיִּזְכְּרוּ תַלְמִידָיו אֶת-הַכָּתוּב כִּי-קִנְאַת בֵּיתְךָ אֲכָלָתְנִי׃(תהילים סט' י') 18 וַיַּעֲנוּ הַיְּהוּדִים וַיֹּאמְרוּ אֵלָיו אֵי-זוֹ אוֹת תַּרְאֵנוּ שֶׁתַּעֲשֶׂה כָזֹאת׃ 19 וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר אֲלֵיהֶם הִרְסוּ אֶת-הַהֵיכָל הַזֶּה וּבִשְׁלשָׁה יָמִים אֲקִימֶנּוּ׃ 20 וַיּאֹמְרוּ הַיְּהוּדִים הִנֵּה אַרְבָּעִים וָשֵׁשׁ שָׁנָה נִבְנָה הַהֵיכָל הַזֶּה וְאַתָּה בִּשְׁלשָׁה יָמִים תְּקִימֶנּוּ׃ 21 וְהוּא דִּבֶּר עַל-הֵיכַל גְּוִיָּתוֹ׃ 22 וְאַחֲרֵי קוּמוֹ מִן-הַמֵּתִים זָכְרוּ תַלְמִידָיו כִּי-זֹאת אָמַר לָהֶם וַיַּאֲמִינוּ בַכָּתוּב וּבַדָּבָר אֲשֶׁר-דִּבֶּר יֵשׁוּעַ׃ 23 וַיְהִי בִּהְיֹתוֹ בִירוּשָׁלַיִם בְּחַג-הַפֶּסַח וַיַּאֲמִינוּ רַבִּים בִּשְׁמוֹ כִּי רָאוּ הָאֹתוֹת אֲשֶׁר עָשָׂה׃(***) 24 וְהוּא יֵשׁוּעַ לֹא הֶאֱמִין לָהֶם עַל-אֲשֶׁר יָדַע אֶת-כֻּלָּם׃ 25 וְלֹא הִצְטָרֵךְ לְעֵדוּת אִישׁ עַל-הָאָדָם כִּי הוּא יָדַע מַה-בְּקֶרֶב הָאָדָם׃(תהילים צד' יא'; ירמיהו יז' י')
