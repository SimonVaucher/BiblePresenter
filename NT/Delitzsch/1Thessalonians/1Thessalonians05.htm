לפרק הקודם

אגרת שאול הראשונה אל התסלוניקים
פרק ה

ביאת האדון -- המשך

1 וְעַל-דְּבַר הָעִתִּים וְהַזְּמַנִּים אֵין-צֹרֶךְ לִכְתֹּב אֲלֵיכֶם אֶחָי׃ 2 הֲלֹא אַף-אַתֶּם יְדַעְתֶּם הֵיטֵב כִּי יוֹם-יְהוָֹה כַּגַּנָּב בַּלַּיְלָה כֵּן יָבוֹא׃ 3 כִּי בְאָמְרָם שָׁלוֹם וְשַׁלְוָה (ירמיהו ו' יד') יָבֹא עֲלֵיהֶם הַשֶּׁבֶר פִּתְאֹם כְּחֶבְלֵי יוֹלֵדָה (ירמיהו ד' לא') וְלֹא יוּכְלוּ הִמָּלֵט׃(ישעיהו י' ג') 4 אֲבָל אַתֶּם אַחַי לֹא בַּחשֶׁךְ אַתֶּם שֶׁיַּשִׂיג אֶתְכֶם הַיּוֹם כַּגַּנָּב׃ 5 אַתֶּם כֻּלְּכֶם בְּנֵי הָאוֹר וּבְנֵי הַיּוֹם לֹא בְנֵי-הַלַּיְלָה אֲנַחְנוּ וְלֹא בְּנֵי הַחשֶׁךְ׃ 6 לָכֵן אַל-נָא נֵרָדֵם כְּמוֹ הָאֲחֵרִים כִּי אִם-נִשְׁקֹד וְנִהְיֶה ְמְפֻכָּחִים׃ 7 כִּי הַנִּרְדָּמִים נִרְדָּמִים בַּלָּיְלָה וְהַמִּשְׁתַּכְּרִים מִשְׁתַּכְּרִים בַּלָּיְלָה׃ 8 וַאֲנַחְנוּ בְּנֵי הַיּוֹם נִתְּפַּכֵּחַ-נָּא וְנִלְבַּשׁ אֶת-שִׁרְיוֹן הָאֱמוּנָה וְהָאַהֲבָה וְכַכּוֹבַע אֶת-תִּקְוַת הַיְשׁוּעָה׃(ישעיהו נט' יז') 9 יַעַן אֲשֶׁר לֹא-יְעָדָנוּ הָאֱלֹהִים לְחָרוֹן כִּי לָרֶשֶׁת אֶת-הַיְשׁוּעָה עַל-יְדֵי אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃ 10 אֲשֶׁר מֵת בַּעֲדֵנוּ לְמַעַן נִחְיֶה עִמּוֹ יָחַד אִם-נִשְׁקֹד וְאִם-נִישָׁן׃ 11 עַל-כֵּן נַחֲמוּ זֶה אֶת-זֶה וּבְנוּ אִישׁ אֶת-אָחִיו כַּאֲשֶׁר עֲשִׂיתֶם׃

הוראות אחרונות

12 וְהִנְנוּ מְבַקְשִׁים מִכֶּם אַחַי לְהַכִּיר אֶת-הָעֲמֵלִים בָּכֶם וְאֵת הַנִּצָּבִים עֲלֵיכֶם בִּדְבַר הָאָדוֹן וּמוֹכִיחִים אֶתְכֶם׃ 13 וְתִנְהֲגוּ-בָם כָּבוֹד גָּדוֹל מְאֹד בְּאַהֲבָה לְמַעַן פְּעֻלָּתָם וִיהִי שָׁלוֹם בֵּינֵיכֶם׃ 14 וְהִנְנוּ מַזְהִירִים אֶתְכֶם אֶחָי הוֹכִיחוּ אֶת-הַמְעַקְּשִׁים דַּרְכֵיהֶם אַמְּצוּ אֶת-רַכֵּי הַלֵּבָב (ישעיהוה לה' ג'-ד') תִּמְכוּ אֶת-הַחַלָּשִׁים וְהַאֲרִיכוּ רוּחַ אֶל-כָּל-אָדָם׃ 15 וּרְאוּ פֶּן-יְשַׁלֵּם אִישׁ לְרֵעֵהוּ רָעָה תַּחַת רָעָה כִּי אִם-רִדְפוּ בְּכָל-עֵת אֶת-הַטּוֹב הֵן לְאִישׁ אִישׁ מִכֶּם הֵן לְכָל-אָדָם׃(משלי כ' כב') 16 שִׂמְחוּ בְּכָל-עֵת׃(תהילים פט' יז'; ישעיהו סא' י') 17 הִתְפַּלְּלוּ כָּל-הַיּוֹם׃ 18 הוֹדוּ עַל-הַכֹּל כִּי-זֶה רְצוֹן אֱלֹהִים בַּמָּשִׁיחַ יֵשׁוּעַ אֲלֵיכֶם׃ 19 אֶת-הָרוּחַ לֹא תְּכַבּוּ׃ 20 אֶת-הַנְּבוּאוֹת לֹא תִמְאָסוּ׃ 21 בַּחֲנוּ כָּל-דָּבָר וּבַטּוֹב אֱחֹזוּ׃ 22 הִתְרַחֲקוּ מִכָּל-הַדּוֹמֶה לָרָע׃(איוב א' א') 23 וֵאלֹהֵי הַשָּׁלוֹם הוּא יְקַדֵּשׁ אֶתְכֶם בִּקְדֻשָּׁה שְׁלֵמָה וְכָל-רוּחֲכֶם וְנַפְשְׁכֶם וְגוּפְכֶם יִשָּׁמֵר תָּמִים בְּבוֹא אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃ 24 נֶאֱמָן הוּא הַקּוֹרֵא אֶתְכֶם אֲשֶׁר גַּם-יַעֲשֶׂה׃ 25 אַחַי הִתְפַּלֲלוּ בַעֲדֵנוּ׃ 26 שַׁאֲלוּ לִשְׁלוֹם כָּל-הָאַחִים בַּנְּשִׁיקָה הַקְּדשָׁה׃ 27 הִנְנִי מַשְׁבִּיעֲכֶם בָּאָדוֹן אֲשֶׁר תִּקָּרֵא הָאִגֶּרֶת הַזֹּאת בְּאָזְנֵי כָּל-הָאַחִים [הַקְּדשִׁים]׃ 28 חֶסֶד יֵשׁוּעַ הַמָּשִׁיחַ אֲדֹנֵינוּ עִמָּכֶם אָמֵן׃
