לפרק הקודם

מעשי השליחים
פרק ז

נאום סטפנוס

1 וַיֹּאמֶר הַכֹּהֵן הַגָּדוֹל הַאֱמֶת דְּבָרָם׃ 2 וַיַּעַן וַיֹּאמֶר אֲנָשִׁים אַחִים וְאָבוֹת שְׁמָעוּ אֱלֹהֵי הַכָּבוֹד (תהילים כט' ג') נִרְאָה אֶל-אַבְרָהָם אָבִינוּ בִּהְיוֹתוֹ בַּאֲרַם נַהֲרַיִם טֶרֶם יֵשֵׁב בְּחָרָן׃ 3 וַיֹּאמֶר אֵלָיו לֶךְ-לְךָ מֵאַרְצְךָ וּמִמּוֹלַדְתֶּךָ וּבֹא אֶל-הָאָרֶץ אֲשֶׁר אַרְאֶךָּ׃(בראשית יב' א') 4 וַיֵּצֵא מֵאֶרֶץ כַּשְׂדִּים וַיֵּשֶׁב בְּחָרָן וְאַחֲרֵי מוֹת אָבִיו הֶעֱבִיר אֹתוֹ מִשָּׁם אֶל-הָאָרֶץ הַזֹּאת אֲשֶׁר אַתֶּם ישְׁבִים בָּהּ כַּיּוֹם הַזֶּה׃(בראשית יא' לב') 5 וְלֹא נָתַן לוֹ בָּהּ נַחֲלָה אַף-לֹא מִדְרַךְ כַּף-רָגֶל וַיַּבְטַח לְתִתָּהּ לַאֲחֻזָּה לוֹ וּלְזַרְעוֹ אַחֲרָיו בְּטֶרֶם הָיָה-לוֹ בֵּן׃(בראשית יז' ח') 6 וַיֹּאמֶר לוֹ אֱלֹהִים כִּי-גֵר יִהְיֶה זַרְעוֹ בְּאֶרֶץ לֹא לָהֶם וַעֲבָדוּם וְעִנּוּ אֹתָם אַרְבַּע מֵאוֹת שָׁנָה׃(בראשית טו' יג') 7 וְגַם אֶת-הַגּוֹי אֲשֶׁר יַעֲבֹדוּ דָּן אָנֹכִי (בראשית טו' יד') אָמַר אֱלֹהִים וְאַחֲרֵי-כֵן יֵצְאוּ וְיַעַבְדוּנִי בַּמָּקוֹם הַזֶּה׃(שמות ג' יב') 8 וַיִּתֶּן-לוֹ אֶת-בְּרִית הַמִּילָה וְכֵן הוֹלִיד אֶת-יִצְחָק וַיָּמָל אֹתוֹ בַּיּוֹם הַשְּׁמִינִי וְיִצְחָק הוֹלִיד אֶת-יַעֲקֹב וְיַעֲקֹב הוֹלִיד אֶת-שְׁנֵים עָשָׂר הָאָבוֹת׃(בראשית יז' י'-יא'; כא' א'-ד') 9 וַיְקַנְאוּ הָאָבוֹת בְּיוֹסֵף וַיִּמְכְּרוּ אוֹתוֹ מִצְרָיְמָה וֵאלֹהִים הָיָה עִמּוֹ׃(בראשית לז' יא'; לז' כח'; לט' ב') 10 וַיַּצִּילֵהוּ מִכָּל-צָרוֹתָיו וַיִּתֶּן-לוֹ חֵן וְחָכְמָה לִפְנֵי פַרְעֹה מֶלֶךְ מִצְרָיִם וַיְשִׂימֵהוּ שַׁלִּיט עַל-מִצְרַיִם וְעַל כָּל-בֵּיתוֹ׃(בראשית מא' מ') 11 וַיְהִי רָעָב עַל-כָּל-אֶרֶץ מִצְרַיִם וּכְנַעַן וְצָרָה גְדוֹלָה וַאֲבוֹתֵינוּ לֹא-מָצְאוּ אֹכֶל׃ 12 וַיִּשְׁמַע יַעֲקֹב כִּי יֶשׁ-שֶׁבֶר בְּמִצְרָיִם וַיִּשְׁלַח-שָׁמָּה אֶת-אֲבוֹתֵינוּ בַּפַּעַם הָרִאשׁוֹנָה׃(בראשית מב' ב') 13 וּבַפַּעַם הַשֵּׁנִית הִתְוַדַּע יוֹסֵף אֶל-אֶחָיו וּמוֹלֶדֶת יוֹסֵף נִגְלְתָה לְפַרְעֹה׃(בראשית מה' ג'; מה' טז') 14 וַיִּשְׁלַח יוֹסֵף וַיִּקְרָא לְיַעֲקֹב אָבִיו וּלְכָל-מִשְׁפַּחְתּוֹ שִׁבְעִים וְחָמֵשׁ נָפֶשׁ׃(בראשית מו' כז') 15 וַיֵּרֶד יַעֲקֹב מִצְרָיְמָה וַיָּמָת הוּא וַאֲבוֹתֵינוּ׃(בראשית לז'; לט'-מט') 16 וַיּוּבְאוּ שְׁכֶמָה וַיּוּשְׂמוּ בַקֶּבֶר אֲשֶׁר קָנָה אַבְרָהָם [וַאֲשֶׁר קָנָה יַעֲקֹב] בְּכַסְפּוֹ מִיַּד בְּנֵי-חֲמוֹר אֲבִי שְׁכֶם׃(יהושוע כד' לב'; בראשית לג' יט') 17 וַיְהִי כַּאֲשֶׁר קָרְבָה עֵת הַשְּׁבוּעָה אֲשֶׁר נִשְׁבַּע אֱלֹהִים לְאַבְרָהָם (***) וַיִּגְדַּל הָעָם וַיִּרֶב בְּמִצְרָיִם׃ 18 עַד כִּי-קָם מֶלֶךְ אַחֵר אֲשֶׁר לֹא-יָדַע אֶת-יוֹסֵף׃(שמות א' ח') 19 וְזֶה הִתְחַכֵּם לְמִשְׁפַּחְתֵּנוּ וַיָּרַע לַאֲבוֹתֵינוּ לְהַשְׁלִיךְ אֶת-עֹלֲלֵיהֶם אֶל-פְּנֵי הַשָּׂדֶה לְבִלְתִּי הַחֲיוֹתָם׃(שמות א' כב') 20 בָּעֵת הַהִיא נוֹלַד משֶׁה וַיְהִי-טוֹב לֵאלֹהִים וַיְהִי אָמוּן בְּבֵית אָבִיו שְׁלשָׁה יְרָחִים׃(שמות ב' ב') 21 וְכַאֲשֶׁר הֻשְׁלַךְ אָסְפָה אֹתוֹ בַּת-פַּרְעֹה וַתְּגַדְּלֵהוּ לָהּ לְבֵן׃(שמות ב' ג'; ב' י') 22 וַיְלֻמַּד משֶׁה בְּכָל-חָכְמַת מִצְרָיִם וַיְהִי גִבּוֹר בִּדְבָרִים וּבְמַעֲשִׂים׃ 23 וּבִמְלֹאת-לוֹ אַרְבָּעִים שָׁנָה עָלָה עַל-לִבּוֹ לִפְקֹד אֶת-אֶחָיו בְּנֵי יִשְׂרָאֵל׃ 24 וַיַּרְא אִישׁ אֶחָד מֻכֶּה חִנָּם וַיּוֹשַׁע לוֹ וַיִּקֹּם נִקְמַת הַמֻּכֶּה בְּהַכּוֹתוֹ אֶת-הַמִּצְרִי׃(שמות ב' יא'-יב') 25 וַיַּחֲשֹׁב בִּלְבָבוֹ כִּי יָבִינוּ אֶחָיו אֲשֶׁר עַל-יָדוֹ יִתֵּן הָאֱלֹהִים לָהֶם תְּשׁוּעָה וְהֵם לֹא הֵבִינוּ׃ 26 וַיְהִי בַּיּוֹם הַשֵּׁנִי וַיָּבֹא בְּתוֹכָם וְהֵם נִצִּים וַיְבַקֵּשׁ לַעֲשׂוֹת שָׁלוֹם בֵּינֵיהֶם לֵאמֹר אֲנָשִׁים אַחִים אַתֶּם וְלָמָּה תַכּוּ אִישׁ אֶת-אָחִיו׃ 27 וְהַמַּכֶּה אֶת-רֵעֵהוּ הֲדָפוֹ לֵאמֹר מִי שָׂמְךָ לְאִישׁ שַׂר וְשֹׁפֵט עָלֵינוּ׃(שמות ב' יג') 28 הַלְהָרְגֵנִי אַתָּה אֹמֵר כַּאֲשֶׁר הָרַגְתָּ אֶתְמֹל אֶת-הַמִּצְרִי׃(שמות ב' יד') 29 וַיִּבְרַח משֶׁה עַל-הַדָּבָר הַזֶּה וַיְהִי-גֵר בְּאֶרֶץ מִדְיָן וַיּוֹלֶד שָׁם שְׁנֵי בָנִים׃(שמות יח' ב'-ג') 30 וּבִמְלֹאת אַרְבָּעִים שָׁנָה נִרְאָה אֵלָיו [מַלְאָךְ] בְּמִדְבַּר הַר סִינַי בְּלַבַּת-אֵשׁ מִתּוֹךְ הַסְּנֶה׃(שמות ג' ב') 31 וַיַּרְא משֶׁה וַיִּתְמַהּ עַל-הַמַּרְאֶה וּבְסוּרוֹ לִרְאוֹת וַיְהִי-קוֹל יְהוָֹה אֵלָיו׃ 32 אָנֹכִי אֱלֹהֵי אֲבוֹתֶיךָ אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֵאלֹהֵי יַעֲקֹב (שמות ג' ו') וַיֶּחֱרַד משֶׁה וַיִּירָא מֵהַבִּיט׃ 33 וַיֹּאמֶר אֵלָיו יְהוָֹה שַׁל נְעָלֶיךָ מֵעַל רַגְלֶיךָ כִּי הַמָּקוֹם אֲשֶׁר אַתָּה עוֹמֵד עָלָיו אַדְמַת-קֹדֶשׁ הוּא׃(שמות ג' ה') 34 רָאֹה רָאִיתִי אֶת-עֳנִי עַמִּי אֲשֶׁר בְּמִצְרַיִם וְאֶת-נַאֲקָתָם שָׁמַעְתִּי וָאֵרֵד לְהַצִּילָם וְעַתָּה לְכָה וְאֶשְׁלָחֲךָ מִצְרָיְמָה׃(שמות ב' כד'; ג' ז'-י') 35 הוּא משֶׁה אֲשֶׁר כִּחֲשׁוּ-בוֹ לֵאמֹר מִי שָׂמְךָ לְשַׂר וְשֹׁפֵט אֹתוֹ שָׁלַח הָאֱלֹהִים לְשַׂר וְגֹאֵל בְּיַד הַמַּלְאָךְ הַנִּרְאֶה אֵלָיו בַּסְּנֶה׃(***) 36 וְהוּא הוֹצִיאָם וַיַּעַשׂ אֹתוֹת וּמֹפְתִים בְּאֶרֶץ מִצְרַיִם וּבְיַם-סוּף וּבַמִדְבָּר אַרְבָּעִים שָׁנָה׃(תהילים קה' כז'-לו'; נחמיה ט' יא') 7 הוּא משֶׁה אֲשֶׁר אָמַר אֶל-בְּנֵי יִשְׂרָאֵל נָבִיא מִקֶּרֶב אֲחֵיכֶם כָּמֹנִי יָקִים לָכֶם יְהוָֹה אֱלֹהֵיכֶם אֵלָיו תִּשְׁמָעוּן׃(דברים יח' טו') 38 הוּא אֲשֶׁר הָיָה בַקָּהָל בַּמִדְבָּר עִם-הַמַּלְאָךְ הַדֹּבֵר אֵלָיו בְּהַר סִינַי וְעִם-אֲבוֹתֵינוּ וַאֲשֶׁר קִבֵּל דְּבָרִים חַיִּים לָתֵת לָנוּ׃(שמות כד' יח') 39 הוּא אֲשֶׁר אֲבוֹתֵינוּ לֹא אָבוּ לִשְׁמֹעַ לוֹ וַיִּמְאֲסוּ אֹתוֹ וּלְבָבָם פָּנָה מִצְרָיְמָה׃(במדבר יד' ג') 40 וַיֹּאמְרוּ אֶל-אַהֲרוֹן עֲשֵׂה-לָנוּ אֱלֹהִים אֲשֶׁר יֵלְכוּ לְפָנֵינוּ כִּי זֶה משֶׁה אֲשֶׁר הֶעֱלָנוּ מֵאֶרֶץ מִצְרַיִם לֹא יָדַעְנוּ מֶה-הָיָה לוֹ׃(שמות לב' א'; לב' כג') 41 וַיַּעֲשׂוּ עֵגֶל בַּיָּמִים הָהֵם וַיִּזְבְּחוּ זְבָחִים לָאֱלִיל וַיִּשְׂמְחוּ בְּמַעֲשֵׂה יְדֵיהֶם׃(שמות לב' ב'-ו') 42 וַיִּפֶן מֵהֶם הָאֱלֹהִים וַיִּתְּנֵם לַעֲבֹד אֶת-צְבָא הַשָּׁמָיִם כַּכָּתוּב בְּסֵפֶר הַנְּבִיאִים הַזְּבָחִים וּמִנְחָה הִגַּשְׁתֶּם-לִי בַמִּדְבָּר אַרְבָּעִים שָׁנָה בֵּית יִשְׂרָאֵל׃ 43 וּנְשָׂאתֶם אֵת סֻכַּת מִלְכֹּם וְאֶת-כּוֹכַב אֱלֹהֵיכֶם [רָמְפָן] הַצְּלָמִים אֲשֶׁר עֲשִׂיתֶם לְהִשְׁתַּחֲוֹת לָהֶם וְהִגְלֵיתִי אֶתְכֶם מֵהָלְאָה לְבָבֶל׃(עמוס ה' כה'-כז') 44 מִשְׁכַּן הָעֵדוּת הָיָה לַאֲבוֹתֵינוּ בַּמִּדְבָּר אֲשֶׁר עָשָׂה משֶׁה כַּתַּבְנִית אֲשֶׁר רָאָה כַּאֲשֶׁר צִוָּה אֹתוֹ הַמְדַבֵּר אֵלָיו׃(שמות כה' מ') 45 וַאֲבוֹתֵינוּ לְקָחֻהוּ וְגַם-הֱבִיאֻהוּ אִתָּם הֵם וִיהוֹשֻׁעַ בְּרִשְׁתָּם אֶת-אַרְצוֹת הַגּוֹיִם אֲשֶׁר גֵּרַשׁ אֹתָם אֱלֹהִים מִפְּנֵי אֲבוֹתֵינוּ עַד-יְמֵי דָוִד׃(יהושוע ג' יד') 46 וְהוּא מָצָא חֵן בְּעֵינֵי אֱלֹהִים וַיִּשְׁאַל לִמְצֹא מִשְׁכָּנוֹת לֵאלֹהֵי יַעֲקֹב׃(תהילים קלב' ה') 47 וּשְׁלֹמֹה בָּנָה-לוֹ בָיִת׃(מלכים א', ו' א'; ח' כז') 48 אֲבָל הָעֶלְיוֹן לֹא יִשְׁכֹּן בְּהֵיכָלוֹת מַעֲשֵׂי יָדָיִם כַּאֲשֶׁר אָמַר הַנָּבִיא׃ 49 הַשָּׁמַיִם כִּסְאִי וְהָאָרֶץ הֲדֹם רַגְלָי אֵי-זֶה בַיִת אֲשֶׁר תִּבְנוּ-לִי אָמַר יְהוָֹה וְאֵי-זֶה מָקוֹם מְנוּחָתִי׃ 50 הֲלֹא אֶת-כָּל-אֵלֶּה יָדִי-עָשָׂתָה׃(ישעיהו סו' א'-ב') 51 קְשֵׁי עֹרֶף וְעַרְלֵי לֵב וְאָזְנָיִם מַמְרִים הֱיִיתֶם תָּמִיד בְּרוּחַ הַקֹּדֶשׁ כַּאֲבוֹתֵיכֶם גַּם-אַתֶּם׃(ירמיהו ו' י'; ישעיהו סג' י') 52 מִי מֵהַנְּבִיאִים אֲשֶׁר לֹא רְדָפוּהוּ אֲבוֹתֵיכֶם אַף-הֵמִיתוּ אֵת הַמְבַשְּׂרִים מִקֶּדֶם בִּיאַת הַצַּדִּיק אֲשֶׁר עַתָּה הֱיִיתֶם אַתֶּם מוֹסְרָיו וּמְרַצְּחָיו׃(נחמיה ט' כו') 53 אַתֶּם אֲשֶׁר קִבַּלְתֶּם אֶת-הַתּוֹרָה עַל-יְדֵי מַלְאֲכוֹת הַמַּלְאָכִים וְלֹא שְׁמַרְתֶּם אוֹתָהּ׃

54 וַיְהִי כְּשָׁמְעָם אֶת-אֵלֶּה וַיִּגָּע בָּהֶם עַד נָפֶשׁ וַיַּחַרְקוּ עָלָיו שִׁנֵּיהֶם׃(תהילים לה' טז') 55 וְהוּא מָלֵא רוּחַ הַקֹּדֶשׁ וַיַּבֵּט הַשָּׁמָיְמָה וַיַּרְא אֶת-כְּבוֹד אֱלֹהִים וְאֶת-יֵשׁוּעַ נִצָּב לִימִין הָאֱלֹהִים׃ 56 וַיֹּאמֶר הִנְנִי רֹאֶה אֶת-הַשָּׁמַיִם נִפְתָּחִים וְאֵת בֶּן-הָאָדָם נִצָּב לִימִין הָאֱלֹהִים׃ 57 וַיִּזְעֲקוּ קוֹל גָּדוֹל וַיַּאַטְמוּ אֶת-אָזְנֵיהֶם וַיִּתְנַפְּלוּ עָלָיו כֻּלָּם יָחַד׃ 58 וַיִּדְחָפֻהוּ חוּצָה לָעִיר וַיִּסְקְלוּ אֹתוֹ בָּאֲבָנִים וְהָעֵדִים פָּשְׁטוּ אֶת-בִּגְדֵיהֶם לְרַגְלֵי בָּחוּר אֶחָד וּשְׁמוֹ שָׁאוּל׃ 59 וַיִּסְקְלוּ אֶת-אִסְטְפָנוֹס וְהוּא מְשַׁוֵּעַ וְאֹמֵר אֲדֹנִי יֵשׁוּעַ קַבֵּל אֶת-רוּחִי׃ 60 וַיִּכְרַע עַל-בִּרְכָּיו וַיִּזְעַק קוֹל גָּדוֹל יְהוָֹה אַל-תִּשְׁמֹר לָהֶם אֶת-הַחֵטְא הַזֶּה וַיְהִי אַחֲרֵי דַבְּרוֹ כַּדָּבָר הַזֶּה וַיִּישָׁן׃
