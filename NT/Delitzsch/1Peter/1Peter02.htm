לפרק הקודם

אגרת פטרוס הראשונה
פרק ב

אבנים חיות ועם קדוש

1 וְעַתָּה הָסִירוּ מֵאִתְּכֶם כָּל-רֶשַׁע וְכָל-מִרְמָה וַחֲנֻפָּה וְקִנְאָה וְכָל-לָשׁוֹן רָע׃ 2 וּכְעֹלֲלִים אֲשֶׁר מִקָּרוֹב נוֹלָדוּ הִתְאַוּוּ לֶחָלָב הַשִׂכְלִי וְהַזָּךְ לְמַעַן תִּגְדְּלוּ-בוֹ [לַתְּשׁוּעָה]׃ 3 אִם-אָמְנָם טְעַמְתֶּם כִּי-טוֹב הָאָדוֹן׃(תהילים לד' ט') 4 אֲשֶׁר נִגַּשְׁתֶּם אֵלָיו אֶל-אֶבֶן חַיָּה אֲשֶׁר מָאֲסוּ-בָהּ בְּנֵי-הָאָדָם וְהִיא נִבְחָרָה וִיקָרָה לֵאלֹהִים׃(תהילים קיח' כב') 5 וְגַם-אַתֶּם כַּאֲבָנִים חַיּוֹת נִבְנֵיתֶם לְמִשְׁכַּן הָרוּחַ לִכְהֻנַּת קֹדֶשׁ לְהַעֲלוֹת זִבְחֵי-רוּחַ לְרָצוֹן לֵאלֹהִים בְּיֵשׁוּעַ הַמָּשִׁיחַ׃(ישעיהו סא' ו') 6 וְזֶה הוּא שֶׁאָמַר הַכָּתוּב הִנְנִי יִסַּד בְּצִיּוֹן אֶבֶן פִּנָּה אֶבֶן בֹּחַן וִיקָרָה וְהַמַּאֲמִין בָּהּ לֹא יֵבוֹשׁ׃(ישעיהו כח' טז') 7 לָכֵן לָכֶם הַמַּאֲמִינִים הִיא אֶבֶן הַיְקָר אֲבָל לַסּוֹרֲרִים הָאֶבֶן אֲשֶׁר מָאֲסוּ הַבּוֹנִים הָיְתָה לְרֹאשׁ פִּנָּה (תהילים קיח' כב') וּלְאֶבֶן נֶגֶף וּלְצוּר מִכְשֹׁל׃(ישעיהו ח' יד') 8 וְהֵם נִכְשְׁלוּ יַעַן לֹא שָׁמְעוּ לַדָּבָר וְלָזֹאת גַּם-נוֹעָדוּ׃ 9 וְאַתֶּם הִנְּכֶם זֶרַע נִבְחָר מַמְלֶכֶת כֹּהֲנִים וְגוֹי קָדוֹשׁ (שמות יט' ו') וְעַם סְגֻלָּה לְמַעַן תְּסַפְּרוּ תְּהִלּוֹת (ישעיהו מג' כא') הַקּוֹרֵא אֶתְכֶם מֵחֹשֶׁךְ אֶל-אוֹרוֹ הַנִּפְלָא׃ 10 אֲשֶׁר לְפָנִים לֹא-עָם הֱיִיתֶם וְעַתָּה עַם אֱלֹהִים וַאֲשֶׁר לְפָנִים לֹא רֻחָמוּ וְעַתָּה מְרֻחָמִים׃(הושע א' ו'; א' ט'; ב' כה')

התנהגו כעבדי אלהים

11 חֲבִיבַי אַזְהִירְכֶם כְּגֵרִים וְתוֹשָׁבִים (תהילים לט' יג') הִנָּזְרוּ מִתַּאֲוֹת הַבָּשָׂר הַמִּתְגָּרוֹת בַּנָּפֶשׁ׃ 12 וְהֵיטִיבוּ דַרְכְּכֶם בַּגּוֹיִם לְמַעַן יַבִּיטוּ אֶל-מַעֲשֵׂיכֶם הַטּוֹבִים וִיכַבְּדוּ אֶת-הָאֱלֹהִים בְּיוֹם הַפְּקֻדָּה (ישעיהו י' ג') תַּחַת אֲשֶׁר חֵרְפוּ אֶתְכֶם כְּפֹעֲלֵי אָוֶן׃ 13 וְהִכָּנְעוּ לְכָל-פְּקֻדַּת אָדָם לְמַעַן הָאָדוֹן אִם-לַמֶּלֶךְ כָּרָאוּי לָרֹאשׁ׃(משלי כד' כא') 14 אִם-לַמֹּשְׁלִים כָּרָאוּי לַשְּׁלוּחִים מֵאִתּוֹ לְנִקְמַת פֹּעֲלֵי אָוֶן וְלִתְהִלַּת עֹשֵׂי טוֹב׃ 15 כִּי כֵן רְצוֹן אֱלֹהִים שֶׁתַּעֲשׂוּ הַטּוֹב וְתִסְכְּרוּ אֶת-פִּי אִוֶּלֶת הָאֲנָשִׁים אֲשֶׁר אֵין-בָּם דָּעַת׃ 16 כַּחָפְשִׁים וְלֹא-כְאִלּוּ הָיְתָה לָכֶם הַחֻפְשָׁה לְמִכְסֵה הָרָעָה כִּי אִם-כְּעַבְדֵי אֱלֹהִים׃ 17 נַהֲגוּ כָבוֹד בְּכָל-אִישׁ אֶהֱבוּ אֵת הָאַחִים יִרְאוּ אֶת-אֱלֹהִים כַּבְּדוּ אֶת הַמֶּלֶךְ׃(קוהלת יב' יג')

המופת של סבלות המשיח

18 הָעֲבָדִים הִכָּנְעוּ לִפְנֵי אֲדֹנֵיכֶם בְּכָל-יִרְאָה לֹא לִפְנֵי הַטּוֹבִים וְהָעֲנָוִים בִּלְבַד כִּי אִם-גַּם-לִפְנֵי הָעִקְּשִׁים׃ 19 כִּי חֶסֶד הוּא לָאִישׁ כִּי-יִשְׂבַּע מַמְּרֹרִים וִיעֻנֶּה חִנָּם לְמַעַן דַּעַת הָאֱלֹהִים׃ 20 כִּי אִם-תֶּחֶטְאוּ וּסְבַלְתֶּם מַכּוֹת אֶגְרוֹף מַה-תִּתְהַלָּלוּ אֲבָל אִם-תְּעֻנּוּ וּסְבַלְתֶּם בַּעֲשׂוֹתְכֶם הַטּוֹב חֶסֶד הוּא מִלִּפְנֵי אֱלֹהִים׃ 21 כִּי לָזֹאת נִקְרֵאתֶם כִּי גַם-הַמָּשִׁיחַ עֻנָּה בַעַדְכֶם וְהִשְׁאִיר לָכֶם מוֹפֵת לָלֶכֶת בְּעִקְּבוֹתָיו׃ 22 אֲשֶׁר לֹא חָמָס עָשָׂה וְלֹא מִרְמָה בְּפִיו׃(ישעיהו נג' ט') 23 אֲשֶׁר שָׁמַע חֶרְפָּתוֹ וְלֹא הֵשִׁיב נַעֲנָה וְלֹא גָעַר כִּי אִם-מָסַר דִּינוֹ לַשֹּׁפֵט צֶדֶק וְאֶת-חַטֹּאתֵנוּ הוּא נָשָׂא בִּגְוִיָּתוֹ עַל-הָעֵץ׃ 24 לְמַעַן נִחְיֶה לַצְּדָקָה אַחֲרֵי חָדַלְנוּ מִן-הַחֲטָאִים׃(ישעיהו נג' ה') 25 אֲשֶׁר בַּחֲבֻרָתוֹ נִרְפָּא לָכֶם כִּי הֱיִיתֶם כְּצֹאן אֹבְדוֹת (ישעיהו נג' ו') וְעַתָּה שַׁבְתֶּם אֶל-הָרֹעֶה פְּקִיד נַפְשֹׁתֵיכֶם׃(ישעיהו מ' יא'; יחזקאל לד' כג')
