לאגרת יעקב

אגרת פטרוס הראשונה
פרק א

1 פֶּטְרוֹס שְׁלִיחַ יֵשׁוּעַ הַמָּשִׁיחַ אֶל-תּוֹשְׁבֵי תְפוּצוֹת פָּנְטוֹס גָּלַטְיָא קַפּוֹדְקִיָּא אַסְיָא וּבִיתִינְיָא׃ 2 הַנִּבְחָרִים מִדַּעַת אֱלֹהִים הָאָב מִקֶּדֶם בְּקִדּוּשׁ הָרוּחַ אֶל-מִשְׁמַעַת יֵשׁוּעַ הַמָּשִׁיחַ וְאֶל-הַזָּיַת דָּמוֹ חֶסֶד וְשָׁלוֹם לְמַכְבִּיר יִהְיוּ לָכֶם׃

תקווה חיה

3 בָּרוּךְ הָאֱלֹהִים אֲבִי אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר כְּרֹב רַחֲמָיו חָזַר וְיָלַד אֹתָנוּ לְתִקְוָה חַיָּה בַּהֲקָמַת יֵשׁוּעַ הַמָּשִׁיחַ מֵעִם הַמֵּתִים׃(תהילים קג' ח') 4 לְנַחֲלָה אֲשֶׁר לֹא תִּשָּׁחֵת וְלֹא תְגֹאַל וְלֹא תִבֹּל הַצְּפוּנָה לָכֶם בַּשָּׁמָיִם׃ 5 הַשְּׁמוּרִים בְּעֹז אֱלֹהִים עַל-יַד הָאֱמוּנָה לַתְּשׁוּעָה הָעֲתִידָה לְהִגָּלוֹת לְעֶת-קֵץ׃(תהילים לז' כח') 6  אֲשֶׁר בָּהּ תָּגִילוּ אַתֶּם הַמִּתְעַצְּבִים עַתָּה מְעַט לְפִי הַצֹּרֶךְ בְּמַסֹּת שֹׁנוֹת׃ 7 לְמַעַן תִּמָּצֵא אֱמוּנַתְכֶם הַמְזֻקָּקָה יְקָרָה הַרְבֵּה מִן-הַזָּהָב הָאֹבֵד הַצָּרוּף בָּאֵשׁ (מלאכי ג' ג') לִתְהִלָּה וּלְכָבוֹד וּלְתִפְאֶרֶת בְּהִתְגַּלּוֹת יֵשׁוּעַ הַמָּשִׁיחַ׃ 8 אֲשֶׁר אֲהַבְתֶּם וְלֹא רְאִיתֶם וַאֲשֶׁר תַּאֲמִינוּ בוֹ עַתָּה וְאֵינְכֶם רֹאִים אֹתוֹ וּבְכֵן תָּגִילוּ בְּשִׂמְחָה מְפֹאָרָה עָצְמָה מִסַּפֵּר׃ 9 לָשֵׂאת עֵקֶב אֱמוּנַתְכֶם תְּשׁוּעָה לְנַפְשֹׁתֵיכֶם׃ 10 אֵת הַתְּשׁוּעָה אֲשֶׁר עָלֶיהָ חָקְרוּ וְדָרְשׁוּ הַנְּבִיאִים הַנִּבְּאִים עַל-הַחֶסֶד הַצָּפוּן לָכֶם׃ 11 לַחְקֹר לָדַעַת מַה-זּוּ וְאֵי-זוּ הָעֵת אֲשֶׁר הוֹדִיעָהּ רוּחַ הַמָּשִׁיחַ אֲשֶׁר בְּקִרְבָּם (דניאל ט' כה'-כו') בְּהַגִּידוֹ מֵרֹאשׁ אֵת עִנּוּיֵי הַמָּשִׁיחַ וְאֶת-הַגְּדֻלּוֹת אֲשֶׁר אַחֲרֵיהֶם׃(תהילים כב' ב'-כב'; ישעיהו ס' א'-ז') 12 וְנִגְלָה לָהֶם אֲשֶׁר לֹא לְנַפְשָׁם כִּי אִם-לָנוּ הָיוּ מְשַׁמְּשִׁים בַּדְּבָרִים הָהֵם אֲשֶׁר הֻגַּד לָכֶם עַתָּה עַל-פִּי הַמְבַשְׂרִים אֶתְכֶם בְּרוּחַ הַקֹּדֶשׁ הַשָּׁלוּחַ מִשָּׁמַיִם דְּבָרִים אֲשֶׁר מַלְאֲכֵי אֱלֹהִים חָמְדוּ לְהַשְׁקִיף אֶל-תּוֹכָם׃

חיי קדוּשה

13 לָכֵן חִגְרוּ מָתְנֵי שִׂכְלְכֶם הִתְעוֹרֲרוּ וְקַוּוּ בְכָל-נַפְשְׁכֶם לַחֶסֶד אֲשֶׁר יְבוֹאֲכֶם בְּהִתְגַּלּוֹת יֵשׁוּעַ הַמָּשִׁיחַ׃ 14 כִּבְנֵי מִשְׁמַעַת אַל-תִּתְנַהֲגוּ כַּתַּאֲוֹת אֲשֶׁר הִתְאַוִּיתֶם בְּעוֹד הֱיוֹתְכֶם בִּבְלִי-דָעַת׃ 15 כִּי אִם-הֱיוּ קְדֹשִׁים בְּכָל-דַּרְכֵיכֶם כַּאֲשֶׁר הַקֹּרֵא אֶתְכֶם קָדוֹשׁ הוּא׃(ישעיהו ו' ג') 16 כִּי עַל-כֵּן כָּתוּב וִהְיִיתֶם קְדֹשִׁים כִּי קָדוֹשׁ אָנִי׃(ויקרא יא' מד'; יט' ב') 17 וְאִם-תִּקְרְאוּ אָב לֵאלֹהִים הַשֹּׁפֵט בִּבְלִי מַשּׂא פָנִים כְּמַעַלְלֵי אִישׁ וָאִישׁ (דברי הימים ב', יט' ז'; תהילים סב' יג') הִתְהַלְּכוּ-נָא בְיִרְאָה בִּימֵי מְגוּרֵיכֶם׃(תהילים ב' יא') 18 מִפְּנֵי שֶׁיֹּדְעִים אַתֶּם כִּי לֹא-בְדָבָר נִפְסָד לֹא בְכֶסֶף וְלֹא בְזָהָב נִפְדֵּיתֶם מִדֶּרֶךְ הַבְלְכֶם אֲשֶׁר הָנְחַלְתֶּם מֵאֵת אֲבוֹתֵיכֶם׃(ישעיהו נב' ג') 19 כִּי אִם-בְּדַם יָקָר שֶׁל-שֶׂה תָמִים שֶׁאֵין-בּוֹ מוּם בְּדַם הַמָּשִׁיחַ׃(ישעיהו נג' ז') 20 הַנּוֹדָע מֵרֹאשׁ לִפְנֵי מוֹסְדוֹת תֵּבֵל וְנִגְלָה בְּאַחֲרִית הַיָּמִים לְמַעַנְכֶם׃ 21 הַמַּאֲמִינִים עַל-פִּיו בֵּאלֹהִים אֲשֶׁר הֵקִים אֹתוֹ מֵעִם הַמֵּתִים וַיִּתֶּן-לוֹ כָבוֹד לְמַעַן הֱיוֹת אֱמוּנַתְכֶם תִּקְוָה לֵאלֹהִים׃ 22 זַכּוּ אֶת-נַפְשֹׁתֵיכֶם עַל-יְדֵי הָרוּחַ בְּשָׁמְעֲכֶם בְּקוֹל הָאֱמֶת לְאַחֲוָה שֶׁאֵין בָּהּ חֲנֻפָּה וַאֲהַבְתֶּם אִישׁ אֶת-רֵעֵהוּ אַהֲבָה עַזָּה בְּלֵב טָהוֹר׃ 23 כַּנּוֹלָדִים שֵׁנִית לֹא מִזֶּרַע נִשְׁחָת כִּי אִם-מִזֶּרַע לֹא יִשָּׁחֵת בַּמַּאֲמָר שֶׁל-אֱלֹהִים הַחַי וְהַקַּיָּם לְעוֹלָם׃ 24 כִּי כָל-בָּשָׂר חָצִיר וְכָל-כְּבוֹד אִישׁ כְּצִיץ הַשָׂדֶה׃ 25 יָבֵשׁ חָצִיר נָבֵל צִיץ וּדְבַר יְהוָֹה יָקוּם לְעוֹלָם (ישעיהו מ' ו'-ח') וְהוּא הַדָּבָר אֲשֶׁר בֻּשַׂר לָכֶם׃
