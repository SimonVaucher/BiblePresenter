אגרת שאול אל האפסים

שרותו של שאול לגויים

1 בַּעֲבוּר זֹאת אֲנִי פוֹלוֹס אֲסִיר הַמָּשִׁיחַ לְמַעַנְכֶם הַגּוֹיִם׃ 2 כִּי-אָמְנָם שְׁמַעְתֶּם הַנְהָגַת חֶסֶד הָאֱלֹהִים אֲשֶׁר-נִתַּן לִי אֲלֵיכֶם׃ 3 כִּי בֶחָזוֹן גִּלָּה-לִי הַסּוֹד אֲשֶׁר כָּתַבְתִּי לָכֶם בִּמְעַט מִלִּין׃ 4 וּבְקָרְאֲכֶם תּוּכְלוּ לְהַכִּיר אֶת-בִּינָתִי בְּסוֹד הַמָּשִׁיחַ׃ 5 אֲשֶׁר בַּדּוֹרֹת הָרִאשֹׁנִים לֹא נוֹדַע לִבְנֵי אָדָם כַּאֲשֶׁר נִגְלָה עַתָּה לִשְׁלִיחָיו הַקְּדשִׁים וְלִנְבִיאָיו בָּרוּחַ׃ 6 שֶׁיִּהְיוּ הַגּוֹיִם גַּם-הֵם נֹחֲלִים וּבְגוּף אֶחָד וְחַבְרֵי הַהַבְטָחָה בַּמָּשִׁיחַ יֵשׁוּעַ עַל-יְדֵי הַבְּשׂרָה׃(ישעיהו מב' ו'; מט' ו') 7 אֲשֶׁר הָיִיתִי לָהּ לִמְשָׁרֵת בְּמַתְּנַת חֶסֶד אֱלֹהִים הַנְּתוּנָה לִי כְּפִי פְּעֻלַּת גְּבוּרָתוֹ׃ 8 לִי אֲנִי צְעִיר הַצְּעִירִים שֶׁבְּכָל-הַקְּדשִׁים נִתַּן הַחֶסֶד הַזֶּה לְבַשֵׂר בַּגּוֹיִם אֶת-עשֶׁר הַמָּשִׁיחַ אֲשֶׁר רַב הוּא לְאֵין חֵקֶר׃ 9 וּלְהָאִיר עֵינֵי כֹל לִרְאוֹת מַה-הַנְהָגַת הַסּוֹד הַנִּסְתָּר מִדֹּרֹת עוֹלָם בֵּאלֹהִים יוֹצֵר הַכֹּל [עַל-יְדֵי יֵשׁוּעַ הַמָּשִׁיחַ]׃ 10 לְמַעַן אֲשֶׁר תִּוָּדַע עַתָּה עַל-יְדֵי הָעֵדָה לַשָׂרִים וְלַשַּׁלִּיטִים אֲשֶׁר בַּמְּרוֹמִים חָכְמַת אֱלֹהִים הַמַּפְלִיאָה דְּרָכֶיהָ׃(דניאל ב' כ') 11 עַל-פִּי עֲצַת עוֹלָמִים אֲשֶׁר יָעַץ בַּמָּשִׁיחַ יֵשׁוּעַ אֲדֹנֵינוּ׃ 12 אֲשֶׁר בּוֹ פִּתְחוֹן פֶּה לָנוּ וּמָבוֹא בְּבִטְחָה עַל-יַד אֱמוּנָתוֹ׃ 13 עַל-כֵּן אֲבַקֵּשׁ שֶׁלּא תֵחַתּוּ בְּצָרוֹתַי לְמַעַנְכֶם כִּי-הִיא תִּפְאַרְתְּכֶם׃

הכרת אהבתו של המשיח

14 בַּעֲבוּר זֹאת אֶכְרְעָה עַל-בִּרְכָּי לִפְנֵי אֲבִי אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃(תהילים צה' ו') 15 אֲשֶׁר נִקְרָא שְׁמוֹ עַל כָּל-מִשְׁפָּחָה שֶׁבַּשָּׁמַיִם וְשֶׁבָּאָרֶץ׃ 16 לָתֵת לָכֶם לְפִי-עשֶׁר כְּבוֹדוֹ שֶׁיִּתְחַזֵּק בִּגְבוּרָה עַל-יְדֵי רוּחוֹ הָאָדָם הַפְּנִימִי׃ 17 שֶׁיִּשְׁכֹּן הַמָּשִׁיחַ בִּלְבַבְכֶם עַל-יְדֵי הָאֱמוּנָה וִהְיִיתֶם מַשְׁרִישִׁים וּמְיֻסָּדִים בְּאַהֲבָה׃ 18 לְמַעַן תּוּכְלוּ לְהָבִין כְּכָל-הַקְּדשִׁים מֶה הָרֹחַב וְהָאֹרֶךְ וְהָעֹמֶק וְהַגֹּבַהּ׃ 19 וִידַעְתֶּם אֶת-אַהֲבַת הַמָּשִׁיחַ הַנַּעֲלָה עַל-כָּל-דָּעַת וְנִמְלֵאתֶם אֶת-כָּל-מְלוֹא הָאֱלֹהִים׃ 20 וְהַיָּכֹל לְהַרְבּוֹת גְּמוּלָיו עָלֵינוּ יוֹתֵר מִכָּל-מִשְׁאֲלוֹתֵינוּ וּמַחְשְׁבוֹתֵינוּ לְפִי הַכֹּחַ הַפֹּעֵל בְּקִרְבֵּנוּ׃ 21 לוֹ הַכָּבוֹד בְּקֶרֶב הַקָּהָל בַּמָּשִׁיחַ יֵשׁוּעַ לְדֹר וָדֹר עַד-עוֹלְמֵי עוֹלָמִים אָמֵן׃
