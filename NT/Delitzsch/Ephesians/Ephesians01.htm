לאגרת שאול אל הגלאטים

אגרת שאול אל האפסים
פרק א

1 פּוֹלוֹס שְׁלִיחַ יֵשׁוּעַ הַמָּשִׁיחַ בִּרְצוֹן אֱלֹהִים אֶל-הַקְּדשִׁים הַנִּמְצְאִים [בְּאֶפְסוֹס] וּמַאֲמִינִים בַּמָּשִׁיחַ יֵשׁוּעַ׃ 2 חֶסֶד לָכֶם וְשָׁלוֹם מֵאֵת הָאֱלֹהִים אָבִינוּ וַאֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

ברכות רוחניות במשיח

3 בָּרוּךְ הָאֱלֹהִים וַאֲבִי אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר בֵּרְכָנוּ בְּכָל-בִּרְכַּת רוּחַ בַּמְּרוֹמִים בַּמָּשִׁיחַ׃ 4 כַּאֲשֶׁר בָּחַר אֹתָנוּ בוֹ לִפְנֵי מוֹסְדוֹת תֵּבֵל לִהְיוֹת קְדשִׁים וּתְמִימִים לְפָנָיו בְּאַהֲבָה׃ 5 יְעָדָנוּ לוֹ לְבָנִים עַל-יְדֵי יֵשׁוּעַ הַמָּשִׁיחַ כִּרְצוֹן חֶפְצוֹ׃ 6 לִתְהִלַּת כְּבוֹד חַסְדּוֹ אֲשֶׁר-חָנַן אֹתָנוּ בִּידִידוֹ׃(תהילים פד' יב') 7 אֲשֶׁר-בּוֹ לָנוּ הַפִּדְיוֹם בְּדָמוֹ וּסְלִיחַת הַפְּשָׁעִים כְּרֹב חַסְדּוֹ׃(תהילים קג' ג'; קל' ד') 8 אֲשֶׁר בָּא עָלֵינוּ בְּשֶׁפַע-חָכְמָה וְהַשְׂכֵּל׃ 9 וְהוֹדִיעַ אֹתָנוּ אֶת-סוֹד חֶפְצוֹ כִּרְצוֹנוֹ וַעֲצָתוֹ מֵרֹאשׁ אֲשֶׁר יָעַץ בּוֹ׃ 10 עַל-דְּבַר הַנְהָגָתוֹ בִּמְלֹאת הַיָּמִים לְקַבֵּל אֶת-הַכֹּל בַּמָּשִׁיחַ מַה-שֶּׁבַּשָּׁמַיִם וּמַה-שֶּׁבָּאָרֶץ׃ 11 וּבוֹ לָקַחְנוּ נַחֲלָתֵנוּ אֲנַחְנוּ הַמְיֻעָדִים לָהּ מִלְּפָנִים בְּמַחֲשֶׁבֶת פֹּעֵל הַכֹּל כַּעֲצַת חֶפְצוֹ׃(ישעיהו מו' י') 12 לִהְיוֹת לִתְהִלַּת כְּבוֹדוֹ אֲנַחְנוּ אֲשֶׁר יִחַלְנוּ אֶל-הַמָּשִׁיחַ מֵאָז׃ 13 וַאֲשֶׁר גַּם-אַתֶּם שְׁתוּלִים בּוֹ אַחֲרֵי שָׁמְעֲכֶם דְּבַר הָאֱמֶת אֶת-בְּשׂוֹרַת יְשׁוּעַתְכֶם וַאֲשֶׁר בּוֹ כְּשֶׁהֶאֱמַנְתֶּם גַּם-נֶחְתַּמְתֶּם בְּרוּחַ הַהַבְטָחָה רוּחַ הַקֹּדֶשׁ׃ 14 כִּי זֶה עֶרְבוֹן נַחֲלָתֵנוּ לִפְדּוֹת-לוֹ עַם סְגֻלָּה לִתְהִלַּת כְּבוֹדוֹ׃

תפילתו של השליח

15 בַּעֲבוּר זֹאת גַּם-אָנֹכִי אַחֲרֵי שָׁמְעִי אֱמוּנַתְכֶם בַּאֲדֹנֵינוּ יֵשׁוּעַ וְהָאַהֲבָה אֲשֶׁר-אֲהַבְתֶּם אֶת-כָּל-הַקְּדשִׁים׃ 16 לֹא אֶחְדַּל מֵהוֹדוֹת בַּעֲבוּרְכֶם בְּהַזְכִּירִי אֶתְכֶם בִּתְפִלּתַי׃ 17 שֶׁיִּתֵּן לָכֶם אֱלֹהֵי אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲבִי הַכָּבוֹד אֶת-רוּחַ הַחָכְמָה וְהֶחָזוֹן לָדַעַת אֹתוֹ׃ 18 וְיָאֵר עֵינֵי לְבַבְכֶם (תהילים קיט' יח') לְמַעַן תֵּדְעוּ מַה-הִיא תּוֹחֶלֶת קְרִיאָתוֹ וּמָה-עשֶׁר כְּבוֹד נַחֲלָתוֹ בַּקְּדשִׁים׃ 19 וּמַה-יִּתְרוֹן גְּדֻלַּת גְּבוּרָתוֹ בָּנוּ הַמַּאֲמִינִים כְּפִי פְעֻלַּת עֹצֶם כֹּחוֹ׃ 20 אֲשֶׁר פָּעַל בַּמָּשִׁיחַ בְּהָעִיר אֹתוֹ מִן-הַמֵּתִים וַיּוֹשִׁיבֶנּוּ לִימִינוֹ בַּמְּרוֹמִים׃(תהילים קי' א') 21 מִמַּעַל לְכָל-שְׂרָרָה וְשָׁלְטָן וּגְבוּרָה וּמֶמְשָׁלָה וְכֹל הַנִּקְרָא בְשֵׁם לֹא-בָעוֹלָם הַזֶּה בִּלְבַד כִּי אִם-גַּם בָּעוֹלָם הַבָּא׃ 22 וַיָּשֶׁת כֹּל תַּחַת רַגְלָיו (תהילים ח' ז'//***) וַיִּתֵּן אֹתוֹ לְרֹאשׁ לְכֹל אֶל-הָעֵדָה׃ 23 אֲשֶׁר הִיא גּוּפוֹ מְלוֹא שֶׁל-הַמְמַלֵּא אֶת-הַכֹּל בַּכֹּל׃
