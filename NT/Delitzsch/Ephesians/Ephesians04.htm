אגרת שאול אל האפסים
פרק ד

אחדות הגוף

1 עַל-כֵּן אַזְהִירְכֶם אֲנִי הָאָסוּר בָּאָדוֹן לְהִתְהַלֵּךְ כַּאֲשֶׁר יָאֲתָה לְמִשְׁמַרְתְּכֶם אֲשֶׁר נִקְרֵאתֶם לָהּ׃ 2 בְּכָל-נְמִיכוּת וַעֲנָוָה וּבְאֹרֶךְ רוּחַ לָשֵׂאת אִישׁ אֶת-רֵעֵהוּ בְּאַהֲבָה׃ 3 וְשִׁקְדוּ לִשְׁמֹר אֶת-אַחֲדוּת הָרוּחַ בַּאֲגֻדַּת הַשָּׁלוֹם׃ 4 גּוּף אֶחָד וְרוּחַ אֶחָד כַּאֲשֶׁר גַּם-נִקְרֵאתֶם בְּתִקְוַת מִשְׁמַרְתְּכֶם הָאֶחָת׃ 5 אָדוֹן אֶחָד אֱמוּנָה אַחַת טְבִילָה אֶחָת׃ 6 אֵל וְאָב אֶחָד לַכֹּל שֶׁהוּא עַל-כֹּל וְאֶת-כֹּל וּבְתוֹךְ כֹּל׃ 7 אֲבָל לְכָל-אֶחָד וְאֶחָד מִמֶּנּוּ נִתַּן הֶחָסֶד כְּמִדַּת מַתְּנַת הַמָּשִׁיחַ׃ 8 עַל-כֵּן הוּא אוֹמֵר עָלָה לַמָּרוֹם שָׁבָה שֶׁבִי וַיִּתֵּן מַתָּנוֹת לָאָדָם׃(תהילים סח' יט') 9 וְעָלָה שֶׁאָמַר מַה-הוּא אִם-לֹא שֶׁיָּרַד מִקֹּדֶם לְתַחְתִּיּוֹת אָרֶץ׃ 10 הוּא הַיֹּרֵד וְהוּא הָעֹלֶה לְמַעְלָה מִכָּל-הַשָּׁמָיִם לְמַעַן יְמַלֵּא אֶת-הַכֹּל׃(ירמיהו כג' כד') 11 וְהוּא נָתַן אֶת-אֵלֶּה שְׁלִיחִים וְאֶת-אֵלֶּה נְבִיאִים וְאֶת-אֵלֶּה מְבַשְׂרִים וְאֶת-אֵלֶּה רֹעִים וּמְלַמְּדִים׃ 12 לְהַשְׁלִים אֶת-הַקְּדשִׁים לְמַעֲשֵׂה הָעֲבוֹדָה לְבִנְיַן גּוּף הַמָּשִׁיחַ׃ 13 עַד כִּי-נַגִּיעַ כֻּלָּנוּ לְאַחֲדוּת הָאֱמוּנָה וְדַעַת בֶּן-הָאֱלֹהִים כְּאִישׁ אֶחָד שָׁלֵם לְשִׁעוּר קוֹמַת מְלוֹא הַמָּשִׁיחַ׃ 14 וְלֹא-נִהְיֶה עוֹד יְלָדִים נִגְרָשִׁים וְנִדָּפִים בְּכָל-רוּחַ הַלִּמּוּד בְּתַרְמִית בְּנֵי אָדָם וּמֹקְשׁוֹתָם אֲשֶׁר שָׁתוּ לְהַתְעוֹת׃ 15  כִּי אִם-נְדַבֵּר הָאֱמֶת בְּאַהֲבָה וְנִגְדְּלָה בְכָל-דָּבָר לוֹ לַמָּשִׁיחַ אֲשֶׁר הוּא הָרֹאשׁ׃ 16 אֲשֶׁר מִמֶּנּוּ כָּל-הַגּוּף הַמֻּרְכָּב וּמְדֻבָּק בְּכָל-חִבּוּר הַשִּׁמּוּשׁ כְּמִדַּת הַכֹּחַ הַנָּתוּן לְכָל-אֵבָר וְאֵבָר יִרְבֶּה וְיִגְדַּל לְהַשְׁלָמַת בִּנְיָנוֹ בְּאַהֲבָה׃

החיים החדשים

17 וְהִנֵּה זֹאת אֲנִי אֹמֵר וּמֵעִיד בָּאָדוֹן כִּי מֵעַתָּה לֹא-תֵלְכוּ עוֹד כְּיֶתֶר הַגּוֹיִם הַהֹלְכִים בְּהַבְלֵי שִׂכְלָם׃ 18 חֲשֻׁכֵי הַדַּעַת וּמוּזָרִים לְחַיֵּי אֱלֹהִים מִפְּנֵי אִוַּלְתָּם אֲשֶׁר בָּהֶם כִּי-טַח מֵהַשְׂכִּיל לְבָבָם׃ 19 אֲשֶׁר הִשְׁמִינוּ וַיִּתְּנוּ אֶת-נַפְשָׁם לַזִּמָּה לַעֲשׂוֹת כָּל-תּוֹעֵבָה לִבְצֹעַ בָּצַע׃ 20 וְאַתֶּם לֹא כֵן לְמַדְתֶּם אֶת-הַמָּשִׁיחַ׃ 21 אִם-אָמְנָם שְׁמַעְתֶּם אֹתוֹ וּבוֹ לֻמַּדְתֶּם כְּפִי הָאֱמֶת אֲשֶׁר בְּיֵשׁוּעַ׃(תהילים מה' ה') 22 לָסוּר מִדַּרְכֵיכֶם הָרִאשֹׁנִים וְלִפְשֹׁט אֶת-הָאָדָם הַיָּשָׁן הַנִּשְׁחָת בְּתַאֲוֹת רְמִיָּה׃ 23 לְהִתְחַדֵּשׁ בְּרוּחַ שִׂכְלְכֶם׃ 24 וְלִלְבּשׁ אֶת-הָאָדָם הֶחָדָשׁ הַנִּבְרָא כִּדְמוּת אֱלֹהִים בִּצְדָקָה וּקְדֻשַּׁת הָאֱמֶת׃ 25 עַל-כֵּן הָסִירוּ מִכֶּם אֶת-הַשֶּׁקֶר וְדַבְּרוּ אֱמֶת אִישׁ אֶת-רֵעֵהוּ (זכריה ח' טז') כִּי אֵבָרִים אֲנַחְנוּ אִישׁ לְאָחִיו׃ 26 רִגְזוּ וְאַל-תֶּחֱטָאוּ (תהילים ד' ה') וְלֹא תָבֹא הַשֶּׁמֶשׁ עַל-רָגְזְכֶם׃(דברים כד' טו') 27 וְאַל-תִּתְּנוּ מָקוֹם לַשָׂטָן׃ 28 מִי שֶׁגָּנַב אַל-יֹסֶף לִגְנֹב כִּי אִם-יִיגַע וּבְיָדָיו יַעֲשֶׂה אֶת-הַטּוֹב לְמַעַן יִהְיֶה-לּוֹ לָתֵת לְאִישׁ מַחְסוֹר׃ 29 וְלֹא-יֵצֵא מִפִּיכֶם כָּל-דְּבַר נִבּוּל כִּי אִם-הַטּוֹב וְהַמּוֹעִיל לִבְנוֹת לְפִי הַצֹּרֶךְ לְמַעַן יִתֵּן חֵן לְשֹׁמְעָיו׃ 30 וְלֹא תְעַצְּבוּ אֶת-רוּחַ הַקֹּדֶשׁ שֶׁל-אֱלֹהִים אֲשֶׁר נֶחְתַּמְתֶּם בּוֹ לְיוֹם הַגְּאֻלָּה׃ 31 כָּל-מְרִירוּת וְחֵמָה וָרֹגֶז וּצְעָקָה וְגִדּוּף תָּסִירוּ מִכֶּם עִם כָּל-הָרִשְׁעָה׃ 32 וִהְיוּ טוֹבִים אִישׁ לְרֵעֵהוּ וְרַחֲמָנִים וְנֹשְׂאִים אִישׁ לְחַטַּאת אָחִיו כַּאֲשֶׁר נָשָׂא לָכֶם הָאֱלֹהִים בַּמָּשִׁיחַ׃
