אגרת שאול אל האפסים
פרק ה

חיים על-פי האור

1 וּבְכֵן לְכוּ בְּדַרְכֵי הָאֱלֹהִים כְּבָנִים חֲבִיבִים׃ 2 וְהִתְהַלְּכוּ בְּאַהֲבָה כַּאֲשֶׁר גַּם-הַמָּשִׁיחַ אָהַב אֹתָנוּ וְהִקְרִיב אֶת-נַפְשׁוֹ בַּעֲדֵנוּ לְקָרְבָּן וָזֶבַח לֵאלֹהִים לְרֵיחַ נִיחוֹחַ׃(בראשית ח' כא'; ויקרא א' ט') 3 אֲבָל זְנוּת וְכָל-טֻמְאָה וְאַהֲבַת בֶּצַע בַּל-יִזָּכֵר בֵּינֵיכֶם כַּאֲשֶׁר נָאוֶה לַקְּדשִׁים׃ 4 גַּם-לֹא נִבּוּל פֶּה וְדִבְרֵי סִכְלוּת לִיצָנוּת מַה-שֶּׁלּא כְהֹגֶן כִּי אִם-קוֹל תּוֹדָה׃ 5 כִּי-זֹאת יָדֹעַ תֵּדְעוּ כִּי כָל-זֹנֶה וְטָמֵא וְאֹהֵב בֶּצַע אֲשֶׁר כְּעֹבֵד אֱלִילִים אֵין-לוֹ חֵלֶק וְנַחֲלָה בְּמַלְכוּת הַמָּשִׁיחַ וְהָאֱלֹהִים׃ 6 אַל-יַשִׂיא אִישׁ אֶתְכֶם בְּדִבְרֵי רִיק כִּי בִגְלַל-אֵלֶּה חֲרוֹן אֱלֹהִים יָבֹא עַל-בְּנֵי הַמֶּרִי׃ 7 עַל-כֵּן אַל-יְהִי חֶלְקְכֶם עִמָּהֶם׃(תהילים א' א') 8 כִּי מִלְּפָנִים הֱיִיתֶם חשֶׁךְ וְעַתָּה הִנְּכֶם אוֹר בַּאֲדֹנֵינוּ הִתְהַלְּכוּ-נָא כִּבְנֵי אוֹר׃ 9 כִּי-פְרִי [הָאוֹר] כָּל-מַעֲשֵׂי חֶסֶד וְצֶדֶק וֶאֱמֶת׃ 10 לִבְחֹן מֶה-הָרָצוּי בְּעֵינֵי אֲדֹנֵינוּ׃ 11 וְאַל-תִּשְׁתַּתְּפוּ בְּמַעֲשֵׂי הַחשֶׁך אֲשֶׁר לֹא-יַעֲשׂוּ פֶרִי כִּי אִם-הוֹכֵחַ תּוֹכִיחוּ אוֹתָם׃ 12 כִּי מַה-שֶּׁהֵם עֹשִׂים בַּסֵּתֶר חֶרְפָּה לְסַפֵּר׃ 13 אֲבָל כָּל-זֹאת יְגֻלֶּה כְּשֶׁיּוּכַח עַל-יְדֵי הָאוֹר כִּי כָל-הַנִּגְלֶה אוֹר הוּא׃ 14 עַל-כֵּן הוּא אוֹמֵר עוּרָה הַיָּשֵׁן וְקוּמָה מִן-הַמֵּתִים (תהילים נז' ט') וְיָאֵר לְךָ הַמָּשִׁיחַ׃ 15 וְעַתָּה רְאוּ וְהִזָּהֲרוּ שֶׁלּא תִתְהַלְּכוּ כַּכְּסִילִים כִּי אִם-כַּחֲכָמִים׃ 16 מוֹקִירִים הַשָּׁעָה כִּי הַיָּמִים רָעִים הֵמָּה׃ 17 עַל-כֵּן אַל-תִּהְיוּ חַסְרֵי דָעַת כִּי אִם-מְבִינִים לָדַעַת מָה רְצוֹן אֲדֹנֵינוּ׃ 18 וְאַל-תִּשְׁתַּכְּרוּ מִיַּיִן הַמֵּבִיא לִידֵי פְרִיצוּת כִּי אִם-הִמָּלְאוּ רוּחַ׃ 19 וּתְהִי שִׂיחַתְכֶם אִישׁ אֶת-רֵעֵהוּ תְּהִלּוֹת וְתִשְׁבָּחוֹת וְשִׁירוֹת רוּחָנִיּוֹת שִׁירוּ וְזַמְּרוּ לַאדֹנָי בִּלְבַבְכֶם׃(תהילים ל' ה'; לד' ב') 20 וְהוֹדוּ בְּכָל-עֵת עַל-הַכֹּל לֵאלֹהִים אָבִינוּ בְּשֵׁם אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ׃

יחסים בין הבריות במשפחה ובחברה

21 הִכָּנְעוּ אִישׁ לְאָחִיו בְּיִרְאַת אֱלֹהִים׃ 22 הַנָּשִׁים הִכָּנַעְנָה לְבַעֲלֵיכֶן כְּמוֹ לַאֲדֹנֵינוּ׃ 23 כִּי הָאִישׁ הוּא-רֹאשׁ הָאִשָּׁה כַּאֲשֶׁר הַמָּשִׁיחַ הוּא רֹאשׁ הָעֵדָה וְהוּא מוֹשִׁיעַ שֶׁל-הַגּוּף׃ 24 אֲבָל כַּאֲשֶׁר תִּכָּנַע הָעֵדָה לַמָּשִׁיחַ אַף כֵּן-הַנָּשִׁים לְבַעֲלֵיהֶן לְכָל-דָּבָר׃ 25 הָאֲנָשִׁים אֶהֱבוּ אֶת-נְשֵׁיכֶם כַּאֲשֶׁר גַּם-הַמָּשִׁיחַ אָהַב אֶת-הָעֵדָה וַיִּתֵּן אֶת-נַפְשׁוֹ בַּעֲדָהּ׃ 26 לְמַעַן קַדְּשָׁהּ בִּדְבָרוֹ אַחֲרֵי אֲשֶׁר-טִהֲרָהּ בִּרְחִיצַת הַמָּיִם׃ 27 לַהֲקִימָהּ בְּכָבוֹד לוֹ לְעֵדָה אֲשֶׁר אֵין-בָּהּ כֶּתֶם וָקֶמֶט וּכְדוֹמֶה כִּי אִם-לְמַעַן תִּהְיֶה קְדוֹשָׁה וּתְמִימָה׃ 28 כֵּן הָאֲנָשִׁים חַיָּבִים לֶאֱהֹב אֶת-נְשֵׁיהֶם כְּגוּפָם כִּי הָאֹהֵב אֶת-אִשְׁתּוֹ אֹהֵב אֶת-עַצְמוֹ׃ 29 כִּי מֵעוֹלָם לֹא-שָׂנֵא אִישׁ אֶת-בְּשָׂרוֹ כִּי אִם-זָן וּמְכַלְכֵּל אֹתוֹ כַּאֲשֶׁר גַּם-הָאָדוֹן אֶת-עֲדָתוֹ׃ 30 כִּי-אֵבָרֵי גוּפוֹ אֲנָחְנוּ מִבְּשָׂרוֹ וּמֵעֲצָמָיו׃(בראשית ב' כג') 31 עַל-כֵּן יַעֲזָב-אִישׁ אֶת-אָבִיו וְאֶת-אִמּוֹ וְדָבַק בְּאִשְׁתּוֹ וְהָיוּ שְׁנֵיהֶם לְבָשָׂר אֶחָד׃(בראשית ב' כד') 32 גָּדוֹל הַסּוֹד הַזֶּה וַאֲנִי פּוֹתֵר אֹתוֹ בַּמָּשִׁיחַ וְהָעֵדָה׃ 33 וְאוּלָם גַּם-אַתֶּם כָּל-אִישׁ מִכֶּם יֶאֱהַב אֶת-אִשְׁתּוֹ כְּנַפְשׁוֹ וְהָאִשָּׁה הִיא תִזָּהֵר וְתִירָא אֶת-בַּעְלָהּ׃
