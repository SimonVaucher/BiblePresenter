לפרק הקודם

אגרת שאול אל הרומים
פרק י

ישראל והבשורה -- המשך

1 אֶחָי חֵפֶץ לְבָבִי וּתְפִילָּתִי לֵאלֹהִים בְּעַד יִשְׂרָאֵל אֲשֶׁר יִוָּשֵׁעוּ׃ 2 כִּי מֵעִיד אֲנִי עֲלֵיהֶם שֶׁמְּקַנְּאִים לֵאלֹהִים אֲבָל לֹא-בְדָעַת׃ 3 כִּי אֶת-צִדְקַת אֱלֹהִים לֹא יָדָעוּ וַיְבַקְשׁוּ לְהָקִים אֶת-צִדְקָתָם וּלְצִדְקַת אֱלֹהִים לֹא נִכְנָעוּ׃ 4 כִּי הַמָּשִׁיחַ סוֹף הַתּוֹרָה לִצְדָקָה לְכָל-הַמַּאֲמִין בּוֹ׃ 5 כִּי-משֶׁה כָּתַב עַל-דְּבַר הַצְּדָקָה מִתּוֹךְ הַתּוֹרָה אֲשֶׁר יַעֲשֶׂה אֹתָם הָאָדָם וָחַי בָּהֶם׃(ויקרא יח' ה') 6 וְהַצְּדָקָה אֲשֶׁר מִתּוֹךְ הָאֱמוּנָה אֹמֶרֶת אַל-תֹּאמַר בִּלְבָבְךָ מִי-יַעֲלֶה הַשָּׁמַיְמָה לְהוֹרִיד אֶת-הַמָּשִׁיחַ׃(דברים ל' יב'-יג') 7 אוֹ מִי יֵרֵד לִתְהוֹם לְהַעֲלוֹת אֶת-הַמָּשִׁיחַ מִן-הַמֵּתִים׃ 8 אֲבָל מַה-תֹּאמַר קָרוֹב אֵלֶיךָ הַדָּבָר בְּפִיךָ וּבִלְבָבֶךָ (דברים ל' יד') הוּא דְּבַר הָאֱמוּנָה אֲשֶׁר אֲנַחְנוּ מְבַשְׂרִים׃ 9 כִּי אִם-בְּפִיךָ תוֹדֶה שֶׁיֵּשׁוּעַ הוּא הָאָדוֹן וְתַאֲמִין בִּלְבָבְךָ שֶׁהָאֱלֹהִים הֱעִירוֹ מִן-הַמֵּתִים תִּוָּשֵׁעַ׃ 10 כִּי בִלְבָבוֹ יַאֲמִין הָאָדָם וְהָיְתָה לּוֹ לִצְדָקָה וּבְפִיהוּ יוֹדֶה וְהָיְתָה-לּוֹ לִישׁוּעָה׃ 11 כִּי הַכָּתוּב אֹמֵר כָּל-הַמַּאֲמִין בּוֹ לֹא יֵבוֹשׁ׃(ישעיהו כח' טז') 12 וְאֵין הַפְרֵשׁ בֵּין הַיְּהוּדִי לַיְּוָנִי כִּי אָדוֹן אֶחָד לְכֻלָּם וְהוּא עָשִׁיר לְכָל-קֹרְאָיו׃(תהילים קמה' יח') 13 כִּי-כֹל אֲשֶׁר-יִקְרָא בְּשֵׁם יְהוָֹה יִמָּלֵט׃(יואל ג' ה') 14 וְעַתָּה אֵיךְ יִקְרְאוּ אֶל-אֲשֶׁר לֹא-הֶאֱמִינוּ בוֹ וְאֵיךְ יַאֲמִינוּ בְּמִי שֶׁלּא שָׁמְעוּ אֶת-שִׁמְעוֹ וְאֵיךְ יִשְׁמְעוּ וְאֵין מַגִּיד׃ 15 וְאֵיךְ יַגִּידוּ כִּי אִם-שְׁלוּחִים כַּכָּתוּב מַה-נָּאווּ רַגְלֵי מְבַשֵׂר שָׁלוֹם מְבַשֵׂר טוֹב׃(ישעיהו נב' ז') 16 אֲבָל לֹא-כֻלָּם שָׁמְעוּ לְקוֹל הַבְּשׂוֹרָה כִּי יְשַׁעְיָהוּ אָמַר יְהוָֹה מִי הֶאֱמִין לִשְׁמֻעָתֵנוּ׃(ישעיהו נג' א') 17 אִם-כֵּן הָאֱמוּנָה בָאָה מִתּוֹךְ הַשְּׁמוּעָה וְהַשְּׁמוּעָה עַל-יְדֵי דְבַר-הַמָּשִׁיחַ׃ 18 וְאֹמַר הֲכִי לֹא שָׁמֵעוּ אָמְנָם בְּכָל-הָאָרֶץ יָצָא קַוָּם וּבִקְצֵה תֵבֵל מִלֵּיהֶם׃(תהילים יט' ה') 19 וְאֹמַר הֲכִי יִשְׂרָאֵל לֹא יָדָע הִנֵּה-כְבָר משֶׁה אָמַר אֲנִי אַקְנִיאֲכֶם בְּלֹא-עָם בְּגוֹי נָבָל אַכְעִיסְכֶם׃(דברים לב' כא') 20 וִישַׁעְיָהוּ מְלָאוֹ לִבּוֹ לֵאמֹר נִמְצֵאתִי לְלֹא בִקְשֻׁנִי נִדְרַשְׁתִּי לְלוֹא שָׁאָלוּ׃(ישעיהו סה' א') 21 וְעַל-יִשְׂרָאֵל הוּא אֹמֵר פֵּרָשְׂתִּי יָדַי כָּל-הַיּוֹם אֶל-עַם סוֹרֵר וּמֹרֶה׃(ישעיהו סה' ב')
