לפרק הקודם

אגרת שאול אל הרומים
פרק ט

ישראל ובחירת אלהים

1 אֱמֶת אֲנִי מְדַבֵּר בַּמָּשִׁיחַ וְלֹא אֲשַׁקֵּר וְדַעְתִּי מְעִידָה לִי בְּרוּחַ הַקֹּדֶשׁ׃ 2 כִּי-גָדוֹל עִצְּבוֹנִי וְאֵין-קֵץ לְדַאֲבוֹן לִבִּי׃ 3 כִּי מִי-יִתֵּן הֱיוֹתִי אֲנִי לְחֵרֶם מִן-הַמָּשִׁיחַ בְּעַד אַחַי שְׁאֵרִי וּבְשָׂרִי׃(שמות לב' לב') 4 אֲשֶׁר הֵם בְּנֵי יִשְׂרָאֵל וְלָהֶם מִשְׁפַּט הַבָּנִים (שמות ד' כב') וְהַכָּבוֹד (שמות מ' לד') וְהַבְּרִיתוֹת (בראשית יז' ז'; שמות כד' ז'; דברים כח' סט'; ירמיהו לא' ל'-לא') וּמַתַּן הַתּוֹרָה (דברים ד' ח') וְהָעֲבוֹדָה (דברים יח' ה') וְהַהַבְטָחוֹת׃(שמות יט' ו') 5 וְלָהֶם הָאָבוֹת וּמֵהֶם יָצָא הַמָּשִׁיחַ לְפִי בְשָׂרוֹ אֲשֶׁר-הוּא אֱלֹהִים עַל-הַכֹּל מְבֹרָךְ לְעוֹלָמִים אָמֵן׃(דברים י' טו') 6 אֲבָל לֹא שֶׁנָּפַל דְּבַר אֱלֹהִים אָרְצָה (במדבר כג' יט') כִּי לֹא-כֹל אֲשֶׁר מִיִּשְׂרָאֵל יִשְׂרָאֵל הֵמָּה׃ 7 וְלֹא מִפְּנֵי שֶׁהֵם זֶרַע אַבְרָהָם כֻּלָּם בָּנִים כִּי בְיִצְחָק יִקָּרֵא לְךָ זָרַע׃(בראשית כא' יב') 8 כְּלוֹמַר לֹא בְנֵי-הַבָּשָׂר הֵמָּה בְּנֵי הָאֱלֹהִים כִּי אִם-בְּנֵי הַהַבְטָחָה הֵם הַנֶּחֱשָׁבִים לְזָרַע׃ 9 כִּי-דְבַר הַהַבְטָחָה הוּא מַה-שֶּׁנֶּאֱמַר לַמּוֹעֵד אָשׁוּב וּלְשָׂרָה בֵן׃(בראשית יח' י'; יח' יד') 10 וְלֹא-עוֹד אֶלָּא שֶׁהָיָה גַּם-בְּרִבְקָה וְהִיא הָרָה לְאֶחָד לְיִצְחָק אָבִינוּ׃(בראשית כה' כא') 11 כִּי בְּטֶרֶם יֻלְּדוּ בָנֶיהָ וְעוֹד לֹא-עֲשׂוּ טוֹב אוֹ-רָע לְמַעַן תָּקוּם עֲצַת הָאֱלֹהִים כְּפִי בְחִירָתוֹ לֹא מִתּוֹךְ מַעֲשִׂים כִּי אִם-כִּרְצוֹן הַקֹּרֵא׃ 12 נֶאֱמַר לָהּ כִּי-רַב יַעֲבֹד צָעִיר׃(בראשית כה' כג') 13 כַּכָּתוּב וָאֹהַב אֶת-יַעֲקֹב וְאֶת-עֵשָׂו שָׂנֵאתִי׃(מלאכי א' ב'-ג') 14 אִם-כֵּן הֲנֹאמַר שֶׁיֶּשׁ-עָוֶל בֵּאלֹהִים חָלִילָה׃(איוב ח' ג') 15 כִּי לְמשֶׁה אָמַר וְחַנֹּתִי אֶת-אֲשֶׁר אָחֹן וְרִחַמְתִּי אֶת-אֲשֶׁר אֲרַחֵם׃(שמות לג' יט') 16 וְעַל-כֵּן אֵין הַדָּבָר לֹא-בְיַד הָרֹצֶה וְלֹא-בְיַד הָרָץ כִּי אִם-בְּיַד הָאֱלֹהִים הַמְרַחֵם׃ 17 כִּי-כֵן הַכָּתוּב אֹמֵר לְפַרְעֹה בַּעֲבוּר זֹאת הֶעֱמַדְתִּיךָ בַּעֲבוּר הַרְאֹתְךָ אֶת-כֹּחִי וּלְמַעַן סַפֵּר שְׁמִי בְּכָל-הָאָרֶץ׃(שמות ט' טז') 18 וְיִוָּדַע בָּזֶה שֶׁמִּי שֶׁיַּחְפֹּץ יְחֻנֶּנּוּ וּמִי שֶׁיַּחְפֹּץ יַקְשֶׁה לִבּוֹ׃(שמות ז' ג')

זעם אלהים ורחמיו

19 וְאִם תֹּאמַר לָמָּה-זֶּה יִפְקֹד עָוֹן כִּי נֶגֶד רְצוֹנוֹ מִי יִתְיַצָּב׃ 20 אֲבָל בֶּן-אָדָם מִי אַתָּה כִּי תָרִיב אֶת-הָאֱלֹהִים הֲיֹאמַר יֵצֶר לְיֹצְרוֹ מַדּוּעַ כָּכָה עֲשִׂיתָנִי׃(ישעיהו מה' ט') 21 אִם-אֵין רְשׁוּת לַיֹּצֵר עַל-הַחֹמֶר לַעֲשׂוֹת הַגֹּלֶם הָאֶחָד כְּלִי כָבוֹד אוֹ כְּלִי קָלוֹן׃(ירמיהו יח' ג'-ו') 22 וּמָה אֵפוֹא אִם-הָאֱלֹהִים הֶחָפֵץ לְהַרְאוֹת זַעְמוֹ וּלְהוֹדִיעַ גְּבוּרָתוֹ נָשָׂא בְּכָל-אֹרֶךְ רוּחוֹ אֶת-כְּלֵי הַזַּעַם הַנְּכוֹנִים לַאֲבַדּוֹן׃(משלי טז' ד') 23 לְהוֹדִיעַ גַּם-אֶת-עשֶׁר כְּבוֹדוֹ בִּכְלֵי הַחֲנִינָה אֲשֶׁר הֵכִין לְכָבוֹד׃ 24 וְהֵם אֲנַחְנוּ אֲשֶׁר קְרָאָנוּ לֹא מִן-הַיְּהוּדִים לְבַדָּם כִּי אַף מִן-הַגּוֹיִם׃ 25 כְּאָמְרוֹ בְהוֹשֵׁעַ אֶקְרָא לְלֹא-עַמִּי עַמִּי וּלְלֹא-רֻחָמָה רֻחָמָה׃(הושע ב' כה') 26 וְהָיָה בִּמְקוֹם אֲשֶׁר-יֵאָמֵר לָהֶם לֹא-עַמִּי אַתֶּם יֵאָמֵר לָהֶם בְּנֵי אֵל-חָי׃(הושע ב' א'; ב' ג') 27 וִישַׁעְיָהוּ קֹרֵא עַל-יִשְׂרָאֵל כִּי אִם-יִהְיֶה מִסְפַּר בְּנֵי יִשְׂרָאֵל כְּחוֹל הַיָּם שְׁאָר יָשׁוּב בּוֹ [כִּלָּיוֹן חָרוּץ שׁוֹטֵף צְדָקָה]׃ 28 כִּי כָלָה וְנֶחֱרָצָה אֲדֹנָי עֹשֶׂה בְּקֶרֶב הָאָרֶץ׃(ישעיהו י' כב'-כג') 29 וְכַאֲשֶׁר אָמַר יְשַׁעְיָהוּ לִפְנֵי מִזֶּה לוּלֵי יְהוָֹה צְבָאוֹת הוֹתִיר לָנוּ שָׂרִיד כִּמְעָט כִּסְדֹם הָיִינוּ לַעֲמֹרָה דָּמִינוּ׃(ישעיהו א' ט')

ישראל והבשורה

30 וְעַתָּה הֲנֹאמַר שֶׁהַגּוֹיִם אֲשֶׁר לֹא רָדְפוּ אַחֲרֵי הַצְּדָקָה הִשִׂיגוּ אֶת-הַצְּדָקָה הִיא הַצְּדָקָה אֲשֶׁר מִתּוֹךְ הָאֱמוּנָה׃ 31 וְיִשְׂרָאֵל בְּרָדְפוֹ תּוֹרַת צְדָקָה לְתוֹרַת הַצְדָקָה לֹא הִגִּיעַ׃ 32 וְעַל-מָה עַל-אֲשֶׁר-לֹא מֵאֱמוּנָה דְּרָשׁוּהָ כִּי אִם-מִמַּעֲשִׂים כִּי הִתְנַגְּפוּ בְּאֶבֶן נָגֶף׃ 33 כַּכָּתוּב הִנְנִי יִסַּד בְּצִיּוֹן אֶבֶן נֶגֶף וְצוּר מִכְשׁוֹל וְכָל-הַמַּאֲמִין בּוֹ לֹא יֵבוֹשׁ׃(ישעיהו ח' יד'; כח' טז')
