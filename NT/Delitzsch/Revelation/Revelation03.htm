פרק ג

המכתב החמישי

1 וְאֶל-מַלְאַךְ קְהַל סַרְדִּיס כְּתֹב כֹּה אָמַר אֲשֶׁר-לוֹ שֶׁבַע רוּחוֹת הָאֱלֹהִים וְשִׁבְעַת הַכּוֹכָבִים יָדַעְתִּי אֶת-מַעֲשֶׂיךָ כִּי-לְךָ שֵׁם כְּאִלּוּ חַי אַתָּה וְהִנְּךָ מֵת׃(תהילים לג' טו') 2 שְׁקֹד וְחַזֵּק אֶת-הַשְּׁאֵרִית הַקְּרוֹבָה לָמוּת כִּי לֹא-מָצָאתִי מַעֲשֶׂיךָ שְׁלֵמִים לִפְנֵי אֱלֹהָי׃ 3 זְכוֹר אֵת אֲשֶׁר-קִבַּלְתָּ וְשָׁמָעְתָּ וְשָׁמְרָה-זֹּאת וְשׁוּבָה וְאִם-לֹא תִשְׁקֹד הִנְנִי בָא עָלֶיךָ כַּגַּנָּב וְלֹא תֵדַע הַשָּׁעָה אֲשֶׁר אָבֹא עָלֶיךָ׃ 4 יֶשׁ-לְךָ גַם-בְּסַרְדִּיס שֵׁמוֹת מְעַט אֲשֶׁר לֹא גֵאֲלוּ אֶת-מַלְבּוּשֵׁיהֶם וְיִתְהַלְּכוּ אִתִּי לְבֻשֵׁי לְבָנִים כִּי רְאוּיִם הֵם לָזֹאת׃ 5 הַמְנַצֵּחַ יִלְבַּשׁ בְּגָדִים לְבָנִים וְלֹא אֶמְחֶה אֶת-שְׁמוֹ מִסֵּפֶר הַחַיִּים וְאוֹדֶה שְׁמוֹ לִפְנֵי אָבִי וְלִפְנֵי מַלְאָכָיו׃(דניאל יב' א') 6 מִי אֲשֶׁר-אֹזֶן לוֹ יִשְׁמַע אֵת אֲשֶׁר הָרוּחַ אֹמֵר לַקְּהִלּוֹת׃

המכתב הששי

7 וְאֶל-מַלְאַךְ קְהַל פִילָדֶלְפִיָּא כְּתֹב כֹּה אָמַר הַקָּדוֹשׁ הָאֲמִתִּי אֲשֶׁר בְּיָדוֹ מַפְתֵּחַ דָּוִד הַפֹּתֵחַ וְאֵין סֹגֵר וְסֹגֵר וְאֵין פֹּתֵחַ׃(ישעיהו כב' כב') 8 יָדַעְתִּי אֶת-מַעֲשֶׂיךָ הִנֵּה נָתַתִּי לְפָנֶיךָ שַׁעַר פָּתוּחַ אֲשֶׁר לֹא-יוּכַל אִישׁ לְסָגְרוֹ כִּי כֹחַ-מְעַט לָךְ וַתִּשְׁמֹר אֶת-דְּבָרִי וְלֹא כִחַשְׁתָּ בִּשְׁמִי׃ 9 הִנְנִי נֹתֵן אֲנָשִׁים מִכְּנֵסֶת הַשָּׂטָן אֲשֶׁר יֹאמְרוּ יְהוּדִים אֲנַחְנוּ וְאֵינָם כִּי-דֹבְרֵי שֶׁקֶר הֵמָּה הִנְנִי עֹשֶׂה אֲשֶׁר יָבֹאוּ לְהִשְׁתַּחֲוֹת לְרַגְלֶיךָ (ישעיהו מט' כג'; ס' יד') וְיָדְעוּ כִּי אֲנִי אֲהַבְתִּיךָ׃(ישעיהו נג' ד') 10 יַעַן שָׁמַרְתָּ דְּבַר סַבְלָנוּתִי אֶשְׁמָרְךָ גַם-אֲנִי מִשְּׁעַת הַנִּסָּיוֹן הָעֲתִידָה לָבוֹא עַל-תֵּבֵל כֻּלָּהּ לְנַסּוֹת אֶת-יֹשְׁבֵי הָאָרֶץ׃ 11 הִנְנִי בָא מְהֵרָה הַחֲזֵק בַּאֲשֶׁר לָךְ לְמַעַן לֹא-יִקַּח אִישׁ אֶת-נִזְרֶךָ׃ 12 הַמְנַצֵּחַ אֶתְּנֶנּוּ לְעַמּוּד בְּהֵיכַל אֱלֹהַי וְלֹא-יֵצֵא עוֹד הַחוּצָה (ישעיהו נו' ה') וְכָתַבְתִּי עָלָיו אֶת-שֵׁם אֱלֹהַי וְאֶת-שֵׁם עִיר אֱלֹהַי יְרוּשָׁלַיִם הַחֲדָשָׁה הַיֹּרֶדֶת מִשָּׁמַיִם מֵעִם-אֱלֹהַי וְאֶת-שְׁמִי הֶחָדָשׁ׃(ישעיהו סה' יז'-יט') 13 מִי אֲשֶׁר-אֹזֶן לוֹ יִשְׁמַע אֵת אֲשֶׁר הָרוּחַ אֹמֵר לַקְּהִלּוֹת׃

המכתב השביעי

14 וְאֶל-מַלְאַךְ קְהַל לוּדְקְיָא כְּתֹב כֹּה אָמַר הָאָמֵן הָעֵד הַנֶּאֱמָן וְהָאֲמִתִּי רֵאשִׁית בְּרִיאַת הָאֱלֹהִים׃ 15 יָדַעְתִּי אֶת-מַעֲשֶׂיךָ כִּי לֹא קַר אַתָּה וְלֹא חָם מִי-יִתֵּן וְהָיִיתָ קַר אוֹ-חָם׃(מלכים א', יח' כא'//***) 16 כִּי עַתָּה פּוֹשֵׁר אַתָּה וְלֹא קַר וְלֹא חָם עַל-כֵּן אֲקִיאֲךָ מִפִּי׃ 17 כִּי אָמַרְתָּ אַךְ עָשַׁרְתִּי מָצָאתִי אוֹן לִי וְלֹא חָסַרְתִּי כֹּל וְלֹא יָדַעְתָּ כִּי-אֻמְלַל אַתָּה וְדַוָּי וְעָנִי וְעִוֵּר וְעֵרֹם׃ 18 אִיעָצְךָ לִקְנוֹת מֵאִתִּי זָהָב צָרוּף בָּאֵשׁ לְמַעַן תַּעֲשִׁיר (ישעיהו נה' א'; משלי ח' יח') וּבְגָדִים לְבָנִים לְמַעַן תִּתְכַּסֶּה בָהֶם וְלֹא תֵרָאֶה בֹּשֶׁת עֶרְוָתְךָ וְקִלּוּרִית לָתֵת עַל-עֵינֶיךָ לְמַעַן תִּרְאֶה׃ 19 אֲנִי אֶת-כֹּל אֲשֶׁר אֵהַב אוֹכִיחַ וַאֲיַסֵּר לָכֵן תִּשְׁקֹד וְתָשׁוּב׃(משלי ג' יא'-יב') 20 הִנְנִי עֹמֵד בַּפֶּתַח וְדֹפֵק וְהָיָה כִּי-יִשְׁמַע אִישׁ לְקוֹלִי וּפָתַח הַפֶּתַח אָבוֹא אֵלָיו וְסָעַדְתִּי עִמּוֹ וְהוּא עִמִּי׃(שיר השירים ה' ב') 21 הַמְנַצֵּחַ אֶתְּנֶנּוּ לָשֶׁבֶת אִתִּי עַל-כִּסְאִי כַּאֲשֶׁר נִצַּחְתִּי גַם-אָנֹכִי וָאֵשֵׁב אֶת-אָבִי עַל-כִּסְאוֹ׃(איוב לו' ז') 22 מִי אֲשֶׁר-אֹזֶן לוֹ יִשְׁמַע אֵת אֲשֶׁר הָרוּחַ אֹמֵר לַקְּהִלּוֹת׃
