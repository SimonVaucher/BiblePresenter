פרק טז

קערות זעם אלהים

1 וָאֶשְׁמַע קוֹל גָּדוֹל מִן-הַהֵיכָל אֹמֵר אֶל-שִׁבְעַת הַמַּלְאָכִים (ישעיהו סו' ו') לְכוּ הָרִיקוּ אֶת-[שֶׁבַע] קַעֲרֹת חֲמַת הָאֱלֹהִים אָרְצָה׃(ירמיהו י' כה'; צפניה ג' ח') 2 וַיֵּלֶךְ הָרִאשׁוֹן וַיָּרֶק אֶת-קַעֲרָתוֹ עַל-הָאָרֶץ וַיְהִי שְׁחִין רַע וּמַכְאִיב בָּאֲנָשִׁים אֲשֶׁר-עֲלֵיהֶם תָּו הַחַיָּה וּבַמִּשְׁתַּחֲוִים לְצַלְמָהּ׃(שמות ט' ט'; דברים כח' לה') 3 וַיָּרֶק הַשֵּׁנִי אֶת-קַעֲרָתוֹ עַל-הַיָּם וַיְהִי לְדָם כְּדַם חָלָל וַתָּמָת כָּל-נֶפֶשׁ חַיָּה אֲשֶׁר בַּיָּם׃(שמות ז' כ'-כא') 4 וַיָּרֶק הַשְּׁלִישִׁי אֶת-קַעֲרָתוֹ בַּנְּהָרוֹת וּבְמַעְיְנוֹת הַמָּיִם וַיִּהְיוּ לְדָם׃ 5 וָאֶשְׁמַע אֶת-מַלְאָךְ הַמַּיִם אֹמֵר צַדִּיק אַתָּה הַהוֶֹה וְהָיָה וְהַקָּדוֹשׁ כִּי-כֵן שָׁפָטְתָּ׃ 6 כִּי דַּם קְדֹשִׁים וּנְבִיאִים שָׁפָכוּ וְדָם הִשְׁקִיתָם כִּי-גְמוּל יָדָם הוּא׃(תהילים עט' ג'; ישעיהו מט' כו') 7 וָאֶשְׁמַע אֶת-הַמִּזְבֵּחַ אֹמֵר אָמְנָם כֵּן יְהוָֹה אֱלֹהִים צְבָאוֹת אֱמֶת וָצֶדֶק מִשְׁפָּטֶיךָ׃(תהילים יט' י') 8 וַיָּרֶק הַמַּלְאָךְ הָרְבִיעִי אֶת-קַעֲרָתוֹ עַל-הַשָּׁמֶשׁ וַיִּנָּתֶן לוֹ לְלַהֵט אֶת-בְּנֵי הָאָדָם בָּאֵשׁ׃(מלאכי ג' יט') 9 וַיְלֹהֲטוּ בְנֵי-אָדָם בְּחֹם גָּדוֹל וַיְגַדְּפוּ אֶת-שֵׁם אֱלֹהִים אֲשֶׁר-לוֹ מֶמְשֶׁלֶת הַמַּכּוֹת הָאֵלֶּה וְלֹא שָׁבוּ לָתֶת-לוֹ הַכָּבוֹד׃(ירמיהו ה' ג') 10 וַיָּרֶק הַחֲמִישִׁי אֶת-קַעֲרָתוֹ עַל-כִּסֵּא הַחַיָּה וַתֶּחְשַׁךְ מַלְכוּתָהּ (שמות י' כא'-כב') וַיְנַשְּׁכוּ אֶת-לְשׁוֹנָם מִכְּאֵב לֵב׃ 11 וַיְגַדְּפוּ אֶת-אֱלֹהֵי הַשָּׁמַיִם (ישעיהו ח' כא') מִמַּכְאֹבָם וּשְׁחִינָם וְלֹא שָׁבוּ מִמַּעֲשֵׂיהֶם׃ 12 וַיָּרֶק הַשִּׁשִּׁי אֶת-קַעֲרָתוֹ עַל-הַנָּהָר הַגָּדוֹל נְהַר-פְּרָת וַיֶּחֶרְבוּ מֵימָיו לְיַשֵּׁר מְסִלָּה לַמְּלָכִים הַבָּאִים מִמִּזְרָח-שָׁמֶשׁ׃(נחום א' ד') 13 וָאֵרֶא וְהִנֵּה מִפִּי הַתַּנִּין וּמִפִּי הַחַיָּה וּמִפִּי נְבִיא הַשֶּׁקֶר יֹצְאוֹת שָׁלֹשׁ רוּחוֹת טְמֵאוֹת דּוֹמֹת לִצְפַרְדְּעִים׃ 14 כִּי רוּחוֹת הַשֵּׁדִים הֵנָּה וְהֵן עֹשׂוֹת אוֹתוֹת וְיֹצְאוֹת אֶל-מַלְכֵי אֶרֶץ וְתֵבֵל כֻּלָּהּ לְאָסְפָם לְמִלְחֶמֶת הַיּוֹם הַהוּא הַגָּדוֹל יוֹם אֱלֹהֵי הַצְּבָאוֹת׃ 15 הִנְנִי בָא כְּגַּנָּב אַשְׁרֵי הַשֹּׁקֵד וְשֹׁמֵר אֶת-בְּגָדָיו פֶּן-יֵלֵךְ עָרֹם וְרָאוּ אֶת-עֶרְוָתוֹ׃ 16 וַיֶּאֱסֹף אֹתָם אֶל-הַמָּקוֹם הַנִּקְרָא בְעִבְרִית הַר מְגִדּוֹן׃ 17 וַיָּרֶק הַמַּלְאָךְ הַשְּׁבִיעִי אֶת-קַעֲרָתוֹ בַּאֲוִיר וַיֵּצֵא קוֹל גָּדוֹל מֵהֵיכַל הַשָּׁמַיִם מִן-הַכִּסֵא וַיֹּאמֶר הָיֹה נִהְיָתָה׃ 18 וַיִּהְיוּ קֹלוֹת וּרְעָמִים וּבְרָקִים וַיְהִי רַעַשׁ גָּדוֹל אֲשֶׁר לֹא-הָיָה כָמֹהוּ לְמִן-הֱיוֹת אָדָם עַל-הָאָרֶץ כִּי גָדוֹל הָרַעַשׁ עַד-מְאֹד׃(שמות יט' טז') 19 וְהָעִיר הַגְּדוֹלָה נֶחְלְקָה לִשְׁלֹשָׁה חֲלָקִים וַתִּפֹּלְנָה עָרֵי הַגּוֹיִם וַתִּזָּכֵר בָּבֶל הַגְּדוֹלָה לִפְנֵי אֱלֹהִים לָתֶת-לָהּ כּוֹס יֵין חֲמַת אַפּוֹ׃(ישעיהו כא' ט'; ירמיהו כה' טו') 20 וַיָּנָס כָּל-אִי וְהֶהָרִים לֹא נִמְצָאוּ׃ 21 וְאַבְנֵי בָרָד כַּכִּכָּר מִשְׁקָלָם יָרְדוּ מִן הַשָּׁמַיִם עַל-בְּנֵי הָאָדָם (יהושוע י' יא') וַיְגַדְּפוּ בְנֵי-הָאָדָם אֶת-הָאֱלֹהִים עַל-אֹדוֹת מַכַּת הַבָּרָד כִּי-כָבְדָה מַכָּתוֹ מְאֹד׃
