פרק יח

מפלת בבל

1 אַחֲרֵי-כֵן רָאִיתִי מַלְאָךְ אַחֵר יוֹרֵד מִן-הַשָּׁמָיִם וְהוּא כַבִּיר-כֹּחַ וְהָאָרֶץ הֵאִירָה מִכְּבֹדוֹ׃ 2 וַיִּקְרָא בְּקוֹל עֹז לֵאמֹר נָפְלָה נָפְלָה בָּבֶל הַגְּדוֹלָה (ישעיהו כא' ט') וַתְּהִי נְוֵה שְׂעִירִים וּמִשְׁמָר לְכָל-רוּחַ טָמֵא וּמִשְׁמָר לְכָל-עוֹף טָמֵא וְנִמְאָס׃(ישעיהו יג' כא'; ירמיהו נ' לט') 3 כִּי מִיֵּין חֲמַת זְנוּתָהּ שָׁתוּ כָּל-הַגּוֹיִם וּמַלְכֵי אֶרֶץ זָנוּ אַחֲרֶיהָ וְסֹחֲרֵי אֶרֶץ מִשִּׁפְעַת תַּעֲנֻגֶיהָ הֶעֱשִׁירוּ׃(נחום ג' ד') 4 וָאֶשְׁמַע קוֹל אַחֵר מִן-הַשָּׁמַיִם אֹמֵר צְאוּ מִמֶּנָּה עַמִּי פֶּן-תִּתְחַבְּרוּ אֶל-חַטֹּאתֶיהָ וּפֶן-תֻּכּוּ מִמַּכּוֹתֶיהָ׃(ירמיהו נא' ו'; נא' מה') 5 כִּי-חַטֹּאתֶיהָ הִגִּיעוּ עַד-לַשָּׁמָיִם וַיִּזְכֹּר אֱלֹהִים אֶת-עֲוֹנוֹתֶיהָ׃(ירמיהו נא' ט') 6 שַׁלְּמוּ-לָהּ גְּמוּלָהּ שֶׁגָּמְלָה לָכֶם וַעֲשׂוּ-לָהּ כִּפְלַיִם כְּפָעֳלָהּ בַּכּוֹס אֲשֶׁר מָסְכָה מִסְכוּ-לָהּ כִּפְלָיִם׃(תהילים קלז' ח'; ירמיהו נ' כט') 7 כַּאֲשֶׁר הִתְרוֹמֲמָה וְהִתְעַנְּגָה כֵּן תְּנוּ-לָהּ חֵבֶל וָאֵבֶל כִּי אָמְרָה בִלְבָבָהּ אֲנִי יָשַׁבְתִּי מַלְכָּה וְלֹא אֶהְיֶה אַלְמָנָה וְאֵבֶל לֹא אֶרְאֶה׃(ישעיהו מז' ח') 8 עַל-כֵּן בְּיוֹם אֶחָד תָּבֹאנָה מַכּוֹתֶיהָ מָוֶת וְאֵבֶל וְרָעָב וְנִשְׂרְפָה בְּמוֹ-אֵשׁ (ישעיהו מז' ט') כִּי חָזָק יְהוָֹה אֱלֹהִים הַשֹּׁפֵט אֹתָהּ׃(ירמיהו נ' לד') 9 וּבָכוּ וְסָפְדוּ עָלֶיהָ מַלְכֵי-אֶרֶץ אֲשֶׁר זָנוּ וְהִתְעַנְּגוּ עִמָּהּ כִּי יִרְאוּ אֶת-עֲשַׁן שְׂרֵפָתָהּ׃ 10 וּמֵרָחוֹק יַעַמְדוּ מִפְּנֵי אֵימַת עִנּוּיָהּ וְאָמְרוּ אוֹי אוֹי לָךְ בָּבֶל הָעִיר הַגְּדוֹלָה הָעִיר הַחֲזָקָה כִּי-בְשָׁעָה אַחַת בָּא מִשְׁפָּטֵךְ׃(יחזקאל כו' טז'-יז') 11 וְסֹחֲרֵי הָאָרֶץ בֹּכִים וּמִתְאַבְּלִים עָלֶיהָ כִּי עַתָּה לֹא-יִקְנֶה עוֹד אִישׁ אֶת-מַשָׂא אֳנִיּוֹתָם׃(יחזקאל כז' לה'-לו') 12 אֶת-מַשָׂא זָהָב וָכֶסֶף וְאֶבֶן יְקָרָה וּפְנִינִים וּבוּץ וְאַרְגָּמָן וּמֶשִׁי וְשָׁנִי וְכָל-עֲצֵי בֹשֶׂם וְכָל-כְּלֵי שֶׁנְהַבִּים וְכָל-כְּלֵי עֵץ יָקָר וּכְלֵי נְחשֶׁת וּבַרְזֶל וָשָׁיִשׁ׃ 13 וְקִנָּמוֹן [וַאֲמוֹמוֹן] וּקְטֹרֶת סַמִּים וּמֹר וּלְבוֹנָה וְיַיִן וָשֶׁמֶן וְסֹלֶת וְחִטִּים וּמִקְנֶה וָצֹאן וְסוּסִים וּמַרְכָּבוֹת וַעֲבָדִים וְנֶפֶשׁ אָדָם׃ 14 וְהַמְּגָדִים מַחְמַד נַפְשֵׁךְ אָזְלוּ מִמֵּךְ וְכָל-שָׁמֵן וּמִַצְהִיר אָבַד מִמֵּךְ וְלֹא תִמְצָא עוֹד׃ 15 וְרֹכְלֵיהֶם אֲשֶׁר הֶעֱשִׁירוּ מִמֶּנָּה יַעַמְדוּ מֵרָחוֹק מִפְּנֵי אֵימַת עִנּוּיָהּ וּבָכוּ וְהִתְאַבָּלוּ׃(***) 16 וְאָמְרוּ אוֹי אוֹי הָעִיר הַגְּדוֹלָה הַמְכֻסָּה שֵׁשׁ וְאַרְגָּמָן וְשָׁנִי וּמֻכְלָלָה זָהָב וְאֶבֶן יְקָרָה וּפְנִינִים כִּי-בְשָׁעָה אַחַת הָחֳרַב הָעֹשֶׁר הַגָּדוֹל הַזֶּה׃ 17 וְכָל-חֹבֵל וְכָל-בַּעַל-מַעֲבֹרֶת וְהַמַּלָּחִים וְכָל-עֹשֵׂי מְלָאכָה בַּיָּם עָמְדוּ מֵרָחוֹק׃ 18 וַיִּצְעֲקוּ בִּרְאֹתָם עֲשַׁן שְׂרֵפָתָהּ וַיֹּאמְרוּ מִי בֶעָרִים כָּעִיר הַגְּדוֹלָה׃ 19 וַיִּזְרְקוּ עָפָר עַל-רָאשֵׁיהֶם (יחזקאל כז' ל') וַיִּצְעֲקוּ בָּכֹה וְסָפוֹד לֵאמֹר אוֹי אוֹי הָעִיר הַגְּדוֹלָה אֲשֶׁר-בָּהּ הֶעֱשִׁירוּ מֵהוֹנָהּ כֹּל אֲשֶׁר-לָהֶם אֳנִיּוֹת בַּיָּם כִּי-בְשָׁעָה אַחַת הָחֳרָבָה׃(יחזקאל כו' יט') 20 רָנּוּ עָלֶיהָ הַשָּׁמַיִם וְהַשְּׁלִיחִים הַקְּדֹשִׁים וְהַנְּבִיאִים כִּי-שָׁפַט אֱלֹהִים אֶת-מִשְׁפַּטְכֶם מִמֶּנָּה׃(ירמיהו נא' מח') 21 וַיִּשָׂא מַלְאָךְ נוֹרָא אֶבֶן גְּדוֹלָה כְּפֶלַח רֶכֶב וַיַּשְׁלִיכֶהָ אֶל-תּוֹךְ הַיָּם וַיֹּאמַר כָּכָה תֻּשְׁלַךְ בְּמַעֲרָצָה בָּבֶל הָעִיר הַגְּדוֹלָה וְלֹא תִּמָּצֵא עוֹד׃(ירמיהו נא' סג'-סד') 22 וְקוֹל הַמְנַגְּנִים בְּכִנּוֹר וְהַמְזַמְּרִים וּמְחַלְּלִים בַּחֲלִילִים וּמַחְצְרִים בַּחֲצֹצְרוֹת בַּל-יִשָּׁמַע עוֹד בְּתוֹכֵךְ וְכָל-חָרָשׁ וְחֹשֵׁב בַּל-יִמָּצֵא בָךְ עוֹד וְקוֹל רֵחַיִם בַּל-יִשָּׁמַע עוֹד בְּקִרְבֵּךְ׃(ישעיהו כד' ח'; יחזקאל כו' יג') 23 וְאוֹר נֵר לֹא-יָאִיר לָךְ עוֹד וְקוֹל חָתָן וְקוֹל כַּלָּה לֹא-יִשָּׁמַע בָּךְ עוֹד (ירמיהו כה' י') כִּי כִנְעָנַיִךְ הָיוּ נִכְבַּדֵּי אֶרֶץ וּבִכְשָׁפַיִךְ נִדְּחוּ כָל-הַגּוֹיִם׃(***) 24 וּבָהּ נִמְצָא דַּם הַנְּבִיאִים וְהַקְּדֹשִׁים וְכָל-הֲרוּגֵי אָרֶץ׃
