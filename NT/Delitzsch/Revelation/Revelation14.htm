פרק יד

השיר החדש של הפדויים

1 וָאֵרֶא וְהִנֵּה-שֶׂה עֹמֵד עַל-הַר צִיּוֹן (תהילים ב' ו') וְעִמּוֹ מְאַת אֶלֶף וְאַרְבָּעִים וְאַרְבָּעָה אֲלָפִים [וּשְׁמוֹ] וְשֵׁם אָבִיו כָּתוּב עַל-מִצְחוֹתָם׃ 2 וָאֶשְׁמַע קוֹל מִן-הַשָּׁמַיִם כְּקוֹל מַיִם רַבִּים (יחזקאל א' כד') וּכְקוֹל רַעַם גָּדוֹל וְהַקּוֹל אֲשֶׁר-שָׁמַעְתִּי כְּקוֹל תֹּפְשֵׂי כִנּוֹר מְנַגְּנִים בְּכִנּוֹרוֹתֵיהֶם׃ 3 וַיָּשִׁירוּ שִׁיר חָדָשׁ לִפְנֵי הַכִּסֵּא וְלִפְנֵי אַרְבַּע הַחַיּוֹת וְלִפְנֵי הַזְּקֵנִים וְאֵין אִישׁ יָכֹל לִלְמֹד אֶת-הַשִּׁיר זוּלָתִי מְאַת הָאֶלֶף וְאַרְבָּעִים וְאַרְבַּעַת הָאֲלָפִים הַנִּקְנִים מִן-הָאָרֶץ׃(תהילים לג' ג'; ישעיהו מב' י') 4 אֵלֶּה הֵם אֲשֶׁר לֹא-נְגֹאֲלוּ בַנָּשִׁים כִּי-כַבְּתוּלוֹת הֵמָּה אֵלֶּה הֵם הַהֹלְכִים אַחֲרֵי הַשֶׂה אֶל-כָּל-אֲשֶׁר יֵלֵךְ אֵלֶּה נִקְנוּ מִתּוֹךְ בְּנֵי הָאָדָם לְרֵאשִׁית בִּכּוּרִים לֵאלֹהִים וְלַשֶׂה׃(ירמיהו ב' ג') 5 וְתַרְמִית לֹא-נִמְצְאָה בְּפִיהֶם כִּי תְּמִימִים הֵמָּה [לִפְנֵי כִּסֵּא הָאֱלֹהִים]׃(צפניה ג' יג'; שיר השירים ד' ז')

הבשורה של שלושת המלאכים

6 וָאֵרֶא מַלְאָךְ אַחֵר מְעוֹפֵף בַּחֲצִי הַשָּׁמָיִם וּבְפִיו בְּשׂוֹרַת עוֹלָם לְבַשֵׂר אֶת-ישְׁבֵי הָאָרֶץ וְאֶת-כָּל-גּוֹי וּמִשְׁפָּחָה וְלָשׁוֹן וָעָם׃ 7 וַיִּקְרָא בְּקוֹל גָּדוֹל יִרְאוּ אֶת-הָאֱלֹהִים וְהָבוּ-לוֹ כָבוֹד (ישעיהו מב' יב'; ירמיהו יג' טז') כִּי בָאָה עֵת מִשְׁפָּטוֹ וְהִשְׁתַּחֲווּ לָעֹשֶׂה שָׁמַיִם וָאָרֶץ אֶת-הַיָּם וּמַעְיְנוֹת הַמָּיִם׃(תהילים קמו' ו'; תהילים צה' ה') 8 וּמַלְאָךְ אַחֵר בָּא אַחֲרָיו וַיֹּאמַר נָפְלָה נָפְלָה בָּבֶל הָעִיר הַגְּדוֹלָה (ישעיהו כא' ט') כִּי הִשְׁקְתָה כָל-הַגּוֹיִם מִיֵּין חֲמַת תַּזְנֻתָהּ׃(ירמיהו נא' ז') 9 וּמַלְאָךְ שְׁלִישִׁי בָּא אַחֲרֵיהֶם וַיִּקְרָא בְּקוֹל גָּדוֹל כָּל-אֲשֶׁר יִשְׁתַּחֲוֶה לִפְנֵי הַחַיָּה וְלִפְנֵי צַלְמָהּ וְנָשָׂא אֶת-תָּוָהּ עַל-מִצְחוֹ אוֹ עַל-יָדוֹ׃ 10 גַּם-הוּא יִשְׁתֶּה מִיֵּין חֲמַת אֱלֹהִים הַנִּמְסָךְ וְלֹא מָהוּל בְּכוֹס זַעְמוֹ וְעֻנָּה בְּאֵשׁ וְגָפְרִית לִפְנֵי הַמַּלְאָכִים הַקְּדֹשִׁים וְלִפְנֵי הַשֶׂה׃(תהילים עה' ט'; ירמיהו כה' טו') 11 וְעָלָה עֲשַׁן עִנּוּיָם לְעוֹלְמֵי-עַד וְהַמִּשְׁתַּחֲוִים לַחַיָּה וּלְצַלְמָהּ וַאֲשֶׁר יִשָׂא אֶת-תָּו שְׁמָהּ לֹא יִמְצְאוּ מְנוּחָה יוֹמָם וָלָיְלָה׃(ישעיהו לד' י') 12 בָּזֹאת סַבְלָנוּתָם שֶׁל-הַקְּדֹשִׁים הַשֹּׁמְרִים אֶת-מִצְוֹת הָאֱלֹהִים וְאֶת-אֱמוּנַת יֵשׁוּעַ׃ 13 וָאֶשְׁמַע קוֹל מִן-הַשָּׁמָיִם מְדַבֵּר אֵלַי כְּתֹב אַשְׁרֵי הַמֵּתִים אֲשֶׁר-יָמוּתוּ בָּאָדוֹן מֵעָתָּה אָמְנָם כֵּן אָמַר הָרוּחַ לְמַעַן יָנוּחוּ מֵעֲמָלָם כִּי מַעֲשֵׂיהֶם הֹלְכִים אַחֲרֵיהֶם׃(ישעיהו נז' ב')

קציר הארץ

14 וָאֵרֶא וְהִנֵּה עָנָן בָּהִיר וְעַל-הֶעָנָן יֹשֵׁב כִּדְמוּת בֶּן-אָדָם (דניאל ז' יג') וַעֲטֶרֶת זָהָב עַל-רֹאשׁוֹ וּבְיָדוֹ מַגָּל מְלֻטָּשׁ׃ 15 וּמַלְאָךְ אַחֵר יָצָא מִן-הָהֵיכָל וַיִּזְעַק בְּקוֹל גָּדוֹל אֶל-הַיּשֵׁב עַל-הֶעָנָן לֵאמֹר שְׁלַח מַגָּלְךָ וּקְצֹר כִּי בָאָה הָעֵת לִקְצֹר כִּי יָבֵשׁ קְצִיר הָאָרֶץ׃ 16 וַיִּשְׁלַח הַיּשֵׁב בֶּעָנָן אֶת-מַגָּלוֹ בָּאָרֶץ וְהָאָרֶץ נִקְצָרָה׃ 17 וּמַלְאָךְ אַחֵר יָצָא מִן-הַהֵיכָל אֲשֶׁר בַּשָּׁמָיִם וְגַם-לוֹ מַגָּל מְלֻטָּשׁ בְּיָדוֹ׃ 18 וַיֵּצֵא מַלְאָךְ אַחֵר מִן-הַמִּזְבֵּחַ וְלוֹ מֶמְשֶׁלֶת הָאֵשׁ וַיִּקְרָא קוֹל גָּדוֹל אֶל-אֲשֶׁר בְּיָדוֹ הַמַּגָּל הַמְלֻטָּשׁ לֵאמֹר שְׁלַח מַגָּלְךָ הַמְלֻטָּשׁ וּבְצֹר אֶת-אַשְׁכְּלֹת גֶּפֶן הָאָרֶץ כִּי-בָשְׁלוּ עִנְּבֵיהֶם׃(עמוס ח' א'-ב') 19 וַיָּנֶף הַמַּלְאָךְ אֶת-מַגָּלוֹ עַל-הָאָרֶץ וַיִּבְצֹר אֶת-אַשְׁכְּלֹת גֶּפֶן הָאָרֶץ וַיַּשְׁלִיכֵם בְּגַת-חֲמַת אֱלֹהִים הַגְּדוֹלָה׃ 20 וַתִּדָּרֵךְ הַגַּת מִחוּץ לָעִיר (ישעיהו סג' ג'; יואל ד' יג') וַיֵּצֵא-דָם מִן-הַגַּת וַיַּעַל עַד רִסְנֵי הַסּוּסִים דֶּרֶךְ-אֶלֶף וְשֵׁשׁ מֵאוֹת רִיס׃
