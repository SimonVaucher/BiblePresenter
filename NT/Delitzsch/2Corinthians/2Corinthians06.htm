לפרק הקודם

אגרת שאול השניה אל הקורינתים
פרק ו

שירות הריצוי -- המשך

1 וַאֲנַחְנוּ כְּעֹזְרִים לוֹ נַזְהִירָה אֶתְכֶם שֶׁלּא תְקַבְּלוּ חֶסֶד אֱלֹהִים וְהָיָה לָרִיק׃ 2 כִּי הוּא אוֹמֵר בְּעֵת רָצוֹן עֲנִיתִיךָ וּבְיוֹם יְשׁוּעָה עֲזַרְתִּיךָ (ישעיהו מט' ח') הִנֵּה עַתָּה עֵת רָצוֹן הִנֵּה עַתָּה יוֹם יְשׁוּעָה׃ 3 וְאֵין אֲנַחְנוּ נֹתְנִים מִכְשׁוֹל בִּמְאוּמָה פֶּן-יִהְיֶה שֵׁרוּתֵנוּ לִנְאָצָה׃ 4 כִּי אִם-נֵרָאֶה בְכָל-דָּבָר כִּמְשָׁרֲתֵי אֱלֹהִים בְּסַבְלָנוּת רַבָּה בְּלַחַץ וּבְצָרוֹת וּבִמְצוּקוֹת׃ 5 בְּמַכּוֹת וּבְמוֹסְרוֹת וּבִמְהוּמוֹת בִּתְלָאוֹת בִּשְׁקִידוֹת וּבְצוֹמוֹת׃ 6 בְּטָהֳרָה וּבְדַעַת וּבְאֹרֶךְ רוּחַ וּבְחֶסֶד וּבְרוּחַ הַקֹּדֶשׁ וּבְאַהֲבָה לֹא צְבוּעָה׃ 7 בִּדְבַר הָאֱמֶת וּבִגְבוּרַת אֱלֹהִים בִּכְלִי נֶשֶׁק הַצְּדָקָה מִיָּמִין וּמִשְּׂמֹאל׃ 8 בְּכָבוֹד וּבְקָלוֹן בְּשֵׁם-רַע וּבְשֵׁם-טוֹב כְּמַתְעִים וְנֶאֱמָנִים׃ 9 כְּאֵינָם יְדוּעִים וְגַם-יְדוּעִים כְּמֵתִים וְהִנְנוּ חַיִּים כִּמְיֻסָּרִים וְלֹא מוּמָתִים׃(תהילים קיח' יח') 10 כְּנֶעֱצָבִים וּבְכָל-עֵת שְׂמֵחִים כְּרָשִׁים וּמַעֲשִׁירֵי רַבִּים כְּמִי שֶׁאֵין-לָהֶם-מְאוּמָה וְיֵשׁ לָהֶם כֹּל׃ 11 פִּינוּ פָּתוּחַ לָכֶם אַנְשֵׁי קוֹרִנְתּוֹס וְרָחַב לִבֵּנוּ׃ 12 לֹא-צַר לָכֶם הַמָּקוֹם בָּנוּ אַךְ-צַר הַמָּקוֹם בִּמְעֵיכֶם׃ 13 וְהָיָה זֶה גְּמוּלִי כְּדַבֵּר אֶל-בָּנִים אָנֹכִי מְדַבֵּר אִם-תַּרְחִיבוּ לְבַבְכֶם גַּם-אַתֶּם׃

היכל אלהים חיים

14 אַל-תִּהְיוּ משְׁכִים בְּעֹל עִם-חַסְרֵי אֱמוּנָה (דברים ז' ב'-ג'; כב' י') כִּי אֵי-זֹה חֶבְרָה לַצְּדָקָה עִם-הָעַוְלָה וּמַה-לָּאוֹר אֶת-הַחשֶׁךְ׃(בראשית א' ד') 15 אֵי-זֹה הַסְכָּמָה לַמָּשִׁיחַ עִם-בְּלִיַּעַל אוֹ מֶה חֵלֶק הַמַּאֲמִין עִם-שֶׁאֵינֶנּוּ מַאֲמִין׃ 16 וְאֵי-זֶה קֶשֶׁר יֵשׁ לְהֵיכַל אֱלֹהִים עִם-הָאֱלִילִים כִּי-אַתֶּם הֵיכַל אֱלֹהִים חַיִּים כְּמוֹ שֶׁאָמַר הָאֱלֹהִים וְשָׁכַנְתִּי וְהִתְהַלַּכְתִּי בְּתוֹכָם וְהָיִיתִי לָהֶם לֵאלֹהִים וְהֵם יִהְיוּ-לִי לָעָם׃(ויקרא כו' יא'-יב'; יחזקאל לז' כז') 17 עַל-כֵּן צְאוּ מִתּוֹכָם וְהִבָּרוּ נְאֻם-יְהוָֹה וְטָמֵא אַל-תִּגָּעוּ (ישעיהו נב' יא'-יב') וַאֲנִי אֲקַבֵּץ אֶתְכֶם׃ 18 וְהָיִיתִי לָכֶם לְאָב וְאַתֶּם תִּהְיוּ-לִי לְבָנִים וּלְבָנוֹת נְאֻם-יְהוָֹה צְבָאוֹת׃(שמואל ב', ז' יד'; ירמיהו לא' ח')
