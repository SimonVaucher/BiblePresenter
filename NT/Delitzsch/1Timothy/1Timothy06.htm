לפרק הקודם

אגרת שאול הראשונה אל טימותיאוס
פרק ו

חובות כלפי המאמינים -- המשך

1 כֹּל אֲשֶׁר-עֹל הָעַבְדוּת עֲלֵיהֶם יִהְיוּ אֲדֹנֵיהֶם בְּעֵינֵיהֶם רְאוּיִם לְכָל-כָּבוֹד לְמַעַן אֲשֶׁר לֹא-יְחֻלַּל שֵׁם הָאֱלֹהִים וְהַלֶּקַח׃ 2 וְאֹתָם שֶׁאֲדֹנֵיהֶם מַאֲמִינִים אַל-יֵקַלּוּ בְעֵינֵיהֶם מִפְּנֵי שֶׁהֵם אַחִים כִּי אִם-יַעַבְדוּ אֹתָם יוֹתֵר מִפְּנֵי שֶׁמַּאֲמִינִים וַאֲהוּבִים הֵם הַמְקַבְּלִים אֶת-הַטּוֹבָה אֶת-זֹאת תְּלַמֵּד וּתְצַוֶּה׃

חיי חסידות

3 הָאִישׁ אֲשֶׁר יוֹרֶה תּוֹרָה אַחֶרֶת וְלֹא יַעֲמֹד בְּדִבְרֵי אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ הַבְּרִיאִים וּבְלֶקַח הַחֲסִידוּת׃ 4 נַפְשׁוֹ עֻפְּלָה וְלֹא יָדַע מְאוּמָה כִּי אִם-חֹלֶה הוּא בִּשְׁאֵלוֹת וְתוֹכְחוֹת מִלִּים הַמּוֹלִידוֹת קִנְאָה וּמְרִיבָה וְגִדּוּפִים וַחֲשָׁד רָע׃ 5 וִכּוּחֵי הֶבֶל מֵאֲנָשִׁים נִשְׁחֲתֵי דַעַת וּמְחֻסְּרֵי אֱמֶת הַשָׂמִים אֶת-הַחֲסִידוּת לִדְבַר בָּצַע [סוּר מֵאֲנָשִׁים כָּאֵלֶּה]׃ 6 וְאָמְנָם בֶּצַע גָּדוֹל הִיא הַחֲסִידוּת עִם-לֵב שָׂמֵחַ בְּחֶלְקוֹ׃ 7 כִּי בָאנוּ לָעוֹלָם וְאֵין בְּיָדֵינוּ מְאוּמָה וְיָדוּעַ שֶׁאַף נֵצֵא מִמֶּנּוּ וְאֵין בְּיָדֵינוּ מְאוּמָה׃(איוב א' כא'; תהילים מט' יח') 8 וְעַל-כֵּן אִם מָזוֹן וּכְסוּת לָנוּ נִסְתַּפְּקָה בָּהֶם׃ 9 אֲבָל הַמְבַקְשִׁים לְהַעֲשִׁיר יָבֹאוּ לִידֵי נִסָּיוֹן וְיִפְּלוּ בְמוֹקְשִׁים וּבְרֹב תַּאֲוֹת סְכָלוֹת וּמַשְׁחִיתוֹת הַמַּשְׁקִיעוֹת אֶת-הָאָדָם בַּשַּׁחַת וּבָאֲבַדּוֹן׃(משלי כג' ד') 10 כִּי שֹׁרֶשׁ כָּל-הָרָעוֹת אַהֲבַת הַכָּסֶף וְיֵשׁ לְהוּטִים אַחֲרָיו אֲשֶׁר סָרוּ מִן-הָאֱמוּנָה וַיַּעֲצִיבוּ אֶת-נַפְשָׁם בְּמַכְאֹבִים רַבִּים׃

המאבק הטוב

11 וְאַתָּה אִישׁ הָאֱלֹהִים בְּרַח-לְךָ מֵאֵלֶּה וּרְדֹף צְדָקָה וַחֲסִידוּת וֶאֱמוּנָה וְאַהֲבָה וְסַבְלָנוּת וַעֲנָוָה׃ 12 הִלָּחֵם הַמִּלְחָמָה הַטּוֹבָה מִלְחֶמֶת הָאֱמוּנָה וֶאֱחֹז אֶת-חַיֵּי הָעוֹלָמִים אֲשֶׁר נִקְרֵאתָ לָהֶם וְהוֹדֵיתָ הוֹדָאָה יָפָה בִּפְנֵי עֵדִים רַבִּים׃ 13 הִנְנִי מְצַוְּךָ נֶגֶד הָאֱלֹהִים הַמְחַיֶּה אֶת-כֹּל וְנֶגֶד הַמָּשִׁיחַ יֵשׁוּעַ אֲשֶׁר הֵעִיד הַהוֹדָאָה הַיָּפָה לִפְנֵי פָּנְטִיּוֹס פִּילָטוֹס׃ 14 אֲשֶׁר תִּשְׁמֹר אֶת-הַמִּצְוָה בְּאֵין-שֶׁמֶץ וָדֹפִי עַד-הוֹפָעַת הַמָּשִׁיחַ יֵשׁוּעַ אֲדֹנֵינוּ׃ 15 אֲשֶׁר יַרְאֶנָּה בְעִתָּהּ הַמְבֹרָךְ וְהַשַּׁלִּיט לְבַדּוֹ מֶלֶךְ הַמְּלָכִים וַאֲדֹנֵי הָאֲדֹנִים׃(דברים י' יז'; דניאל ב' מז') 16 אֲשֶׁר הוּא לְבַדּוֹ חַי וְקַיָּם וְהוּא שֹׁכֵן אוֹר נִשְׂגָּב (תהילים צ' ב'; קד' ב') וְאִישׁ לֹא רָאָהוּ וְלֹא יוּכַל לִרְאוֹתוֹ וְלוֹ הַכָּבוֹד וּגְבוּרַת עוֹלָמִים אָמֵן׃(שמות לג' כ') 17 אֶת-עֲשִׁירֵי הָעוֹלָם הַזֶּה תְּצַוֶּה שֶׁלּא יָרוּם לְבָבָם וְלֹא-יִבְטְחוּ בָּעֹשֶׂר הַבּוֹגֵד כִּי אִם-בֵּאלֹהִים חַיִּים הַמַּסְפִּיק לָנוּ דַּי וְהוֹתֵר לִשְׂבֹּעַ׃(תהילים קמו' ה'; סב' יא') 18 וְיִגְמְלוּ טוֹב וְיַעְשִׁירוּ בְּמַעֲשִׂים טוֹבִים לָתֵת מֵהוֹנָם וְלַעֲזֹר לְזוּלָתָם׃(תהילים קיב' ט') 19 וְיַאַצְרוּ לָהֶם אוֹצָר לִיסוֹד טוֹב לֶעָתִיד לָבֹא לְמַעַן יַשִׂיגוּ אֶת-חַיֵּי [הָאֱמֶת]׃(משלי יט' יז') 20 אַתָּה טִימוֹתִיּוֹס שְׁמֹר אֶת-הַפִּקָּדוֹן וְהִתְרַחֵק מִדִּבְרֵי הֶבֶל הַפְּסוּלִים וּמִן-וִכּוּחֵי הַמַּדָּע הַנִּקְרָא כֵן בְּטָעוּת׃ 21 אֲשֶׁר יֵשׁ מִתְפָּאֲרִים בּוֹ וַיִּתְעוּ מִן-הָאֱמוּנָה. הַחֶסֶד עִמָּךְ אָמֵן׃
