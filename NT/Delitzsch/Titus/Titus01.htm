לאגרת שאול השניה אל טימותיאוס

אגרת שאול אל טיטוס
פרק א

1 פּוֹלוֹס עֶבֶד אֱלֹהִים וּשְׁלִיחַ יֵשׁוּעַ הַמָּשִׁיחַ לְפִי אֱמוּנַת בְּחִירֵי אֱלֹהִים וְדַעַת הָאֱמֶת אֲשֶׁר לַחֲסִידוּת׃ 2 עֲלֵי-תִקְוַת חַיֵּי עוֹלָם אֲשֶׁר לִפְנֵי יְמוֹת עוֹלָם הִבְטִיחַ הָאֵל אֲשֶׁר לֹא יְכַזֵּב׃(במדבר כג' יט') 3 וְגִלָּה אֶת-דְּבָרוֹ בְּמוֹעֲדוֹ עַל-יְדֵי הַקְּרִיאָה הַמֻּפְקָדָה בְּיָדִי בְּמִצְוַת הָאֱלֹהִים מוֹשִׁיעֵנוּ׃ 4 אֶל-טִיטוֹס בְּנִי הָאֲמִתִּי בֶּאֱמוּנָה אֶחָת חֶסֶד וְרַחֲמִים וְשָׁלוֹם מֵאֵת הָאֱלֹהִים אָבִינוּ וַאֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ מוֹשִׁיעֵנוּ׃

פעולתו של טיטוס בכריתים

5 בַּעֲבוּר זֹאת הִנַּחְתִּיךָ בִקְרֵיטִי בַּעֲבוּר אֲשֶׁר תַּשְׁלִים אֶת-הֶחָסֵר וּתְמַנֶּה זְקֵנִים בְּכָל-עִיר וָעִיר כַּאֲשֶׁר צִוִּיתִיךָ׃ 6 אִם-יִמָּצֵא אִישׁ תָּם וּבַעַל-אִשָּׁה אֶחָת וְיֶשׁ-לוֹ בָּנִים מַאֲמִינִים וְאֵין עֲלֵיהֶם טַעֲנַת פְּרִיצוּת וְאֵינָם סוֹרֲרִים׃ 7 כִּי-הֶגְמוֹן הָעֵדָה צָרִיךְ לִהְיוֹת אִישׁ תָּם כְּסֹכֵן לֵאלֹהִים לֹא עֹמֵד עַל-דַּעְתּוֹ וְלֹא רַגְזָן וְלֹא אֹהֵב יַיִן וְלֹא-בַעַל אֶגְרֹף וְלֹא בֹצֵעַ בֶּצַע רָע׃ 8 כִּי אִם-יְהִי מַכְנִיס אֹרְחִים וְאֹהֵב טוֹב וְצָנוּעַ וְצַדִּיק וְקָדוֹשׁ וְכֹבֵשׁ אֶת-יִצְרוֹ׃ 9 וּמַחֲזִיק בַּדָּבָר הַנֶּאֱמָן כְּפִי הַהוֹרָאָה לְמַעַן יִהְיֶה-בּוֹ כֹחַ לְהַזְהִיר בַּלֶּקַח הַבָּרִיא וּלְהוֹכִיחַ אֶת-הַמְּרִיבִים׃ 10 כִּי-יֵשׁ הַרְבֵּה מֹרְדִים דֹּבְרֵי הֶבֶל וּמַתְעֵי נָפֶשׁ וְרֻבָּם מִן-הַנִּמּוֹלִים׃ 11 אֲשֶׁר סָכוֹר יִסָּכֵר פִּיהֶם הַהֹפְכִים בָּתִּים כֻּלָּם בְּהוֹרוֹתָם דְּבָרִים אֲשֶׁר לֹא-כֵן עֵקֶב בָּצַע׃(ירמיהו ח' י') 12 וּכְבָר אָמַר אֶחָד מֵהֶם נְבִיאָם מִקִּרְבָּם בְּנֵי קְרֵיטִי מְשַׁקְּרִים מֵעוֹלָם וְחַיּוֹת רָעוֹת הֵמָּה וְגַם כְּרֵשִׂים עֲצֵלִים׃ 13 וְהָעֵדוּת הַזֹּאת אֱמֶת וּבַעֲבוּר-כֵּן תּוֹכִיחֵם תּוֹכֵחָה קָשָׁה לְמַעַן יִהְיוּ בְרִיאִים בָּאֱמוּנָה׃ 14 וְלֹא יָשִׂימוּ לֵב אֶל-הַגָּדוֹת הַיְּהוּדִים וְאֶל-מִצְוֹת הָאֲנָשִׁים הַסָּרִים מִן-הָאֱמֶת׃ 15 הַכֹּל טָהוֹר לַטְּהוֹרִים אֲבָל לַנִּטְמְאִים וּלְחַסְרֵי אֱמוּנָה אֵין-דָּבָר טָהוֹר כִּי נִטְמְאָה דַּעְתָּם וְרוּחָם׃ 16 אֹמְרִים הֵמָּה יָדַעְנוּ אֶת-הָאֱלֹהִים וּבְמַעֲשֵׂיהֶם כּוֹפְרִים בּוֹ כִּי מְתֹעָבִים וּמַמְרִים הֵם וְלֹא-יִצְלְחוּ לְכָל-מַעֲשֶׂה טוֹב׃(ירמיהו ד' כב')
