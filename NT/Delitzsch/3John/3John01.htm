לאגרת יוחנן השניה

אגרת יוחנן השלישית
פרק א

1 הַזָּקֵן אֶל-גָּיוֹס יְדִידִי אֲשֶׁר אֲנִי אֹהֵב אֹתוֹ בֶּאֱמֶת׃ 2 יְדִידִי רְצוֹנִי שֶׁיִּיטַב לְךָ לְכָל-דָּבָר וְתִהְיֶה בָרִיא כְּטוֹב לְנַפְשֶׁךָ׃ 3 כִּי שָׂמַחְתִּי מְאֹד כַּאֲשֶׁר בָּאוּ אַחִים וַיָּעִידוּ עַל-אֲמִתֶּךָ אֲשֶׁר מִתְהַלֵּךְ אַתָּה בָּאֱמֶת׃ 4 אֵין לִי שִׂמְחָה גְדוֹלָה מִלִּשְׁמֹעַ אֶת-בָּנַי מִתְהַלְּכִים בָּאֱמֶת׃ 5 יְדִידִי בֶּאֱמוּנָה כָּל-מַעֲשֶׂיךָ עִם-הָאַחִים וְעִם-הָאֹרְחִים׃ 6 אֲשֶׁר הֵעִידוּ עַל-אַהֲבָתְךָ בִּפְנֵי הַקָּהָל וְיָפֶה תַעֲשֶׂה לְשַׁלַּח אֹתָם כָּרָאוּי לִפְנֵי אֱלֹהִים׃ 7 כִּי לְמַעַן שְׁמוֹ יָצָאוּ וְלֹא לָקְחוּ דָבָר מִן-הַגּוֹיִם׃ 8 עַל-כֵּן עָלֵינוּ לְקַבֵּל אֹתָם לְמַעַן נִהְיֶה עֹזְרִים לָאֱמֶת׃ 9 אֲנִי כָתַבְתִּי אֶל-הַקְּהִלָּה אֲבָל דִּיּוֹטְרִיפַס הַמִּתְאַוֶּה לִהְיוֹת לָהֶם לְרֹאשׁ אֵינֶנּוּ מְקַבֵּל אֹתָנוּ׃ 10 עַל-כֵּן בְּבֹאִי אַזְכִּיר אֶת-מַעֲשָׂיו אֲשֶׁר הוּא עֹשֶׂה לְבַטֵּא עָלֵינוּ דְּבָרִים רָעִים וְלֹא דַּי-לוֹ בָזֶה כִּי גַם אֶת-הָאַחִים לֹא יְקַבֵּל וְיִמְנַע אֶת-הַחֲפֵצִים לְקַבְּלָם וַיְגָרֲשֵׁם מִתּוֹךְ הַקָּהָל׃ 11 יְדִידִי אַל-תֵּלֵךְ בְּעִקְּבוֹת הָרָעָה כִּי אִם-בְּעִקְּבוֹת הַטּוֹב הָעֹשֶׂה טוֹב הוּא מֵאֱלֹהִים וְהָעֹשֶׂה רַע לֹא רָאָה אֶת-הָאֱלֹהִים׃(תהילים לד' טו') 12 עַל-דְּמַטְרִיּוֹס הֵעִידוּ הַכֹּל וְגַם-הָאֱמֶת עַצְמָהּ וְגַם-אֲנַחְנוּ מְעִידִים עָלָיו וִידַעְתֶּם כִּי עֵדוּתֵנוּ נֶאֱמָנָה׃ 13 הַרְבֵּה יֶשׁ-לִי לִכְתֹּב וְלֹא חָפַצְתִּי לִכְתֹּב אֵלֶיךָ בִּדְיוֹ וּבְקוֹלָמוֹס׃ 14 אֲבָל אֲקַוֶּה לִרְאוֹתְךָ בִּמְהֵרָה וּפֶה אֶל-פֶּה נְדַבֵּר׃ 15 שָׁלוֹם לָךְ הָרֵעִים שֹׁאֲלִים לִשְׁלוֹמֶךָ שְׁאַל לִשְׁלוֹם הָרֵעִים לְאִישׁ אִישׁ בִּשְׁמוֹ׃(במדבר ו' כו')
