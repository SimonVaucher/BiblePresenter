פרק ד

ישוע עומד בניסיון

(מתי ד' 11-1;מרקוס א' 13-12)

1 וְיֵשׁוּעַ שָׁב מִן-הַיַּרְדֵּן וְהוּא מָלֵא רוּחַ הַקֹּדֶשׁ וַיִּשָׂאֵהוּ הָרוּחַ הַמִּדְבָּרָה׃(דברים ט' ט') 2 וַיְנַסֵּהוּ הַשָׂטָן אַרְבָּעִים יוֹם וְלֹא אָכַל מְאוּמָה בַּיָּמִים הָהֵם וְאַחֲרֵי אֲשֶׁר-תַּמּוּ וַיִּרְעָב׃ 3 וַיֹּאמֶר אֵלָיו הַשָׂטָן אִם בֶּן-הָאֱלֹהִים אַתָּה דַּבֵּר אֶל-הָאֶבֶן הַזֹּאת וּתְהִי לְלָחֶם׃ 4 וַיַּעַן אֹתוֹ יֵשׁוּעַ הֵן כָּתוּב כִּי לֹא עַל-הַלֶּחֶם לְבַדּוֹ יִחְיֶה הָאָדָם [כִּי עַל-כָּל-מוֹצָא פִי-יְהוָֹה]׃(דברים ח' ג') 5 וַיַּעֲלֵהוּ הַשָׂטָן עַל-הַר גָּבֹהַּ וַיַּרְאֵהוּ בְּרֶגַע אֶחָד אֵת כָּל-מַמְלְכוֹת תֵּבֵל׃ 6 וַיֹּאמֶר אֵלָיו הַשָׂטָן לְךָ אֶתֵּן אֶת-כָּל-הַמֶּמְשָׁלָה הַזֹּאת וְאֶת-כְּבוֹדָן כִּי-נִמְסְרָה בְיָדִי וּנְתַתִּיהָ לַאֲשֶׁר אֶחְפָּץ׃ 7 וְעַתָּה אִם-תִּשְׁתַּחֲוֶה לְפָנָי הַכֹּל יִהְיֶה-לָּךְ׃ 8 וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר אֵלָיו [סוּר מִמֶּנִּי הַשָׂטָן כִּי] כָתוּב לַיְהוָֹה אֱלֹהֶיךָ תִּשְׁתַּחֲוֶה וְאֹתוֹ לְבַדּוֹ תַּעֲבֹד׃(דברים ו' יג'; י' יב') 9 וַיְבִיאֵהוּ יְרוּשָׁלַיִם וַיַּעֲמִידֵהוּ עַל-פִּנַּת גַּג בֵּית-הַמִּקְדָּשׁ וַיֹּאמֶר אֵלָיו אִם בֶּן-הָאֱלֹהִים אַתָּה נְפֹל מִזֶּה מָטָּה׃ 10 כִּי כָתוּב כִּי-מַלְאָכָיו יְצַוֶּה-לָּךְ לְשָׁמְרֶךָ׃ 11 וְכִי עַל-כַּפַּיִם יִשָׂאוּנְךָ פֶּן-תִּגֹּף בָּאֶבֶן רַגְלֶךָ׃(תהילים צא' יא'-יב') 12 וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר אֵלָיו הֵן נֶאֱמַר לֹא תְנַסֶּה אֵת יְהוָֹה אֱלֹהֶיךָ׃(דברים ו' טז') 13 וּכְכַלּוֹת הַשָׂטָן כָּל-מַסָּה וַיִּרֶף מִמֶּנּוּ עַד-עֵת׃ 14 וַיָּשָׁב יֵשׁוּעַ בִּגְבוּרַת הָרוּחַ אֶל-הַגָּלִיל וַיֵּצֵא שָׁמְעוֹ בְּכָל-הַכִּכָּר׃ 15 וְהוּא מְלַמֵּד בִּכְנֵסִיּוֹתֵיהֶם וַיְהַלֲלֻהוּ כֻּלָּם׃

אין נביא בעירו

(מתי יג' 58-53;מרקוס ו' 6-1)

16 וַיָּבֹא אֶל-נְצֶרֶת אֲשֶׁר גֻּדַּל-שָׁם וַיֵּלֶךְ כְּמִשְׁפָּטוֹ בְּיוֹם הַשַּׁבָּת אֶל-בֵּית הַכְּנֵסֶת וַיָּקָם לִקְרֹא בַסֵּפֶר׃ 17 וַיֻּתַּן-לוֹ סֵפֶר יְשַׁעֲיָה הַנָּבִיא וַיִּפְתַּח אֶת-הַסֵּפֶר וַיִּמְצָא אֶת-הַמָּקוֹם אֲשֶׁר הָיָה-כָתוּב בּוֹ׃ 18 רוּחַ אֲדֹנָי יֱהוֹה עָלָי יַעַן מָשַׁח אֹתִי לְבַשֵׂר עֲנָוִים׃ 19 שְׁלָחַנִי לַחֲבשׁ לְנִשְׁבְּרֵי-לֵב לִקְרֹא לִשְׁבוּיִם דְּרוֹר וּלְעִוְרִים פְּקַח-קוֹחַ לְשַׁלַּח רְצוּצִים חָפְשִׁים לִקְרֹא שְׁנַת-רָצוֹן לַיְהוָֹה׃(ישעיהו סא' א'-ב') 20 וַיְהִי כַּאֲשֶׁר גָּלַל אֶת-הַסֵּפֶר וַיְשִׁיבֵהוּ אֶל-הַחַזָּן וַיֵּשֶׁב וְעֵינֵי כָּל-אֲשֶׁר בְּבֵית הַכְּנֵסֶת נְשֻׂאוֹת אֵלָיו׃ 21 וַיָּחֶל וַיֹּאמֶר אֲלֵיהֶם הַיּוֹם נִתְמַלֵּא הַכָּתוּב הַזֶּה בְּאָזְנֵיכֶם׃ 22 וְכֻלָּם הֱעִידֻהוּ וְתָמְהוּ עַל-דִּבְרֵי חֵן הַיֹּצְאִים מִפִּיהוּ וַיֹּאמְרוּ הֲלֹא-זֶה הוּא בֶּן-יוֹסֵף׃ 23 וַיֹּאמֶר אֲלֵיהֶם הֵן תֹּאמְרוּ לִי אֶת-הַמָּשָׁל הַזֶּה רֹפֵא רְפָא אֶת-עַצְמֶךָ וּכְכֹל אֲשֶׁר שָׁמַעְנוּ שֶׁנַּעֲשָׂה בִּכְפַר-נַחוּם עֲשֵׂה-כֵן גַּם-פֹּה בְּעִיר מוֹלַדְתֶּךָ׃ 24 וַיֹּאמַר אָמֵן אֹמֵר אֲנִי לָכֶם אֵין-נָבִיא רָצוּי בְּאֶרֶץ מוֹלַדְתּוֹ׃ 25 וֶאֱמֶת אַגִּיד לָכֶם אַלְמָנוֹת רַבּוֹת הָיוּ בְיִשְׂרָאֵל בִּימֵי אֵלִיָּהוּ בְּהֵעָצֵר הַשָּׁמַיִם שָׁלֹשׂ שָׁנִים וְשִׁשָּׁה חֳדָשִׁים וַיְהִי רָעָב גָּדוֹל בְּכָל-הָאָרֶץ׃ 26 וְאֵלִיָּהוּ לֹא-נִשְׁלַח אֶל-אַחַת מֵהֵנָּה זוּלָתִי צָרְפַתָה אֲשֶׁר לְצִידוֹן אֶל-אִשָּׁה אַלְמָנָה׃(מלכים א', יז' א'; יז' ט') 27 וּמְצֹרָעִים רַבִּים הָיוּ בְיִשְׂרָאֵל בִּימֵי אֱלִישָׁע הַנָּבִיא וְלֹא טֹהַר אֶחָד מֵהֶם זוּלָתִי נַעֲמָן הָאֲרַמִּי׃(מלכים ב', ה' יד') 28 וְכֹל אֲשֶׁר בְּבֵית הַכְּנֵסֶת כְּשָׁמְעָם אֶת-הַדְּבָרִים הָאֵלֶּה וַיִּמָּלְאוּ חֵמָה׃ 29 וַיָּקוּמוּ וַיַּדִּיחוּ אוֹתוֹ אֶל-מִחוּץ לָעִיר וַיְבִיאֻהוּ עַד-גַּב הָהָר אֲשֶׁר נִבְנְתָה עִירָם עָלָיו לְמַעַן הַשְׁלִיכוֹ מָטָּה׃ 30 אַךְ-הוּא עָבַר בְּתוֹכָם וַיֵּלֶךְ לְדַרְכּוֹ׃

גירוש רוח טמאה

(מרקוס א' 28-21)

31 וַיֵּרֶד אֶל-כְּפַר-נַחוּם עִיר הַגָּלִיל וַיְלַמְּדֵם בַּשַּׁבָּתוֹת׃ 32 וַיִּשְׁתּוֹמֲמוּ עַל-תּוֹרָתוֹ כִּי דְּבָר שִׁלְטוֹן דְּבָרוֹ׃(ירמיהו כג' כט') 33 וּבְבֵית הַכְּנֵסֶת הָיָה אִישׁ וּבוֹ רוּחַ שֵׁד טָמֵא וַיִּזְעַק בְּקוֹל גָּדוֹל לֵאמֹר׃ 34 אֲהָהּ מַה-לָּנוּ וָלָךְ יֵשׁוּעַ הַנָּצְרִי כִּי-בָאתָ לְהַאֲבִידֵנוּ יְדַעְתִּיךָ מִי אַתָּה קְדוֹשׁ הָאֱלֹהִים׃ 35 וַיִּגְעַר-בּוֹ יֵשׁוּעַ לֵאמֹר הֵאָלֵם וְצֵא מִמֶּנּוּ וַיַּפִּילֵהוּ הַשֵּׁד בְּתוֹכָם וַיֵּצֵא מִמֶּנּוּ לֹא הֵרַע לוֹ׃ 36 וַתִּפֹּל אֵימָה עַל-כֻּלָּם וַיְדַבְּרוּ אִישׁ אֶל-רֵעֵהוּ לֵאמֹר מָה הַדָּבָר הַזֶּה כִּי-בְשָׁלְטָן וּבִגְבוּרָה מְצַוֶּה לְרוּחוֹת הַטֻּמְאָה וְהֵמָּה יֹצְאִים׃ 37 וְשָׁמְעוֹ הוֹלֵךְ בְּכָל-מְקֹמוֹת הַכִּכָּר׃

ריפוי חולים רבים

(מתי ח' 17-14;מרקוס א' 34-29)

38 וַיָּקָם מִבֵּית הַכְּנֵסֶת וַיָּבֹא בֵּיתָה שִׁמְעוֹן וְחֹתֶנֶת שִׁמְעוֹן אֲחָזַתָּה קַדַּחַת רָעָה וַיִּפְגְּעוּ בוֹ בַּעֲדָהּ׃ 39 וַיִּתְיַצֵּב עָלֶיהָ וַיִּגְעַר בַּקַּדַּחַת וַתִּרֶף מִמֶּנָּה וַתָּקָם מְהֵרָה וַתְּשָׁרֵת אֹתָם׃ 40 וְהָיָה כֹּל אֲשֶׁר לָהֶם חֹלִים חֳלָיִים שׁוֹנִים וַיְבִיאוּם אֵלָיו כְּבוֹא הַשָּׁמֶשׁ וַיָּשֶׂם אֶת-יָדָיו עַל-כָּל-אֶחָד מֵהֶם וַיְרַפֵּא אוֹתָם׃ 41 וְגַם-שֵׁדִים יָצְאוּ מֵרַבִּים וְהֵם קֹרְאִים וְאֹמְרִים אַתָּה הוּא [הַמָּשִׁיחַ] בֶּן-הָאֱלֹהִים וַיִּגְעַר-בָּם וְלֹא נְתָנָם לְדַבֵּר כִּי יָדְעוּ אֲשֶׁר הוּא הַמָּשִׁיחַ׃

ישוע יוצא למסע הטפה

(מרקוס א' 39-35)

42 וּכְאוֹר הַבֹּקֶר יָצָא וַיֵּלֶךְ-לוֹ אֶל-מְקוֹם שׁוֹמֵם וַהֲמוֹן הָעָם בִּקְשֻׁהוּ וַיָּבֹאוּ עָדָיו וַיִּפְצְרוּ-בוֹ לְבִלְתִּי סוּר מֵהֶם׃ 43 וַיֹּאמֶר לָהֶם הֵן עָלַי לְבַשֵׂר גַּם-לֶעָרִים הָאֲחֵרוֹת אֶת-בְּשׂוֹרַת מַלְכוּת הָאֱלֹהִים כִּי לָזֹאת שֻׁלָּחְתִּי׃ 44 וַיְהִי קוֹרֵא בְּבָתֵּי כְנֵסִיּוֹת שֶׁבַּגָּלִיל׃
