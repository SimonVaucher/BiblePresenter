פרק יח

המשל על האלמנה והשופט

1 וְגַם-מָשָׁל דִּבֶּר אֲלֵיהֶם לְהִתְפַּלֵּל תָּמִיד וְלֹא לְהִתְרַפּוֹת׃ 2 וַיֹּאמַר שׁוֹפֵט הָיָה בְּעִיר אֶחָת אֲשֶׁר לֹא יָרֵא אֶת-הָאֱלֹהִים וְלֹא-נָשָׂא פְּנֵי אָדָם׃ 3 וְאַלְמָנָה הָיְתָה בָּעִיר הַהִיא וַתָּבֹא אֵלָיו לֵאמֹר דִּינָה אֶת-דִּינִי מִמְּרִיבִי׃(ישעיהו א' יז') 4 וַיְמָאֵן יוֹם וָיוֹם וּמִקֵּץ יָמִים אָמַר בְּנַפְשׁוֹ גַּם-כִּי לֹא אִירָא אֶת-הָאֱלֹהִים וְלֹא-אֶשָׂא פָנִים לְאָדָם׃(ישעיהו א' כג') 5 אֶעֱשֶׂה אֶת-מִשְׁפַּט הָאַלְמָנָה הַזֹּאת עַל-הַלְאוֹתָהּ אֹתִי פֶּן-תָּבוֹא תָמִיד וְדִכְּאַתְנִי בְּמִלִּים׃ 6 וַיֹּאמֶר הָאָדוֹן שִׁמְעוּ אֶת-אֲשֶׁר אֹמֵר דַּיַּן הָעַוְלָה׃ 7 וְהָאֱלֹהִים הַהוּא לֹא יַעֲשֶׂה מִשְׁפַּט בְּחִירָיו הַקֹּרְאִים אֵלָיו יוֹמָם וָלָיְלָה וַאֲלֵיהֶם יִתְאַפָּק׃(תהילים נה' יח'; פח' ב') 8 אֲנִי אֹמֵר לָכֶם כִּי-עָשׂוֹ יַעֲשֶׂה אֶת-מִשְׁפָּטָם בִּמְהֵרָה אַךְ בֶּן-הָאָדָם בְּבֹאוֹ הֲיִמְצָא אֱמוּנָה בָּאָרֶץ׃

המשל על הפרוש והמוכס

9 וַיּוֹסֶף וַיִּשָׂא מְשָׁלוֹ אֶל-אֲנָשִׁים בֹּטְחִים בְּנַפְשָׁם כִּי צַדִּיקִים הֵמָּה וַאֲחֵרִים נִבְזִים בְּעֵינֵיהֶם וַיֹּאמַר׃ 10 שְׁנֵי אֲנָשִׁים עָלוּ אֶל-הַמִּקְדָּשׁ לְהִתְפַּלֵּל אֶחָד פָּרוּשׁ וְאֶחָד מוֹכֵס׃ 11 וַיַּעֲמֹד הַפָּרוּשׁ לְבַדּוֹ וַיִּתְפַּלֵּל לֵאמֹר אוֹדְךָ אֱלֹהִים עַל כִּי אֵינֶנִּי כְּיֶתֶר הָאָדָם הַגֹּזְלִים וְהָעשְׁקִים וְהַנֹּאֲפִים וְגַם-לֹא כַּמֹּכֵס הַזֶּה׃ 12 אֲנִי צָם פַּעֲמַיִם בַּשָּׁבוּעַ אֲנִי מְעַשֵׂר אֵת כָּל-אֲשֶׁר אֲנִי קֹנֶה׃ 13 וְהַמּוֹכֵס עָמַד מֵרָחוֹק וְלֹא אָבָה לָשֵׂאת אֶת-עֵינָיו הַשָּׁמַיְמָה וְתוֹפֵף עַל-לִבּוֹ וַיֹּאמַר אֱלֹהִים סְלַח-לִי אֲנִי הַחוֹטֵא׃(עזרא ט' ו'; תהילים נא' ג') 14 אֲנִי אֹמֵר לָכֶם כִּי-יָרַד זֶה לְבֵיתוֹ נִצְדָּק מִזֶּה כִּי כָּל-הַמֵּרִים נַפְשׁוֹ יִשָּׁפֵל וַאֲשֶׁר יַשְׁפִּילָהּ יְרוֹמָם׃(איוב כב' כט')

ישוע מקבל את הילדים הקטנים

(מתי יט' 15-13;מרקוס י' 16-13)

15 וַיָּבִיאוּ אֵלָיו גַּם אֶת-הַיְלָדִים לְמַעַן יִגַּע בָּהֶם וַיִּרְאוּ הַתַּלְמִידִים וַיִּגְעֲרוּ בָם׃ 16 וַיִּקְרָא אֹתָם יֵשׁוּעַ אֵלָיו וַיֹּאמַר הַנִּיחוּ לַיְלָדִים לָבוֹא אֵלָי וְאַל-תִּמְנָעוּם כִּי לְאֵלֶּה מַלְכוּת הָאֱלֹהִים׃ 17 אָמֵן אֹמֵר אֲנִי לָכֶם כֹּל אֲשֶׁר לֹא יְקַבֵּל אֶת-מַלְכוּת הָאֱלֹהִים כַּיָּלֶד הוּא לֹא-יָבֹא בָהּ׃

מכשול העושר

(מתי יט' 30-16;מרקוס י' 31-17)

18 וַיִּשְׁאָלֵהוּ קָצִין אֶחָד לֵאמֹר רַבִּי הַטּוֹב מָה-אֶעֱשֶׂה וְאִירַשׁ חַיֵּי עוֹלָם׃ 19 וַיֹּאמֶר אֵלָיו יֵשׁוּעַ מַדּוּעַ קָרָאתָ לִּי טוֹב אֵין טוֹב כִּי אִם-אֶחָד וְהוּא הָאֱלֹהִים׃ 20 אֶת-הַמִּצְוֹת אַתָּה יוֹדֵעַ לֹא תִנְאָף לֹא תִרְצָח לֹא תִגְנֹב לֹא-תַעֲנֶה עֵד שָׁקֶר כַּבֵּד אֶת-אָבִיךָ וְאֶת-אִמֶּךָ׃(שמות כ' יא'-טז'; דברים ה' טו'-כ') 21 וַיֹּאמַר אֶת-כָּל-אֵלֶּה שָׁמַרְתִּי מִנְּעוּרָי׃ 22 וַיִּשְׁמַע יֵשׁוּעַ וַיֹּאמֶר עוֹד אַחַת חָסָרְתָּ מְכֹר אֶת-כֹּל אֲשֶׁר-לְךָ וְחַלֵּק לָעֲנִיִּים וִיהִי לְךָ אוֹצָר בַּשָּׁמָיִם וּבוֹא וְלֵךְ אַחֲרָי׃ 23 וַיְהִי כְּשָׁמְעוֹ אֶת-זֹאת וַיֵּעָצֵב מְאֹד כִּי-עשֶׁר גָּדוֹל הָיָה לוֹ׃ 24 וַיַּרְא יֵשׁוּעַ כִּי נֶעֱצַב וַיֹּאמַר כַּמָּה יִקְשֶׁה לְבַעֲלֵי נְכָסִים לָבוֹא אֶל-מַלְכוּת הָאֱלֹהִים׃(תהילים סב' יא') 25 כִּי נָקֵל לַגָּמָל עֲבֹר בְּתוֹךְ-נֶקֶב הַמַּחַט מִבּוֹא עָשִׁיר אֶל-מַלְכוּת הָאֱלֹהִים׃ 26 וַיֹּאמְרוּ הַשֹּׁמְעִים וּמִי יוּכַל לְהִוָּשֵׁעַ׃ 27 וַיֹּאמֶר אֲשֶׁר יִפָּלֵא מִבְּנֵי אָדָם לֹא יִפָּלֵא מֵאֱלֹהִים׃(ירמיהו לב' יז'; איוב מב' ב'; בראשית יח' יד') 28 וַיֹּאמֶר פֶּטְרוֹס הֵן אֲנַחְנוּ עָזַבְנוּ אֶת-הַכֹּל וַנֵּלֶך אַחֲרֶיךָ׃ 29 וַיֹּאמֶר אֲלֵיהֶם אָמֵן אֹמֵר אֲנִי לָכֶם אֵין אִישׁ אֲשֶׁר עָזַב אֶת-בֵּיתוֹ אוֹ אֶת-אֲבוֹתָיו אוֹ אֶת-אֶחָיו אוֹ אֶת-אִשְׁתּוֹ אוֹ אֶת-בָּנָיו לְמַעַן מַלְכוּת הָאֱלֹהִים׃ 30 וְלֹא-יִקַּח תַּחְתֵּיהֶם כִּפְלֵי כִפְלַיִם בָּעוֹלָם הַזֶּה וְלָעוֹלָם הַבָּא חַיֵּי עוֹלָם׃

ישוע מדבר פעם שלישית על מותו ותחייתו

(מתי כ' 19-17;מרקוס י' 34-32)

31 וַיִּקַּח אֵלָיו אֶת-שְׁנֵים הֶעָשָׂר וַיֹּאמֶר לָהֶם הִנְנוּ עֹלִים יְרוּשָׁלָיְמָה וְכָל-הַכָּתוּב בִּידֵי הַנְּבִיאִים עַל בֶּן-הָאָדָם יִמָּלֵא׃ 32 כִּי יִמָּסֵר לַגּוֹיִם וְהֵתֵלּוּ בוֹ וְהִתְעַלְּלוּ וְיָרְקוּ בְּפָנָיו׃(ישעיהו נ' ו') 33 וְהִכּוּהוּ בַשּׁוֹטִים וֶהֱמִיתוּהוּ וּבַיּוֹם הַשְּׁלִישִׁי קוֹם יָקוּם׃ 34 וְהֵם לֹא הֵבִינוּ דָבָר מִכָּל-אֵלֶּה וַיְהִי הַדָּבָר הַזֶּה נֶעְלָם מֵהֶם וְלֹא יָדְעוּ אֶת-אֲשֶׁר דִּבֶּר אֲלֵיהֶם׃

ריפוי איש עיוור

(מתי כ' 34-29;מרקוס י' 52-46)

35 וַיְהִי בְּקָרְבוֹ אֶל-יְרִיחוֹ וְהִנֵּה-אִישׁ עִוֵּר ישֵׁב עַל-יַד הַדֶּרֶךְ לִשְׁאֹל צְדָקָה׃ 36 וַיִּשְׁמַע אֶת-קוֹל הָעָם הָעֹבֵר וַיִּדְרשׁ לָדַעַת מַה-הוּא׃ 37 וַיַּגִּידוּ לוֹ כִּי-יֵשׁוּעַ הַנָּצְרִי עוֹבֵר׃ 38 וַיִּזְעַק לֵאמֹר יֵשׁוּעַ בֶּן-דָּוִד חָנֵּנִי׃ 39 וְהַהֹלְכִים לְפָנָיו גָּעֲרוּ-בוֹ לְהַחֲשֹׁתוֹ וְהוּא הִרְבָּה עוֹד לִזְעֹק יֵשׁוּעַ בֶּן-דָּוִד חָנֵּנִי׃ 40 וַיַּעֲמֹד יֵשׁוּעַ וַיְצַו לַהֲבִיאוֹ אֵלָיו וַיְהִי כַּאֲשֶׁר קָרַב וַיִּשְׁאָלֵהוּ לֵאמֹר׃ 41 מַה-תִּרְצֶה שֶׁאֶעֱשֶׂה-לָּךְ וַיֹּאמֶר אֲדֹנִי אֲשֶׁר תִּפָּקַחְנָה עֵינָי׃ 42 וַיֹּאמֶר אֵלָיו תִּפָּקַחְנָה עֵינֶיךָ אֱמוּנָתְךָ הוֹשִׁיעָה לָּךְ׃ 43 וּפִתְאֹם נִפְקְחוּ עֵינָיו וַיֵּלֶךְ אַחֲרָיו הָלֹךְ וְשַׁבֵּחַ אֶת-הָאֱלֹהִים וְכָל-הָעָם רָאוּ וַיּוֹדוּ לֵאלֹהִים׃(ישעיהו לה' ה'; תהילים סו' ד'-ה')
