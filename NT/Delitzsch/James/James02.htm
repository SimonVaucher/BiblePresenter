לפרק הקודם

אגרת יעקב
פרק ב

הזהרה על משוא פנים

1 אַחַי אַל-יְהִי מַשּׂא פָנִים בֶּאֱמוּנַתְכֶם בְּיֵשׁוּעַ הַמָּשִׁיחַ אֲדֹנֵינוּ אֲדוֹן הַכָּבוֹד׃(ויקרא יט' טו'; דברי הימים ב', יט' ז') 2 כִּי אִם-יָבוֹא אִישׁ לְבֵית הַכְּנֵסֶת שֶׁלָּכֶם וְטַבְּעֹת זָהָב עַל-יָדָיו וְהוּא לָבוּשׁ לְבֻשֵׁי מִכְלוֹל וּבָא שָׁמָּה גַּם-אִישׁ עָנִי בִּבְגָדִים צוֹאִים׃ 3 וּפְנִיתֶם אֶל-הַלָּבוּשׁ לְבֻשֵׁי מִכְלוֹל וַאֲמַרְתֶּם לוֹ שֶׁב-לְךָ הֵנָּה בְּכָבוֹד וְלֶעָנִי תֹאמְרוּ עֲמָד-שָׁם אוֹ שֶׁב-פֹּה מִתַּחַת לַהֲדֹם רַגְלָי׃ 4 הֲלֹא לֵב וָלֵב לָכֶם וְהִנְּכֶם שֹׁפְטִים בַּעֲלֵי מַחֲשָׁבוֹת רָעוֹת׃ 5 שִׁמְעוּ אַחַי אֲהוּבָי הֲלֹא בַעֲנִיֵּי הָעוֹלָם הַזֶּה בָּחַר הָאֱלֹהִים (שמואל א', ב' ח') לִהְיוֹת עֲשִׁירִים בָּאֱמוּנָה וְיֹרְשֵׁי הַמַּלְכוּת אֲשֶׁר הִבְטִיחַ לְאֹהֲבָיו׃ 6 וְאַתֶּם הִכְלַמְתֶּם אֶת-הֶעָנִי הֲלֹא הָעֲשִׁירִים הֵם הָעֹשְׁקִים אֶתְכֶם וְהֵם הַסֹּחֲבִים אֶתְכֶם אֶל-בָּתֵּי דִין׃ 7 הֲלֹא הֵם הַמְגַדְּפִים אֶת-הַשֵּׁם הַטּוֹב הַנִּקְרָא עֲלֵיכֶם׃ 8 הֵן בַּעֲשׂוֹתְכֶם אֶת-הַמִּצְוָה הַמֹּלֶכֶת עַל-כֻּלָּן כְּפִי הַכָּתוּב וְאָהַבְתָּ לְרֵעֲךָ כָּמוֹךָ (ויקרא יט' יח') תֵּיטִיבוּ לַעֲשׂוֹת׃ 9 אוּלָם אִם-תִּשְׂאוּ פָנִים חַטָּאִים אַתֶּם וְהַתּוֹרָה תוֹכִיחֲכֶם כְּעֹבְרִים עָלֶיהָ׃(דברים א' יז') 10 כִּי אִישׁ אֲשֶׁר יְקַיֵּם אֶת-כָּל-הַתּוֹרָה וְנִכְשַׁל בְּאַחַת מִמִּצְוֹתֶיהָ נִדּוֹן עַל-כֻּלָּן׃(יחזקאל לג' יג') 11 כִּי הָאֹמֵר לֹא תִנְאָף הוּא הָאֹמֵר לֹא תִרְצָח (שמות כ' יב'; דברים ה' טז') וְאִם-אֵינְךָ נֹאֵף וְאַתָּה רוֹצֵחַ הִנְּךָ עֹבֵר עַל-הַתּוֹרָה׃ 12 כֵּן דַּבְּרוּ וְכֵן עֲשׂוּ כְּמִי שֶׁעֲתִידִים לְהִשָּׁפֵט עַל-פִּי תּוֹרַת הַחֵרוּת׃ 13 כִּי אֵין חֶסֶד בַּמִּשְׁפָּט לַאֲשֶׁר לֹא-עָשָׂה חָסֶד וְהַחֶסֶד יִתְגָּאֶה עַל-הַמִּשְׁפָּט׃

אמונה ומעשים

14 אַחַי מַה-יּוֹעִיל לְאִישׁ שֶׁיֹּאמַר כִּי אֱמוּנָה בוֹ וּמַעֲשִׂים אֵין בּוֹ הֲתוּכַל הָאֱמוּנָה לְהוֹשִׁיעוֹ׃ 15 אָח אוֹ אָחוֹת כִּי-יִהְיוּ בְעֵירֹם וְאֵין לָהֶם לֶחֶם חֻקָּם׃(דברים טו' יא') 16 וְאִישׁ מִכֶּם יֹאמַר אֲלֵיהֶם לְכוּ לְשָׁלוֹם וְהִתְחַמֲמוּ וּשְׂבָעוּ וְלֹא-תִתְּנוּ לָהֶם צָרְכֵי גוּפָם מַה-תּוֹעִיל זֹאת׃(ישעיהו נח' ז') 17 כָּכָה גַּם-הָאֱמוּנָה אִם-אֵין בָּהּ מַעֲשִׂים מֵתָה הִיא בְּעַצְמָהּ׃ 18 וְאִם-יֹאמַר אִישׁ אַתָּה אֱמוּנָה בְךָ וְלִי מַעֲשִׂים הַרְאֵנִי נָא אֶת-אֱמוּנָתְךָ [בִּבְלִי מַעֲשִׂים] וְאַרְאֲךָ אֲנִי מִתּוֹךְ מַעֲשַׂי אֶת-אֱמוּנָתִי׃ 19 אַתָּה מַאֲמִין שֶׁהָאֱלֹהִים אֶחָד הוּא הֱטִיבוֹתָ לְהַאֲמִין גַּם הַשֵּׁדִים מַאֲמִינִים בּוֹ וְרֹעֲדִים׃ 20 וְאַתָּה אִישׁ-בָּעַר הֲתַחְפֹּץ לָדַעַת כִּי הָאֱמוּנָה בְּאֵין מַעֲשִׂים מֵתָה הִיא׃ 21 אַבְרָהָם אָבִינוּ הֲלֹא בְּמַעֲשָׂיו נִצְדָּק בְּהַעֲלֹתוֹ אֶת-יִצְחָק בְּנוֹ עַל-הַמִּזְבֵּחַ׃(בראשית כב' ט'-י') 22 הִנְּךָ רֹאֶה כִּי-הָאֱמוּנָה עֹזֶרֶת לְמַעֲשָׂיו וּמִתּוֹךְ הַמַּעֲשִׂים הָשְׁלְמָה הָאֱמוּנָה׃ 23 וַיִּמָּלֵא הַכָּתוּב הָאֹמֵר וְהֶאֱמִן אַבְרָהָם בַּיהוָֹה וַתֵּחָשֶׁב-לוֹ לִצְדָקָה (בראשית טו' ו') וַיִּקָּרֵא אֹהֵב יְהוָֹה׃(דברי הימים ב', כ' ז'; ישעיהו מא' ח') 24 הִנְּכֶם רֹאִים כִּי בְּמַעֲשִׂים יִצְדַּק הָאִישׁ וְלֹא בֶאֱמוּנָה לְבַדָּהּ׃ 25 וְכֵן גַּם-רָחָב הַזּוֹנָה הֲלֹא נִצְדְּקָה בְּמַעֲשִׂים בְּאָסְפָהּ אֶת-הַמַּלְאָכִים אֶל-בֵּיתָהּ וַתְּשַׁלְּחֵם בְּדֶרֶךְ אַחֵר׃(יהושוע ו' יז') 26 כִּי כַּאֲשֶׁר הַגּוּף מִבְּלִי נְשָׁמָה מֵת הוּא כֵּן גַּם-הָאֱמוּנָה מִבְּלִי-מַעֲשִׂים מֵתָה׃
