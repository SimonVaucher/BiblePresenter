לפרק הקודם

אגרת יעקב
פרק ה

אזהרה לעשירים

1 הוֹי הָעֲשִׁירִים בְּכוּ וְהֵילִילוּ עַל-הַצָּרוֹת אֲשֶׁר תָּבֹאנָה עֲלֵיכֶם׃ 2 עָשְׁרְכֶם בָּלָה וּבִגְדֵיכֶם אֲכָלָם עָשׁ׃ 3 זְהַבְכֶם וְכַסְפְּכֶם כִּסָּתַם חֶלְאָה וְהָיְתָה חֶלְאָתָם בָּכֶם לְעֵדוּת וְאָכְלָה כְּמוֹ-אֵשׁ אֶת-בְּשַׂרְכֶם זֶה הוּא הָאוֹצָר אֲצַרְתֶּם לָכֶם לְקֵץ הַיָּמִים׃ 4 הִנֵּה שְׂכַר הַפֹּעֲלִים אֹסְפֵי קְצִיר שְׂדֹתֵיכֶם אֲשֶׁר עֲשַׁקְתֶּם צֹעֵק עֲלֵיכֶם וְצַעֲקַת הַקּוֹצְרִים בָּאָה בְאָזְנֵי יְהוָֹה צְבָאוֹת׃(ויקרא יט' יג'; דברים כד' טו') 5 הִתְעַדַּנְתֶּם בָּאָרֶץ וְהִתְעַנַּגְתֶּם וְהִשְׁמַנְתֶּם אֶת-לִבְּכֶם כְּמוֹ לְיוֹם טִבְחָה׃ 6 הִרְשַׁעְתֶּם וַהֲמִתֶּם אֶת-הַצַּדִּיק וְהוּא נַעֲנֶה לֹא-יִפְתַּח פִּיו׃(תהילים צד' כא')

סבלנות ותפילה

7 לָכֵן אַחַי דֹּמּוּ וְהוֹחִילוּ עַד-בּוֹא הָאָדוֹן הִנֵּה הָאִכָּר מְחַכֶּה לְטוּב תְּבוּאַת הָאֲדָמָה וּמְיַחֵל כִּי-יֵרֵד עָלֶיהָ גֶּשֶׁם יוֹרֶה וּמַלְקוֹשׁ׃(דברים יא' יד') 8 כֵּן הוֹחִילוּ גַם-אַתֶּם וְאַמְּצוּ לְבַבְכֶם כִּי קָרוֹב הָאָדוֹן לָבֹא׃ 9 אֶחָי אַל-תִּתְאוֹנֲנוּ אִישׁ עַל-רֵעֵהוּ פֶּן-תִּשָּׁפֵטוּ הִנֵּה הַדַּיָּן עֹמֵד בַּפָּתַח׃ 10 אֶחָי הַנְּבִיאִים אֲשֶׁר דִּבְּרוּ בְּשֵׁם יְהוָֹה הֵם יִהְיוּ לָכֶם לְמוֹפֵת הָעֳנִי וְהַתּוֹחֶלֶת׃(ירמיהו כו' יד'-טז') 11 הִנֵּה מְאַשְּׁרִים אֲנַחְנוּ אֶת-הַסֹּבְלִים שְׁמַעְתֶּם סַבְלָנוּת אִיּוֹב וְאֶת-אַחֲרִית הָאָדוֹן (איוב מב' י'-יז') רְאִיתֶם כִּי-רַחוּם וְחַנּוּן יְהוָֹה׃(תהילים קג' יג'; איכה ג' לב') 12 וְרֹאשׁ דָּבָר אַחַי לֹא תִשָּׁבְעוּ לֹא בַשָּׁמַיִם וְלֹא בָאָרֶץ וְלֹא כָּל-שְׁבוּעָה אַחֶרֶת וִיהִי הֵן שֶׁלָּכֶם הֵן וְלֹא שֶׁלָּכֶם לֹא פֶּן-תִּפְּלוּ בִּידֵי הַדִּין׃ 13 אִישׁ מִכֶּם כִּי-יֵצַר לוֹ יִתְפַּלֵּל וַאֲשֶׁר יִיטַב לִבּוֹ הוּא יְזַמֵּר׃(תהילים נ' טו'; נז' ח') 14 אִישׁ מִכֶּם כִּי-יֶחֱלֶה יִקְרָא אֶת-זִקְנֵי הַקְּהִלָּה וְיִתְפַּלֲלוּ בַעֲדוֹ וִיסוּכֻהוּ שֶׁמֶן בְּשֵׁם יְהוָֹה׃ 15 וּתְפִלַּת הָאֱמוּנָה תּוֹשִׁיעַ אֶת-הַחוֹלֶה וַיְהוָֹה יְקִימֶנּוּ וַאֲשֶׁר חָטָא יִסָּלַח לוֹ׃(ישעיהו לח' יז') 16 הִתְוַדּוּ עֲוֹנוֹתֵיכֶם אִישׁ לִפְנֵי רֵעֵהוּ וְהִתְפַּלֲלוּ אִישׁ בְּעַד רֵעֵהוּ לְמַעַן תֵּרָפֵאוּ כִּי-גָדוֹל כֹּחַ תְּפִלַּת הַצַּדִּיק הַקֹּרֵא אֶל-אֱלֹהִים בְּחָזְקָה׃(משלי כח' יג'; בראשית כ' יז') 17 אֵלִיָּהוּ אֱנוֹשׁ אָנוּשׁ הָיָה כָמֹנוּ וְהִתְפַּלֵּל תְּפִלָּה שֶׁלּא יִהְיֶה מָטָר וְלֹא-הָיָה מָטָר בָּאָרֶץ שָׁלֹשׁ שָׁנִים וְשִׁשָּׁה חֳדָשִׁים׃ 18 וַיָּשָׁב וַיִּתְפַּלֵּל וְהַשָּׁמַיִם נָתְנוּ מָטָר וְהָאָרֶץ הִצְמִיחָה אֶת-פִּרְיָהּ׃(מלכים א', יח' מב'-מה') 19 אֶחָי כִּי יִתְעֶה אִישׁ בָּכֶם מִן-הָאֱמֶת וְאִישׁ אַחֵר יְשִׁיבֶנּוּ׃ 20 יֵדַע-נָא כִּי הַמֵּשִׁיב אֶת-הַחוֹטֵא מֵעִקְּשׁוּת דַּרְכּוֹ הוּא יוֹשִׁיעַ אֶת-נַפְשׁוֹ מִמָּוֶת וִיכַסֶּה עַל-הֲמוֹן פְּשָׁעִים׃(משלי י' יב'; ישעיהו נה' ז')
