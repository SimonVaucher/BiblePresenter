לפרק הקודם

אגרת פטרוס הראשונה
פרק ג

התנהגות האישה ובעלה

1 כֵּן גַּם אַתֶּן, הַנָּשִׁים, הִכָּנַעְנָה לְבַעֲלֵיכֶן; וְכָךְ, אִם יֵשׁ שֶׁאֵינָם מְצַיְּתִים לַדָּבָר, יִקָּנֶה לִבָּם לֹא עַל-יְדֵי דִּבּוּר אֶלָּא עַל-יְדֵי הִתְנַהֲגוּת הַנָּשִׁים, 2 בִּרְאוֹתָם אֶת הִתְנַהֲגוּתְכֶן הַטְּהוֹרָה הַמְלֻוָּה בְּיִרְאָה. 3 וּפְאֵרְכֶן אַל יְהֵא פְּאֵר חִיצוֹנִי שֶׁל מַחְלְפוֹת שֵׂעָר וַעֲדִי זָהָב וּבְגָדִים,(ישעיהו ג' טז'-כד'; משלי לא' ל') 4 אֶלָּא הָאָדָם אֲשֶׁר בְּסֵתֶר הַלֵּב, הַפְּאֵר הַבִּלְתִּי נִשְׁחָת שֶׁל רוּחַ עֲנָוָה וּשְׁקֵטָה אֲשֶׁר יְקָרָה הִיא מְאֹד בְּעֵינֵי אֱלֹהִים. 5 כָּךְ הִתְקַשְּׁטוּ בֶּעָבָר גַּם הַנָּשִׁים הַקְּדוֹשׁוֹת וְהַמְיַחֲלוֹת לֵאלֹהִים, בְּהִכָּנְעָן לְבַעֲלֵיהֶן; 6 כְּשָׂרָה אֲשֶׁר שָׁמְעָה בְּקוֹל אַבְרָהָם וְקָרְאָה לוֹ אָדוֹן.(בראשית יח' יב') וְאַתֶּן נִהְיֵיתֶן בָּנוֹת לָהּ, אִם אַתֶּן עוֹשׂוֹת אֶת הַטּוֹב וְאֵינְכֶן פוֹחֲדוֹת פַּחַד.(תהילים קיב' ז'; משלי ג' כה') 7 וְכֵן אַתֶּם, הַבְּעָלִים, חֲיוּ יַחַד עִמָּהֶן בִּידִיעָה שֶׁהָאִשָּׁה הִיא כְּלִי חַלָּשׁ יוֹתֵר, וּתְנוּ לָהֶן כָּבוֹד כְּשֻׁתָּפוֹת בְּנַחֲלַת חֶסֶד הַחַיִּים, לְמַעַן לֹא תְּעֻכַּבְנָה תְּפִלּוֹתֵיכֶם.

הסבל של ההולכים בדרך הצדק

8 סוֹף דָּבָר, הֱיוּ כֻּלְּכֶם לֵב אֶחָד, שֻׁתָּפִים לְצַעַר הַזּוּלַת, אוֹהֲבֵי הָאַחִים, רַחֲמָנִים וַעֲנָוִים. 9 אַל תְּשַׁלְּמוּ רָעָה תַּחַת רָעָה וְלֹא חֵרוּף תַּחַת חֵרוּף; אַדְּרַבָּא -- תְּבָרְכוּ, שֶׁהֲרֵי לְכָךְ נִקְרֵאתֶם, לְמַעַן תִּירְשׁוּ אֶת הַבְּרָכָה; 10 כִּי "הָאִישׁ הֶחָפֵץ חַיִּים אֹהֵב יָמִים לִרְאוֹת טוֹב, יִנְצֹר לְשׁוֹנוֹ מֵרָע, וּשְׂפָתָיו מִדַּבֵּר מִרְמָה. 11 יָסוּר מֵרָע וְיַעֲשֶׂה-טוֹב, יְבַקֵּשׁ שָׁלוֹם וְיִרְדְּפֵהוּ", 12 כִּי "עֵינֵי יהוה אֶל-צַדִּיקִים, וְאָזְנָיו אֶל-שַׁוְעָתָם; וּפְנֵי יהוה בְּעֹשֵׂי רָע."(תהילים לד' יג'-יז') 13 וּמִי יָרֵעַ לָכֶם אִם תִּהְיוּ קַנָּאִים לַטּוֹב? 14 אַךְ גַּם אִם תִּסְבְּלוּ לְמַעַן הַצְּדָקָה -- אַשְׁרֵיכֶם. אַל תִּפְחֲדוּ מֵאֵימָתָם וְאַל תִּבָּהֲלוּ. 15 אֶת הָאָדוֹן הַמָּשִׁיחַ הַקְדִּישׁוּ בִּלְבַבְכֶם,(ישעיהו ח' יב'-יג') וְתָמִיד הֱיוּ מוּכָנִים לְהָשִׁיב בַּעֲנָוָה וְיִרְאָה לְכָל מִי שֶׁמְּבַקֵּשׁ מִכֶּם דִּין וְחֶשְׁבּוֹן בִּדְבַר הַתִּקְוָה שֶׁבְּלִבְּכֶם. 16 יְהֵא מַצְפּוּנְכֶם טָהוֹר, כְּדֵי שֶׁיֵּבוֹשׁוּ הַמַּשְׁמִיצִים מֵאוֹתָם דְּבָרִים שֶׁבִּגְלָלָם הֵם מְגַנִּים אֶת הִתְנַהֲגוּתְכֶם הַטּוֹבָה בַּמָּשִׁיחַ. 17 מוּטָב שֶׁתִּסְבְּלוּ, אִם זֶהוּ רְצוֹן אֱלֹהִים, בַּעֲשׂוֹתְכֶם אֶת הַטּוֹב מִשֶּׁתִּסְבְּלוּ בַּעֲשׂוֹתְכֶם רַע. 18 הֵן גַּם הַמָּשִׁיחַ מֵת לְכַפָּרַת חֲטָאִים, אַחַת וּלְתָמִיד, הַצַּדִּיק בְּעַד הָרְשָׁעִים, כְּדֵי לְקָרֵב אוֹתָנוּ אֶל הָאֱלֹהִים. הוּא הוּמַת בִּבְשָׂרוֹ, אַךְ הָחֳיָה בָּרוּחַ 19 וּבָה גַּם הָלַךְ וּבִשֵּׂר לָרוּחוֹת הַכְּלוּאוֹת בַּמַּאֲסָר, 20 לְאֵלֶּה שֶׁלֹּא צִיְּתוּ כַּאֲשֶׁר חִכָּה אֱלֹהִים בְּאֹרֶךְ אַפּוֹ בִּיְמֵי נֹחַ. בְּאוֹתָם יָמִים נִבְנְתָה הַתֵּבָה אֲשֶׁר מְעַטִּים בְּתוֹכָהּ -- שְׁמוֹנֶה נְפָשׁוֹת -- נוֹשְׁעוּ בַּמַּיִם;(בראשית ז' ז') 21 וְזֶה מְסַמֵּל אֶת הַטְּבִילָה הַמּוֹשִׁיעָה כָּעֵת גַּם אֶתְכֶם -- אֵין הִיא הֲסָרַת זֻהֲמַת הַגּוּף, אֶלָּא הִתְחַיְּבוּת שֶׁל מַצְפּוּן טָהוֹר כְּלַפֵּי אֱלֹהִים -- בִּתְחִיַּת יֵשׁוּעַ הַמָּשִׁיחַ 22 אֲשֶׁר עָלָה הַשָּׁמַיְמָה וְהוּא לְימִין אֱלֹהִים,(תהילים קי' א') וּלְפָנָיו הֻכְנְעוּ מַלְאָכִים וְרָשֻׁיּוֹת וּגְבוּרוֹת.(דניאל ז' יד')
