לפרק הקודם

הבשורה על-פי מרקוס
פרק יג

ישוע מנבא חורבן וצרות

(מתי כד' 14-1; לוקס כא' 19-5)

1 בְּצֵאתוֹ מִבֵּית הַמִּקְדָּשׁ אָמַר אֵלָיו אֶחָד מִתַּלְמִידָיו: "רַבִּי, רְאֵה אֵיזֶה אֲבָנִים וְאֵיזֶה בִּנְיָנִים!" 2 אָמַר לוֹ יֵשׁוּעַ: "אַתָּה רוֹאֶה אֶת הַבִּנְיָנִים הַגְּדוֹלִים הָאֵלֶּה? לֹא תִּשָּׁאֵר אֶבֶן עַל אֶבֶן אֲשֶׁר לֹא תֻּפַּל אַרְצָה." 3 כְּשֶׁיָּשַׁב עַל הַר הַזֵּיתִים, מוּל בֵּית הַמִּקְדָּשׁ, שְׁאָלוּהוּ כֵּיפָא וְיַעֲקֹב וְיוֹחָנָן וְאַנְדְּרֵי בִּהְיוֹתָם לְבַדָּם אִתּוֹ: 4 "אֱמֹר נָא לָנוּ, מָתַי יִהְיֶה הַדָּבָר הַזֶּה וּמַהוּ הָאוֹת לִפְנֵי שֶׁכָּל זֶה יִתְגַּשֵּׁם?" 5 אָמַר לָהֶם יֵשׁוּעַ: "הִזָּהֲרוּ שֶׁלֹּא יַתְעֶה אֶתְכֶם אִישׁ.(ירמיהו כט' ח') 6 רַבִּים יָבוֹאוּ בִּשְׁמִי וְיֹאמְרוּ 'אֲנִי הוּא' וְיַתְעוּ רַבִּים. 7 וּבְשָׁמְעֲכֶם מִלְחָמוֹת וּשְׁמוּעוֹת מִלְחָמָה אַל תִּבָּהֲלוּ; זֶה מַה שֶּׁצָרִיךְ לִהְיוֹת, וּבְכָל זֹאת עוֹד לֹא בָּא הַקֵּץ. 8 גּוֹי יָקוּם עַל גּוֹי וּמַמְלָכָה עַל מַמְלָכָה (ישעיהו יט' ב'; חגי ב' כב') וְתִהְיֶינָה רְעִידוֹת אֲדָמָה בִּמְקוֹמוֹת רַבִּים וְיִהְיֶה רָעָב. אֵלֶּה רֵאשִׁית הַצָּרוֹת. 9 וְאַתֶּם הִזָּהֲרוּ; יִמְסְרוּ אֶתְכֶם לְסַנְהֶדְרִיּוֹת, יַכּוּ אֶתְכֶם בְּבָתֵּי כְּנֶסֶת, וְלִפְנֵי מוֹשְׁלִים וּמְלָכִים יַעֲמִידוּ אֶתְכֶם בִּגְלָלִי, לְעֵדוּת לָהֶם. 10 אֲבָל תְּחִלָּה צָרִיךְ שֶׁהַבְּשׂוֹרָה תֻּכְרַז לְכָל הַגּוֹיִים. 11 כַּאֲשֶׁר יוֹלִיכוּ אֶתְכֶם כְּדֵי לְהַסְגִּירְכֶם, אַל תִּדְאֲגוּ לְמַה שֶּׁתְּדַבְּרוּ, כִּי מַה שֶּׁיִּנָּתֵן לָכֶם בְּאוֹתָהּ שָׁעָה אֶת זֶה דַּבְּרוּ; שֶׁכֵּן לֹא אַתֶּם תְּדַבְּרוּ אֶלָּא רוּחַ הַקֹּדֶשׁ. 12 אָז יַסְגִּיר אָח אֶת אָחִיו לַמָּוֶת וְאָב יַסְגִּיר אֶת בְּנוֹ. בָּנִים יָקוּמוּ עַל הוֹרֵיהֶם וְיָמִיתוּ אוֹתָם.(מיכה ז' ו') 13 וְתִהְיוּ שְׂנוּאִים עַל הַכֹּל בִּגְלַל שְׁמִי, אַךְ הַמַּחֲזִיק מַעֲמָד עַד קֵץ הוּא יִוָּשַׁע."

צרה גדולה

(מתי כד' 28-15; לוקס כא' 24-20)

14 "וּכְשֶׁתִּרְאוּ אֶת הַשִּׁקּוּץ הַמְשׁוֹמֵם עוֹמֵד בְּמָקוֹם שֶׁאֵינוֹ מְקוֹמוֹ (דניאל ט' כז'; יא' לא'; יב' יא') -- עַל הַקּוֹרֵא לְהָבִין -- אֲזַי הַנִּמְצָאִים בִּיהוּדָה, שֶׁיָּנוּסוּ אֶל הֶהָרִים; 15 מִי שֶׁעַל הַגַּג אַל יֵרֵד וְאַל יִכָּנֵס לָקַחַת דָּבָר מִבֵּיתוֹ; 16 וּמִי שֶׁבַּשָּׂדֶה אַל יַחֲזֹר לָקַחַת אֶת בִּגְדוֹ. 17 וְאוֹי לֶהָרוֹת וְלַמֵּינִיקוֹת בַּיָּמִים הָהֵם! 18 הִתְפַּלְּלוּ שֶׁלֹּא יִהְיֶה זֶה בַּחֹרֶף, 19 כִּי הַיָּמִים הָהֵם יִהְיִוּ צָרָה אֲשֶׁר לֹא הָיְתָה כָּמוֹהָ מֵרֵאשִׁית הַבְּרִיאָה שֶׁבָּרָא אֱלֹהִים וְעַד עַתָּה; אַף לֹא תִּהְיֶה כָּמוֹהָ.(דניאל יב' א'; יואל ב' ב') 20 וְלוּלֵא קִצֵּר אֲדֹנָי אֶת הַיָּמִים לֹא הָיָה נִצָּל כָּל בָּשָׂר, אַךְ לְמַעַן הַבְּחִירִים אֲשֶׁר בָּחַר בָּהֶם קִצֵּר אֶת הַיָּמִים.(ישעיהו סה' ח') 21 וְאָז אִם יֹאמַר לָכֶם אִישׁ, 'הִנֵּה פֹּה הַמָּשִׁיחַ, הִנֵּה שָׁם,' אַל תַּאֲמִינוּ לוֹ; 22 כִּי יָקוּמוּ מְשִׁיחֵי שֶׁקֶר וּנְבִיאֵי שֶׁקֶר(דברים יג' ב'-ד'; ירמיהו יד' יד') וְיַעֲשׂוּ אוֹתוֹת וּמוֹפְתִים לְהַתְעוֹת, אִם אֶפְשָׁר, אֶת הַבְּחִירִים. 23 וְאַתֶּם הִזָּהֲרוּ; אָמַרְתִּי לָכֶם אֶת הַכֹּל מֵרֹאשׁ."

בואו של בן-האדם

(מתי כד' 44-29; לוקס כא' 33-25)

24 "בַּיָּמִים הָהֵם, אַחֲרֵי הַצָּרָה הַהִיא, תֶּחְשַׁךְ הַשֶּׁמֶשׁ וְהַיָּרֵחַ לֹא יַגִּיהַּ אוֹרוֹ,(ישעיהו יג' י'; יואל ב' ב'; ב' י'; ג' ג'-ד') 25 הַכּוֹכָבִים יִפְּלוּ מִן הַשָּׁמַיִם וְכֹחוֹת הַשָּׁמַיִם יִזְדַּעְזְעוּ.(ישעיהו לד' ד') 26 אָז יִרְאוּ אֶת בֶּן-הָאָדָם בָּא בַּעֲנָנִים בִּגְבוּרָה רַבָּה וּבְכָבוֹד.(דניאל ז' יג'-יד') 27 וְאָז יִשְׁלַח אֶת הַמַּלְאָכִים וִיקַבֵּץ אֶת בְּחִירָיו מֵאַרְבַּע הָרוּחוֹת, מִקְּצֵה הָאָרֶץ עַד קְצֵה הַשָּׁמַיִם.(דברים ל' ד'; ישעיהו יא' יב') 28 לִמְדוּ אֶת הַמָּשָׁל מִן הַתְּאֵנָה: כַּאֲשֶׁר עֲנָפֶיהָ מִתְרַכְּכִים וְהֶעָלִים צָצִים יוֹדְעִים אַתֶּם שֶׁהַקַּיִץ קָרוֹב. 229 כֵּן גַּם אַתֶּם, כְּשֶׁתִּרְאוּ אֶת הַדְּבָרִים הָאֵלֶּה מִתְקַיְּמִים דְּעוּ כִּי קָרוֹב הוּא בַּפֶּתַח. 30 אָמֵן. אוֹמֵר אֲנִי לָכֶם, לֹא יַעֲבֹר הַדּוֹר הַזֶּה עַד אֲשֶׁר יִהְיוּ כָּל אֵלֶּה. 31 הַשָּׁמַיִם וְהָאָרֶץ יַעַבְרוּ וּדְבָרַי לֹא יַעַבְרוּ.(ישעיהו נא' ו'; תהילים קב' כז'-כח') 32 אֲבָל אֶת הַיּוֹם הַהוּא אוֹ הַשָּׁעָה אֵין אִישׁ יוֹדֵעַ, גַּם לֹא הַמַּלְאָכִים בַּשָּׁמַיִם וְגַם לֹא הַבֵּן, אֶלָּא הָאָב. 33 הִזָּהֲרוּ, הֱיוּ עֵרִים, כִּי אֵינְכֶם יוֹדְעִים מָתַי הָעֵת. 34 דּוֹמֶה הַדָּבָר לְאִישׁ הַיּוֹצֵא לְמַסָּע. כְּשֶׁעָזַב אֶת בֵּיתוֹ נָתַן סַמְכוּת לַעֲבָדָיו, לְכָל אֶחָד אֶת מְלַאכְתּוֹ, וְעַל הַשּׁוֹעֵר צִוָּה לַעֲמֹד עַל הַמִּשְׁמָר. 35 לָכֵן עִמְדוּ עַל הַמִּשְׁמָר, כִּי אֵינְכֶם יוֹדְעִים מָתַי יָבוֹא בַּעַל הַבַּיִת, אִם בָּעֶרֶב אוֹ בַּחֲצוֹת הַלַּיְלָה, אִם בְּעֵת קְרִיאַת הַתַּרְנְגוֹל אוֹ בַּבֹּקֶר, 36 פֶּן יָבוֹא פִּתְאוֹם וְיִמְצָא אֶתְכֶם יְשֵׁנִים. 37 אֶת אֲשֶׁר אֲנִי אוֹמֵר לָכֶם אוֹמֵר אֲנִי לַכֹּל: עִמְדוּ עַל הַמִּשְׁמָר!"
