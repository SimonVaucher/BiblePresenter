לאגרת פטרוס השניה

אגרת יוחנן הראשונה
פרק א

דבר החיים

1 אֶת אֲשֶׁר הָיָה מֵרֵאשִׁית, אֶת אֲשֶׁר שָׁמַעְנוּ, אֶת אֲשֶׁר רָאִינוּ בְּעֵינֵינוּ, אֲשֶׁר הִבַּטְנוּ בּוֹ וְיָדֵינוּ מִשְּׁשׁוּ אוֹתוֹ, עַל-אוֹדוֹת דְּבַר הַחַיִּים, -- 2 וְהַחַיִּים נִגְלוּ וַאֲנַחְנוּ רָאִינוּ, וַהֲרֵינוּ מְעִידִים וּמוֹדִיעִים לָכֶם אֶת חַיֵּי הָעוֹלָמִים אֲשֶׁר הָיוּ אֵצֶל הָאָב וְנִגְלוּ לָנוּ -- 3 אֶת אֲשֶׁר רָאִינוּ וְשָׁמַעְנוּ מוֹדִיעִים אָנוּ גַּם לָכֶם, לְמַעַן תִּתְחַבְּרוּ עִמָּנוּ גַּם אַתֶּם.(תהילים עח' ג'-ד') וְאָכֵן הִתְחַבְּרוּתֵנוּ הִיא עִם הָאָב וְעִם בְּנוֹ יֵשׁוּעַ הַמָּשִׁיחַ. 4 זֹאת אֲנַחְנוּ כּוֹתְבִים לְמַעַן תִּהְיֶה שִׂמְחָתֵנוּ שְׁלֵמָה.

האלהים הוא אור

5 וְזֶה דְּבַר הַבְּשׂוֹרָה אֲשֶׁר שָׁמַעְנוּ מִמֶּנּוּ וַאֲנַחְנוּ מַשְׁמִיעִים לָכֶם, שֶׁהָאֱלֹהִים אוֹר הוּא וְכָל חֹשֶׁךְ אֵין בּוֹ.(תהילים קד' ב') 6 אִם נֹאמַר שֶׁהִתְחַבְּרוּת לָנוּ אִתּוֹ וְנִתְהַלֵּךְ בַּחֹשֶׁךְ, דּוֹבְרֵי שֶׁקֶר אֲנַחְנוּ וְאֵינֶנּוּ מְקַיְּמִים אֶת הָאֱמֶת. 7 אֲבָל אִם נִתְהַלֵּךְ בָּאוֹר, כְּמוֹ שֶׁהוּא בָּאוֹר, כִּי אָז הִתְחַבַּרְנוּ זֶה עִם זֶה וְדַם יֵשׁוּעַ הַמָּשִׁיחַ בְּנוֹ מְטַהֵר אוֹתָנוּ מִכָּל חֵטְא. 8 אִם נֹאמַר שֶׁאֵין בָּנוּ חֵטְא, מַתְעִים אָנוּ אֶת עַצְמֵנוּ וְהָאֱמֶת אֵינֶנָּה בָּנוּ.(תהילים נא' ז') 9 אִם נִתְוַדֶּה עַל חֲטָאֵינוּ, נֶאֱמָן הוּא וְצַדִּיק לִסְלֹחַ לָנוּ עַל חֲטָאֵינוּ וּלְטַהֵר אוֹתָנוּ מִכָּל עַוְלָה.(משלי כח' יג'; תהילים לב' ה') 10 אִם נֹאמַר שֶׁלֹּא חָטָאנוּ, לְכוֹזֵב שָׂמְנוּ אוֹתוֹ וּדְבָרוֹ אֵינוֹ בָּנוּ.
