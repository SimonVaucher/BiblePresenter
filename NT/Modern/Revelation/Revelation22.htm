פרק כב

ירושלים החדשה -- המשך

1 הוּא הֶרְאָה לִי נְהַר מַיִם חַיִּים, מַבְהִיק כִּבְדֹלַח, יוֹצֵא מִכִּסֵּא הָאֱלֹהִים וְהַשֶּׂה.(יחזקאל מז' א'; יואל ד' יח'; זכריה יד' ח') 2 בְּאֶמְצַע רְחוֹב הָעִיר וְעַל שְׂפַת הַנָּהָר מִזֶּה וּמִזֶּה -- עֵץ חַיִּים עוֹשֶׂה פֵּרוֹת שְׁתֵּים-עֶשְׂרֵה פְּעָמִים, בְּתִתּוֹ בְּכָל חֹדֶשׁ וְחֹדֶשׁ אֶת פִּרְיוֹ, וַעֲלֵה הָעֵץ לְמַרְפֵּא הַגּוֹיִים.(בראשית ג' כב'; ג' כד'; יחזקאל מז' יב') 3 שׁוּם קְלָלָה לֹא תִּהְיֶה עוֹד. כִּסֵּא אֱלֹהִים וְהַשֶּׂה יִהְיֶה בָּהּ וַעֲבָדָיו יְשָׁרְתוּהוּ.(זכריה יד' יא'; תהילים מו' ו'; יחזקאל מח' לה') 4 הֵם יִרְאוּ אֶת פָּנָיו, וּשְׁמוֹ עַל מִצְחוֹתָם.(תהילים יא' ז'; יז' טו') 5 וְלַיְלָה לֹא יִהְיֶה עוֹד וְלֹא יִצְטָרְכוּ לְאוֹר מְנוֹרָה וּלְאוֹר שֶׁמֶשׁ, כִּי יהוה אֱלֹהִים יָאִיר עֲלֵיהֶם (ישעיהו ס' א'-ב'; תהילים פד' יב') וְיִמְלְכוּ לְעוֹלְמֵי עוֹלָמִים.(דניאל ז' יח'; ז' כז')

ביאת ישוע המשיח

6 עוֹד אָמַר אֵלַי: "הַדְּבָרִים הָאֵלֶּה נֶאֱמָנִים וַאֲמִתִּיִּים וַיהוה אֱלֹהֵי רוּחוֹת הַנְּבִיאִים שָׁלַח אֶת מַלְאָכוֹ לְהַרְאוֹת לַעֲבָדָיו אֶת אֲשֶׁר צָרִיךְ לִהְיוֹת בִּמְהֵרָה."(דניאל ב' כח') 7 "הִנְנִי בָּא מַהֵר. אַשְׁרֵי הַשּׁוֹמֵר אֶת דִּבְרֵי הַנְּבוּאָה שֶׁל הַסֵּפֶר הַזֶּה."(תהילים קיא' ב') 8 וַאֲנִי, יוֹחָנָן, הָרוֹאֶה וְשׁוֹמֵעַ אֶת אֵלֶּה. וְכַאֲשֶׁר שָׁמַעְתִּי וְרָאִיתִי, נָפַלְתִּי לְהִשְׁתַּחֲווֹת לְרַגְלֵי הַמַּלְאָךְ אֲשֶׁר הִרְאַנִי אֶת אֵלֶּה. 9 אָמַר אֵלַי: "רְאֵה, אַל-נָא; עֶבֶד-חָבֵר אֲנִי לְךָ וּלְאַחֶיךָ הַנְּבִיאִים וְלַשּׁוֹמְרִים אֶת דִּבְרֵי הַסֵּפֶר הַזֶּה. לֵאלֹהִים תִּשְׁתַּחֲוֶה!"(דברים ו' יג'; תהילים כט' ב') 10 עוֹד אָמַר אֵלַי: "אַל תַּחְתֹּם אֶת דִּבְרֵי הַנְּבוּאָה שֶׁל הַסֵּפֶר הַזֶּה, כִּי קָרוֹב הַמּוֹעֵד. 11 הַמְעַוֵּל -- שֶׁיּוֹסִיף לַעֲשׂוֹת עָוֶל; הַטָּמֵא -- שֶׁיִּטָּמֵא עוֹד; הַצַּדִּיק -- שֶׁיִּצָּדֵק עוֹד; הַקָּדוֹשׁ -- שֶׁיִּתְקַדֵּשׁ עוֹד." 12 "הִנְנִי בָּא מַהֵר וְאִתִּי הַשָּׂכָר לְשַׁלֵּם לְכָל אִישׁ כְּמַעֲשֵׂהוּ.(ישעיהו מ' י'; סב' יא'; איוב לד' יא') 13 אֲנִי הָאָלֶף אַף אֲנִי הַתָּו, הָרִאשׁוֹן אַף הָאַחֲרוֹן, הָרֵאשִׁית וְהַתַּכְלִית."(ישעיהו מד' ו'; מח' יב') 14 אַשְׁרֵי הַמְכַבְּסִים אֶת גְּלִימוֹתֵיהֶם, לְמַעַן תִּהְיֶה לָהֶם זְכוּת עַל עֵץ הַחַיִּים וְיִכָּנְסוּ הָעִירָה דֶּרֶךְ הַשְּׁעָרִים.(***) 15 בַּחוּץ יִהְיוּ הַכְּלָבִים וְהַמְכַשְּׁפִים, הַזּוֹנִים וְהַמְרַצְּחִים, עוֹבְדֵי אֱלִילִים וְכָל אוֹהֵב שֶׁקֶר וְעוֹשֵׂהוּ.(ישעיהו נו' י'-יא') 16 "אֲנִי יֵשׁוּעַ שָׁלַחְתִּי אֶת מַלְאָכִי לְהָעִיד לָכֶם אֶת הַדְּבָרִים הָאֵלֶּה בַּקְּהִלּוֹת. אָנֹכִי שֹׁרֶשׁ דָּוִד וְצֶאֱצָאוֹ, אַיֶּלֶת הַשַּׁחַר."(ישעיהו יא' א'; יא' י') 17 הָרוּחַ וְהַכַּלָּה אוֹמְרוֹת: "בּוֹא!", וְהַשּׁוֹמֵעַ יֹאמַר נָא "בּוֹא!"; הַצָּמֵא יָבוֹא נָא, וְהֶחָפֵץ יִקַּח נָא מַיִם חַיִּים חִנָּם.(ישעיהו נה' א') 18 אָנֹכִי מֵעִיד בְּכָל מִי שֶׁשּׁוֹמֵעַ אֶת דִּבְרֵי נְבוּאַת הַסֵּפֶר הַזֶּה: אִישׁ אִם יוֹסִיף עֲלֵיהֶם, יוֹסִיף עָלָיו אֱלֹהִים אֶת הַמַּכּוֹת הַכְּתוּבוֹת בַּסֵּפֶר הַזֶּה.(משלי ל' ו') 19 וְאִישׁ אִם יִגְרַע מִדִּבְרֵי סֵפֶר הַנְּבוּאָה הַזֹּאת, יִגְרַע הָאֱלֹהִים אֶת חֶלְקוֹ מֵעֵץ הַחַיִּים וּמֵעִיר הַקֹּדֶשׁ, מִן הַדְּבָרִים הַכְּתוּבִים בַּסֵּפֶר הַזֶּה.(דברים ד' ב'//***) 20 הַמֵּעִיד אֶת הַדְּבָרִים הָאֵלֶּה אוֹמֵר: "אָכֵן, אֲנִי בָּא מַהֵר." אָמֵן. בּוֹא נָא הָאָדוֹן יֵשׁוּעַ. 21 חֶסֶד הָאָדוֹן יֵשׁוּעַ עִם כָּל הַקְּדוֹשִׁים. אָמֵן.
