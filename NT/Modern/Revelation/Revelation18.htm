לפרק הקודם

התגלות
פרק יח

מפלת בבל

1 אַחֲרֵי כֵן רָאִיתִי מַלְאָךְ אַחֵר יוֹרֵד מִן הַשָּׁמַיִם; רַבָּה הָיְתָה סַמְכוּתוֹ וְהָאָרֶץ הֵאִירָה מִכְּבוֹדוֹ. 2 הוּא קָרָא בְּקוֹל חָזָק וְאָמַר: "נָפְלָה נָפְלָה בָּבֶל הַגְּדוֹלָה (ישעיהו כא' ט') וְהָיְתָה לִמְעוֹן שֵׁדִים וּלְבֵית כֶּלֶא לְכָל רוּחַ טְמֵאָה וּבֵית כֶּלֶא לְכָל עוֹף טָמֵא וּמָאוּס,(ישעיהו יג' כא'; ירמיהו נ' לט') 3 כִּי מִיֵּין חֲרוֹן תַּזְנוּתָהּ שָׁתוּ כָּל הַגּוֹיִים, וּמַלְכֵי הָאָרֶץ זָנוּ עִמָּהּ וְסוֹחֲרֵי הָאָרֶץ הִתְעַשְּׁרוּ מֵעֹצֶם מוֹתְרוֹתֶיהָ."(נחום ג' ד') 4 שָׁמַעְתִּי קוֹל אַחֵר מִן הַשָּׁמַיִם אוֹמֵר: "צְאוּ מִמֶּנָּה, עַמִּי, פֶּן תִּשְׁתַּתְּפוּ בְּחַטֹּאתֶיהָ וּפֶן תְּקַבְּלוּ מִמַּכּוֹתֶיהָ,(ירמיהו נא' ו'; נא' מה') 5 כִּי הִגִּיעוּ חַטֹּאתֶיהָ עַד הַשָּׁמַיִם וְזָכַר אֱלֹהִים אֶת פְּשָׁעֶיהָ.(ירמיהו נא' ט') 6 שַׁלְּמוּ לָהּ כַּגְּמוּל שֶׁשִּׁלְּמָה גַּם הִיא וְכִפְלוּ לָהּ כִּפְלַיִם כְּמַעֲלָלֶיהָ; בַּכּוֹס אֲשֶׁר מָסְכָה מִסְכוּ לָהּ כִּפְלַיִם.(תהילים קלז' ח'; ירמיהו נ' כט') 7 בַּמִּדָּה שֶׁרוֹמְמָה עַצְמָהּ וְהִתְעַנְּגָה כֵּן תְּנוּ לָהּ מַכְאוֹב וָאֵבֶל, כִּי אָמְרָה בִּלְבָבָהּ: 'יוֹשֶׁבֶת אֲנִי, מַלְכָּה; אֵינֶנִּי אַלְמָנָה וְאֵבֶל לֹא אֶרְאֶה.'(ישעיהו מז' ח') 8 עַל כֵּן בְּיוֹם אֶחָד תָּבֹאנָה מַכּוֹתֶיהָ: מָוֶת, אֵבֶל וְרָעָב, וּבָאֵשׁ תִּשָּׂרֵף;(ישעיהו מז' ט') כִּי חָזָק יהוה אֱלֹהִים הַשּׁוֹפֵט אוֹתָהּ."(ירמיהו נ' לד') 9 וּמַלְכֵי הָאָרֶץ אֲשֶׁר זָנוּ וְהִתְעַנְּגוּ אִתָּהּ יִבְכּוּ וְיִסְפְּדוּ עָלֶיהָ כִּרְאוֹתָם אֶת עֲשַׁן שְׂרֵפָתָהּ. 10 מֵרָחוֹק יַעַמְדוּ מִפְּנֵי אֵימַת עִנּוּיָהּ וְיֹאמְרוּ: "אוֹי, אוֹי, בָּבֶל הָעִיר הַגְּדוֹלָה, הָעִיר הַחֲזָקָה, כִּי בְּשָׁעָה אַחַת בָּא מִשְׁפָּטֵךְ."(יחזקאל כו' טז'-יז') 11 גַּם סוֹחֲרֵי הָאָרֶץ בּוֹכִים וּמִתְאַבְּלִים עָלֶיהָ, כִּי אֵין קוֹנֶה עוֹד אֶת מִטְעַן סְחוֹרָתָם,(יחזקאל כז' לה'-לו') 12 מִטְעַן שֶׁל זָהָב וְכֶסֶף וְאֶבֶן יְקָרָה וּפְנִינִים וּבוּץ וְאַרְגָּמָן וּמֶשִׁי וְשָׁנִי; כָּל עֵץ בֹּשֶׂם וְכָל כְּלִי שֶׁנְהָב, כָּל כְּלִי מֵעֵץ יָקָר וּמִנְּחֹשֶׁת, מִבַּרְזֶל וּמִשַּׁיִשׁ; 13 קִנָּמוֹן וַאֲמוֹמוֹן, קְטֹרֶת סַמִּים, מוֹר וּלְבוֹנָה, יַיִן וְשֶׁמֶן, סֹלֶת וְחִטָּה, בָּקָר וְצֹאן, סוּסִים וּמֶרְכָּבוֹת, עֲבָדִים וְנֶפֶשׁ אָדָם. 14 "סָר מִמֵּךְ הַפְּרִי חֶמְדַּת נַפְשֵׁךְ; כָּל הַמּוֹתָרוֹת וְכָל נוֹצֵץ אָבְדוּ מִמֵּךְ; לֹא יִמְצְאוּם עוֹד." 15 סוֹחֲרֵי הַדְּבָרִים הָאֵלֶּה, אֲשֶׁר הִתְעַשְּׁרוּ מִמֶּנָּה, יַעַמְדוּ מֵרָחוֹק מִפְּנֵי אֵימַת עִנּוּיָהּ, בּוֹכִים וּמִתְאַבְּלִים (***) 16 וְאוֹמְרִים: "אוֹי, אוֹי, הָעִיר הַגְּדוֹלָה הַלְּבוּשָׁה שֵׁשׁ וְאַרְגָּמָן וְשָׁנִי וּמְקֻשֶּׁטֶת בְּזָהָב וְאֶבֶן יְקָרָה וּפְנִינִים, 17 כִּי בְּשָׁעָה אַחַת הָחֳרַב עֹשֶׁר גָּדוֹל כָּזֶה!" וְכָל רַב-חוֹבֵל וְכָל הַנּוֹסֵעַ לִמְחוֹז חֶפְצוֹ, גַּם מַלָּחִים וְכָל אֲשֶׁר מְלַאכְתָּם בַּיָּם, עָמְדוּ מֵרָחוֹק 18 וְצָעֲקוּ בִּרְאוֹתָם אֶת עֲשַׁן שְׂרֵפָתָהּ. "מִי בֶּעָרִים כָּעִיר הַגְּדוֹלָה?" אָמְרוּ. 19 הֵם זָרְקוּ עָפָר עַל רָאשֵׁיהֶם (יחזקאל כז' ל') וְצָעֲקוּ בִּבְכִי וְיָגוֹן: "אוֹי, אוֹי, הָעִיר הַגְּדוֹלָה אֲשֶׁר עָשְׁרוּ בָּה מֵהוֹנָהּ כָּל בַּעֲלֵי הָאֳנִיּוֹת בַּיָּם, כִּי בְּשָׁעָה אַחַת הָחְרְבָה!"(יחזקאל כו' יט') 20 גִּילוּ עָלֶיהָ שָׁמַיִם, גַּם קְדוֹשִׁים וּשְׁלִיחִים וּנְבִיאִים, כִּי שָׁפַט אֱלֹהִים אֶת מִשְׁפַּטְכֶם מִמֶּנָּה!(ירמיהו נא' מח') 21 וּמַלְאָךְ חָזָק אֶחָד הֵרִים אֶבֶן, כְּאֶבֶן רֵיחַיִם גְּדוֹלָה, וְהִשְׁלִיךְ אוֹתָהּ לַיָּם בְּאָמְרוֹ: "כָּכָה תֻּשְׁלַךְ בְּשֶׁצֶף בָּבֶל הָעִיר הַגְּדוֹלָה וְלֹא תִּמָּצֵא עוֹד.(ירמיהו נא' סג'-סד') 22 צְלִילֵי פּוֹרְטֵי נֵבֶל, צְלִילֵי נוֹגְנִים וּמְחַלְּלִים בַּחֲלִילִים, וּצְלִילֵי תּוֹקְעִים בַּחֲצוֹצְרוֹת -- לֹא יִשָּׁמְעוּ בָּךְ עוֹד. וְכָל אֻמָּן מִכָּל אֻמָּנוּת לֹא יִמָּצֵא בָּךְ עוֹד, וְלֹא יִשָּׁמַע עוֹד בְּקִרְבֵּךְ קוֹל רֵיחַיִם.(ישעיהו כד' ח'; יחזקאל כו' יג') 23 אוֹר מְנוֹרָה לֹא יָאִיר בָּךְ עוֹד, וְקוֹל חָתָן וְכַלָּה לֹא יִשָּׁמַע בָּךְ עוֹד;(ירמיהו כה' י') כִּי סוֹחֲרַיִךְ הָיוּ נִכְבְּדֵי הָאָרֶץ וּבְכִשּׁוּפַיִךְ הֻתְעוּ כָּל הַגּוֹיִים.(***) 24 בָּהּ נִמְצָא דַּם נְבִיאִים וּקְדוֹשִׁים וְדַם כָּל אֲשֶׁר נִטְבְּחוּ עֲלֵי אָרֶץ."
