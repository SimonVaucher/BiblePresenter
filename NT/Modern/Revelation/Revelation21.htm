פרק כא

השמים החדשים והארץ החדשה

1 רָאִיתִי שָׁמַיִם חֲדָשִׁים וְאֶרֶץ חֲדָשָׁה, כִּי הַשָּׁמַיִם הָרִאשׁוֹנִים וְהָאָרֶץ הָרִאשׁוֹנָה עָבְרוּ וְהַיָּם אֵינֶנּוּ עוֹד.(ישעיהו סה' יז'; סו' כב') 2 רָאִיתִי אֶת עִיר הַקֹּדֶשׁ,(ישעיהו נב' א') יְרוּשָׁלַיִם הַחֲדָשָׁה, יוֹרֶדֶת מִן הַשָּׁמַיִם מֵאֵת הָאֱלֹהִים, מוּכָנָה כְּכַלָּה מְקֻשֶּׁטֶת לְבַעֲלָהּ.(ישעיהו סא' י') 3 וְשָׁמַעְתִּי קוֹל גָּדוֹל מִן הַכִּסֵּא -- אוֹמֵר: "הִנֵּה מִשְׁכַּן הָאֱלֹהִים עִם בְּנֵי אָדָם וְיִשְׁכֹּן עִמָּהֶם; הֵמָּה יִהְיוּ לוֹ לְעַם וְהוּא הָאֱלֹהִים יִהְיֶה עִמָּהֶם,(יחזקאל מג' ז'; לז' כז') 4 וְיִמְחֶה כָּל דִּמְעָה מֵעֵינֵיהֶם וְהַמָּוֶת לֹא יִהְיֶה עוֹד;(ישעיהו כה' ח') גַּם אֵבֶל וּזְעָקָה וּכְאֵב לֹא יִהְיוּ עוֹד, כִּי הָרִאשׁוֹנוֹת עָבְרוּ."(ישעיהו לה' י'; נא' יא') 5 אָמַר הַיּוֹשֵׁב עַל הַכִּסֵּא: "הִנְנִי עוֹשֶׂה הַכֹּל חָדָשׁ." אָמַר אֵלַי: "כְּתֹב, כִּי הַדְּבָרִים הָאֵלֶּה נֶאֱמָנִים וַאֲמִתִּיִּים הֵם."(ישעיהו מג' יט') 6 הוֹסִיף וְאָמַר אֵלַי: "הָיֹה נִהְיָה. אֲנִי הָאָלֶף וְהַתָּו, הָרֵאשִׁית וְהַתַּכְלִית. אֲנִי אֶתֵּן לַצָּמֵא מִמַּעְיַן הַמַּיִם שֶׁל הַחַיִּים -- חִנָּם.(ישעיהו נה' א') 7 הַמְנַצֵּחַ יִירַשׁ אֶת אֵלֶּה וַאֲנִי אֶהְיֶה לוֹ לֵאלֹהִים וְהוּא יִהְיֶה לִי לְבֵן.(זכריה ח' ח'; שמואל ב', ז' יד') 8 אֲבָל מוּגֵי הַלֵּב וְהַבִּלְתִּי מַאֲמִינִים, הַמְתֹעָבִים וְהַמְרַצְּחִים, הַזּוֹנִים וְהַמְכַשְּׁפִים, עוֹבְדֵי הָאֱלִילִים וְכָל הַמְשַׁקְּרִים -- חֶלְקָם בָּאֲגַם הַבּוֹעֵר בְּאֵשׁ וְגָפְרִית אֲשֶׁר הוּא הַמָּוֶת הַשֵּׁנִי."(דניאל ז' יא')

ירושלים החדשה

9 אֶחָד מִשִּׁבְעַת הַמַּלְאָכִים, אֲשֶׁר לָהֶם שֶׁבַע הַקְּעָרוֹת הַמְּלֵאוֹת בְּשֶׁבַע הַמַּכּוֹת הָאַחֲרוֹנוֹת, בָּא וְדִבֵּר אִתִּי. אָמַר "בּוֹא אַרְאֶה לְךָ אֶת הַכַּלָּה אֵשֶׁת הַשֶּׂה". 10 הוּא נָשָׂא אוֹתִי, בָּרוּחַ, אֶל הַר גָּדוֹל וְגָבוֹהַּ וְהֶרְאָה לִי אֶת עִיר הַקֹּדֶשׁ, יְרוּשָׁלַיִם, יוֹרֶדֶת מִן הַשָּׁמַיִם מֵאֵת אֱלֹהִים (***//יחזקאל מ' ב') 11 וּכְבוֹד אֱלֹהִים לָהּ.(תהילים פד' יב') אוֹר נָגְהָהּ כְּאֶבֶן יְקָרָה מְאֹד, כְּאֶבֶן יָשְׁפֶה זַכָּה כִּבְדֹלַח, 12 וְיֵשׁ לָהּ חוֹמָה גְּדוֹלָה וּגְבוֹהָה עִם שְׁנֵים-עָשָׂר שְׁעָרִים, וְעַל הַשְּׁעָרִים שְׁנֵים-עָשָׂר מַלְאָכִים וְשֵׁמוֹת כְּתוּבִים עֲלֵיהֶם -- שְׁנֵים-עָשָׂר שִׁבְטֵי בְּנֵי יִשְׂרָאֵל;(שמות כח' כא') 13 מִמִּזְרָח שְׁלוֹשָׁה שְׁעָרִים, מִצָּפוֹן שְׁלוֹשָׁה שְׁעָרִים, מִדָּרוֹם שְׁלוֹשָׁה שְׁעָרִים, מִמַּעֲרָב שְׁלוֹשָׁה שְׁעָרִים.(יחזקאל מח' לא'-לד') 14 וּלְחוֹמַת הָעִיר שְׁנֵים-עָשָׂר יְסוֹדוֹת וַעֲלֵיהֶם שְׁנֵים-עָשָׂר שֵׁמוֹת שֶׁל שְׁנֵים-עָשָׂר שְׁלִיחֵי הַשֶּׂה. 15 הַמְדַבֵּר אֵלַי הֶחֱזִיק קְנֵה-מִדָּה מִזָּהָב כְּדֵי לִמְדֹּד אֶת הָעִיר וְאֶת שְׁעָרֶיהָ וְאֶת חוֹמָתָהּ.(יחזקאל מ' ג'; מ' ה'; זכריה ב' ה'-ו') 16 וְהָעִיר מְרֻבַּעַת, אָרְכָּהּ כְּרָחְבָּהּ. הוּא מָדַד אֶת הָעִיר בְּקָנֶה -- שְׁנֵים-עָשָׂר אֶלֶף רִיס; אָרְכָּהּ וְרָחְבָּהּ וְקוֹמָתָהּ שָׁוִים.(יחזקאל מח' טז'-יז') 17 גַּם אֶת חוֹמָתָהּ מָדַד, וְהַמִּדָּה מֵאָה וְאַרְבָּעִים וְאַרְבַּע אַמּוֹת לְפִי מִדָּה אֱנוֹשִׁית אֲשֶׁר הִיא מִדָּה שֶׁל מַלְאָךְ. 18 הַחוֹמָה בִּנְיָנָהּ אֶבֶן יָשְׁפֶה, וְהָעִיר זָהָב טָהוֹר כִּזְכוּכִית טְהוֹרָה. 19 יְסוֹדוֹת חוֹמַת הָעִיר מְקֻשָּׁטִים בְּכָל אֶבֶן יְקָרָה; הַיְסוֹד הָרִאשׁוֹן יָשְׁפֶה, הַשֵּׁנִי סַפִּיר, הַשְּׁלִישִׁי שְׁבוֹ, הָרְבִיעִי בָּרֶקֶת,(ישעיהו נד' יא'-יב') 20 הַחֲמִישִׁי יַהֲלוֹם, הַשִּׁשִּׁי אֹדֶם, הַשְּׁבִיעִי תַּרְשִׁישׁ, הַשְּׁמִינִי שֹׁהַם, הַתְּשִׁיעִי פִּטְדָה, הָעֲשִׂירִי נֹפֶךְ, הָאַחַד-עָשָׂר לֶשֶׁם, הַשְּׁנֵים-עָשָׂר אַחְלָמָה. 21 וּשְׁנֵים-עָשָׂר הַשְּׁעָרִים -- שְׁתֵּים-עֶשְׂרֵה מַרְגָּלִיּוֹת; כָּל שַׁעַר וְשַׁעַר הָיָה מִמַּרְגָּלִית אַחַת, וּרְחוֹב הָעִיר זָהָב טָהוֹר כִּזְכוּכִית זַכָּה. 22 הֵיכָל לֹא רָאִיתִי בָּהּ, כִּי הֵיכָלָהּ הוּא יהוה אֱלֹהֵי צְבָאוֹת וְהַשֶּׂה. 23 וְהָעִיר אֵינֶנָּה צְרִיכָה לַשֶּׁמֶשׁ וְלַיָּרֵחַ שֶׁיָּאִירוּ בָּהּ, כִּי כְּבוֹד אֱלֹהִים הֵאִיר אוֹתָהּ וְהַשֶּׂה הוּא מְנוֹרָתָהּ.(***//ישעיהו ס' יט') 24 הַגּוֹיִים יֵלְכוּ לְאוֹרָהּ וּמַלְכֵי הָאָרֶץ מְבִיאִים אֶת כְּבוֹדָם אֵלֶיהָ.(תהילים פט' טז'; ישעיהו ס' ג'; ס' ה') 25 שְׁעָרֶיהָ לֹא יִסָּגְרוּ בַּיּוֹם; וַהֲרֵי לַיְלָה לֹא יִהְיֶה שָׁם.(ישעיהו ס' יא') 26 אֶת כְּבוֹד הַגּוֹיִים וִיקָרָם יָבִיאוּ אֵלֶיהָ,(ישעיהו סו' יב') 27  וְלֹא יִכָּנֵס אֵלֶיהָ כָּל טָמֵא וְעוֹשֵׂה תּוֹעֵבָה וָשֶׁקֶר, כִּי אִם הַכְּתוּבִים בְּסֵפֶר הַחַיִּים שֶׁל הַשֶּׂה.(ישעיהו נב' א')
