לפרק הקודם

התגלות
פרק ו

החותמות

1 רָאִיתִי כַּאֲשֶׁר פָּתַח הַשֶּׂה אֶת אֶחָד מִשִּׁבְעַת הַחוֹתָמוֹת, וְשָׁמַעְתִּי אַחַת מֵאַרְבַּע הַחַיּוֹת אוֹמֶרֶת כְּמוֹ בְּקוֹל רַעַם: "בּוֹא!" 2 רָאִיתִי וְהִנֵּה סוּס לָבָן וְרוֹכְבוֹ נוֹשֵׂא קֶשֶׁת; עֲטֶרֶת נִתְּנָה לוֹ וְהוּא יָצָא מְנַצֵּחַ וּלְמַעַן יְנַצֵּחַ.(זכריה א' ח') 3 כַּאֲשֶׁר פָּתַח אֶת הַחוֹתָם הַשֵּׁנִי שָׁמַעְתִּי אֶת הַחַיָּה הַשְּׁנִיָּה אוֹמֶרֶת: "בּוֹא!" 4 סוּס אַחֵר יָצָא, אָדֹם כָּאֵשׁ.(***) לְרוֹכְבוֹ הֻתַּר לְהָסִיר אֶת הַשָּׁלוֹם מִן הָאָרֶץ כְּדֵי שֶׁיַּהַרְגוּ זֶה אֶת זֶה; וְחֶרֶב גְּדוֹלָה נִתְּנָה לוֹ.(יחזקאל כא' יד'-טז') 5 כַּאֲשֶׁר פָּתַח אֶת הַחוֹתָם הַשְּׁלִישִׁי שָׁמַעְתִּי אֶת הַחַיָּה הַשְּׁלִישִׁית אוֹמֶרֶת: "בּוֹא!" רָאִיתִי וְהִנֵּה סוּס שָׁחוֹר וְרוֹכְבוֹ מַחֲזִיק מֹאזְנַיִם בְּיָדוֹ. 6 וְשָׁמַעְתִּי כְּמוֹ קוֹל בֵּין אַרְבַּע הַחַיּוֹת אוֹמֵר: "קַב חִטִּים בְּדִינָר וּשְׁלוֹשָׁה קַבִּין שְׂעוֹרִים בְּדִינָר, וְלַשֶּׁמֶן וְלַיַּיִן אַל תַּזִּיק." 7 כַּאֲשֶׁר פָּתַח אֶת הַחוֹתָם הָרְבִיעִי שָׁמַעְתִּי אֶת קוֹל הַחַיָּה הָרְבִיעִית אוֹמֵר: "בּוֹא!" 8 רָאִיתִי וְהִנֵּה סוּס יְרַקְרַק וְרוֹכְבוֹ שְׁמוֹ מָוֶת,(זכריה ו' ב'-ג') וּשְׁאוֹל הָלְכָה אַחֲרָיו. וְנִתְּנָה לָהֶם סַמְכוּת עַל רֶבַע הָאָרֶץ, לַהֲרֹג בַּחֶרֶב וּבָרָעָב וּבַמָּוֶת וְעַל-יְדֵי חַיּוֹת הָאָרֶץ.(יחזקאל ה' יב'; יחזקאל יד' כא') 9 כַּאֲשֶׁר פָּתַח אֶת הַחוֹתָם הַחֲמִישִׁי רָאִיתִי מִתַּחַת לַמִּזְבֵּחַ אֶת נַפְשׁוֹת הַטְּבוּחִים עַל דְּבַר הָאֱלֹהִים וְעַל הָעֵדוּת שֶׁהֶחֱזִיקוּ בָּהּ. 10 וְהֵם צָעֲקוּ בְּקוֹל גָּדוֹל: "עַד מָתַי, אֲדֹנָי הַקָּדוֹשׁ וַאֲמִתִּי, לֹא תִּשְׁפֹּט וְתִנְקֹם אֶת דָּמֵינוּ מִיּוֹשְׁבֵי הָאָרֶץ?"(תהילים ו' ד'; דברים לב' מג') 11 לְכָל אֶחָד מֵהֶם נִתְּנָה גְּלִימָה לְבָנָה וְנֶאֱמַר לָהֶם לָנוּחַ עוֹד זְמַן מְעַט, עַד שֶׁיִּמָּלֵא גַּם מִסְפַּר חַבְרֵיהֶם הָעֲבָדִים וַאֲחֵיהֶם הָעֲתִידִים לֵהָרֵג כְּמוֹהֶם. 12 רָאִיתִי כְּשֶׁפָּתַח אֶת הַחוֹתָם הַשִּׁשִּׁי הִתְחוֹלְלָה רְעִידַת אֲדָמָה גְּדוֹלָה; הַשֶּׁמֶשׁ הִשְׁחִירָה כְּשַׂק שֵׂעָר וְהַיָּרֵחַ כֻּלּוֹ הָיָה כְּדָם;(ישעיהו יג' י'; יואל ג' ד') 13 כּוֹכְבֵי הַשָּׁמַיִם נָפְלוּ אַרְצָה, כִּתְאֵנָה הַמַּשְׁלִיכָה אֶת פַּגֶּיהָ כַּאֲשֶׁר הִיא מִטַּלְטֶלֶת בְּרוּחַ חֲזָקָה; 14 הַשָּׁמַיִם נָסוֹגוּ כִּמְגִלָּה נִגְלֶלֶת; כָּל הַר וְאִי נֶעֶתְקוּ מִמְּקוֹמָם.(ישעיהו לד' ד') 15 וּמַלְכֵי הָאָרֶץ וְהַנִּכְבָּדִים, שָׂרֵי הָאֲלָפִים וְהָעֲשִׁירִים וּבַעֲלֵי הָעָצְמָה, כָּל עֶבֶד וְכָל בֶּן-חוֹרִין, הִתְחַבְּאוּ בַּמְּעָרוֹת וּבְסַלְעֵי הֶהָרִים, 16 וְאָמְרוּ לֶהָרִים וְלַסְּלָעִים: "נִפְלוּ עָלֵינוּ וְהַסְתִּירוּ אוֹתָנוּ (ישעיהו ב' יט') מִפְּנֵי הַיּוֹשֵׁב עַל הַכִּסֵּא וּמִזַּעַם הַשֶּׂה,(תהילים ב' יב') 17 כִּי בָּא הַיּוֹם הַגָּדוֹל, יוֹם חֲרוֹנָם, וּמִי יָכוֹל לַעֲמֹד?"(תהילים קי' ה'; עו' ח'; יואל ב' יא')
