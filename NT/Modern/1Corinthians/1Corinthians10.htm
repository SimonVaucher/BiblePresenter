לפרק הקודם

אגרת שאול הראשונה אל הקורינתים
פרק י

אזהרה על עבודת אלילים

1 אַחַי, אֵינֶנִּי רוֹצֶה שֶׁיֵּעָלֵם מִכֶּם שֶׁאֲבוֹתֵינוּ כֻּלָּם הָיוּ תַּחַת הֶעָנָן,(שמות יג' כא') וְכֻלָּם עָבְרוּ בְּתוֹךְ הַיָּם,(שמות יד' כא'-כב') 2 וְכֻלָּם נִטְבְּלוּ לְמֹשֶׁה בֶּעָנָן וּבַיָּם. 3 כֻּלָּם אָכְלוּ אוֹתוֹ מַאֲכָל רוּחָנִי,(דברים ח' ג') 4 וְכֻלָּם שָׁתוּ אוֹתוֹ מַשְׁקֶה רוּחָנִי, כִּי שָׁתוּ מִן הַצּוּר הָרוּחָנִי הַהוֹלֵךְ עִמָּהֶם וְהַצּוּר הוּא הַמָּשִׁיחַ.(שמות יז' ו'; תהילים עח' טו') 5 אֲבָל בְּרֻבָּם לֹא רָצָה אֱלֹהִים וְהֵם הוּמְתוּ בַּמִּדְבָּר.(במדבר יד' כט') 6 כָּל זֶה הָיָה דֻּגְמָה לָנוּ, לְמַעַן לֹא נִתְאַוֶּה לְרָעוֹת כְּשֵׁם שֶׁהִתְאַוּוּ הֵם.(במדבר יא' ד') 7 אַל תִּהְיוּ עוֹבְדֵי אֱלִילִים כְּשֵׁם שֶׁהָיוּ כַּמָּה מֵהֶם, כַּכָּתוּב: "וַיֵּשֶׁב הָעָם לֶאֱכֹל וְשָׁתוֹ וַיָּקֻמוּ לְצַחֵק."(שמות לב' ו') 8 גַּם אַל נִזְנֶה כְּשֵׁם שֶׁזָּנוּ כַּמָּה מֵהֶם וּבְיוֹם אֶחָד נָפְלוּ עֶשְׂרִים וּשְׁלוֹשָׁה אֶלֶף.(במדבר כה' א'-ט') 9 וְאַל נְנַסֶּה אֶת יהוה כְּשֵׁם שֶׁנִּסּוּהוּ אֲנָשִׁים מֵהֶם וְהוּמְתוּ עַל-יְדֵי הַנְּחָשִׁים.(במדבר כא' ה'-ו') 10 אַף אַל תִּתְלוֹנְנוּ כְּשֵׁם שֶׁהִתְלוֹנְנוּ כַּמָּה מֵהֶם וּמֵתוּ בִּידֵי הַמַּלְאָךְ הַמַּשְׁחִית.(במדבר יד' ב'; יד' לז') 11 מַה שֶּׁקָּרָה לָהֶם הָיָה לֶקַח לְדֻגְמָה וְזֶה נִכְתַּב כְּדֵי לְהַזְהִיר אוֹתָנוּ,אֲנַחְנוּ אֲשֶׁר קִצֵּי הָעוֹלָמִים הִגִּיעוּ אֵלֵינוּ. 12 לָכֵן מִי שֶׁחוֹשֵׁב כִּי עֲמִידָתוֹ אֵיתָנָה, יִזָּהֵר פֶּן יִפֹּל. 13 שׁוּם נִסָּיוֹן לֹא בָּא עֲלֵיכֶם מִלְּבַד נִסָּיוֹן אֱנוֹשִׁי רָגִיל. נֶאֱמָן הוּא הָאֱלֹהִים (דברים ז' ט'; ישעיהו מט' ז') וְלֹא יַנִּיחַ לָכֶם לְהִתְנַסּוֹת לְמַעְלָה מִיכָלְתְּכֶם, אֶלָּא עִם הַנִּסָּיוֹן יָכִין גַּם אֶת דֶּרֶךְ הַמּוֹצָא כְּדֵי שֶׁתּוּכְלוּ לַעֲמֹד בּוֹ.(תהילים לד' כ') 14 עַל כֵּן, חֲבִיבַי, תִּבְרְחוּ מֵעֲבוֹדַת אֱלִילִים. 15 כְּאֶל אֲנָשִׁים נְבוֹנִים אֲנִי מְדַבֵּר; שִׁפְטוּ אַתֶּם מַה שֶּׁאֲנִי אוֹמֵר: 16 כּוֹס הַבְּרָכָה שֶׁאָנוּ מְבָרְכִים עָלֶיהָ, הַאֵין הִיא הִתְחַבְּרוּת לְדַם הַמָּשִׁיחַ? הַלֶּחֶם שֶׁאָנוּ בּוֹצְעִים, הַאֵין הוּא הִתְחַבְּרוּת לְגוּף הַמָּשִׁיחַ? 17 מִשּׁוּם שֶׁהַלֶּחֶם אֶחָד, אֲנַחְנוּ הָרַבִּים גּוּף אֶחָד; שֶׁכֵּן כֻּלָּנוּ מִשְׁתַּתְּפִים בַּלֶּחֶם הָאֶחָד. 18 רְאוּ נָא אֶת עַם יִשְׂרָאֵל. הַאִם אֵין הָאוֹכְלִים מִן הַקָּרְבָּנוֹת שֻׁתָּפִים בַּמִּזְבֵּחַ?(ויקרא ז' ו'; ויקרא ז' טו') 19 וּבְכֵן מָה אֲנִי אוֹמֵר -- שֶׁיֵּשׁ מַמָּשׁ בְּזִבְחֵי אֱלִילִים? שֶׁיֵּשׁ מַמָּשׁ בָּאֱלִיל? 20 לֹא! אֶלָּא מַה שֶׁהַגּוֹיִים מַקְרִיבִים, לְשֵׁדִים הֵם מַקְרִיבִים וְלֹא לֵאלֹהִים; וְאֵינֶנִּי רוֹצֶה שֶׁתִּהְיוּ שֻׁתָּפִים לַשֵּׁדִים.(דברים לב' יז') 21 אֵינְכֶם יְכוֹלִים לִשְׁתּוֹת מִכּוֹס הָאָדוֹן וּמִכּוֹס הַשֵּׁדִים; אֵינְכֶם יְכוֹלִים לְהִשְׁתַּתֵּף בְּשֻׁלְחַן הָאָדוֹן וּבְשֻׁלְחַן הַשֵּׁדִים. 22 הַאִם נָעֵז לְעוֹרֵר אֶת קִנְאַת הָאָדוֹן? כְּלוּם חֲזָקִים אֲנַחְנוּ מִמֶּנּוּ?(דברים לב' כא') 23 הַכֹּל מֻתָּר, אַךְ לֹא הַכֹּל מוֹעִיל; הַכֹּל מֻתָּר, אַךְ לֹא הַכֹּל בּוֹנֶה. 24 אִישׁ אַל יְבַקֵּשׁ אֶת טוֹבַת עַצְמוֹ אֶלָּא אֶת טוֹבַת זוּלָתוֹ. 25 אִכְלוּ כָּל מַה שֶּׁנִּמְכָּר בַּשּׁוּק וְאַל תִּשְׁאֲלוּ שׁוּם שְׁאֵלָה שֶׁל מַצְפּוּן, 26 כִּי לַיהוה הָאָרֶץ וּמְלוֹאָהּ.(תהילים כד' א') 27 אִם יַזְמִינְכֶם אִישׁ בִּלְתִּי מַאֲמִין וּרְצוֹנְכֶם לָלֶכֶת אֵלָיו, אִכְלוּ כָּל מַה שֶּׁיֻּגַּשׁ לָכֶם וְאַל תִּשְׁאֲלוּ שׁוּם שְׁאֵלָה שֶׁל מַצְפּוּן. 28 אֲבָל אִם מִישֶׁהוּ יֹאמַר לָכֶם: "זֶה נִזְבַּח לֶאֱלִיל", אַל תֹּאכְלוּ לְמַעַן הָאִישׁ שֶׁהוֹדִיעַ לָכֶם וּלְמַעַן הַמַּצְפּוּן. 29 מַצְפּוּן, אֲנִי אוֹמֵר -- לֹא זֶה שֶׁלְּךָ אֶלָּא שֶׁל הָאַחֵר,שֶׁכֵּן לָמָּה שֶׁתִּשָּׁפֵט חֵרוּתִי עַל-יְדֵי מַצְפּוּנוֹ שֶׁל אַחֵר? 30 אִם אֲנִי מְבָרֵךְ וְאוֹכֵל לָמָּה שֶׁיְּדֻבַּר בִּי רָעוֹת בִּגְלַל מַה שֶּׁאֲנִי מְבָרֵךְ עָלָיו? 31 וּבְכֵן אִם תֹּאכְלוּ אוֹ תִּשְׁתּוּ, אוֹ כָּל מַה שֶּׁתַּעֲשׂוּ -- עֲשׂוּ אֶת הַכֹּל לְמַעַן כְּבוֹד אֱלֹהִים. 32 אַל תִּהְיוּ מִכְשׁוֹל לֹא לַיְּהוּדִים וְלֹא לַגּוֹיִים וְלֹא לִקְהִלַּת אֱלֹהִים, 33 כְּדֶרֶךְ שֶׁגַּם אֲנִי מִשְׁתַּדֵּל לִהְיוֹת רָצוּי לְכֻלָּם בְּכָל דָּבָר וְאֵינֶנִּי מְבַקֵּשׁ אֶת טוֹבָתִי אֲנִי, אֶלָּא אֶת טוֹבַת הָרַבִּים לְמַעַן יִוָּשְׁעוּ.
