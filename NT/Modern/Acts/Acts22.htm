מעשי השליחים
פרק כב

שאול נושא את דברו אל העם -- המשך

1 "אֲנָשִׁים אַחִים וְאָבוֹת, שִׁמְעוּ נָא כָּעֵת אֶת דְּבַר הִצְטַדְּקוּתִי לִפְנֵיכֶם" -- 2 כַּאֲשֶׁר שָׁמְעוּ אוֹתוֹ מְדַבֵּר אֲלֵיהֶם בַּשָּׂפָה הָעִבְרִית הָיְתָה שְׁתִיקָתָם עֲמֻקָּה יוֹתֵר, וְהוּא הִמְשִׁיךְ -- 3 "אִישׁ יְהוּדִי אֲנִי, יְלִיד טַרְסוֹס אֲשֶׁר בְּקִילִיקְיָה, אֲבָל גֻּדַּלְתִּי בָּעִיר הַזֹּאת לְרַגְלֵי גַּמְלִיאֵל. חֻנַּכְתִּי עַל-פִּי דִּקְדּוּקֵי תּוֹרַת אֲבוֹתֵינוּ וְהָיִיתִי קַנַּאי לֵאלֹהִים כְּמוֹ כֻּלְּכֶם הַיּוֹם. 4 רָדַפְתִּי אֶת הַדֶּרֶך הַזֹּאת עַד מָוֶת. כָּבַלְתִּי אֲנָשִׁים וְנָשִׁים וְהִסְגַּרְתִּי אוֹתָם לְבָתֵּי כֶּלֶא, 5 דָּבָר שֶׁגַּם הַכֹּהֵן הַגָּדוֹל יָעִיד עָלַי וְכָל הַסַּנְהֶדְרִין; מֵהֶם גַּם קִבַּלְתִּי אִגְּרוֹת אֶל אַחֵינוּ, וְהָלַכְתִּי לְדַמֶּשֶׂק לַעֲצֹר אֶת הַנִּמְצָאִים שָׁם וְלַהֲבִיאָם בִּכְבָלִים לִירוּשָׁלַיִם כְּדֵי שֶׁיֵּעָנְשׁוּ. 6 וְהִנֵּה כְּשֶׁהָלַכְתִּי וְהִתְקָרַבְתִּי לְדַמֶּשֶׂק, בִּשְׁעַת הַצָּהֳרַיִם בְּעֵרֶךְ, לְפֶתַע פִּתְאוֹם נָגַהּ סְבִיבִי אוֹר גָּדוֹל מִן הַשָּׁמַיִם. 7 נָפַלְתִּי אַרְצָה וְשָׁמַעְתִּי קוֹל אוֹמֵר אֵלַי, 'שָׁאוּל, שָׁאוּל, לָמָּה אַתָּה רוֹדֵף אוֹתִי?' 8 שָׁאַלְתִּי, 'מִי אַתָּה, אֲדוֹנִי?' הֵשִׁיב וְאָמַר אֵלַי, 'אֲנִי יֵשׁוּעַ מִנָּצְרַת אֲשֶׁר אַתָּה רוֹדֵף.' 9 הָאֲנָשִׁים שֶׁהָיוּ אִתִּי אָמְנָם רָאוּ אֶת הָאוֹר, אַךְ לֹא שָׁמְעוּ אֶת קוֹלוֹ שֶׁל הַמְדַבֵּר אֵלַי.(דניאל י' ז') 10 אָמַרְתִּי, 'מֶה עָלַי לַעֲשׂוֹת, אֲדוֹנִי?' וְהָאָדוֹן אָמַר אֵלַי, 'קוּם, לֵךְ אֶל דַּמֶּשֶׂק וְשָׁם יֵאָמֵר לְךָ כָּל אֲשֶׁר יֻעַד לְךָ לַעֲשׂוֹת.' 11 מֵאַחַר שֶׁלֹּא יָכֹלְתִּי לִרְאוֹת מִפְּנֵי הַזֹּהַר שֶׁל אוֹתוֹ הָאוֹר, בָּאתִי לְדַמֶּשֶׂק כְּשֶׁמְּלַוַּי מוֹבִילִים אוֹתִי בַּיָּד. 12 אִישׁ אֶחָד, חֲנַנְיָה שְׁמוֹ, אִישׁ חָסִיד עַל-פִּי הַתּוֹרָה וְשֵׁם טוֹב לוֹ בְּקֶרֶב כָּל הַתּוֹשָׁבִים הַיְּהוּדִים, 13 בָּא אֵלַי וְעָמַד לְיָדִי. אָמַר אֵלַי, 'שָׁאוּל אָחִי, רְאֵה שׁוּב!' בְּאוֹתָהּ שָׁעָה רָאִיתִי אוֹתוֹ. 14 הוֹסִיף וְאָמַר, 'אֱלֹהֵי אֲבוֹתֵינוּ (שמות ג' ו') בָּחַר בְּךָ לָדַעַת אֶת רְצוֹנוֹ וְלִרְאוֹת אֶת הַצַּדִיק וְלִשְׁמֹעַ קוֹל מִפִּיו,(ישעיהו נג' יא') 15 כִּי תִּהְיֶה לוֹ לְעֵד אֶל כָּל בְּנֵי אָדָם עַל הַדְּבָרִים אֲשֶׁר רָאִיתָ וְשָׁמַעְתָּ. 16 וְעַכְשָׁו לָמָּה אַתָּה מִתְמַהְמֵהַּ? קוּם, הִטָּבֵל וְהִתְרַחֵץ מֵחֲטָאֶיךָ בְּקָרְאֲךָ בִּשְׁמוֹ.' 17 כַּאֲשֶׁר חָזַרְתִּי לִירוּשָׁלַיִם וְהִתְפַּלַּלְתִּי בַּמִּקְדָּשׁ הָיְתָה עָלַי יַד-יהוה 18 וְרָאִיתִי אוֹתוֹ אוֹמֵר אֵלַי, 'הִזְדָּרֵז וְצֵא מַהֵר מִירוּשָׁלַיִם, כִּי לֹא יְקַבְּלוּ אֶת עֵדוּתְךָ עָלַי.' 19 וַאֲנִי אָמַרְתִּי, 'אֲדוֹנִי, הֲרֵי הֵם יוֹדְעִים שֶׁאֲנִי הִשְׁלַכְתִּי אֶת הַמַּאֲמִינִים בְּךָ לַכֶּלֶא וְהִלְקֵיתִים בְּבָתֵּי הַכְּנֶסֶת; 20 וְכַאֲשֶׁר נִשְׁפַּךְ דַּם סְטֵפָנוֹס עֵדְךָ -- אַף אֲנִי עָמַדְתִּי שָׁם בְּהַסְכָּמָה לְמַּעֲשֶׂה וְשָׁמַרְתִּי אֶת הַבְּגָדִים שֶׁל הוֹרְגָיו.' 21 אַךְ הוּא אָמַר אֵלַי, 'לֵךְ, כִּי אֲנִי אֶשְׁלַח אוֹתְךָ הַרְחֵק -- אֶל הַגּוֹיִים.'"

שאול מוכנס למצודה

22 עַד לַדָּבָר הַזֶּה הִקְשִׁיבוּ אֵלָיו וְאָז הֵרִימוּ קוֹלָם וְקָרְאוּ: "הָסֵר מֵעַל הָאֲדָמָה אֶחָד שֶׁכָּזֶה! הוּא לֹא רָאוּי לִחְיוֹת!" 23 הֵם צָעֲקוּ, הִשְׁלִיכוּ אֶת בִּגְדֵיהֶם וְהֵטִילוּ עָפָר בָּאֲוִיר. 24 בִּתְגוּבָה צִוָּה שַׂר הָאֶלֶף לַהֲבִיאוֹ אֶל תּוֹךְ הַמְּצוּדָה וְהוֹרָה לַחֲקֹר אוֹתוֹ בְּמַלְקוֹת כְּדֵי לְבָרֵר מַדּוּעַ צָעֲקוּ עָלָיו כָּךְ. 25 אֲבָל בְּעֵת שֶׁקָּשְׁרוּ אוֹתוֹ בִּרְצוּעוֹת אָמַר שָׁאוּל אֶל שַׂר הַמֵּאָה הָעוֹמֵד לְיָדוֹ: "הַאִם מֻתָּר לָכֶם לְהַלְקוֹת אֶזְרָח רוֹמִי בְּלֹא מִשְׁפָּט?" 26 שָׁמַע שַׂר הַמֵּאָה, הָלַךְ אֶל שַׂר הָאֶלֶף וְאָמַר לוֹ: "מָה אַתָּה עוֹמֵד לַעֲשׂוֹת? הֲרֵי הָאִישׁ הַזֶּה רוֹמִי!" 27 בָּא שַׂר הָאֶלֶף וְאָמַר אֵלָיו: "הַגֵּד לִי, הַאִם אַתָּה רוֹמִי?" הֵשִׁיב שָׁאוּל: "כֵּן." 28 אָמַר שַׂר הָאֶלֶף: "אֲנִי בְּכֶסֶף רַב קָנִיתִי אֶת הָאֶזְרָחוּת הַזֹּאת." אָמַר שָׁאוּל: "אֲבָל אֲנִי נוֹלַדְתִּי בָּהּ." 29 בְּאוֹתוֹ רֶגַע הִרְפּוּ מִמֶּנּוּ הָאֲנָשִׁים שֶׁעָמְדוּ לַחֲקֹר אוֹתוֹ; אֲפִלּוּ שַׂר הָאֶלֶף פָּחַד כְּשֶׁנּוֹכַח כִּי הוּא רוֹמִי וְכִי כָּבַל אוֹתוֹ.

שאול לפני הסנהדרין

30 לְמָחֳרָת, כֵּיוָן שֶׁרָצָה לָדַעַת נְכוֹנָה מַדּוּעַ הֶאֱשִׁימוּהוּ בְּנֵי עַמּוֹ, הִתִּיר אֶת כְּבָלָיו וְצִוָּה עַל רָאשֵׁי הַכֹּהֲנִים וְכָל הַסַּנְהֶדְרִין לְהִתְאַסֵּף. הוּא הֵבִיא אֶת שָׁאוּל וְהֶעֱמִידוֹ לִפְנֵיהֶם.
