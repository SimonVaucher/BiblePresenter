לפרק הקודם

מעשי השליחים
פרק ז

נאום סטפנוס

1 שָׁאַל הַכֹּהֵן הַגָּדוֹל: הַאִם נְכוֹנִים הַדְּבָרִים הָאֵלֶּה? 2 הֵשִׁיב סְטֵפָנוֹס וְאָמַר: אֲנָשִׁים, אַחִים וְאָבוֹת, שִׁמְעוּ נָא. אֱלֹהֵי הַכָּבוֹד (תהילים כט' ג') נִרְאָה אֶל אַבְרָהָם אָבִינוּ בִּהְיוֹתוֹ בַּאֲרַם נַהֲרַיִם, לִפְנֵי שֶׁגָּר בְּחָרָן 3 וְאָמַר אֵלָיו, 'לֶךְ-לְךָ מֵאַרְצְךָ וּמִמּוֹלַדְתְּךָ וּבוֹא אֶל-הָאָרֶץ אֲשֶׁר אַרְאֶךָּ.'(בראשית יב' א') 4 אָז יָצָא מֵאֶרֶץ כַּשְׂדִּים וְגָר בְּחָרָן, וְאַחֲרֵי מוֹת אָבִיו הֶעֱבִיר אוֹתוֹ אֱלֹהִים מִשָּׁם אֶל הָאָרֶץ הַזֹּאת אֲשֶׁר אַתֶּם גָּרִים בָּהּ כָּעֵת.(בראשית יא' לב') 5 וְלֹא נָתַן לוֹ בָּהּ נָחֲלָה, אַף לֹא מִדְרַךְ כַּף רֶגֶל. הוּא הִבְטִיחַ לָתֵת אוֹתָהּ לַאֲחֻזָּה לוֹ וּלְזַרְעוֹ אַחֲרָיו בְּטֶרֶם הָיָה לוֹ בֵּן.(בראשית יז' ח') 6 כְּמוֹ כֵן אָמַר לוֹ הָאֱלֹהִים כִּי גֵר יִהְיֶה זַרְעוֹ בְּאֶרֶץ נָכְרִיָּה וְיַעֲבִידוּ אוֹתוֹ וִיעַנּוּהוּ אַרְבַּע מֵאוֹת שָׁנָה.(בראשית טו' יג') 7 'וְגַם אֶת-הַגּוֹי אֲשֶׁר יַעֲבֹדוּ דָּן אָנֹכִי',(בראשית טו' יד') אָמַר אֱלֹהִים, 'וְאַחֲרֵי-כֵן יֵצְאוּ וְיַעַבְדוּנִי בַּמָּקוֹם הַזֶּה.'(שמות ג' יב') 8 גַּם נָתַן לוֹ אֶת בְּרִית הַמִּילָה וּכְשֶׁהוֹלִיד אֶת יִצְחָק מָל אוֹתוֹ בַּיּוֹם הַשְּׁמִינִי; וְיִצְחָק אֶת יַעֲקֹב, וְיַעֲקֹב אֶת שְׁנֵים-עָשָׂר הָאָבוֹת.(בראשית יז' י'-יא'; כא' א'-ד') 9 הָאָבוֹת קִנְּאוּ בְּיוֹסֵף וּמָכְרוּ אוֹתוֹ מִצְרַיְמָה, אַךְ אֱלֹהִים הָיָה אִתּוֹ (בראשית לז' יא'; לז' כח'; לט' ב') 10 וְהִצִּיל אוֹתוֹ מִכָּל צָרוֹתָיו; גַּם נָתַן לוֹ חֵן וְחָכְמָה לִפְנֵי פַּרְעֹה מֶלֶךְ מִצְרַיִם, וְהַלָּה מִנָּה אוֹתוֹ עַל מִצְרַיִם וְעַל כָּל בֵּיתוֹ.(בראשית מא' מ') 11 אֲבָל רָעָב בָּא עַל כָּל אֶרֶץ מִצְרַיִם וְאֶרֶץ כְּנַעַן וְגַם צָרָה גְּדוֹלָה, וַאֲבוֹתֵינוּ לֹא מָצְאוּ אֹכֶל. 12 כַּאֲשֶׁר שָׁמַע יַעֲקֹב כִּי יֵשׁ חִטָּה בְּמִצְרַיִם, שָׁלַח אֶת אֲבוֹתֵינוּ בַּפַּעַם הָרִאשׁוֹנָה.(בראשית מב' ב') 13 בַּפַּעַם הַשְּׁנִיָּה הִתְוַדַּע יוֹסֵף אֶל אֶחָיו, וּמוֹצָאוֹ שֶׁל יוֹסֵף נִתְגַּלָּה לְפַרְעֹה.(בראשית מה' ג'; מה' טז') 14 שָׁלַח יוֹסֵף וְקָרָא לְיַעֲקֹב אָבִיו וּלְכָל מִשְׁפַּחְתוֹ, שִׁבְעִים וְחָמֵשׁ נְפָשׁוֹת.(בראשית מו' כז') 15 יַעֲקֹב יָרַד מִצְרַיְמָה וְשָׁם מֵת הוּא וַאֲבוֹתֵינוּ.(בראשית מו' ו'; מט'-לג'; שמות א' ו') 16 הֵם הוּבְאוּ לִשְׁכֶם וְנִטְמְנוּ בַּקֶּבֶר שֶׁקָּנָה אַבְרָהָם בִּמְחִיר כֶּסֶף מִבְּנֵי חֲמוֹר אֲשֶׁר בִּשְׁכֶם.(יהושוע כד' לב'; בראשית לג' יט') 17 כְּשֶׁקָּרַב מוֹעֵד קִיּוּם הַהַבְטָחָה שֶׁנִּשְׁבַּע אֱלֹהִים לְאַבְרָהָם,(***) גָּדַל הָעָם וְהִתְרַבָּה בְּמִצְרַיִם, 18 עַד שֶׁקָּם מֶלֶךְ-חָדָשׁ אֲשֶׁר לֹא-יָדַע אֶת-יוֹסֵף.(שמות א' ח') 19 זֶה הִתְחַכֵּם לְעַמֵּנוּ וְהֵרַע לַאֲבוֹתֵינוּ בְּהַכְרִיחוֹ אוֹתָם לְהַפְקִיר אֶת עוֹלְלֵיהֶם כְּדֵי שֶׁלֹּא יִחְיוּ.(שמות א' כב') 20 בָּעֵת הַהִיא נוֹלַד מֹשֶׁה, שֶׁהָיָה טוֹב בְּעֵינֵי אֱלֹהִים. שְׁלוֹשָׁה חֳדָשִׁים גֻּדַּל בְּבֵית אָבִיו (שמות ב' ב') 21 וְכַאֲשֶׁר הֻפְקַר אָסְפָה אוֹתוֹ בַּת פַּרְעֹה וְגִדְּלָה אוֹתוֹ לָהּ לְבֵן.(שמות ב' ג'; ב' י') 22 מֹשֶׁה לֻמַּד בְּכָל חָכְמַת הַמִּצְרִים וְהָיָה בֶּן-חַיִל בִּדְבָרָיו וּבְמַעֲשָׂיו. 23 כְּשֶׁמָּלְאוּ לוֹ אַרְבָּעִים שָׁנָה עָלָה עַל לִבּוֹ לִפְקֹד אֶת אֶחָיו בְּנֵי יִשְׂרָאֵל. 24 כִּרְאוֹתוֹ אִישׁ אֶחָד מֻכֶּה הֵגֵן עָלָיו וְנָקַם אֶת נִקְמָתוֹ שֶׁל הַנִּפְגָּע בַּהֲמִיתוֹ אֶת הַמִּצְרִי.(שמות ב' יא'-יב') 25 הוּא חָשַׁב שֶׁאֶחָיו יָבִינוּ כִּי עַל-יָדָיו יִתֵּן לָהֶם אֱלֹהִים תְּשׁוּעָה, אֶלָּא שֶׁהֵם לֹא הֵבִינוּ. 26 בַּיּוֹם הַשֵּׁנִי הוֹפִיעַ בֵּינֵיהֶם כַּאֲשֶׁר רָבוּ, וּבִקֵּשׁ לְפַיֵּס אוֹתָם לְשָׁלוֹם בְּאָמְרוֹ, אֲנָשִׁים אַחִים אַתֶּם, לָמָּה תָּרֵעוּ זֶה לָזֶה? 27 אֲבָל הַפּוֹגֵעַ בְּרֵעֵהוּ הָדַף אוֹתוֹ וְאָמַר, 'מִי שָׂמְךָ לְשַׂר וְשֹׁפֵט עָלֵינוּ?(שמות ב' יג') 28 הַלְהָרְגֵנִי אַתָּה אֹמֵר כַּאֲשֶׁר הָרַגְתָּ אֶתְמוֹל אֶת-הַמִּצְרִי?'(שמות ב' יד') 29 בִּגְלַל הַדָּבָר הַזֶּה בָּרַח מֹשֶׁה וְהָיָה לְגֵר בְּאֶרֶץ מִדְיָן וְשָׁם הוֹלִיד שְׁנֵי בָּנִים.(שמות יח' ב'-ג') 30 כַּעֲבֹר אַרְבָּעִים שָׁנָה נִרְאָה אֵלָיו מַלְאָךְ בְּמִדְבַּר הַר סִינַי, בְּלַבַּת אֵשׁ מִתּוֹךְ הַסְּנֶה.(שמות ג' ב') 31 רָאָה מֹשֶׁה וְתָמַהּ עַל הַמַּרְאֶה, אַךְ כְּשֶׁנִּגַּשׁ לְהַבִּיט הָיָה אֵלָיו קוֹל אֲדֹנָי: 32 'אָנֹכִי אֱלֹהֵי אֲבֹתֶיךָ, אֱלֹהֵי אַבְרָהָם, יִצְחָק וְיַעֲקֹב.'(שמות ג' ו') מֹשֶׁה נֶחֱרַד וְלֹא הֵעֵז לְהַבִּיט. 33 אָמַר אֵלָיו אֱלֹהִים, 'שַׁל-נְעָלֶיךָ מֵעַל רַגְלֶיךָ, כִּי הַמָּקוֹם אֲשֶׁר אַתָּה עוֹמֵד עָלָיו אַדְמַת-קֹדֶשׁ הוּא.(שמות ג' ה') 34 רָאֹה רָאִיתִי אֶת-עֳנִי עַמִּי אֲשֶׁר בְּמִצְרַיִם וְאֶת נַאֲקָתָם שָׁמַעְתִּי וָאֵרֵד לְהַצִּילָם; וְעַתָּה לְכָה וְאֶשְׁלָחֲךָ מִצְרָיְמָה.'(שמות ב' כד'; ג' ז'-י') 35 מֹשֶׁה זֶה, אֲשֶׁר כָּפְרוּ בּוֹ בְּאָמְרָם 'מִי שָׂמְךָ לְשַׂר וְשֹׁפֵט'(***) -- אוֹתוֹ שָׁלַח אֱלֹהִים בְּיַד הַמַּלְאָךְ הַנִּרְאֶה אֵלָיו בַּסְּנֶה, לִהְיוֹת לְשַׂר וְגוֹאֵל. 36 הוּא הוֹצִיאָם בַּעֲשׂוֹתוֹ אוֹתוֹת וּמוֹפְתִים בְּאֶרֶץ מִצְרַיִם וּבְיַם סוּף, וְכֵן גַּם בַּמִּדְבָּר בְּמֶשֶׁךְ אַרְבָּעִים שָׁנָה.(תהילים קה' כז'-לו'; נחמיה ט' יא') 37 זֶהוּ אוֹתוֹ מֹשֶׁה אֲשֶׁר אָמַר לִבְנֵי יִשְׂרָאֵל, נָבִיא מִקֶּרֶב אֲחֵיכֶם, כָּמוֹנִי, יָקִים לָכֶם אֱלֹהִים.(דברים יח' טו') 38 הוּא הָאִישׁ שֶׁהָיָה בַּקָּהָל בַּמִּדְבָּר עִם הַמַּלְאָךְ הַדּוֹבֵר אֵלָיו בְּהַר סִינַי וְעִם אֲבוֹתֵינוּ, הָאִישׁ שֶׁקִּבֵּל דִּבְרֵי חַיִּים לָתֵת לָנוּ.(שמות כד' יח') 39 וַאֲבוֹתֵינוּ לֹא רָצוּ לִשְׁמֹעַ בְּקוֹלוֹ כִּי אִם דָּחוּ אוֹתוֹ וּבִלְבָבָם שָׁבוּ מִצְרָיְמָה,(במדבר יד' ג') 40 בְּאָמְרָם אֶל אַהֲרֹן 'עֲשֵׂה-לָנוּ אֱלֹהִים אֲשֶׁר יֵלְכוּ לְפָנֵינוּ, כִּי-זֶה מֹשֶׁה אֲשֶׁר הֶעֱלָנוּ מֵאֶרֶץ מִצְרַיִם לֹא יָדַעְנוּ מֶה-הָיָה לוֹ.'(שמות לב' א'; לב' כג') 41 וְאָכֵן עָשׂוּ עֵגֶל בְּאוֹתָם יָמִים וְהִקְרִיבוּ זֶבַח לָאֱלִיל, וְשָׂמְחוּ בְּמַעֲשֵׂי יְדֵיהֶם.(שמות לב' ב'-ו') 42 אָז פָּנָה הָאֱלֹהִים וּמָסַר אוֹתָם לְפֻלְחַן צְבָא הַשָּׁמַיִם כַּכָּתוּב בְּסֵפֶר הַנְּבִיאִים, 'הַזְּבָחִים וּמִנְחָה הִגַּשְׁתֶּם-לִי בַמִּדְבָּר אַרְבָּעִים שָׁנָה, בֵּית יִשְׂרָאֵל? 43 וּנְשָׂאתֶם אֶת סֻכַּת מֹלֶךְ וְאֶת כּוֹכַב אֱלֹהֵיכֶם רֵיפָן, הַצְּלָמִים אֲשֶׁר עֲשִׂיתֶם לְהִשְׁתַּחֲווֹת לָהֶם. וְהִגְלֵיתִי אֶתְכֶם מֵהָלְאָה לְבָבֶל.'(עמוס ה' כה'-כז') 44 מִשְׁכַּן הָעֵדוּת הָיָה לַאֲבוֹתֵינוּ בַּמִּדְבָּר, כְּמוֹ שֶׁצִּוָּה הַמְדַבֵּר אֶל מֹשֶׁה לַעֲשׂוֹתוֹ עַל-פִּי הַתַּבְנִית שֶׁרָאָה.(שמות כה' מ') 45 אֲבוֹתֵינוּ קִבְּלוּ אוֹתוֹ וּבְהַנְהָגַת יְהוֹשֻׁעַ הֱבִיאוּהוּ לַאֲחֻזַּת הַגּוֹיִים אֲשֶׁר גֵּרְשָׁם אֱלֹהִים מִפְּנֵי אֲבוֹתֵינוּ, עַד שֶׁהִגִּיעוּ יְמֵי דָּוִד.(יהושוע ג' יד') 46 הוּא מָצָא חֵן בְּעֵינֵי אֱלֹהִים וּבִקֵּשׁ לִמְצֹא מִשְׁכָּן לֵאלֹהֵי יַעֲקֹב.(תהילים קלב' ה') 47 אָמְנָם שְׁלֹמֹה בָּנָה לוֹ בַּיִת,(מלכים א', ו' א'; ח' כז') 48 אַךְ הָעֶלְיוֹן אֵינֶנּוּ שׁוֹכֵן בְּהֵיכָלוֹת מַעֲשֵׂי יָדַיִם, כְּמוֹ שֶׁאוֹמֵר הַנָּבִיא, 49 'הַשָּׁמַיִם כִּסְאִי וְהָאָרֶץ הֲדֹם רַגְלָי. אֵי-זֶה בַיִת אֲשֶׁר תִּבְנוּ-לִי, נְאֻם יהוה, וְאֵי-זֶה מָקוֹם מְנוּחָתִי? 50 הֲלֹא אֶת-כָּל-אֵלֶּה יָדִי עָשָׂתָה.'(ישעיהו סו' א'-ב') 51 קְשֵׁי עֹרֶף וְעַרְלֵי לֵב וְאָזְנַיִם,(ירמיהו ט' כד'-כה'; יחזקאל מד' ו'-ח') תָּמִיד מִתְנַגְּדִים אַתֶּם לְרוּחַ הַקֹּדֶשׁ, כַּאֲבוֹתֵיכֶם כֵּן גַּם אַתֶּם.(ירמיהו ו' י'; ישעיהו סג' י') 52 אֶת מִי מֵהַנְּבִיאִים לֹא רָדְפוּ אֲבוֹתֵיכֶם? גַּם הָרְגוּ אֶת אֵלֶּה שֶׁהִגִּידוּ מֵרֹאשׁ אֶת דְּבַר בּוֹא הַצַּדִּיק אֲשֶׁר כָּעֵת אַתֶּם הֱיִיתֶם מַסְגִּירָיו וְהוֹרְגָיו;(נחמיה ט' כו') 53 אַתֶּם שֶׁקִּבַּלְתֶּם אֶת הַתּוֹרָה בְּאֶמְצָעוּת מַלְאָכִים וְלֹא שְׁמַרְתֶּם אוֹתָהּ."

סטפנוס נסקל

54 כְּשָׁמְעָם אֶת הַדְּבָרִים הָאֵלֶּה כָּעֲסוּ עַד עֹמֶק לִבָּם וְחָרְקוּ עָלָיו שִׁנֵּיהֶם;(תהילים לה' טז') 55 וְהוּא מָלֵא רוּחַ הַקֹּדֶשׁ וּמַבִּיט הַשָּׁמַיְמָה. רָאָה אֶת כְּבוֹד אֱלֹהִים וְאֶת יֵשׁוּעַ עוֹמֵד לִימִין הָאֱלֹהִים 56 וְאָמַר: "הִנְנִי רוֹאֶה אֶת הַשָּׁמַיִם פְּתוּחִים וְאֶת בֶּן-הָאָדָם עוֹמֵד לִימִין הָאֱלֹהִים." 57 אַךְ הֵם צָעֲקוּ בְּקוֹל גָּדוֹל, בְּאָטְמָם אֶת אָזְנֵיהֶם, וּכְאִישׁ אֶחָד הִסְתָּעֲרוּ עָלָיו, 58 הֲדָפוּהוּ אֶל מִחוּץ לָעִיר וְסָקְלוּ אוֹתוֹ בַּאֲבָנִים. הָעֵדִים הִנִּיחוּ אֶת בִּגְדֵיהֶם לְרַגְלֵי בָּחוּר אֶחָד, שָׁאוּל שְׁמוֹ, 59 וּסְטֵפָנוֹס נִסְקַל כְּשֶׁהוּא זוֹעֵק וְאוֹמֵר: "אֲדוֹנִי יֵשׁוּעַ, קַבֵּל אֶת רוּחִי!" 60 כְּמוֹ כֵן כָּרַע עַל בִּרְכָּיו וְצָעַק בְּקוֹל גָּדוֹל: "אֲדֹנָי, אַל תִּפְקֹד עֲלֵיהֶם אֶת הַחֵטְא הַזֶּה!" אָמַר זֹאת וָמֵת.
