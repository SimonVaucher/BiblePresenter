פרק כג

שאול לפני הסנהדרין -- המשך

1 הִבִּיט שָׁאוּל בַּסַּנְהֶדְרִין וְאָמַר: "אֲנָשִׁים אַחִים, אֲנִי בְּמַצְפּוּן נָקִי לְגַמְרֵי הִתְהַלַּכְתִּי לִפְנֵי הָאֱלֹהִים עַד הַיּוֹם הַזֶּה." 2 אָז צִוָּה חֲנַנְיָה הַכֹּהֵן הַגָּדוֹל עַל הָעוֹמְדִים לְיָדוֹ לְהַכּוֹתוֹ עַל פִּיו. 3 אָמַר לוֹ שָׁאוּל: "עָתִיד אֱלֹהִים לְהַכּוֹת אוֹתְךָ, קִיר מְטֻיָּח שֶׁכְּמוֹתְךָ! אַתָּה יוֹשֵׁב לִשְׁפֹּט אוֹתִי עַל-פִּי הַתּוֹרָה וּמְצַוֶּה לְהַכּוֹתֵנִי שֶׁלֹּא כַּתּוֹרָה?" 4 אָמְרוּ הָעוֹמְדִים לְיָדוֹ: "אֶת הַכֹּהֵן הַגָּדוֹל לֵאלֹהִים אַתָּה מְחָרֵף!" 5 הֵשִׁיב שָׁאוּל: "אַחַי, לֹא יָדַעְתִּי שֶׁהוּא הַכֹּהֵן הַגָּדוֹל; הֵן כָּתוּב 'נָשִׂיא בְעַמְּךָ לֹא תָאֹר.'"(שמות כב' כז') 6 הוֹאִיל וְשָׁאוּל יָדַע כִּי חֵלֶק מֵהֶם צְדוֹקִים וְחֵלֶק פְּרוּשִׁים הִכְרִיז לִפְנֵי הַסַּנְהֶדְרִין: "אֲנָשִׁים אַחִים, פָּרוּשׁ בֶּן-פָּרוּשׁ אָנֹכִי וְעַל הַתִּקְוָה לִתְחִיַּת הַמֵּתִים אֲנִי נִשְׁפָּט!" 7 כַּאֲשֶׁר אָמַר זֹאת הֵחֵלָּה הִתְנַצְּחוּת בֵּין הַפְּרוּשִׁים לַצְּדוֹקִים וְהַקָּהָל נִתְפַּלֵג לִשְׁנַיִם, 8 כִּי הַצְּדוֹקִים אוֹמְרִים שְׁאֵין תְּחִיָּה וְאֵין מַלְאָךְ וְאֵין רוּחַ, אַךְ הַפְּרוּשִׁים מוֹדִים בְכָל אֵלֶּה. 9 הָיְתָה הֲמֻלָּה גְּדוֹלָה וְאָז קָמוּ סוֹפְרִים אֲחָדִים מִכַּת הַפְּרוּשִׁים וְטָעֲנוּ בְּתֹקֶף: אֵין אָנוּ מוֹצְאִים שׁוּם רַע בָּאִישׁ הַזֶּה. וְאִם רוּחַ דִּבְּרָה אֵלָיו אוֹ מַלְאָךְ [אַל נִלָּחֵם בֵּאלֹהִים]." 10 הַהִתְנַצְּחוּת גָּבְרָה מְאֹד וְשַׂר הָאֶלֶף חָשַׁשׁ שֶׁמָּא יְשַׁסְּעוּ אֶת שָׁאוּל. לְפִיכָךְ צִוָּה עַל הַחַיָּלִים לָרֶדֶת וְלַחֲטֹף אוֹתוֹ מִתּוֹכָם וְלַהֲבִיאוֹ אֶל הַמְּצוּדָה. 11 בַּלַּיְלָה נִצַּב הָאָדוֹן עַל שָׁאוּל וְאָמַר: "הִתְחְזֵּק! כְּמוֹ שֶׁהֲעִידוֹתָ עָלַי בִּירוּשָׁלַיִם כָּךְ עָלֶיךָ לְהָעִיד גַּם בְּרוֹמָא."

התנכלות להמית את שאול

12 בַּבֹּקֶר חָבְרוּ יְהוּדִים יַחַד וְקִבְּלוּ עֲלֵיהֶם בִּשְׁבוּעָה שֶׁלֹּא יֹאכְלוּ וְלֹא יִשְׁתּוּ עַד אֲשֶׁר יַהַרְגוּ אֶת שָׁאוּל. 13 מִסְפַּר הַמִּשְׁתַּתְּפִים בַּמְּזִמָּה הַזֹּאת עָלָה עַל אַרְבָּעִים. 14 הֵם בָּאוּ אֶל רָאשֵׁי הַכֹּהֲנִים וְאֶל הַזְּקֵנִים וְאָמְרוּ: "קִבַּלְנוּ עָלֵינוּ שְׁבוּעָה שֶׁלֹּא לִטְעֹם מְאוּמָה עַד שֶׁנַּהֲרֹג אֶת שָׁאוּל. 15 וְעַכְשָׁו אַתֶּם וְהַסַּנְהֶדְרִין הַגִּידוּ לְשַׂר הָאֶלֶף שֶׁיּוֹרִיד אוֹתוֹ אֲלֵיכֶם, כְּאִלּוּ שֶׁכַּוָּנַתְכֶם לְבָרֵר בְּיֶתֶר דִּיּוּק אֶת עִנְיָנוֹ, וַאֲנַחְנוּ מוּכָנִים לַהֲרֹג אוֹתוֹ לִפְנֵי שֶׁיַגִּיעַ." 16 כַּאֲשֶׁר שָׁמַע בֶּן אֲחוֹתוֹ שֶׁל שָׁאוּל עַל-דְּבַר הַמַּאֲרָב הָלַךְ וְנִכְנַס לַמְּצוּדָה וְהוֹדִיעַ לְשָׁאוּל. 17 קָרָא שָׁאוּל לְאֶחָד מִשָּׂרֵי הַמֵּאוֹת וְאָמַר: "הָבֵא נָא אֶת הַבָּחוּר הַזֶּה אֶל שַׂר הָאֶלֶף, כִּי יֵשׁ לוֹ דָּבָר לְהוֹדִיעוֹ." 18 הוּא הֵבִיא אוֹתוֹ אֶל שַׂר הָאֶלֶף וְאָמַר: "הָאָסִיר שָׁאוּל קָרָא לִי וּבִקֵּשׁ מִמֶּנִּי לְהָבִיא אֵלֶיךָ אֶת הַבָּחוּר הַזֶּה, כִּי יֵשׁ לוֹ אֵיזֶה דָּבָר לוֹמַר לְךָ." 19 הֶחֱזִיק שַׂר הָאֶלֶף בְּיָדוֹ, הוֹבִיל אוֹתוֹ הַצִּדָּה וְשָׁאַל: "מַה יֵּשׁ לְךָ לְהוֹדִיעַ לִי?" 20 הֵשִׁיב הַבָּחוּר: "רָאשֵׁי הַיְּהוּדִים נוֹעֲצוּ יַחְדָּיו לְבַקֵּשׁ מִמְּךָ לְהוֹרִיד מָחָר אֶת שָׁאוּל אֶל הַסַּנְהֶדְרִין כְּאִלּוּ הֵם מִתְכַּוְּנִים לַחֲקֹר אֶת עִנְיָנוֹ בְּיֶתֶר דִּיּוּק. 21 לָכֵן אַל נָא תִּשְׁמַע לָהֶם, כִּי יוֹתֵר מֵאַרְבָּעִים אִישׁ מֵהֶם אוֹרְבִים לוֹ. הַלָּלוּ קִבְּלוּ עֲלֵיהֶם בִּשְׁבוּעָה שֶׁלֹּא לֶאֱכֹל וְלֹא לִשְׁתּוֹת עַד שֶׁיַּהַרְגוּ אוֹתוֹ וְכָעֵת הֵם מוּכָנִים בְּצִפִּיָּה לְהַבְטָחָה מִמְּךָ." 22 שִׁלַּח שַׂר הָאֶלֶף אֶת הַבָּחוּר בְּצַוּוֹתוֹ עָלָיו: "אַל תַּגִּיד לְאִישׁ שֶׁהוֹדַעְתָּ לִי אֶת הַדָּבָר הַזֶּה!"

שאול מועבר לקיסריה

23 הוּא קָרָא לִשְׁנֵי שָׂרֵי מֵאוֹת וְאָמַר: "הָכִינוּ מָאתַיִם חַיָּלִים כְּדֵי שֶׁיֵּצְאוּ בְּשָׁעָה תֵּשַׁע בַּלַּיְלָה וְיֵלְכוּ לְקֵיסַרְיָה, וְכֵן גַּם שִׁבְעִים פָּרָשִׁים וּמָאתַיִם נוֹשְׂאֵי כִּידוֹן, 24 וּתְנוּ בְּהֵמוֹת לְהַרְכִּיב אֶת שָׁאוּל וְלַהֲבִיאוֹ בְּשָׁלוֹם אֶל פֶלִיקְס הַנָּצִיב." 25 כְּמוֹ כֵן כָּתַב אִגֶּרֶת בְּזֶה הַלָּשׁוֹן: 26 "קְלוֹדְיוֹס לִיסִיאַס אֶל הוֹד מַעֲלָתוֹ הַנָּצִיב פֶלִיקְס -- שָׁלוֹם. 27 אֶת הָאִישׁ הַזֶּה תָּפַס הָמוֹן יְהוּדִי וְעָמַד לַהֲרֹג אוֹתוֹ. בָּאתִי עִם הַחַיָּלִים וְחִלַּצְתִּי אוֹתוֹ בְּהִוָּדַע לִי שֶׁהוּא רוֹמִי. 28 הוֹאִיל וְחָפַצְתִּי לָדַעַת מַדּוּעַ הֶאֱשִׁימוּ אוֹתוֹ, הוֹרַדְתִּיו אֶל הַסַּנְהֶדְרִין שֶׁלָּהֶם 29 וּמָצָאתִי שֶׁהֶאֱשִׁימוּ אוֹתוֹ בְּעִנְיָנִים הַנּוֹגְעִים לְתוֹרָתָם, אַךְ לֹא הָיְתָה שׁוּם אַשְׁמָה שֶׁתְּחַיֵּב מָוֶת אוֹ מַאֲסָר. 30 כַּאֲשֶׁר הֻגַּד לִי כִּי נִרְקֶמֶת מְזִּמָּה נֶגֶד הָאִישׁ, שְׁלַחְתִּיו אֵלֶיךָ מִיָּד וְגַם צִוִּיתִי עַל מַאֲשִׁימָיו לוֹמַר לְפָנֶיךָ מַה שֶּׁיֵּשׁ לָהֶם לוֹמַר עָלָיו. הֱיֵה שָׁלוֹם." 31 עַל-פִּי הַפְּקֻדָּה שֶׁצֻּוּוּ לָקְחוּ הַחַיָּלִים אֶת שָׁאוּל וֶהֱבִיאוּהוּ בַּלַּיְלָה לְאַנְטִיפַּטְרִיס. 32 לְמָחֳרָת הִנִּיחוּ לַפָּרָשִׁים לְהַמְשִׁיךְ אִתּוֹ וְחָזְרוּ לַמְּצוּדָה. 33 אֵלֶּה בָּאוּ לְקֵיסַרְיָה, מָסְרוּ אֶת הָאִגֶּרֶת לַנָּצִיב וְהֶעֱמִידוּ אֶת שָׁאוּל לְפָנָיו. 34 הַנָּצִיב קָרָא אֶת הָאִגֶּרֶת וְשָׁאַל מֵאֵיזֶה אֵזוֹר מִמְשָׁל הוּא. כְּשָׁמְעוֹ שֶׁהוּא מִקִּילִיקְיָה אָמַר: 35 "אֶשְׁמָע אוֹתְךָ כַּאֲשֶׁר גַּם מַאֲשִׁימֶיךָ יָבוֹאוּ הֵנָּה", וְצִוָּה לְהַחֲזִיקוֹ בְּאַרְמוֹן הוֹרְדוֹס.
