לפרק הקודם

אגרת שאול השניה אל טימותיאוס
פרק ב

חייל טוב של המשיח ישוע

1 וְאַתָּה, בְּנִי, הִתְחַזֵּק בַּחֶסֶד אֲשֶׁר בַּמָּשִׁיחַ יֵשׁוּעַ.(תהילים כז' יד') 2 וְהַדְּבָרִים שֶׁשָּׁמַעְתָּ מִמֶּנִּי בְּמַעֲמַד עֵדִים רַבִּים, הַפְקֵד אוֹתָם בִּידֵי אֲנָשִׁים נֶאֱמָנִים הַמֻּכְשָׁרִים לְלַמֵּד גַּם אֲנָשִׁים אֲחֵרִים. 3 הִשְׁתַּתֵּף בַּסֵּבֶל כְּחַיָּל טוֹב שֶׁל הַמָּשִׁיחַ יֵשׁוּעַ. 4 אִישׁ הַמְשָׁרֵת בַּצָּבָא לֹא יִתְעָרֵב בְּעִסְקֵי הַחַיִּים, וְזֹאת כְּדֵי לְהַשְׂבִּיעַ אֶת רְצוֹן מְפַקְּדוֹ. 5 וְגַם הַמִּשְׁתַּתֵּף בְּתַחֲרוּת לֹא יֻכְתַּר אִם לֹא יִתְחָרֶה לְפִי הַכְּלָלִים. 6 הָאִכָּר הֶעָמֵל, מִן הַדִּין שֶׁיִּהְיֶה רִאשׁוֹן לָקַחַת חֵלֶק מִן הַפְּרִי. 7 שִׂים לִבְּךָ לְמַה שֶּׁאֲנִי אוֹמֵר, וְהָאָדוֹן יִתֵּן לְךָ בִּינָה בַּכֹּל.(תהילים צד' י') 8 זְכֹר אֶת יֵשׁוּעַ הַמָּשִׁיחַ שֶׁנֵּעוֹר מִן הַמֵּתִים, אֲשֶׁר הוּא מִזֶּרַע דָּוִד כִּדְבַר בְּשׂוֹרָתִי, 9 הִיא הַבְּשׂוֹרָה שֶׁבַּעֲבוּרָהּ אֲנִי סוֹבֵל רָעוֹת עַד כְּדֵי שִׁבְתִּי בִּכְבָלִים כְּעוֹשֵׂה עָוֶל. אוּלָם דְּבַר הָאֱלֹהִים אֵינֶנּוּ בִּכְבָלִים. 10 עַל כֵּן אֲנִי סוֹבֵל הַכֹּל לְמַעַן הַנִּבְחָרִים, כְּדֵי שֶׁגַּם הֵם יַשִּׂיגוּ תְּשׁוּעָה בַּמָּשִׁיחַ יֵשׁוּעַ עִם כְּבוֹד עוֹלָמִים. 11 מְהֵימָן הַדָּבָר: אִם מַתְנוּ אִתּוֹ, גַּם נִחְיֶה אִתּוֹ; 12 אִם נַחֲזִיק מַעֲמָד, גַּם נִמְלֹךְ אִתּוֹ; אִם נִתְכַּחֵשׁ, גַּם הוּא יִתְכַּחֵשׁ לָנוּ; 13 אִם אֵינֶנּוּ נֶאֱמָנִים, הוּא נִשְׁאָר נֶאֱמָן, כִּי לֹא יוּכַל לְהִתְכַּחֵשׁ לְעַצְמוֹ.(במדבר כג' יט'; תהילים פט' לד')

פועל נאמן לפני האלהים

14 הַזְכֵּר לָהֶם אֶת הַדְּבָרִים הָאֵלֶּה וְהַזְהֵר אוֹתָם לִפְנֵי הָאֱלֹהִים, שֶׁלֹּא יִתְוַכְּחוּ עַל דְּבָרִים שֶׁאֵין בָּהֵם מוֹעִיל וְאֵינָם אֶלָּא הוֹרְסִים אֶת הַשּׁוֹמְעִים. 15 הֱיֵה שָׁקוּד לְהִתְיַצֵּב נֶאֱמָן לִפְנֵי אֱלֹהִים -- פּוֹעֵל לֹא יֵבוֹשׁ, הַמְחַלֵּק נְכוֹנָה אֶת דְּבַר הָאֱמֶת. 16 רְחַק מִדִּבּוּרִים תְּפֵלִים וְחַסְרֵי קְדֻשָּׁה, כִּי בַּעֲלֵיהֶם יוֹסִיפוּ רֶשַׁע עַל רֶשַׁע, 17 וּדְבָרָם כְּרָקָב יֹאכַל; עִמָּהֶם נִמְנִים הִימֵנֵאוֹס וּפִילִיטוֹס 18 אֲשֶׁר תָּעוּ מִן הָאֱמֶת, בְּאָמְרָם כִּי תְּחִיַּת הַמֵּתִים כְּבָר הָיְתָה, וְהֵם מְמוֹטְטִים אֱמוּנַת כַּמָּה אֲנָשִׁים. 19 אֲבָל הַיְסוֹד הֶחָזָק שֶׁהִנִּיחַ אֱלֹהִים עוֹמֵד אֵיתָן וְיֵשׁ לוֹ הַחוֹתָם הַזֶּה: "וְיֹדַע יהוה אֶת אֲשֶׁר לוֹ",(במדבר טז' ה'; נחום א' ז') וְגַם "יָסוּר מֵעָוֶל כָּל הַקּוֹרֵא בְּשֵׁם יהוה".(ישעיהו נב' יא') 20 הִנֵּה בְּבַיִת גָּדוֹל לֹא רַק כְּלֵי זָהָב וְכֶסֶף, אֶלָּא גַּם כְּלֵי עֵץ וְחֶרֶס, וּמֵהֶם כַּמָּה לְכָבוֹד וְכַמָּה לְקָלוֹן. 21 לְפִיכָךְ אִם אִישׁ יְטַהֵר אֶת עַצְמוֹ מֵאֵלֶּה יִהְיֶה כְּלִי לְכָבוֹד, מְקֻדָּשׁ וּמוֹעִיל לְבַעַל הַבַּיִת וּמוּכָן לְכָל מַעֲשֶׂה טוֹב. 22 בְּרַח לְךָ מִתַּאֲווֹת הַנְּעוּרִים וּרְדֹף צֶדֶק, אֱמוּנָה, אַהֲבָה וְשָׁלוֹם, עִם כָּל הַקּוֹרְאִים אֶל יהוה בְּלֵב טָהוֹר. 23 רְחַק מִשְּׁאֵלוֹת אֱוִילִיּוֹת וְנִבְעָרוֹת; אַתָּה יוֹדֵעַ שֶׁהֵן גּוֹרְמוֹת מְרִיבוֹת. 24 וְעֶבֶד הָאָדוֹן מִן הַדִּין שֶׁלֹּא יָרִיב, אֶלָּא יְהֵא נֹחַ לַכֹּל, מֻכְשָׁר לְלַמֵּד, סַבְלָן, 25 וּמוֹכִיחַ בַּעֲנָוָה אֶת הַמִּתְנַגְּדִים; אוּלַי יִתֵּן לָהֶם הָאֱלֹהִים לַחֲזֹר בִּתְשׁוּבָה כְּדֵי שֶׁיַּכִּירוּ אֶת הָאֱמֶת 26 וְיִתְפַּכְּחוּ וְיֵצְאוּ מִמַּלְכֹּדֶת הַשָּׂטָן אֲשֶׁר לְכָדָם כִּרְצוֹנוֹ.
