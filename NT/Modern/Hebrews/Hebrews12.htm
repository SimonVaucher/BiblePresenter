לפרק הקודם

אגרת אל העברים
פרק יב

מוסר ה'

1 לָכֵן גַּם אֲנַחְנוּ, אֲשֶׁר עֲנַן עֵדִים כָּזֶה אוֹפֵף אוֹתָנוּ, נָסִירָה נָא כָּל מַעֲמָסָה וְגַם אֶת הַחֵטְא הַלּוֹכֵד עַל נְקַלָּה, וּבְסַבְלָנוּת נָרוּצָה אֶת הַמֵּרוֹץ הֶעָרוּךְ לְפָנֵינוּ, 2 בְּהַבִּיטֵנוּ אֶל יֵשׁוּעַ מְכוֹנֵן הָאֱמוּנָה וּמַשְׁלִימָהּ,(תהילים כה' טו') אֲשֶׁר בְּעַד הַשִּׂמְחָה הָעֲרוּכָה לְפָנָיו סָבַל אֶת הַצְּלָב וּבָז לַחֶרְפָּה, וְיָשַׁב לִימִין כִּסֵּא הָאֱלֹהִים.(תהילים קי' א') 3 הִתְבּוֹנְנוּ אֶל הָאֶחָד שֶׁסָּבַל הִתְנַגְּדוּת כָּזֹאת מֵאֵת הַחוֹטְאִים, כְּדֵי שֶׁלֹּא תִּתְעַיְּפוּ וְלֹא תִּרְפֶּה נַפְשְׁכֶם. 4 עֲדַיִן לֹא נִלְחַמְתֶּם עַד לַדָּם בְּמַאֲבַקְכֶם עִם הַחֵטְא, 5 וּשְׁכַחְתֶּם אֶת מִלּוֹת הַנֹּחַם הַמְדַבְּרוֹת אֲלֵיכֶם כְּאֶל בָּנִים: "מוּסַר יהוה, בְּנִי, אַל-תִּמְאָס, וְאַל-תָּקֹץ בְּתוֹכַחְתּוֹ;(איוב ה' יז') 6 כִּי אֶת אֲשֶׁר יֶאֱהַב יהוה יוֹכִיחַ, יַכְאִיב אֶת-בֵּן יִרְצֶה."(משלי ג' יא'-יב') 7 לְשֵׁם מוּסָר סוֹבְלִים אַתֶּם, וֵאלֹהִים נוֹהֵג עִמָּכֶם כְּעִם בָּנִים; כִּי אֵיזֶהוּ הַבֵּן אֲשֶׁר אָבִיו אֵינוֹ מְיַסֵּר אוֹתוֹ? 8 אַךְ אִם אַתֶּם בְּלֹא מוּסָר, אֲשֶׁר הַכֹּל נוֹטְלִים בּוֹ חֶלְקָם, אֲזַי מַמְזֵרִים אַתֶּם וְלֹא בָּנִים. 9 וְעוֹד, הָאָבוֹת שֶׁהוֹלִידוּ אֶת גּוּפֵנוּ יִסְּרוּ אוֹתָנוּ וְאָנוּ יָרֵאנוּ מֵהֶם, הֲלֹא כָּל שֶׁכֵּן נִכָּנַע לִפְנֵי אֲבִי הָרוּחוֹת וְנִחְיֶה? 10 הֵם אָמְנָם יִסְּרוּנוּ יָמִים מְעַטִּים עַל-פִּי מַה שֶּׁהֵם רָאוּ לְנָכוֹן, אַךְ הוּא לְהוֹעִיל מְיַסֵּר אוֹתָנוּ, לְמַעַן יִהְיֶה לָנוּ חֵלֶק בִּקְדֻשָּׁתוֹ. 11 כָּל מוּסָר בְּעֵת בּוֹאוֹ אֵינוֹ נֶחְשָׁב לְשִׂמְחָה כִּי אִם לְעֵצֶב, אֲבָל אַחֲרֵי כֵן יִתֵּן פְּרִי שָׁלוֹם וּצְדָקָה לַמִּתְחַנְּכִים בּוֹ. 12 עַל כֵּן אַמְּצוּ יָדַיִם רָפוֹת וְחַזְּקוּ בִּרְכַּיִם כּוֹשְׁלוֹת;(ישעיהו לה' ג') 13 וְכוֹנְנוּ נְתִיבוֹת יְשָׁרוֹת לְרַגְלֵיכֶם,(משלי ד' כו') לְמַעַן לֹא יִסְטֶה הַצּוֹלֵעַ כִּי אִם יֵרָפֵא.

הזהרות

14 רִדְפוּ שָׁלוֹם עִם כָּל אָדָם, וְגַם אֶת הַקְּדֻשָּׁה שֶׁבִּלְעָדֶיהָ לֹא יִרְאֶה אִישׁ אֶת הָאָדוֹן.(תהילים לד' טו') 15 הִשָּׁמְרוּ פֶּן יִגָּרַע אִישׁ מֵחֶסֶד אֱלֹהִים; פֶּן יַפְרִיעַ אֶתְכֶם שֹׁרֶשׁ פּוֹרֶה מְרִירוּת (דברים כט' יז') וְיִטָּמְאוּ בּוֹ רַבִּים; 16 פֶּן יִהְיֶה אִישׁ זוֹנֶה אוֹ חֲסַר קְדֻשָּׁה כְּעֵשָׂו אֲשֶׁר בְּנָזִיד אֶחָד מָכַר אֶת בְּכוֹרָתוֹ.(בראשית כה' לג') 17 יוֹדְעִים אַתֶּם כִּי אַחֲרֵי כֵן, כַּאֲשֶׁר רָצָה לָרֶשֶׁת אֶת הַבְּרָכָה, נִדְחָה; הוּא לֹא מָצָא מָקוֹם לַחֲזֹר בּוֹ, אַף כִּי בִּקֵּשׁ זֹאת בִּדְמָעוֹת.(בראשית כז' לח') 18 הֵן לֹא בָּאתֶם אֶל הַר מוּחָשִׁי וּבוֹעֵר בָּאֵשׁ, וְלֹא אֶל אֲפֵלָה וְחֹשֶׁךְ וּסְעָרָה,(דברים ד' יא') 19 וְלֹא לְקוֹל שׁוֹפָר וּלְקוֹל דְּבָרִים אֲשֶׁר שׁוֹמְעָיו בִּקְשׁוּ שֶׁלֹּא יוֹסִיף לְדַבֵּר עִמָּם,(שמות יט' יט'; כ' טו') 20 כִּי לֹא יָכְלוּ לָשֵׂאת אֶת מַה שֶּׁצֻּוָּה, "גַּם אִם בְּהֵמָה תִּגַּע בָּהָר סָקוֹל תִּסָּקֵל";(שמות יט' יב'-יג') 21 וְהַמַּרְאֶה הָיָה נוֹרָא מְאֹד עַד כִּי מֹשֶׁה אָמַר, "יָרֵא אֲנִי וְחָרֵד."(דברים ט' יט') 22 אַתֶּם בָּאתֶם אֶל הַר צִיּוֹן וְאֶל עִיר אֱלֹהִים חַיִּים, אֶל יְרוּשָׁלַיִם הַשְּׁמֵימִית וְאֶל רִבְבוֹת הַמַּלְאָכִים, אֶל עֲצֶרֶת,(דניאל ז' י') 23 אֶל קְהִלַּת הַבְּכוֹרִים הַכְּתוּבִים בַּשָּׁמַיִם,(דניאל יב' א') אֶל אֱלֹהִים שׁוֹפֵט הַכֹּל, אֶל רוּחוֹת הַצַּדִּיקִים שֶׁנַּעֲשׂוּ מֻשְׁלָמִים,(בראשית יח' כה') 24 אֶל יֵשׁוּעַ מְתַוֵּךְ הַבְּרִית הַחֲדָשָׁה וְאֶל דַּם הַהַזָּיָה הַמֵּיטִיב דַּבֵּר מִדַּם הֶבֶל.(ירמיהו לא' ל'-לד') 25 לָכֵן הִזָּהֲרוּ פֶּן תְּסָרְבוּ לַמְדַבֵּר, שֶׁכֵּן אִם לֹא נִמְלְטוּ אֵלֶּה שֶׁסֵּרְבוּ לַמְדַבֵּר עַל הָאָרֶץ, עַל אַחַת כַּמָּה וְכַמָּה לֹא נִמָּלֵט אֲנַחְנוּ אִם נָסוּר מֵהַמְדַבֵּר מִן הַשָּׁמַיִם, 26 אֲשֶׁר קוֹלוֹ הִרְעִישׁ אָז אֶת הָאָרֶץ וְעַתָּה הִבְטִיחַ בְּאָמְרוֹ, "עוֹד אַחַת וַאֲנִי מַרְעִישׁ לֹא אֶת-הָאָרֶץ בִּלְבַד, אֶלָּא גַּם אֶת-הַשָּׁמַיִם."(חגי ב' ו') 27 "עוֹד אַחַת" מְצַיֵּן אֶת הֲסָרַת הַדְּבָרִים הַמִּתְמוֹטְטִים -- כְּגוֹן דְּבָרִים שֶׁנַּעֲשׂוּ -- לְמַעַן יִשָּׁאֲרוּ הַדְּבָרִים אֲשֶׁר לֹא יִמּוֹטוּ.(דניאל ב' מד') 28 וְכֵיוָן שֶׁאָנוּ מְקַבְּלִים מַלְכוּת אֲשֶׁר לֹא תִּמּוֹט, נִהְיֶה נָא מַכִּירֵי טוֹבָה וּבְהַכָּרָה זֹאת נַעֲבֹד אֶת אֱלֹהִים כִּרְצוֹנוֹ, בַּחֲסִידוּת וּבְיִרְאָה; 29 כִּי אֱלֹהֵינוּ אֵשׁ אוֹכְלָה הוּא.(דברים ד' כד')
