לפרק הקודם

אגרת אל העברים
פרק ה

ישוע הכהן הגדול העליון -- המשך

1 כָּל כֹּהֵן גָּדוֹל הַלָּקוּחַ מִקֶּרֶב בְּנֵי אָדָם, מְמֻנֶּה לְמַעַן בְּנֵי אָדָם עַל עִנְיְנֵי אֱלֹהִים כְּדֵי לְהַקְרִיב מְנָחוֹת וּזְבָחִים עַל חֲטָאִים. 2 הוּא יָכוֹל לַחֲמֹל עַל הַשּׁוֹגְגִים וְהַתּוֹעִים, מִשּׁוּם שֶׁגַּם הוּא מֻקָּף חֻלְשָׁה. 3 מִסִּבָּה זֹאת הוּא חַיָּב לְהַקְרִיב עַל חֲטָאִים גַּם בְּעַד הָעָם וְגַם בְּעַד עַצְמוֹ.(ויקרא ט' ז') 4 וְאֵין אִישׁ לוֹקֵחַ לְעַצְמוֹ אֶת הַכָּבוֹד זוּלָתִי הַנִּקְרָא מִטַּעַם אֱלֹהִים, כְּאַהֲרֹן.(שמות כח' א') 5 כֵּן גַּם הַמָּשִׁיחַ לֹא נָטַל לְעַצְמוֹ אֶת הַכָּבוֹד לִהְיוֹת כֹּהֵן גָּדוֹל, אֶלָּא קִבֵּל זֹאת מֵאֵת הָאוֹמֵר אֵלָיו, "בְּנִי אַתָּה, אֲנִי הַיּוֹם יְלִדְתִּיךָ";(תהילים ב' ז') 6 כְּמוֹ שֶׁגַּם בְּמָקוֹם אַחֵר הוּא אוֹמֵר, "אַתָּה-כֹהֵן לְעוֹלָם עַל-דִּבְרָתִי מַלְכִּי-צֶדֶק."(תהילים קי' ד') 7 וּבִימֵי הֱיוֹתוֹ בְּגוּף בָּשָׂר וָדָם, הִקְרִיב תְּפִלּוֹת וְתַחֲנוּנִים בִּצְעָקָה גְּדוֹלָה וּבִדְמָעוֹת אֶל הַיָּכוֹל לְהוֹשִׁיעוֹ מִמָּוֶת, וְאָמְנָם נִשְׁמַע בִּגְלַל יִרְאַת הָאֱלֹהִים שֶׁבּוֹ.(תהילים כב' כה') 8 וְאַף כִּי הָיָה הַבֵּן, מִסִּבְלוֹתָיו לָמַד לְצַיֵּת.(ישעיהו נ' ה') 9 וְכַאֲשֶׁר הֻשְׁלַם הָיָה לְמָקוֹר שֶׁל יְשׁוּעַת עוֹלָמִים לְכָל הַמְצַיְּתִים לוֹ,(ישעיהו מה' יז') 10 וֵאלֹהִים קָרָא לוֹ, "כֹּהֵן גָּדוֹל עַל-דִּבְרָתִי מַלכִּי-צֶדֶק."(***)

יש להתקדם ולא לסגת

11 עַל זֹאת יֵשׁ לָנוּ לוֹמַר דְּבָרִים רַבִּים, אַךְ קָשֶׁה לְהַסְבִּירָם מִפְּנֵי שֶׁכָּבְדוּ אָזְנֵיכֶם. 12 הִנֵּה בִּזְמַן שֶׁעֲלֵיכֶם לִהְיוֹת כְּבָר מוֹרִים, שׁוּב צְרִיכִים אַתֶּם לְמִישֶׁהוּ שֶׁיְּלַמֵּד אֶתְכֶם אֶת רֵאשִׁית יְסוֹדוֹת דִּבְרֵי אֱלֹהִים וְנִהְיֵיתֶם זְקוּקִים לֶחָלָב וְלֹא לְמָזוֹן מוּצָק. 13 הֵן כָּל מִי שֶׁמַּאֲכָלוֹ חָלָב אֵינֶנּוּ מֵבִין בְּדִבְרֵי צֶדֶק, כִּי עוֹדֶנּוּ תִּינוֹק. 14 אַךְ הַמַּאֲכָל הַמּוּצָק הוּא לַמְבֻגָּרִים, אֵלֶּה אֲשֶׁר חוּשֵׁיהֶם הֻרְגְּלוּ עַל-יְדֵי הַנִּסָּיוֹן הַמַּעֲשִׂי לְהַבְחִין בֵּין טוֹב לְרַע.(מלכים א', ג' ט')
